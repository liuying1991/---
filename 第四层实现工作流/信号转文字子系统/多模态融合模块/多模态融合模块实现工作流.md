### 核心功能实现

#### 1. 多模态融合管理器实现

创建 `src/core/fusion/fusion_engine.py`：
```python
from typing import Dict, List, Optional, Any, Tuple
import logging
import time
import numpy as np
from abc import ABC, abstractmethod

from src.models.multimodal_models import MultiModalData, ModalityFeature
from src.models.fusion_models import FusionResult, FusionConfig
from src.core.fusion.fusion_strategies import FusionStrategyFactory
from src.core.fusion.attention_mechanism import AttentionMechanism
from src.utils.evaluation import FusionEvaluator

logger = logging.getLogger(__name__)

class FusionEngine:
    """多模态融合引擎"""
    
    def __init__(self, config: Dict[str, Any]):
        """初始化融合引擎"""
        self.config = config
        self.strategy_factory = FusionStrategyFactory()
        self.attention_mechanism = AttentionMechanism(config.get('attention', {}))
        self.evaluator = FusionEvaluator(config.get('evaluation', {}))
        
        # 默认融合策略
        self.default_strategy = config.get('default_strategy', 'attention_hybrid')
        
        # 性能监控
        self.performance_metrics = {
            'fusion_count': 0,
            'total_fusion_time': 0.0,
            'average_fusion_time': 0.0
        }
        
        logger.info(f"Initialized FusionEngine with strategy: {self.default_strategy}")
    
    def fuse(self, 
             modal_features: Dict[str, ModalityFeature], 
             strategy: Optional[str] = None,
             context: Optional[Dict[str, Any]] = None) -> FusionResult:
        """
        融合多模态特征
        
        Args:
            modal_features: 各模态特征字典
            strategy: 融合策略，如果为None则使用默认策略
            context: 融合上下文信息
            
        Returns:
            FusionResult: 融合结果
        """
        start_time = time.time()
        
        try:
            # 获取融合策略
            fusion_strategy = strategy or self.default_strategy
            strategy_instance = self.strategy_factory.get_strategy(fusion_strategy)
            
            # 验证输入
            self._validate_modal_features(modal_features)
            
            # 预处理特征
            processed_features = self._preprocess_features(modal_features)
            
            # 计算模态权重
            modality_weights = self._calculate_modality_weights(
                processed_features, 
                context
            )
            
            # 执行融合
            if fusion_strategy.startswith('attention'):
                fused_vector, attention_weights = strategy_instance.fuse(
                    processed_features, 
                    modality_weights,
                    self.attention_mechanism
                )
            else:
                fused_vector = strategy_instance.fuse(
                    processed_features, 
                    modality_weights
                )
                attention_weights = None
            
            # 创建融合结果
            result = FusionResult(
                fused_vector=fused_vector,
                modality_weights=modality_weights,
                attention_weights=attention_weights,
                strategy=fusion_strategy,
                confidence=self._calculate_confidence(fused_vector, processed_features),
                metadata={
                    'modalities': list(modal_features.keys()),
                    'feature_dimensions': {
                        mod: feat.feature_vector.shape 
                        for mod, feat in modal_features.items()
                    },
                    'fusion_time': time.time() - start_time,
                    'context': context or {}
                }
            )
            
            # 更新性能指标
            self._update_performance_metrics(time.time() - start_time)
            
            logger.info(f"Successfully fused {len(modal_features)} modalities using {fusion_strategy}")
            return result
            
        except Exception as e:
            logger.error(f"Error during fusion: {str(e)}")
            raise
    
    def batch_fuse(self, 
                   modal_features_batch: List[Dict[str, ModalityFeature]], 
                   strategy: Optional[str] = None,
                   context: Optional[Dict[str, Any]] = None) -> List[FusionResult]:
        """
        批量融合多模态特征
        
        Args:
            modal_features_batch: 多模态特征批次
            strategy: 融合策略
            context: 融合上下文
            
        Returns:
            List[FusionResult]: 融合结果列表
        """
        results = []
        
        for i, modal_features in enumerate(modal_features_batch):
            try:
                result = self.fuse(modal_features, strategy, context)
                result.metadata['batch_index'] = i
                results.append(result)
            except Exception as e:
                logger.error(f"Error fusing batch item {i}: {str(e)}")
                # 创建错误结果
                error_result = FusionResult(
                    fused_vector=np.array([]),
                    modality_weights={},
                    attention_weights=None,
                    strategy=strategy or self.default_strategy,
                    confidence=0.0,
                    metadata={
                        'batch_index': i,
                        'error': str(e),
                        'status': 'failed'
                    }
                )
                results.append(error_result)
        
        return results
    
    def stream_fuse(self, 
                    modal_features_stream, 
                    strategy: Optional[str] = None,
                    context: Optional[Dict[str, Any]] = None):
        """
        流式融合多模态特征
        
        Args:
            modal_features_stream: 多模态特征流
            strategy: 融合策略
            context: 融合上下文
            
        Yields:
            FusionResult: 融合结果
        """
        for modal_features in modal_features_stream:
            try:
                result = self.fuse(modal_features, strategy, context)
                yield result
            except Exception as e:
                logger.error(f"Error in stream fusion: {str(e)}")
                # 创建错误结果
                error_result = FusionResult(
                    fused_vector=np.array([]),
                    modality_weights={},
                    attention_weights=None,
                    strategy=strategy or self.default_strategy,
                    confidence=0.0,
                    metadata={
                        'error': str(e),
                        'status': 'failed'
                    }
                )
                yield error_result
    
    def _validate_modal_features(self, modal_features: Dict[str, ModalityFeature]) -> None:
        """验证模态特征"""
        if not modal_features:
            raise ValueError("No modal features provided")
        
        if len(modal_features) > self.config.get('max_modalities', 10):
            raise ValueError(f"Too many modalities: {len(modal_features)} > {self.config.get('max_modalities', 10)}")
        
        # 验证每个特征
        for modality, feature in modal_features.items():
            if not isinstance(feature, ModalityFeature):
                raise TypeError(f"Invalid feature type for {modality}: {type(feature)}")
            
            if feature.feature_vector is None or feature.feature_vector.size == 0:
                raise ValueError(f"Empty feature vector for {modality}")
    
    def _preprocess_features(self, modal_features: Dict[str, ModalityFeature]) -> Dict[str, ModalityFeature]:
        """预处理模态特征"""
        processed_features = {}
        
        for modality, feature in modal_features.items():
            # 标准化特征向量
            normalized_vector = self._normalize_feature_vector(feature.feature_vector)
            
            # 创建处理后的特征
            processed_feature = ModalityFeature(
                modality=feature.modality,
                feature_vector=normalized_vector,
                feature_type=feature.feature_type,
                quality_score=feature.quality_score,
                timestamp=feature.timestamp,
                metadata=feature.metadata
            )
            
            processed_features[modality] = processed_feature
        
        return processed_features
    
    def _normalize_feature_vector(self, feature_vector: np.ndarray) -> np.ndarray:
        """标准化特征向量"""
        # L2归一化
        norm = np.linalg.norm(feature_vector)
        if norm > 0:
            return feature_vector / norm
        return feature_vector
    
    def _calculate_modality_weights(self, 
                                   modal_features: Dict[str, ModalityFeature], 
                                   context: Optional[Dict[str, Any]]) -> Dict[str, float]:
        """计算模态权重"""
        weights = {}
        
        # 基于质量分数的基础权重
        for modality, feature in modal_features.items():
            weights[modality] = feature.quality_score
        
        # 归一化权重
        total_weight = sum(weights.values())
        if total_weight > 0:
            weights = {mod: w / total_weight for mod, w in weights.items()}
        else:
            # 如果所有质量分数都为0，则使用均等权重
            equal_weight = 1.0 / len(modal_features)
            weights = {mod: equal_weight for mod in modal_features.keys()}
        
        # 根据上下文调整权重
        if context and 'modality_preferences' in context:
            preferences = context['modality_preferences']
            for modality, preference in preferences.items():
                if modality in weights:
                    weights[modality] *= preference
            
            # 重新归一化
            total_weight = sum(weights.values())
            if total_weight > 0:
                weights = {mod: w / total_weight for mod, w in weights.items()}
        
        return weights
    
    def _calculate_confidence(self, 
                            fused_vector: np.ndarray, 
                            modal_features: Dict[str, ModalityFeature]) -> float:
        """计算融合置信度"""
        # 基于模态质量分数的平均值
        quality_scores = [feat.quality_score for feat in modal_features.values()]
        avg_quality = sum(quality_scores) / len(quality_scores)
        
        # 基于融合向量的范数
        vector_norm = np.linalg.norm(fused_vector)
        
        # 综合置信度
        confidence = (avg_quality * 0.7) + (min(vector_norm, 1.0) * 0.3)
        
        return float(confidence)
    
    def _update_performance_metrics(self, fusion_time: float) -> None:
        """更新性能指标"""
        self.performance_metrics['fusion_count'] += 1
        self.performance_metrics['total_fusion_time'] += fusion_time
        self.performance_metrics['average_fusion_time'] = (
            self.performance_metrics['total_fusion_time'] / 
            self.performance_metrics['fusion_count']
        )
    
    def get_performance_metrics(self) -> Dict[str, Any]:
        """获取性能指标"""
        return self.performance_metrics.copy()
    
    def reset_performance_metrics(self) -> None:
        """重置性能指标"""
        self.performance_metrics = {
            'fusion_count': 0,
            'total_fusion_time': 0.0,
            'average_fusion_time': 0.0
        }
```

#### 2. 融合策略实现

创建 `src/core/fusion/fusion_strategies.py`：
```python
from abc import ABC, abstractmethod
from typing import Dict, List, Optional, Any, Tuple
import numpy as np
import logging

from src.models.multimodal_models import ModalityFeature
from src.core.fusion.attention_mechanism import AttentionMechanism

logger = logging.getLogger(__name__)

class FusionStrategy(ABC):
    """融合策略抽象基类"""
    
    @abstractmethod
    def fuse(self, 
             modal_features: Dict[str, ModalityFeature], 
             modality_weights: Dict[str, float],
             attention_mechanism: Optional[AttentionMechanism] = None) -> np.ndarray:
        """融合多模态特征"""
        pass

class EarlyFusionStrategy(FusionStrategy):
    """早期融合策略"""
    
    def __init__(self, config: Dict[str, Any] = None):
        """初始化早期融合策略"""
        self.config = config or {}
        self.concat_dim = self.config.get('concat_dim', 0)
        logger.info("Initialized EarlyFusionStrategy")
    
    def fuse(self, 
             modal_features: Dict[str, ModalityFeature], 
             modality_weights: Dict[str, float],
             attention_mechanism: Optional[AttentionMechanism] = None) -> np.ndarray:
        """执行早期融合"""
        # 收集所有特征向量
        feature_vectors = []
        
        for modality in sorted(modal_features.keys()):
            feature = modal_features[modality]
            weighted_feature = feature.feature_vector * modality_weights[modality]
            feature_vectors.append(weighted_feature)
        
        # 拼接所有特征向量
        fused_vector = np.concatenate(feature_vectors, axis=self.concat_dim)
        
        return fused_vector

class LateFusionStrategy(FusionStrategy):
    """晚期融合策略"""
    
    def __init__(self, config: Dict[str, Any] = None):
        """初始化晚期融合策略"""
        self.config = config or {}
        self.fusion_method = self.config.get('fusion_method', 'weighted_average')
        logger.info(f"Initialized LateFusionStrategy with method: {self.fusion_method}")
    
    def fuse(self, 
             modal_features: Dict[str, ModalityFeature], 
             modality_weights: Dict[str, float],
             attention_mechanism: Optional[AttentionMechanism] = None) -> np.ndarray:
        """执行晚期融合"""
        # 收集所有特征向量
        feature_vectors = []
        weights = []
        
        for modality in sorted(modal_features.keys()):
            feature = modal_features[modality]
            feature_vectors.append(feature.feature_vector)
            weights.append(modality_weights[modality])
        
        # 确保所有特征向量具有相同的形状
        max_dims = max(vec.shape[0] for vec in feature_vectors)
        normalized_vectors = []
        
        for vec in feature_vectors:
            if vec.shape[0] < max_dims:
                # 填充较小的向量
                padded_vec = np.zeros(max_dims)
                padded_vec[:vec.shape[0]] = vec
                normalized_vectors.append(padded_vec)
            else:
                normalized_vectors.append(vec)
        
        # 执行融合
        if self.fusion_method == 'weighted_average':
            # 加权平均
            weights_array = np.array(weights).reshape(-1, 1)
            fused_vector = np.average(normalized_vectors, axis=0, weights=weights.flatten())
        elif self.fusion_method == 'max_pooling':
            # 最大池化
            stacked_vectors = np.stack(normalized_vectors, axis=0)
            fused_vector = np.max(stacked_vectors, axis=0)
        elif self.fusion_method == 'attention_weighted':
            # 注意力加权
            attention_weights = np.array(weights).reshape(-1, 1)
            fused_vector = np.sum(normalized_vectors * attention_weights, axis=0)
        else:
            raise ValueError(f"Unknown fusion method: {self.fusion_method}")
        
        return fused_vector

class HybridFusionStrategy(FusionStrategy):
    """混合融合策略"""
    
    def __init__(self, config: Dict[str, Any] = None):
        """初始化混合融合策略"""
        self.config = config or {}
        self.early_ratio = self.config.get('early_ratio', 0.5)
        self.late_ratio = 1.0 - self.early_ratio
        logger.info(f"Initialized HybridFusionStrategy with early_ratio: {self.early_ratio}")
    
    def fuse(self, 
             modal_features: Dict[str, ModalityFeature], 
             modality_weights: Dict[str, float],
             attention_mechanism: Optional[AttentionMechanism] = None) -> np.ndarray:
        """执行混合融合"""
        # 早期融合
        early_fusion = EarlyFusionStrategy(self.config)
        early_result = early_fusion.fuse(modal_features, modality_weights)
        
        # 晚期融合
        late_fusion = LateFusionStrategy(self.config)
        late_result = late_fusion.fuse(modal_features, modality_weights)
        
        # 确保两个结果具有相同的形状
        max_dims = max(early_result.shape[0], late_result.shape[0])
        
        if early_result.shape[0] < max_dims:
            early_padded = np.zeros(max_dims)
            early_padded[:early_result.shape[0]] = early_result
        else:
            early_padded = early_result
            
        if late_result.shape[0] < max_dims:
            late_padded = np.zeros(max_dims)
            late_padded[:late_result.shape[0]] = late_result
        else:
            late_padded = late_result
        
        # 混合两种融合结果
        fused_vector = (early_padded * self.early_ratio) + (late_padded * self.late_ratio)
        
        return fused_vector

class AttentionFusionStrategy(FusionStrategy):
    """注意力融合策略"""
    
    def __init__(self, config: Dict[str, Any] = None):
        """初始化注意力融合策略"""
        self.config = config or {}
        self.attention_heads = self.config.get('attention_heads', 8)
        self.dropout = self.config.get('dropout', 0.1)
        logger.info(f"Initialized AttentionFusionStrategy with heads: {self.attention_heads}")
    
    def fuse(self, 
             modal_features: Dict[str, ModalityFeature], 
             modality_weights: Dict[str, float],
             attention_mechanism: Optional[AttentionMechanism] = None) -> np.ndarray:
        """执行注意力融合"""
        if attention_mechanism is None:
            raise ValueError("Attention mechanism is required for attention fusion")
        
        # 收集所有特征向量
        feature_vectors = []
        modality_names = []
        
        for modality in sorted(modal_features.keys()):
            feature = modal_features[modality]
            feature_vectors.append(feature.feature_vector)
            modality_names.append(modality)
        
        # 应用注意力机制
        fused_vector, attention_weights = attention_mechanism.apply_attention(
            feature_vectors, 
            modality_weights,
            self.attention_heads,
            self.dropout
        )
        
        return fused_vector

class AttentionHybridFusionStrategy(FusionStrategy):
    """注意力混合融合策略"""
    
    def __init__(self, config: Dict[str, Any] = None):
        """初始化注意力混合融合策略"""
        self.config = config or {}
        self.early_ratio = self.config.get('early_ratio', 0.3)
        self.late_ratio = self.config.get('late_ratio', 0.3)
        self.attention_ratio = 1.0 - self.early_ratio - self.late_ratio
        logger.info(f"Initialized AttentionHybridFusionStrategy with ratios: early={self.early_ratio}, late={self.late_ratio}, attention={self.attention_ratio}")
    
    def fuse(self, 
             modal_features: Dict[str, ModalityFeature], 
             modality_weights: Dict[str, float],
             attention_mechanism: Optional[AttentionMechanism] = None) -> np.ndarray:
        """执行注意力混合融合"""
        if attention_mechanism is None:
            raise ValueError("Attention mechanism is required for attention hybrid fusion")
        
        # 早期融合
        early_fusion = EarlyFusionStrategy(self.config)
        early_result = early_fusion.fuse(modal_features, modality_weights)
        
        # 晚期融合
        late_fusion = LateFusionStrategy(self.config)
        late_result = late_fusion.fuse(modal_features, modality_weights)
        
        # 注意力融合
        attention_fusion = AttentionFusionStrategy(self.config)
        attention_result = attention_fusion.fuse(modal_features, modality_weights, attention_mechanism)
        
        # 确保所有结果具有相同的形状
        max_dims = max(
            early_result.shape[0], 
            late_result.shape[0], 
            attention_result.shape[0]
        )
        
        # 填充结果
        def pad_vector(vec):
            if vec.shape[0] < max_dims:
                padded = np.zeros(max_dims)
                padded[:vec.shape[0]] = vec
                return padded
            return vec
        
        early_padded = pad_vector(early_result)
        late_padded = pad_vector(late_result)
        attention_padded = pad_vector(attention_result)
        
        # 混合三种融合结果
        fused_vector = (
            early_padded * self.early_ratio + 
            late_padded * self.late_ratio + 
            attention_padded * self.attention_ratio
        )
        
        return fused_vector

class FusionStrategyFactory:
    """融合策略工厂"""
    
    _strategies = {
```

```python
from typing import List, Dict, Tuple, Optional, Any
import numpy as np
import torch
import torch.nn as nn
import torch.nn.functional as F
import logging

logger = logging.getLogger(__name__)

class MultiHeadAttention(nn.Module):
    """多头注意力机制"""
    
    def __init__(self, feature_dim: int, num_heads: int = 8, dropout: float = 0.1):
        """初始化多头注意力机制"""
        super(MultiHeadAttention, self).__init__()
        
        assert feature_dim % num_heads == 0, "feature_dim must be divisible by num_heads"
        
        self.feature_dim = feature_dim
        self.num_heads = num_heads
        self.head_dim = feature_dim // num_heads
        self.dropout = dropout
        
        # 线性变换层
        self.query = nn.Linear(feature_dim, feature_dim)
        self.key = nn.Linear(feature_dim, feature_dim)
        self.value = nn.Linear(feature_dim, feature_dim)
        
        # 输出层
        self.out = nn.Linear(feature_dim, feature_dim)
        
        # Dropout层
        self.dropout_layer = nn.Dropout(dropout)
        
        # 初始化权重
        self._init_weights()
    
    def _init_weights(self):
        """初始化权重"""
        for module in [self.query, self.key, self.value, self.out]:
            nn.init.xavier_uniform_(module.weight)
            nn.init.constant_(module.bias, 0)
    
    def forward(self, 
                query: torch.Tensor, 
                key: torch.Tensor, 
                value: torch.Tensor,
                mask: Optional[torch.Tensor] = None) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        前向传播
        
        Args:
            query: 查询张量 [batch_size, seq_len, feature_dim]
            key: 键张量 [batch_size, seq_len, feature_dim]
            value: 值张量 [batch_size, seq_len, feature_dim]
            mask: 掩码张量 [batch_size, seq_len, seq_len]
            
        Returns:
            output: 输出张量 [batch_size, seq_len, feature_dim]
            attention_weights: 注意力权重 [batch_size, num_heads, seq_len, seq_len]
        """
        batch_size, seq_len, _ = query.size()
        
        # 线性变换
        Q = self.query(query)  # [batch_size, seq_len, feature_dim]
        K = self.key(key)      # [batch_size, seq_len, feature_dim]
        V = self.value(value)  # [batch_size, seq_len, feature_dim]
        
        # 重塑为多头格式
        Q = Q.view(batch_size, seq_len, self.num_heads, self.head_dim).transpose(1, 2)
        K = K.view(batch_size, seq_len, self.num_heads, self.head_dim).transpose(1, 2)
        V = V.view(batch_size, seq_len, self.num_heads, self.head_dim).transpose(1, 2)
        
        # 计算注意力分数
        scores = torch.matmul(Q, K.transpose(-2, -1)) / np.sqrt(self.head_dim)
        
        # 应用掩码
        if mask is not None:
            mask = mask.unsqueeze(1).expand(-1, self.num_heads, -1, -1)
            scores = scores.masked_fill(mask == 0, -1e9)
        
        # 计算注意力权重
        attention_weights = F.softmax(scores, dim=-1)
        attention_weights = self.dropout_layer(attention_weights)
        
        # 应用注意力权重
        context = torch.matmul(attention_weights, V)
        
        # 重塑回原始格式
        context = context.transpose(1, 2).contiguous().view(
            batch_size, seq_len, self.feature_dim
        )
        
        # 输出层
        output = self.out(context)
        
        return output, attention_weights

class CrossModalityAttention(nn.Module):
    """跨模态注意力机制"""
    
    def __init__(self, modality_dims: Dict[str, int], num_heads: int = 8, dropout: float = 0.1):
        """初始化跨模态注意力机制"""
        super(CrossModalityAttention, self).__init__()
        
        self.modalities = list(modality_dims.keys())
        self.modality_dims = modality_dims
        self.num_heads = num_heads
        self.dropout = dropout
        
        # 为每个模态创建多头注意力层
        self.attention_layers = nn.ModuleDict()
        for modality in self.modalities:
            self.attention_layers[modality] = MultiHeadAttention(
                modality_dims[modality], 
                num_heads, 
                dropout
            )
        
        # 模态间交互层
        self.interaction_layers = nn.ModuleDict()
        for i, mod1 in enumerate(self.modalities):
            for j, mod2 in enumerate(self.modalities):
                if i != j:
                    self.interaction_layers[f"{mod1}_to_{mod2}"] = nn.Linear(
                        modality_dims[mod1], 
                        modality_dims[mod2]
                    )
        
        # 融合层
        max_dim = max(modality_dims.values())
        self.fusion_layer = nn.Linear(max_dim * len(self.modalities), max_dim)
        
        logger.info(f"Initialized CrossModalityAttention with modalities: {self.modalities}")
    
    def forward(self, 
                modality_features: Dict[str, torch.Tensor],
                modality_weights: Dict[str, float]) -> Tuple[torch.Tensor, Dict[str, torch.Tensor]]:
        """
        前向传播
        
        Args:
            modality_features: 模态特征字典
            modality_weights: 模态权重字典
            
        Returns:
            fused_features: 融合特征
            attention_weights: 注意力权重字典
        """
        batch_size = 1  # 假设批处理大小为1
        
        # 为每个模态添加序列维度
        modality_seqs = {}
        for modality, features in modality_features.items():
            if isinstance(features, np.ndarray):
                features = torch.from_numpy(features).float()
            
            # 确保特征是3D张量 [batch_size, seq_len, feature_dim]
            if features.dim() == 1:
                features = features.unsqueeze(0).unsqueeze(0)  # [1, 1, feature_dim]
            elif features.dim() == 2:
                features = features.unsqueeze(0)  # [1, seq_len, feature_dim]
            
            modality_seqs[modality] = features
        
        # 计算每个模态的自注意力
        self_attention_weights = {}
        for modality in self.modalities:
            if modality in modality_seqs:
                _, attn_weights = self.attention_layers[modality](
                    modality_seqs[modality],
                    modality_seqs[modality],
                    modality_seqs[modality]
                )
                self_attention_weights[modality] = attn_weights
        
        # 计算跨模态注意力
        cross_attention_weights = {}
        for modality in self.modalities:
            if modality in modality_seqs:
                query = modality_seqs[modality]
                
                # 使用其他模态作为键和值
                keys = []
                values = []
                for other_modality in self.modalities:
                    if other_modality != modality and other_modality in modality_seqs:
                        # 将其他模态的特征转换为当前模态的维度
                        transformed = self.interaction_layers[f"{other_modality}_to_{modality}"](
                            modality_seqs[other_modality]
                        )
                        keys.append(transformed)
                        values.append(transformed)
                
                if keys:
                    # 拼接所有键和值
                    key = torch.cat(keys, dim=1)
                    value = torch.cat(values, dim=1)
                    
                    # 计算跨模态注意力
                    _, attn_weights = self.attention_layers[modality](
                        query,
                        key,
                        value
                    )
                    cross_attention_weights[modality] = attn_weights
        
        # 融合所有模态特征
        max_seq_len = max(seq.size(1) for seq in modality_seqs.values())
        max_feature_dim = max(self.modality_dims.values())
        
        # 填充所有模态特征到相同大小
        padded_features = []
        for modality in self.modalities:
            if modality in modality_seqs:
                seq = modality_seqs[modality]
                
                # 填充序列长度
                if seq.size(1) < max_seq_len:
                    padding = torch.zeros(
                        batch_size, 
                        max_seq_len - seq.size(1), 
                        seq.size(2)
                    )
                    seq = torch.cat([seq, padding], dim=1)
                
                # 填充特征维度
                if seq.size(2) < max_feature_dim:
                    padding = torch.zeros(batch_size, max_seq_len, max_feature_dim - seq.size(2))
                    seq = torch.cat([seq, padding], dim=2)
                
                # 应用模态权重
                weighted_seq = seq * modality_weights.get(modality, 1.0)
                padded_features.append(weighted_seq)
        
        # 拼接所有模态特征
        if padded_features:
            concatenated = torch.cat(padded_features, dim=2)
            fused_features = self.fusion_layer(concatenated)
        else:
            fused_features = torch.zeros(batch_size, max_seq_len, max_feature_dim)
        
        # 合并注意力权重
        all_attention_weights = {**self_attention_weights, **cross_attention_weights}
        
        return fused_features, all_attention_weights

class AttentionMechanism:
    """注意力机制"""
    
    def __init__(self, config: Dict[str, Any]):
        """初始化注意力机制"""
        self.config = config
        self.num_heads = config.get('num_heads', 8)
        self.dropout = config.get('dropout', 0.1)
        
        # 初始化跨模态注意力
        self.cross_modal_attention = None
        
        logger.info(f"Initialized AttentionMechanism with {self.num_heads} heads")
    
    def apply_attention(self, 
                       feature_vectors: List[np.ndarray], 
                       modality_weights: Dict[str, float],
                       num_heads: Optional[int] = None,
                       dropout: Optional[float] = None) -> Tuple[np.ndarray, Dict[str, np.ndarray]]:
        """
        应用注意力机制
        
        Args:
            feature_vectors: 特征向量列表
            modality_weights: 模态权重字典
            num_heads: 注意力头数
            dropout: Dropout率
            
        Returns:
            fused_vector: 融合后的特征向量
            attention_weights: 注意力权重字典
        """
        num_heads = num_heads or self.num_heads
        dropout = dropout or self.dropout
        
        # 创建模态名称列表
        modality_names = [f"modality_{i}" for i in range(len(feature_vectors))]
        
        # 创建模态维度字典
        modality_dims = {
            name: vec.shape[0] if len(vec.shape) == 1 else vec.shape[1]
            for name, vec in zip(modality_names, feature_vectors)
        }
        
        # 初始化跨模态注意力
        if self.cross_modal_attention is None or self.cross_modal_attention.modalities != modality_names:
            self.cross_modal_attention = CrossModalityAttention(
                modality_dims, 
                num_heads, 
                dropout
            )
        
        # 创建模态特征字典
        modality_features = {
            name: vec for name, vec in zip(modality_names, feature_vectors)
        }
        
        # 应用跨模态注意力
        with torch.no_grad():
            fused_features, attention_weights = self.cross_modal_attention(
                modality_features, 
                modality_weights
            )
        
        # 转换为numpy数组
        fused_vector = fused_features.squeeze().cpu().numpy()
        
        # 转换注意力权重为numpy数组
        attention_weights_np = {}
        for modality, weights in attention_weights.items():
            # 取平均注意力权重
            avg_weights = weights.mean(dim=1).squeeze().cpu().numpy()
            attention_weights_np[modality] = avg_weights
        
        return fused_vector, attention_weights_np
    
    def calculate_modality_importance(self, 
                                     feature_vectors: List[np.ndarray], 
                                     target_vector: np.ndarray) -> Dict[str, float]:
        """
        计算模态重要性
        
        Args:
            feature_vectors: 特征向量列表
            target_vector: 目标向量
            
        Returns:
            importance_scores: 模态重要性分数字典
        """
        modality_names = [f"modality_{i}" for i in range(len(feature_vectors))]
        importance_scores = {}
        
        for i, (name, vec) in enumerate(zip(modality_names, feature_vectors)):
            # 计算特征向量与目标向量的余弦相似度
            dot_product = np.dot(vec, target_vector)
            norm_product = np.linalg.norm(vec) * np.linalg.norm(target_vector)
            
            if norm_product > 0:
                similarity = dot_product / norm_product
            else:
                similarity = 0.0
            
            importance_scores[name] = float(similarity)
        
        return importance_scores
```

```python
@router.post("/batch_fuse", response_model=BatchFusionResponse)
async def batch_fuse_modalities(
    request: BatchFusionRequest,
    fusion_manager: FusionEngine = Depends(get_fusion_manager)
):
    """
    批量融合多模态数据
    
    Args:
        request: 批量融合请求
        fusion_manager: 融合管理器依赖
        
    Returns:
        BatchFusionResponse: 批量融合结果响应
    """
    try:
        # 转换请求数据为模态特征批次
        modal_features_batch = []
        
        for batch_item in request.batch:
            modal_features = {}
            for modality, data in batch_item.modalities.items():
                # 转换numpy数组
                feature_vector = np.array(data.feature_vector)
                
                # 创建模态特征
                modal_features[modality] = ModalityFeature(
                    modality=modality,
                    feature_vector=feature_vector,
                    feature_type=FeatureType(data.feature_type),
                    quality_score=data.quality_score,
                    timestamp=data.timestamp,
                    metadata=data.metadata
                )
            
            modal_features_batch.append(modal_features)
        
        # 执行批量融合
        start_time = time.time()
        fusion_results = fusion_manager.batch_fuse(
            modal_features_batch=modal_features_batch,
            strategy=request.strategy,
            context=request.context
        )
        batch_fusion_time = time.time() - start_time
        
        # 构建响应
        batch_results = []
        
        for i, fusion_result in enumerate(fusion_results):
            if 'error' in fusion_result.metadata:
                # 错误结果
                result = {
                    "batch_index": i,
                    "status": "failed",
                    "error": fusion_result.metadata.get('error', 'Unknown error'),
                    "data_id": None,
                    "fused_vector": [],
                    "modality_weights": {},
                    "attention_weights": None,
                    "strategy": fusion_result.strategy,
                    "confidence": 0.0,
                    "fusion_time": 0.0,
                    "metadata": fusion_result.metadata
                }
            else:
                # 成功结果
                result = {
                    "batch_index": i,
                    "status": "success",
                    "error": None,
                    "data_id": f"fusion_{int(time.time())}_{i}",
                    "fused_vector": fusion_result.fused_vector.tolist(),
                    "modality_weights": fusion_result.modality_weights,
                    "attention_weights": {
                        mod: weights.tolist() if weights is not None else None
                        for mod, weights in fusion_result.attention_weights.items()
                    } if fusion_result.attention_weights else None,
                    "strategy": fusion_result.strategy,
                    "confidence": fusion_result.confidence,
                    "fusion_time": fusion_result.metadata.get('fusion_time', 0.0),
                    "metadata": fusion_result.metadata
                }
            
            batch_results.append(result)
        
        # 计算成功和失败数量
        success_count = sum(1 for r in batch_results if r["status"] == "success")
        failed_count = len(batch_results) - success_count
        
        # 构建响应
        response = BatchFusionResponse(
            status="completed",
            batch_size=len(batch_results),
            success_count=success_count,
            failed_count=failed_count,
            batch_fusion_time=batch_fusion_time,
            results=batch_results,
            metadata={
                "strategy": request.strategy,
                "total_modalities": sum(len(item.modalities) for item in request.batch)
            }
        )
        
        logger.info(f"Successfully processed batch fusion: {success_count}/{len(batch_results)} succeeded")
        return response
        
    except Exception as e:
        logger.error(f"Error during batch fusion: {str(e)}")
        raise HTTPException(status_code=500, detail=f"Batch fusion error: {str(e)}")
```

```python
import requests
import json

# 设置API基础URL
base_url = "http://localhost:8000/fusion"

# 设置请求头
headers = {
    "Content-Type": "application/json",
    "Authorization": "Bearer your_token_here"
}

# 准备请求数据
data = {
    "modalities": {
        "visual": {
            "feature_vector": [0.1, 0.2, 0.3] * 85,  # 256维特征向量
            "feature_type": "processed",
            "quality_score": 0.9
        },
        "audio": {
            "feature_vector": [0.4, 0.5, 0.6] * 42,   # 128维特征向量
            "feature_type": "processed",
            "quality_score": 0.8
        },
        "text": {
            "feature_vector": [0.7, 0.8, 0.9] * 170,  # 512维特征向量
            "feature_type": "processed",
            "quality_score": 0.95
        }
    },
    "strategy": "attention_hybrid",
    "context": {
        "modality_preferences": {
            "visual": 1.5,
            "audio": 1.0,
            "text": 0.8
        }
    }
}

# 发送请求
response = requests.post(
    f"{base_url}/fuse",

    data=json.dumps(data)

if response.status_code == 200:
```

```javascript
    "Content-Type": "application/json",
    "Authorization": "Bearer your_token_here"
};

// 准备请求数据
const data = {
    modalities: {
        visual: {
            feature_vector: Array(256).fill(0).map((_, i) => (i % 3 + 1) / 10),
            feature_type: "processed",
            quality_score: 0.9
        },
        audio: {
            feature_vector: Array(128).fill(0).map((_, i) => (i % 3 + 4) / 10),

        },

            feature_type: "processed",
            quality_score: 0.95
        }
    },
    strategy: "attention_hybrid",
    context: {
        modality_preferences: {
            visual: 1.5,
            audio: 1.0,
            text: 0.8
        }
    }
};

// 发送请求
fetch(`${baseUrl}/fuse`, {
    method: "POST",
    headers: headers,
    body: JSON.stringify(data)
})
.then(response => {
    if (response.ok) {
        return response.json();
    } else {
        throw new Error(`Error: ${response.status} - ${response.statusText}`);
    }
})
.then(result => {
    console.log(`Fusion successful! Data ID: ${result.data_id}`);
    console.log(`Fused vector dimension: ${result.fused_vector.length}`);
    console.log(`Confidence: ${result.confidence}`);
})
.catch(error => {
    console.error(error);
});
```

1. **特征向量维度**: 不同模态的特征向量维度可以不同，系统会自动处理维度对齐。

3. **融合策略**: 不同的融合策略适用于不同的场景，建议根据实际需求选择合适的策略。

4. **批处理**: 批量融合可以提高处理效率，但会增加内存使用。

5. **性能监控**: 定期检查性能指标，确保系统运行正常。
