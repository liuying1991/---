# 交互历史管理模块6A工作流

## 模块概述

### 模块名称
交互历史管理模块

### 所属子系统
交互表达子系统

### 模块功能描述
交互历史管理模块负责管理交互表达子系统的交互历史数据，包括交互记录、交互内容、交互效果和交互反馈等。该模块提供交互历史的存储、检索、分析和可视化功能，支持交互历史的长期保存和高效访问，为交互策略优化和个性化交互提供历史数据支持。

### 模块职责
1. 历史存储：存储和管理交互历史数据
2. 历史检索：提供交互历史数据的检索和查询
3. 历史分析：分析交互历史数据和模式
4. 历史可视化：可视化交互历史数据和趋势
5. 历史清理：定期清理和归档历史数据
6. 历史学习：从历史数据中学习交互模式和规律

### 模块接口
- **输入接口**：交互记录、交互内容、交互效果、交互反馈
- **输出接口**：历史数据、分析结果、可视化图表、模式规律
- **依赖模块**：交互管理模块、反馈处理模块、交互效果评估模块
- **被依赖模块**：交互策略模块、交互内容生成模块、交互控制模块

        self.history_collector = history_collector

        self.history_visualizer = history_visualizer
        self.history_cleaner = history_cleaner
        self.history_learner = history_learner
        self.history_manager = history_manager
        self.privacy_protector = privacy_protector
        self.data_exporter = data_exporter
        
        # 初始化数据库连接
        self._init_db_connections()
        
    async def collect_history(self, baby_id, session_id, interaction_data):
        try:
            # 验证交互数据
            validated_data = await self.history_collector.validate_data(interaction_data)
            
            # 转换数据格式
            transformed_data = await self.history_collector.transform_data(validated_data)
            
            # 应用隐私保护
            protected_data = await self.privacy_protector.protect_data(transformed_data)
            
            # 存储历史数据
            storage_result = await self.history_storage.store_history(
                baby_id, session_id, protected_data
            )
            
            # 构建收集结果
            collection_result = {
                "baby_id": baby_id,
                "session_id": session_id,
                "interaction_data": protected_data,
                "storage_result": storage_result,
                "timestamp": time.time()
            }
            
            # 保存收集结果
            await self.history_manager.save_collection_result(collection_result)
            
            return collection_result
            
        except Exception as e:
            # 处理异常
            error_result = {
                "baby_id": baby_id,
                "session_id": session_id,
                "error": str(e),
                "timestamp": time.time()
            }
            
            return error_result
    
    async def retrieve_history(self, baby_id, query_params):
        try:
            # 构建查询条件
            query = await self.history_retriever.build_query(baby_id, query_params)
            
            # 执行查询
            raw_results = await self.history_storage.execute_query(query)
            
            # 应用隐私保护
            protected_results = await self.privacy_protector.protect_results(raw_results)
            
            # 格式化结果
            formatted_results = await self.history_retriever.format_results(
                protected_results, query_params
            )
            
            # 构建检索结果
            retrieval_result = {
                "baby_id": baby_id,
                "query_params": query_params,
                "query": query,
                "raw_results": raw_results,
                "protected_results": protected_results,
                "formatted_results": formatted_results,
                "result_count": len(formatted_results),
                "timestamp": time.time()
            }
            
            # 记录检索操作
            await self.history_manager.log_retrieval_operation(retrieval_result)
            
            return retrieval_result
            
        except Exception as e:
            # 处理异常
            error_result = {
                "baby_id": baby_id,
                "query_params": query_params,
                "error": str(e),
                "timestamp": time.time()

                "baby_id": baby_id,
                "export_params": export_params,
                "error": str(e),

class HistoryAnalyzer:
    def __init__(self, pattern_miner, trend_analyzer):
        self.pattern_miner = pattern_miner
        self.trend_analyzer = trend_analyzer
        
    async def analyze_history(self, history_data, analysis_params):
        try:
            # 获取分析类型
            analysis_type = analysis_params.get("type", "general")
            
            # 根据分析类型执行不同的分析
            if analysis_type == "pattern":
                analysis_result = await self._analyze_patterns(history_data, analysis_params)
            elif analysis_type == "trend":
                analysis_result = await self._analyze_trends(history_data, analysis_params)
            elif analysis_type == "statistics":
                analysis_result = await self._analyze_statistics(history_data, analysis_params)
            elif analysis_type == "correlation":
                analysis_result = await self._analyze_correlations(history_data, analysis_params)
            else:
                analysis_result = await self._analyze_general(history_data, analysis_params)
            
            return {
                "analysis_type": analysis_type,
                "analysis_params": analysis_params,
                "analysis_result": analysis_result,
                "data_count": len(history_data)
            }
            
        except Exception as e:
            return {
                "error": str(e),
                "analysis_type": analysis_params.get("type", "general"),
                "analysis_params": analysis_params
            }
    
    async def _analyze_patterns(self, history_data, analysis_params):
        # 挖掘交互模式
        interaction_patterns = await self.pattern_miner.mine_interaction_patterns(
            history_data, analysis_params
        )
        
        # 挖掘时间模式
        temporal_patterns = await self.pattern_miner.mine_temporal_patterns(
            history_data, analysis_params
        )
        
        # 挖掘序列模式
        sequential_patterns = await self.pattern_miner.mine_sequential_patterns(
            history_data, analysis_params
        )
        
        return {
            "interaction_patterns": interaction_patterns,
            "temporal_patterns": temporal_patterns,
            "sequential_patterns": sequential_patterns
        }
    
    async def _analyze_trends(self, history_data, analysis_params):
        # 分析交互频率趋势
        frequency_trends = await self.trend_analyzer.analyze_frequency_trends(
            history_data, analysis_params
        )
        
        # 分析交互效果趋势
        effect_trends = await self.trend_analyzer.analyze_effect_trends(
            history_data, analysis_params
        )
        
        # 分析交互内容趋势
        content_trends = await self.trend_analyzer.analyze_content_trends(
            history_data, analysis_params
        )
        
        return {
            "frequency_trends": frequency_trends,
            "effect_trends": effect_trends,
            "content_trends": content_trends
        }
    
    async def _analyze_statistics(self, history_data, analysis_params):
        # 计算基本统计指标
        basic_stats = await self._calculate_basic_statistics(history_data)
        
        # 计算交互类型统计
        interaction_type_stats = await self._calculate_interaction_type_statistics(history_data)
        
        # 计算交互效果统计
        effect_stats = await self._calculate_effect_statistics(history_data)
        
        # 计算时间统计
        temporal_stats = await self._calculate_temporal_statistics(history_data)
        
        return {
            "basic_stats": basic_stats,
            "interaction_type_stats": interaction_type_stats,
            "effect_stats": effect_stats,
            "temporal_stats": temporal_stats
        }
    
    async def _analyze_correlations(self, history_data, analysis_params):
        # 分析交互类型与效果的相关性
        type_effect_correlation = await self._analyze_type_effect_correlation(history_data)
        
        # 分析时间与效果的相关性
        time_effect_correlation = await self._analyze_time_effect_correlation(history_data)
        
        # 分析内容与效果的相关性
        content_effect_correlation = await self._analyze_content_effect_correlation(history_data)
        
        return {
            "type_effect_correlation": type_effect_correlation,
            "time_effect_correlation": time_effect_correlation,
            "content_effect_correlation": content_effect_correlation
        }
    
    async def _analyze_general(self, history_data, analysis_params):
        # 综合分析
        pattern_analysis = await self._analyze_patterns(history_data, analysis_params)
        trend_analysis = await self._analyze_trends(history_data, analysis_params)
        statistics_analysis = await self._analyze_statistics(history_data, analysis_params)
        
        return {
            "pattern_analysis": pattern_analysis,
            "trend_analysis": trend_analysis,
            "statistics_analysis": statistics_analysis
        }
    
    async def _calculate_basic_statistics(self, history_data):
        # 计算交互总数
        total_interactions = len(history_data)
        
        # 计算平均交互时长
        interaction_durations = [
            interaction.get("duration", 0) for interaction in history_data
        ]
        avg_duration = sum(interaction_durations) / len(interaction_durations) if interaction_durations else 0
        
        # 计算平均交互效果
        effect_scores = [
            interaction.get("effect", {}).get("score", 0.5) for interaction in history_data
        ]
        avg_effect = sum(effect_scores) / len(effect_scores) if effect_scores else 0.5
        
        return {
            "total_interactions": total_interactions,
            "avg_duration": avg_duration,
            "avg_effect": avg_effect
        }
    
    async def _calculate_interaction_type_statistics(self, history_data):
        # 统计交互类型
        interaction_types = {}
        for interaction in history_data:
            interaction_type = interaction.get("type", "unknown")
            interaction_types[interaction_type] = interaction_types.get(interaction_type, 0) + 1
        
        # 计算类型占比
        total_interactions = len(history_data)
        type_percentages = {
            type_name: count / total_interactions
            for type_name, count in interaction_types.items()
        }
        
        return {
            "type_counts": interaction_types,
            "type_percentages": type_percentages
        }
    
    async def _calculate_effect_statistics(self, history_data):
        # 统计效果分数
        effect_scores = [
            interaction.get("effect", {}).get("score", 0.5) for interaction in history_data
        ]
        
        # 计算基本统计量
        import statistics
        avg_effect = statistics.mean(effect_scores) if effect_scores else 0.5
        median_effect = statistics.median(effect_scores) if effect_scores else 0.5
        min_effect = min(effect_scores) if effect_scores else 0.5
        max_effect = max(effect_scores) if effect_scores else 0.5
        
        # 计算标准差
        stdev_effect = statistics.stdev(effect_scores) if len(effect_scores) > 1 else 0.0
        
        return {
            "avg_effect": avg_effect,
            "median_effect": median_effect,
            "min_effect": min_effect,
            "max_effect": max_effect,
            "stdev_effect": stdev_effect
        }
    
    async def _calculate_temporal_statistics(self, history_data):
        # 统计时间分布
        hours = {}
        days = {}
        months = {}
        
        for interaction in history_data:
            timestamp = interaction.get("timestamp", 0)
            date_time = datetime.datetime.fromtimestamp(timestamp)
            
            hour = date_time.hour
            day = date_time.strftime("%A")
            month = date_time.strftime("%B")
            
            hours[hour] = hours.get(hour, 0) + 1
            days[day] = days.get(day, 0) + 1
            months[month] = months.get(month, 0) + 1
        
        return {
            "hour_distribution": hours,
            "day_distribution": days,
            "month_distribution": months
        }
    
    async def _analyze_type_effect_correlation(self, history_data):
        # 分析交互类型与效果的相关性
        type_effects = {}
        
        for interaction in history_data:
            interaction_type = interaction.get("type", "unknown")
            effect_score = interaction.get("effect", {}).get("score", 0.5)
            
            if interaction_type not in type_effects:
                type_effects[interaction_type] = []
            
            type_effects[interaction_type].append(effect_score)
        
        # 计算每种类型的平均效果
        type_avg_effects = {
            type_name: sum(effects) / len(effects)
            for type_name, effects in type_effects.items()
        }
        
        return {
            "type_effects": type_effects,
            "type_avg_effects": type_avg_effects
        }
    
    async def _analyze_time_effect_correlation(self, history_data):
        # 分析时间与效果的相关性
        time_effects = []
        
        for interaction in history_data:
            timestamp = interaction.get("timestamp", 0)
            effect_score = interaction.get("effect", {}).get("score", 0.5)
            
            time_effects.append({
                "timestamp": timestamp,
                "effect_score": effect_score
            })
        
        # 按时间排序
        time_effects.sort(key=lambda x: x["timestamp"])
        
        return {
            "time_effects": time_effects
        }
    
    async def _analyze_content_effect_correlation(self, history_data):
        # 分析内容与效果的相关性
        content_effects = {}
        
        for interaction in history_data:
            content = interaction.get("content", {})
            effect_score = interaction.get("effect", {}).get("score", 0.5)
            
            # 提取内容特征
            content_features = await self._extract_content_features(content)
            
            for feature, value in content_features.items():
                if feature not in content_effects:
                    content_effects[feature] = []
                
                content_effects[feature].append({
                    "value": value,
                    "effect_score": effect_score
                })
        
        # 计算每个特征与效果的相关性
        feature_correlations = {}
        for feature, data in content_effects.items():
            values = [item["value"] for item in data]
            effects = [item["effect_score"] for item in data]
            
            # 计算相关系数
            correlation = await self._calculate_correlation(values, effects)
            
            feature_correlations[feature] = correlation
        
        return {
            "content_effects": content_effects,
            "feature_correlations": feature_correlations
        }
    
    async def _extract_content_features(self, content):
        # 提取内容特征
        features = {}
        
        # 提取语音特征
        if "voice" in content:
            voice = content["voice"]
            features["voice_duration"] = voice.get("duration", 0)
            features["voice_volume"] = voice.get("volume", 0.5)
            features["voice_pitch"] = voice.get("pitch", 0.5)
        
        # 提取表情特征
        if "expression" in content:
            expression = content["expression"]
            features["expression_type"] = hash(expression.get("type", "neutral")) % 100
            features["expression_intensity"] = expression.get("intensity", 0.5)
        
        # 提取动作特征
        if "action" in content:
            action = content["action"]
            features["action_type"] = hash(action.get("type", "none")) % 100
            features["action_duration"] = action.get("duration", 0)
        
        return features
    
    async def _calculate_correlation(self, x, y):
        # 计算相关系数
        import statistics
        
        n = len(x)
        if n < 2:
            return 0.0
        
        mean_x = statistics.mean(x)
        mean_y = statistics.mean(y)
        
        sum_xy = sum((x[i] - mean_x) * (y[i] - mean_y) for i in range(n))
        sum_xx = sum((x[i] - mean_x) ** 2 for i in range(n))
        sum_yy = sum((y[i] - mean_y) ** 2 for i in range(n))
        
        if sum_xx == 0 or sum_yy == 0:
            return 0.0
        
        correlation = sum_xy / ((sum_xx * sum_yy) ** 0.5)
        
        return correlation
