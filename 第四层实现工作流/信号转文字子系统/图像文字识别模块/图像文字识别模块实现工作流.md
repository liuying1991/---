# 图像文字识别核心类设计
import asyncio
import logging
from typing import Dict, List, Optional, Any, Tuple, Iterator
from dataclasses import dataclass
from enum import Enum
import time
import cv2
import numpy as np
from datetime import datetime

from image_text_recognition.models.image_models import ImageData
from image_text_recognition.models.text_models import TextRegion, OCRResult
from image_text_recognition.models.result_models import RecognitionResult
from image_text_recognition.services.detection_service import DetectionService
from image_text_recognition.services.recognition_service import RecognitionService
from image_text_recognition.services.correction_service import CorrectionService

logger = logging.getLogger(__name__)

class RecognitionLanguage(Enum):
    """识别语言类型"""
    CHINESE = "chinese"
    ENGLISH = "english"
    AUTO_DETECT = "auto"
    MIXED = "mixed"

@dataclass
class ImageData:
    """图像数据"""
    image_id: str
    image_path: str
    image_format: str
    image_size: Tuple[int, int]
    image_quality: float
    metadata: Dict[str, Any]
    created_at: datetime

@dataclass
class TextRegion:
    """文字区域"""
    region_id: str
    bbox: Tuple[int, int, int, int]  # (x, y, width, height)
    confidence: float
    text_type: str  # "handwritten", "printed"
    language: Optional[str] = None

@dataclass
class OCRResult:
    """OCR识别结果"""
    image_id: str
    text_content: str
    regions: List[TextRegion]
    confidence: float
    processing_time: float
    language: str
    created_at: datetime

class ImageTextRecognitionManager:
    """图像文字识别管理器"""
    
    def __init__(self, config: Dict):
        """初始化图像文字识别管理器"""
        self.config = config
        
        # 初始化服务
        self.detection_service = DetectionService(config.get('detection', {}))
        self.recognition_service = RecognitionService(config.get('recognition', {}))
        self.correction_service = CorrectionService(config.get('correction', {}))
        
        # 初始化模型
        self._load_models()
        
        logger.info("ImageTextRecognitionManager initialized")
    
    def recognize_text(self, image_data: bytes, options: Dict = None) -> Dict:
        """
        识别图像中的文字
        
        Args:
            image_data: 图像二进制数据
            options: 识别选项
            
        Returns:
            Dict: 识别结果
        """
        try:
            start_time = time.time()
            
            # 解析选项
            options = options or {}
            language = options.get('language', 'auto')
            detect_regions_only = options.get('detect_regions_only', False)
            
            # 转换图像数据
            nparr = np.frombuffer(image_data, np.uint8)
            image = cv2.imdecode(nparr, cv2.IMREAD_COLOR)
            
            # 创建图像数据对象
            image_data_obj = ImageData(
                image_id=f"img_{datetime.now().strftime('%Y%m%d_%H%M%S')}",
                image_path="",
                image_format="",
                image_size=(image.shape[1], image.shape[0]),
                image_quality=self._calculate_image_quality(image),
                metadata=options,
                created_at=datetime.now()
            )
            
            # 预处理图像
            preprocessed_image = self._preprocess_image(image)
            
            # 检测文字区域
            regions = self.detection_service.detect_text_regions(preprocessed_image)
            
            # 如果只需要检测区域，直接返回
            if detect_regions_only:
                processing_time = (time.time() - start_time) * 1000
                result = {
                    'image_id': image_data_obj.image_id,
                    'regions': [self._region_to_dict(r) for r in regions],
                    'processing_time': processing_time,
                    'status': 'success',
                    'timestamp': datetime.now().isoformat()
                }
                return result
            
            # 识别文字内容
            ocr_result = self.recognition_service.recognize_text(
                preprocessed_image, regions, language
            )
            
            # 校正文本
            if language != 'none':
                corrected_text = self.correction_service.correct_text(
                    ocr_result.text_content, language
                )
                ocr_result.text_content = corrected_text
            
            # 计算处理时间
            processing_time = (time.time() - start_time) * 1000
            
            # 返回结果
            result = {
                'image_id': image_data_obj.image_id,
                'text_content': ocr_result.text_content,
                'regions': [self._region_to_dict(r) for r in ocr_result.regions],
                'confidence': ocr_result.confidence,
                'language': ocr_result.language,
                'processing_time': processing_time,
                'status': 'success',
                'timestamp': datetime.now().isoformat()
            }
            
            logger.info(f"Text recognition completed for image {image_data_obj.image_id}")
            return result
            
        except Exception as e:
            logger.error(f"Error recognizing text: {str(e)}")
            return {
                'status': 'error',
                'error': str(e),
                'timestamp': datetime.now().isoformat()
            }
    
    async def batch_recognize(self, image_list: List[bytes], options: Dict = None) -> List[Dict]:
        """
        批量识别图像中的文字
        
        Args:
            image_list: 图像二进制数据列表
            options: 识别选项
            
        Returns:
            List[Dict]: 识别结果列表
        """
        tasks = []
        for image_data in image_list:
            task = asyncio.create_task(
                asyncio.to_thread(self.recognize_text, image_data, options)
            )
            tasks.append(task)
        
        results = await asyncio.gather(*tasks)
        return results
    
    def recognize_video_stream(self, video_stream: bytes, options: Dict = None) -> Iterator[Dict]:
        """
        识别视频流中的文字
        
        Args:
            video_stream: 视频流数据
            options: 识别选项
            
        Yields:
            Dict: 识别结果
        """
        # 将视频流转换为帧序列
        frames = self._extract_frames(video_stream)
        
        for frame_id, frame in enumerate(frames):
            # 识别当前帧的文字
            result = self.recognize_text(frame, options)
            result['frame_id'] = frame_id
            yield result
    
    def _load_models(self):
        """加载OCR模型"""
        try:
            # 加载文字检测模型
            self.detection_model = self.detection_service.load_model(
                self.config.get('models', {}).get('detection', {})
            )
            
            # 加载文字识别模型
            self.recognition_model = self.recognition_service.load_model(
                self.config.get('models', {}).get('recognition', {})
            )
            
            # 加载语言模型
            self.language_model = self.correction_service.load_model(
                self.config.get('models', {}).get('language', {})
            )
            
            logger.info("OCR models loaded successfully")
        except Exception as e:
            logger.error(f"Error loading OCR models: {str(e)}")
            raise
    
    def _preprocess_image(self, image: np.ndarray) -> np.ndarray:
        """预处理图像"""
        try:
            # 图像增强
            enhanced_image = self._enhance_image(image)
            
            # 图像校正
            corrected_image = self._correct_image(enhanced_image)
            
            return corrected_image
        except Exception as e:
            logger.error(f"Error preprocessing image: {str(e)}")
            return image
    
    def _enhance_image(self, image: np.ndarray) -> np.ndarray:
        """增强图像"""
        # 转换为灰度图
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        
        # 自适应阈值处理
        binary = cv2.adaptiveThreshold(
            gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, 11, 2
        )
        
        # 降噪
        denoised = cv2.medianBlur(binary, 3)
        
        return denoised
    
    def _correct_image(self, image: np.ndarray) -> np.ndarray:
        """校正图像"""
        # 检测边缘
        edges = cv2.Canny(image, 50, 150, apertureSize=3)
        
        # 检测直线
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 100, minLineLength=100, maxLineGap=10)
        
        if lines is not None:
            # 计算倾斜角度
            angle = 0
            count = 0
            
            for line in lines:
                x1, y1, x2, y2 = line[0]
                if x2 != x1:
                    angle += np.arctan((y2-y1)/(x2-x1)) * 180 / np.pi
                    count += 1
            
            if count > 0:
                angle = angle / count
                
                # 旋转图像
                center = (image.shape[1]//2, image.shape[0]//2)
                rotation_matrix = cv2.getRotationMatrix2D(center, angle, 1.0)
                rotated = cv2.warpAffine(image, rotation_matrix, (image.shape[1], image.shape[0]))
                
                return rotated
        
        return image
    
    def _calculate_image_quality(self, image: np.ndarray) -> float:
        """计算图像质量"""
        # 转换为灰度图
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        
        # 计算清晰度（拉普拉斯方差）
        sharpness = cv2.Laplacian(gray, cv2.CV_64F).var()
        
        # 归一化到0-1范围
        quality = min(sharpness / 1000.0, 1.0)
        
        return quality
    
    def _region_to_dict(self, region: TextRegion) -> Dict:
        """将文字区域转换为字典"""
        return {
            'region_id': region.region_id,
            'bbox': region.bbox,
            'confidence': region.confidence,
            'text_type': region.text_type,
            'language': region.language
        }
    
    def _extract_frames(self, video_stream: bytes) -> List[np.ndarray]:
        """从视频流中提取帧"""
        # 将视频流写入临时文件
        with tempfile.NamedTemporaryFile(suffix='.mp4', delete=False) as tmp:
            tmp.write(video_stream)
            tmp_path = tmp.name
        
        try:
            # 读取视频
            cap = cv2.VideoCapture(tmp_path)
            frames = []
            
            while True:
                ret, frame = cap.read()
                if not ret:
                    break
                
                # 转换为二进制数据
                _, buffer = cv2.imencode('.jpg', frame)
                frame_data = buffer.tobytes()
                
                frames.append(frame_data)
            
            cap.release()
            return frames
        finally:
            # 删除临时文件
            os.unlink(tmp_path)

- `RATE_LIMIT_EXCEEDED`: 请求频率超过限制

## 风险评估与应对措施

### 1. 技术风险
- **风险**: OCR模型准确率不达标
  - **应对**: 采用多种模型集成，增加训练数据多样性
- **风险**: 处理速度不满足要求
  - **应对**: 优化算法，使用GPU加速，实现分布式处理
- **风险**: 复杂场景下识别效果差
  - **应对**: 增强图像预处理，针对不同场景优化识别策略

### 2. 资源风险
- **风险**: 计算资源不足
  - **应对**: 使用云服务弹性扩展，优化资源使用效率
- **风险**: 存储空间不足
  - **应对**: 实现数据压缩和清理策略，使用分布式存储

### 3. 安全风险
- **风险**: 敏感信息泄露
  - **应对**: 实现敏感信息自动过滤，加强数据加密
- **风险**: 恶意图像攻击
  - **应对**: 增强恶意图像检测能力，实现异常行为识别
