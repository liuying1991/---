# 中期记忆模块实现工作流

## 1. 模块概述

### 1.1 模块功能与职责

中期记忆模块是婴儿AI管家系统记忆存储子系统的核心组件之一，负责存储和管理中期记忆数据。中期记忆作为短期记忆和长期记忆之间的桥梁，具有以下主要功能：

1. **记忆过渡管理**：管理从短期记忆到长期记忆的过渡过程
2. **记忆筛选与强化**：根据重要性和访问频率筛选和强化记忆
3. **记忆组织与关联**：组织记忆内容并建立记忆间的关联
4. **记忆检索与更新**：提供高效的记忆检索和更新机制
5. **记忆容量管理**：管理中期记忆的容量，防止溢出

### 1.2 模块在系统中的位置

中期记忆模块位于记忆存储子系统中，与短期记忆模块和长期记忆模块紧密协作：

```
记忆存储子系统
├── 短期记忆模块 (临时存储最近感知和交互信息)
├── 中期记忆模块 (存储和管理中期记忆数据)  ← 当前实现
└── 长期记忆模块 (持久化存储重要记忆)
```

### 1.3 关键技术选型

1. **存储技术**：混合存储架构，结合内存数据库和持久化存储
2. **检索技术**：多模态检索，支持语义、关键词、时间和关联检索
3. **记忆管理**：基于重要性和访问频率的记忆管理算法
4. **关联技术**：图数据库存储记忆关联关系
5. **缓存技术**：多级缓存提高访问性能

## 3. 核心组件实现

### 3.1 记忆存储组件

```python
# core/memory_store.py
import uuid
import time
import numpy as np
from typing import List, Dict, Any, Optional, Tuple
from datetime import datetime, timedelta
import logging

from models.memory_data import MemoryData, MemoryType, MemoryImportance
from storage.memory_storage import MemoryStorage
from storage.persistent_storage import PersistentStorage
from storage.graph_storage import GraphStorage
from utils.memory_encoder import MemoryEncoder
from utils.memory_analyzer import MemoryAnalyzer
from utils.cache_manager import CacheManager
from config.mid_term_memory_config import MidTermMemoryConfig

logger = logging.getLogger(__name__)

class MemoryStore:
    """记忆存储组件"""
    
    def __init__(self, config: MidTermMemoryConfig):
        self.config = config
        self.memory_storage = MemoryStorage(config.memory_storage)
        self.persistent_storage = PersistentStorage(config.persistent_storage)
        self.graph_storage = GraphStorage(config.graph_storage)
        self.encoder = MemoryEncoder(config.encoder)
        self.analyzer = MemoryAnalyzer(config.analyzer)
        self.cache = CacheManager(config.cache)
        
        # 记忆过渡队列
        self.transition_queue = []
        
        # 记忆访问统计
        self.access_stats = {}
        
        logger.info("MemoryStore initialized")
    
    def store_memory(self, memory_data: MemoryData) -> str:
        """存储记忆"""
        try:
            # 生成记忆ID
            if not memory_data.memory_id:
                memory_data.memory_id = str(uuid.uuid4())
            
            # 设置创建时间
            if not memory_data.created_at:
                memory_data.created_at = datetime.now()
            
            # 编码记忆内容
            if not memory_data.content_vector:
                memory_data.content_vector = self.encoder.encode(memory_data.content)
            
            # 分析记忆重要性
            if memory_data.importance == MemoryImportance.UNKNOWN:
                memory_data.importance = self.analyzer.analyze_importance(memory_data)
            
            # 存储到内存存储
            self.memory_storage.add_memory(memory_data)
            
            # 存储到持久化存储
            self.persistent_storage.add_memory(memory_data)
            
            # 添加到图存储
            self.graph_storage.add_memory_node(memory_data)
            
            # 初始化访问统计
            self.access_stats[memory_data.memory_id] = {
                "access_count": 0,
                "last_access": datetime.now(),
                "access_frequency": 0.0
            }
            
            # 检查是否需要过渡到长期记忆
            self._check_transition(memory_data)
            
            # 更新缓存
            self.cache.set(f"memory:{memory_data.memory_id}", memory_data)
            
            logger.info(f"Memory stored: {memory_data.memory_id}")
            return memory_data.memory_id
            
        except Exception as e:
            logger.error(f"Error storing memory: {e}")
            raise
    
    def retrieve_memory(self, query: str, limit: int = 10, 
                       memory_type: Optional[str] = None) -> List[MemoryData]:
        """检索记忆"""
        try:
            # 检查缓存
            cache_key = self.cache.get_memory_cache_key(query, memory_type)
            cached_result = self.cache.get(cache_key)
            if cached_result:
                logger.info(f"Retrieved {len(cached_result)} memories from cache")
                return cached_result
            
            # 编码查询
            query_vector = self.encoder.encode(query)
            
            # 从内存存储检索
            memories = self.memory_storage.search_memories(
                query_vector, limit, memory_type
            )
            
            # 更新访问统计
            for memory in memories:
                self._update_access_stats(memory.memory_id)
            
            # 缓存结果
            self.cache.set(cache_key, memories, expire=300)  # 5分钟过期
            
            logger.info(f"Retrieved {len(memories)} memories")
            return memories
            
        except Exception as e:
            logger.error(f"Error retrieving memories: {e}")
            raise
    
    def get_memory(self, memory_id: str) -> Optional[MemoryData]:
        """获取单个记忆"""
        try:
            # 检查缓存
            cache_key = f"memory:{memory_id}"
            memory = self.cache.get(cache_key)
            if memory:
                self._update_access_stats(memory_id)
                return memory
            
            # 从内存存储获取
            memory = self.memory_storage.get_memory(memory_id)
            
            if memory:
                # 更新访问统计
                self._update_access_stats(memory_id)
                
                # 更新缓存
                self.cache.set(cache_key, memory)
            
            return memory
            
        except Exception as e:
            logger.error(f"Error getting memory: {e}")
            raise
    
    def update_memory(self, memory_id: str, updates: Dict[str, Any]) -> bool:
        """更新记忆"""
        try:
            # 获取当前记忆
            memory = self.get_memory(memory_id)
            if not memory:
                return False
            
            # 更新记忆属性
            for key, value in updates.items():
                if hasattr(memory, key):
                    setattr(memory, key, value)
            
            # 如果内容更新，重新编码
            if "content" in updates:
                memory.content_vector = self.encoder.encode(memory.content)
                
                # 重新分析重要性
                memory.importance = self.analyzer.analyze_importance(memory)
            
            # 更新修改时间
            memory.updated_at = datetime.now()
            
            # 更新内存存储
            self.memory_storage.update_memory(memory)
            
            # 更新持久化存储
            self.persistent_storage.update_memory(memory)
            
            # 更新图存储
            self.graph_storage.update_memory_node(memory)
            
            # 清除缓存
            self.cache.delete(f"memory:{memory_id}")
            
            # 检查是否需要过渡到长期记忆
            self._check_transition(memory)
            
            logger.info(f"Memory updated: {memory_id}")
            return True
            
        except Exception as e:
            logger.error(f"Error updating memory: {e}")
            raise
    
    def delete_memory(self, memory_id: str) -> bool:
        """删除记忆"""
        try:
            # 从内存存储删除
            self.memory_storage.delete_memory(memory_id)
            
            # 从持久化存储删除
            self.persistent_storage.delete_memory(memory_id)
            
            # 从图存储删除
            self.graph_storage.delete_memory_node(memory_id)
            
            # 删除访问统计
            if memory_id in self.access_stats:
                del self.access_stats[memory_id]
            
            # 清除缓存
            self.cache.delete(f"memory:{memory_id}")
            
            logger.info(f"Memory deleted: {memory_id}")
            return True
            
        except Exception as e:
            logger.error(f"Error deleting memory: {e}")
            raise
    
    def list_memories(self, memory_type: Optional[str] = None, 
                     importance: Optional[str] = None,
                     limit: int = 100, offset: int = 0) -> List[MemoryData]:
        """列出记忆"""
        try:
            memories = self.memory_storage.list_memories(
                memory_type, importance, limit, offset
            )
            
            # 更新访问统计
            for memory in memories:
                self._update_access_stats(memory.memory_id)
            
            return memories
            
        except Exception as e:
            logger.error(f"Error listing memories: {e}")
            raise
    
    def get_statistics(self) -> Dict[str, Any]:
        """获取统计信息"""
        try:
            # 获取记忆统计
            memory_stats = self.memory_storage.get_statistics()
            
            # 添加访问统计
            access_stats = {
                "total_accesses": sum(
                    stats["access_count"] for stats in self.access_stats.values()
                ),
                "unique_memories_accessed": len(self.access_stats),
                "average_access_frequency": np.mean([
                    stats["access_frequency"] for stats in self.access_stats.values()
                ]) if self.access_stats else 0.0
            }
            
            # 合并统计
            statistics = {**memory_stats, **access_stats}
            
            return statistics
            
        except Exception as e:
            logger.error(f"Error getting statistics: {e}")
            raise
    
    def _update_access_stats(self, memory_id: str):
        """更新访问统计"""
        if memory_id not in self.access_stats:
            self.access_stats[memory_id] = {
                "access_count": 0,
                "last_access": datetime.now(),
                "access_frequency": 0.0
            }
        
        stats = self.access_stats[memory_id]
        stats["access_count"] += 1
        stats["last_access"] = datetime.now()
        
        # 计算访问频率 (访问次数 / 存在天数)
        memory = self.get_memory(memory_id)
        if memory and memory.created_at:
            days_alive = (datetime.now() - memory.created_at).days
            if days_alive > 0:
                stats["access_frequency"] = stats["access_count"] / days_alive
    
    def _check_transition(self, memory_data: MemoryData):
        """检查是否需要过渡到长期记忆"""
        # 计算过渡分数
        transition_score = self._calculate_transition_score(memory_data)
        
        # 如果分数超过阈值，添加到过渡队列
        if transition_score >= self.config.transition_threshold:
            self.transition_queue.append({
                "memory_id": memory_data.memory_id,
                "score": transition_score,
                "timestamp": datetime.now()
            })
            
            logger.info(f"Memory {memory_data.memory_id} queued for transition with score {transition_score}")
    
    def _calculate_transition_score(self, memory_data: MemoryData) -> float:
        """计算过渡分数"""
        # 基础分数 (基于重要性)
        importance_scores = {
            MemoryImportance.LOW: 0.2,
            MemoryImportance.MEDIUM: 0.5,
            MemoryImportance.HIGH: 0.8,
            MemoryImportance.CRITICAL: 1.0
        }
        base_score = importance_scores.get(memory_data.importance, 0.5)
        
        # 访问频率加成
        if memory_data.memory_id in self.access_stats:
            access_frequency = self.access_stats[memory_data.memory_id]["access_frequency"]
            frequency_bonus = min(access_frequency / 10, 0.3)  # 最多0.3的加成
        else:
            frequency_bonus = 0.0
        
        # 时间加成 (记忆存在时间越长，分数越高)
        if memory_data.created_at:
            days_alive = (datetime.now() - memory_data.created_at).days
            time_bonus = min(days_alive / 30, 0.2)  # 最多0.2的加成
        else:
            time_bonus = 0.0
        
        # 计算总分数
        total_score = base_score + frequency_bonus + time_bonus
        
        return min(total_score, 1.0)  # 确保分数不超过1.0
    
    def get_transition_queue(self) -> List[Dict[str, Any]]:
        """获取过渡队列"""
        return self.transition_queue.copy()
    
    def process_transition_queue(self) -> List[str]:
        """处理过渡队列，返回已过渡的记忆ID列表"""
        transitioned_memories = []
        
        # 按分数排序
        sorted_queue = sorted(
            self.transition_queue, 
            key=lambda x: x["score"], 
            reverse=True
        )
        
        # 处理队列中的记忆
        for item in sorted_queue[:self.config.max_transition_batch]:
            memory_id = item["memory_id"]
            
            try:
                # 获取记忆
                memory = self.get_memory(memory_id)
                if not memory:
                    continue
                
                # 这里应该调用长期记忆模块的API存储记忆
                # 为了简化，我们只是标记为已过渡
                memory.transitioned = True
                memory.transitioned_at = datetime.now()
                
                # 更新记忆
                self.update_memory(memory_id, {
                    "transitioned": True,
                    "transitioned_at": datetime.now()
                })
                
                transitioned_memories.append(memory_id)
                
            except Exception as e:
                logger.error(f"Error transitioning memory {memory_id}: {e}")
        
        # 从队列中移除已处理的记忆
        self.transition_queue = [
            item for item in self.transition_queue 
            if item["memory_id"] not in transitioned_memories
        ]
        
        return transitioned_memories
    
    def create_association(self, source_memory_id: str, target_memory_id: str, 
                          association_type: str, strength: float = 0.5,
                          metadata: Dict[str, Any] = None) -> str:
        """创建记忆关联"""
        try:
            # 验证记忆存在
            source_memory = self.get_memory(source_memory_id)
            target_memory = self.get_memory(target_memory_id)
            
            if not source_memory or not target_memory:
                raise ValueError("Source or target memory not found")
            
            # 创建关联
            association_id = self.graph_storage.create_association(
                source_memory_id, target_memory_id, association_type, strength, metadata
            )
            
            logger.info(f"Association created: {association_id}")
            return association_id
            
        except Exception as e:
            logger.error(f"Error creating association: {e}")
            raise
    
    def get_memory_associations(self, memory_id: str, 
                               association_type: Optional[str] = None) -> List[Dict[str, Any]]:
        """获取记忆关联"""
        try:
            associations = self.graph_storage.get_associations(memory_id, association_type)
            return associations
            
        except Exception as e:
            logger.error(f"Error getting associations: {e}")
            raise
    
    def delete_association(self, association_id: str) -> bool:
        """删除记忆关联"""
        try:
            success = self.graph_storage.delete_association(association_id)
            
            if success:
                logger.info(f"Association deleted: {association_id}")
            
            return success
            
        except Exception as e:
            logger.error(f"Error deleting association: {e}")
            raise
```

```python
# core/memory_retrieval.py
import numpy as np
from typing import List, Dict, Any, Optional, Tuple
from datetime import datetime, timedelta
import logging

from models.memory_data import MemoryData, MemoryType, MemoryImportance
from models.memory_query import MemoryQuery, QueryType, QueryResult
from storage.memory_storage import MemoryStorage
from storage.graph_storage import GraphStorage
from utils.memory_encoder import MemoryEncoder
from utils.cache_manager import CacheManager
from config.mid_term_memory_config import MidTermMemoryConfig

logger = logging.getLogger(__name__)

class MemoryRetrieval:
    """记忆检索组件"""
    
    def __init__(self, config: MidTermMemoryConfig):
        self.config = config
        self.memory_storage = MemoryStorage(config.memory_storage)
        self.graph_storage = GraphStorage(config.graph_storage)
        self.encoder = MemoryEncoder(config.encoder)
        self.cache = CacheManager(config.cache)
        
        logger.info("MemoryRetrieval initialized")
    
    def search_memories(self, query: MemoryQuery) -> QueryResult:
        """搜索记忆"""
        try:
            # 检查缓存
            cache_key = self.cache.get_query_cache_key(query)
            cached_result = self.cache.get(cache_key)
            if cached_result:
                logger.info(f"Retrieved query result from cache: {query.query_id}")
                return cached_result
            
            # 根据查询类型执行不同的搜索策略
            if query.query_type == QueryType.SEMANTIC:
                results = self._semantic_search(query)
            elif query.query_type == QueryType.KEYWORD:
                results = self._keyword_search(query)
            elif query.query_type == QueryType.TEMPORAL:
                results = self._temporal_search(query)
            elif query.query_type == QueryType.ASSOCIATIVE:
                results = self._associative_search(query)
            elif query.query_type == QueryType.HYBRID:
                results = self._hybrid_search(query)
            else:
                raise ValueError(f"Unsupported query type: {query.query_type}")
            
            # 创建查询结果
            query_result = QueryResult(
                query_id=query.query_id,
                query=query,
                results=results,
                total_count=len(results),
                execution_time=(datetime.now() - query.created_at).total_seconds(),
                created_at=datetime.now()
            )
            
            # 缓存结果
            self.cache.set(cache_key, query_result, expire=600)  # 10分钟过期
            
            logger.info(f"Query executed: {query.query_id}, found {len(results)} results")
            return query_result
            
        except Exception as e:
            logger.error(f"Error searching memories: {e}")
            raise
    
    def _semantic_search(self, query: MemoryQuery) -> List[Dict[str, Any]]:
        """语义搜索"""
        # 编码查询文本
        query_vector = self.encoder.encode(query.text)
        
        # 从内存存储搜索
        memories = self.memory_storage.search_memories(
            query_vector, query.limit, query.memory_type
        )
        
        # 计算相似度分数
        results = []
        for memory in memories:
            similarity = self._calculate_similarity(query_vector, memory.content_vector)
            
            # 如果相似度低于阈值，跳过
            if similarity < self.config.semantic_similarity_threshold:
                continue
            
            results.append({
                "memory": memory,
                "score": similarity,
                "match_type": "semantic",
                "explanation": f"语义相似度: {similarity:.2f}"
            })
        
        # 按分数排序
        results.sort(key=lambda x: x["score"], reverse=True)
        
        # 限制结果数量
        return results[:query.limit]
    
    def _keyword_search(self, query: MemoryQuery) -> List[Dict[str, Any]]:
        """关键词搜索"""
        # 提取关键词
        keywords = self._extract_keywords(query.text)
        
        # 从内存存储搜索
        memories = self.memory_storage.keyword_search(
            keywords, query.limit, query.memory_type
        )
        
        # 计算匹配分数
        results = []
        for memory in memories:
            score = self._calculate_keyword_score(keywords, memory.content)
            
            results.append({
                "memory": memory,
                "score": score,
                "match_type": "keyword",
                "explanation": f"关键词匹配: {score:.2f}"
            })
        
        # 按分数排序
        results.sort(key=lambda x: x["score"], reverse=True)
        
        # 限制结果数量
        return results[:query.limit]
    
    def _temporal_search(self, query: MemoryQuery) -> List[Dict[str, Any]]:
        """时间搜索"""
        # 解析时间范围
        start_time, end_time = self._parse_time_range(query.time_range)
        
        # 从内存存储搜索
        memories = self.memory_storage.temporal_search(
            start_time, end_time, query.limit, query.memory_type
        )
        
        # 计算时间相关性分数
        results = []
        for memory in memories:
            score = self._calculate_temporal_score(memory, query)
            
            results.append({
                "memory": memory,
                "score": score,
                "match_type": "temporal",
                "explanation": f"时间相关性: {score:.2f}"
            })
        
        # 按分数排序
        results.sort(key=lambda x: x["score"], reverse=True)
        
        # 限制结果数量
        return results[:query.limit]
    
    def _associative_search(self, query: MemoryQuery) -> List[Dict[str, Any]]:
        """关联搜索"""
        # 如果没有提供种子记忆，无法执行关联搜索
        if not query.seed_memory_ids:
            return []
        
        # 获取种子记忆
        seed_memories = []
        for memory_id in query.seed_memory_ids:
            memory = self.memory_storage.get_memory(memory_id)
            if memory:
                seed_memories.append(memory)
        
        if not seed_memories:
            return []
        
        # 通过关联图查找相关记忆
        related_memories = []
        for seed_memory in seed_memories:
            # 获取关联的记忆
            associations = self.graph_storage.get_associations(seed_memory.memory_id)
            
            for association in associations:
                # 获取关联的记忆
                target_memory_id = association["target_memory_id"]
                target_memory = self.memory_storage.get_memory(target_memory_id)
                
                if not target_memory:
                    continue
                
                # 计算关联分数
                score = association["strength"]
                
                # 如果有查询文本，计算语义相似度
                if query.text:
                    query_vector = self.encoder.encode(query.text)
                    similarity = self._calculate_similarity(query_vector, target_memory.content_vector)
                    score = score * 0.7 + similarity * 0.3  # 关联权重0.7，语义权重0.3
                
                related_memories.append({
                    "memory": target_memory,
                    "score": score,
                    "match_type": "associative",
                    "explanation": f"关联强度: {association['strength']:.2f}",
                    "association": association
                })
        
        # 去重
        unique_memories = {}
        for item in related_memories:
            memory_id = item["memory"].memory_id
            if memory_id not in unique_memories or item["score"] > unique_memories[memory_id]["score"]:
                unique_memories[memory_id] = item
        
        # 转换为列表并排序
        results = list(unique_memories.values())
        results.sort(key=lambda x: x["score"], reverse=True)
        
        # 限制结果数量
        return results[:query.limit]
    
    def _hybrid_search(self, query: MemoryQuery) -> List[Dict[str, Any]]:
        """混合搜索"""
        # 执行各种搜索
        semantic_results = self._semantic_search(query) if query.text else []
        keyword_results = self._keyword_search(query) if query.text else []
        temporal_results = self._temporal_search(query) if query.time_range else []
        associative_results = self._associative_search(query) if query.seed_memory_ids else []
        
        # 合并结果
        all_results = {}
        
        # 添加语义搜索结果
        for item in semantic_results:
            memory_id = item["memory"].memory_id
            if memory_id not in all_results:
                all_results[memory_id] = {
                    "memory": item["memory"],
                    "scores": {},
                    "match_types": [],
                    "explanations": []
                }
            all_results[memory_id]["scores"]["semantic"] = item["score"]
            all_results[memory_id]["match_types"].append("semantic")
            all_results[memory_id]["explanations"].append(item["explanation"])
        
        # 添加关键词搜索结果
        for item in keyword_results:
            memory_id = item["memory"].memory_id
            if memory_id not in all_results:
                all_results[memory_id] = {
                    "memory": item["memory"],
                    "scores": {},
                    "match_types": [],
                    "explanations": []
                }
            all_results[memory_id]["scores"]["keyword"] = item["score"]
            all_results[memory_id]["match_types"].append("keyword")
            all_results[memory_id]["explanations"].append(item["explanation"])
        
        # 添加时间搜索结果
        for item in temporal_results:
            memory_id = item["memory"].memory_id
            if memory_id not in all_results:
                all_results[memory_id] = {
                    "memory": item["memory"],
                    "scores": {},
                    "match_types": [],
                    "explanations": []
                }
            all_results[memory_id]["scores"]["temporal"] = item["score"]
            all_results[memory_id]["match_types"].append("temporal")
            all_results[memory_id]["explanations"].append(item["explanation"])
        
        # 添加关联搜索结果
        for item in associative_results:
            memory_id = item["memory"].memory_id
            if memory_id not in all_results:
                all_results[memory_id] = {
                    "memory": item["memory"],
                    "scores": {},
                    "match_types": [],
                    "explanations": []
                }
            all_results[memory_id]["scores"]["associative"] = item["score"]
            all_results[memory_id]["match_types"].append("associative")
            all_results[memory_id]["explanations"].append(item["explanation"])
        
        # 计算综合分数
        results = []
        for memory_id, data in all_results.items():
            # 计算加权平均分数
            weights = {
                "semantic": 0.4,
                "keyword": 0.2,
                "temporal": 0.2,
                "associative": 0.2
            }
            
            total_score = 0.0
            total_weight = 0.0
            
            for score_type, score in data["scores"].items():
                if score_type in weights:
                    total_score += score * weights[score_type]
                    total_weight += weights[score_type]
            
            # 如果有分数，计算平均分数
            if total_weight > 0:
                average_score = total_score / total_weight
            else:
                average_score = 0.0
            
            results.append({
                "memory": data["memory"],
                "score": average_score,
                "match_type": "hybrid",
                "explanation": f"混合搜索: {', '.join(data['explanations'])}",
                "scores": data["scores"],
                "match_types": data["match_types"]
            })
        
        # 按分数排序
        results.sort(key=lambda x: x["score"], reverse=True)
        
        # 限制结果数量
        return results[:query.limit]
    
    def _calculate_similarity(self, vector1: np.ndarray, vector2: np.ndarray) -> float:
        """计算向量相似度"""
        # 使用余弦相似度
        dot_product = np.dot(vector1, vector2)
        norm1 = np.linalg.norm(vector1)
        norm2 = np.linalg.norm(vector2)
        
        if norm1 == 0 or norm2 == 0:
            return 0.0
        
        return dot_product / (norm1 * norm2)
    
    def _extract_keywords(self, text: str) -> List[str]:
        """提取关键词"""
        # 简单的关键词提取，实际应用中可以使用更复杂的NLP技术
        import re
        
        # 移除标点符号并转换为小写
        cleaned_text = re.sub(r'[^\w\s]', '', text.lower())
        
        # 分词并过滤停用词
        stop_words = {'the', 'a', 'an', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'for', 'of', 'with', 'by', 'is', 'are', 'was', 'were', 'be', 'been', 'being', 'have', 'has', 'had', 'do', 'does', 'did', 'will', 'would', 'could', 'should', 'may', 'might', 'must', 'can', 'this', 'that', 'these', 'those', 'i', 'you', 'he', 'she', 'it', 'we', 'they', 'me', 'him', 'her', 'us', 'them', 'my', 'your', 'his', 'her', 'its', 'our', 'their'}
        
        words = [word for word in cleaned_text.split() if word not in stop_words and len(word) > 2]
        
        # 返回前10个关键词
        return words[:10]
    
    def _calculate_keyword_score(self, keywords: List[str], content: str) -> float:
        """计算关键词匹配分数"""
        content_lower = content.lower()
        match_count = 0
        
        for keyword in keywords:
            if keyword.lower() in content_lower:
                match_count += 1
        
        # 计算匹配比例
        if not keywords:
            return 0.0
        
        return match_count / len(keywords)
    
    def _parse_time_range(self, time_range: str) -> Tuple[datetime, datetime]:
        """解析时间范围"""
        now = datetime.now()
        
        if not time_range:
            # 默认为最近30天
            start_time = now - timedelta(days=30)
            end_time = now
        elif time_range == "today":
            start_time = now.replace(hour=0, minute=0, second=0, microsecond=0)
            end_time = now
        elif time_range == "yesterday":
            yesterday = now - timedelta(days=1)
            start_time = yesterday.replace(hour=0, minute=0, second=0, microsecond=0)
            end_time = yesterday.replace(hour=23, minute=59, second=59, microsecond=999999)
        elif time_range == "this_week":
            start_time = now - timedelta(days=now.weekday())
            start_time = start_time.replace(hour=0, minute=0, second=0, microsecond=0)
            end_time = now
        elif time_range == "last_week":
            start_time = now - timedelta(days=now.weekday() + 7)
            start_time = start_time.replace(hour=0, minute=0, second=0, microsecond=0)
            end_time = start_time + timedelta(days=6, hours=23, minutes=59, seconds=59)
        elif time_range == "this_month":
            start_time = now.replace(day=1, hour=0, minute=0, second=0, microsecond=0)
            end_time = now
        elif time_range == "last_month":
            if now.month == 1:
                start_time = now.replace(year=now.year-1, month=12, day=1, hour=0, minute=0, second=0, microsecond=0)
            else:
                start_time = now.replace(month=now.month-1, day=1, hour=0, minute=0, second=0, microsecond=0)
            
            # 计算上个月的最后一天
            if now.month == 1:
                end_time = now.replace(year=now.year-1, month=12, day=31, hour=23, minute=59, second=59, microsecond=999999)
            else:
                # 简化处理，假设每个月都是30天
                end_time = start_time + timedelta(days=29, hours=23, minutes=59, seconds=59)
        else:
            # 尝试解析自定义时间范围，格式如 "2023-01-01,2023-01-31"
            try:
                dates = time_range.split(',')
                if len(dates) == 2:
                    start_time = datetime.strptime(dates[0].strip(), "%Y-%m-%d")
                    end_time = datetime.strptime(dates[1].strip(), "%Y-%m-%d")
                    end_time = end_time.replace(hour=23, minute=59, second=59, microsecond=999999)
                else:
                    # 如果解析失败，默认为最近30天
                    start_time = now - timedelta(days=30)
                    end_time = now
            except:
                # 如果解析失败，默认为最近30天
                start_time = now - timedelta(days=30)
                end_time = now
        
        return start_time, end_time
    
    def _calculate_temporal_score(self, memory: MemoryData, query: MemoryQuery) -> float:
        """计算时间相关性分数"""
        if not memory.created_at:
            return 0.0
        
        # 计算记忆创建时间与查询时间的距离
        time_diff = abs((datetime.now() - memory.created_at).total_seconds())
        
        # 将时间差转换为天数
        days_diff = time_diff / (24 * 3600)
        
        # 时间越近，分数越高
        # 使用指数衰减函数
        score = np.exp(-days_diff / 30)  # 30天的半衰期
        
        return score
    
    def get_memory_clusters(self, cluster_count: int = 10) -> List[Dict[str, Any]]:
        """获取记忆聚类"""
        try:
            # 获取所有记忆
            memories = self.memory_storage.list_memories(limit=1000)
            
            if len(memories) < cluster_count:
                cluster_count = len(memories)
            
            # 提取记忆向量
            vectors = []
            memory_ids = []
            
            for memory in memories:
                if memory.content_vector is not None:
                    vectors.append(memory.content_vector)
                    memory_ids.append(memory.memory_id)
            
            if not vectors:
                return []
            
            # 使用K-means聚类
            from sklearn.cluster import KMeans
            
            kmeans = KMeans(n_clusters=cluster_count, random_state=42)
            cluster_labels = kmeans.fit_predict(vectors)
            
            # 组织聚类结果
            clusters = []
            for i in range(cluster_count):
                cluster_memories = []
                for j, label in enumerate(cluster_labels):
                    if label == i:
                        memory_id = memory_ids[j]
                        memory = self.memory_storage.get_memory(memory_id)
                        if memory:
                            cluster_memories.append(memory)
                
                # 计算聚类中心
                center = kmeans.cluster_centers_[i]
                
                # 生成聚类标签
                cluster_label = self._generate_cluster_label(cluster_memories)
                
                clusters.append({
                    "cluster_id": i,
                    "label": cluster_label,
                    "center": center.tolist(),
                    "memories": cluster_memories,
                    "memory_count": len(cluster_memories)
                })
            
            return clusters
            
        except Exception as e:
            logger.error(f"Error getting memory clusters: {e}")
            raise
    
    def _generate_cluster_label(self, memories: List[MemoryData]) -> str:
        """生成聚类标签"""
        if not memories:
            return "空聚类"
        
        # 提取所有记忆的内容
        contents = [memory.content for memory in memories]
        
        # 使用TF-IDF提取关键词
        from sklearn.feature_extraction.text import TfidfVectorizer
        
        vectorizer = TfidfVectorizer(max_features=10, stop_words='english')
        tfidf_matrix = vectorizer.fit_transform(contents)
        
        # 获取特征名称
        feature_names = vectorizer.get_feature_names_out()
        
        # 计算每个关键词的平均TF-IDF分数
        mean_scores = tfidf_matrix.mean(axis=0).tolist()[0]
        
        # 按分数排序
        keyword_scores = list(zip(feature_names, mean_scores))
        keyword_scores.sort(key=lambda x: x[1], reverse=True)
        
        # 取前3个关键词作为聚类标签
        top_keywords = [keyword for keyword, score in keyword_scores[:3]]
        
        return ", ".join(top_keywords)
    
    def get_memory_timeline(self, memory_type: Optional[str] = None) -> List[Dict[str, Any]]:
```

    def transition_memory(self, memory_id: str) -> bool:

            evaluation = self.evaluate_memory_for_transition(memory_id)

            # 获取记忆

            if not memory:
                raise ValueError(f"Memory not found: {memory_id}")
            
            # 应用过渡策略
            policy = self._select_transition_policy(memory)
            
            # 执行过渡
            success = self._execute_transition(memory, policy)
            
            if success:

                memory.transitioned = True
                memory.transitioned_at = datetime.now()
                
                # 更新记忆
                self.memory_storage.update_memory(memory)

    def get_transition_statistics(self) -> Dict[str, Any]:

```from datetime import datetime, timedelta
        # 使用余弦相似度
        query_np = np.array(query_vector)
        memory_np = np.array(memory_vector)
        
        # 计算余弦相似度
        dot_product = np.dot(query_np, memory_np)
        norm_query = np.linalg.norm(query_np)
        norm_memory = np.linalg.norm(memory_np)
        
        if norm_query == 0 or norm_memory == 0:
            return 0.0
        
        similarity = dot_product / (norm_query * norm_memory)
        
        # 确保结果在[0, 1]范围内
        return max(0.0, min(1.0, similarity))

        """计算关键词相关性"""
        query_words = set(query.lower().split())
        content_words = set(content.lower().split())
        
        if not query_words:
            return 0.0
        
        # 计算交集
        intersection = query_words.intersection(content_words)
        
        # 计算相关性分数
        relevance = len(intersection) / len(query_words)
        
        return relevance
    
    def _calculate_temporal_relevance(self, memory: MemoryData, 
                                    start_time: datetime, 
                                    end_time: datetime) -> float:
        """计算时间相关性"""
        if not memory.created_at:
            return 0.0
        
        # 计算记忆时间与查询时间范围的关系
        if start_time <= memory.created_at <= end_time:
            # 在时间范围内，根据距离中心点的远近计算相关性
            center_time = start_time + (end_time - start_time) / 2
            distance = abs((memory.created_at - center_time).total_seconds())
            max_distance = (end_time - start_time).total_seconds() / 2
            
            if max_distance == 0:
                return 1.0
            
            relevance = 1.0 - (distance / max_distance)
            return max(0.0, relevance)
        else:
            # 不在时间范围内，根据距离时间范围的远近计算相关性
            if memory.created_at < start_time:
                distance = (start_time - memory.created_at).total_seconds()
            else:
                distance = (memory.created_at - end_time).total_seconds()
            
            # 距离越远，相关性越低
            relevance = max(0.0, 1.0 - (distance / (7 * 24 * 3600)))  # 7天为基准
            
            return relevance

            association = self.graph_storage.get_association(seed_id, memory_id)
```

```python
        self.graph_storage = GraphStorage(config.graph_storage)
        self.analyzer = MemoryAnalyzer(config.analyzer)
        self.cache = CacheManager(config.cache)
        
        # 过渡规则
        self.transition_rules = self._load_transition_rules()
        
        # 过渡历史
        self.transition_history = []
        
        logger.info("MemoryTransition initialized")
    
    def evaluate_transition(self, memory_data: MemoryData) -> Dict[str, Any]:
        """评估记忆是否需要过渡"""
        try:
            # 计算过渡分数
            transition_score = self._calculate_transition_score(memory_data)
            
            # 应用过渡规则
            applicable_rules = self._find_applicable_rules(memory_data, transition_score)
            
            # 确定过渡状态
            if transition_score >= self.config.transition_threshold:
                status = TransitionStatus.READY
            elif transition_score >= self.config.transition_threshold * 0.7:
                status = TransitionStatus.PENDING
            else:
                status = TransitionStatus.NOT_READY
            
            # 创建评估结果
            result = {
                "memory_id": memory_data.memory_id,
                "transition_score": transition_score,
                "status": status,
                "applicable_rules": applicable_rules,
                "evaluation_time": datetime.now()
            }
            
            # 缓存评估结果
            cache_key = f"transition_eval:{memory_data.memory_id}"
            self.cache.set(cache_key, result, expire=3600)  # 1小时过期
            
            return result
            
        except Exception as e:
            logger.error(f"Error evaluating transition: {e}")
            raise
    
    def transition_memory(self, memory_id: str, target_system: str = "long_term_memory") -> bool:
        """过渡记忆到目标系统"""
        try:
            # 获取记忆
            memory = self.memory_storage.get_memory(memory_id)
            if not memory:
                logger.error(f"Memory not found: {memory_id}")
                return False
            
            # 评估过渡状态
            evaluation = self.evaluate_transition(memory)
            if evaluation["status"] != TransitionStatus.READY:
                logger.warning(f"Memory {memory_id} not ready for transition: {evaluation['status']}")
                return False
            
            # 执行过渡
            success = self._execute_transition(memory, target_system)
            
            if success:
                # 更新记忆状态
                memory.transitioned = True
                memory.transitioned_at = datetime.now()
                memory.transition_target = target_system
                
                # 更新存储
                self.memory_storage.update_memory(memory)
                self.persistent_storage.update_memory(memory)
                
                # 记录过渡历史
                self._record_transition(memory_id, target_system, evaluation)
                
                logger.info(f"Memory {memory_id} transitioned to {target_system}")
                return True
            else:
                logger.error(f"Failed to transition memory {memory_id}")
                return False
                
        except Exception as e:
            logger.error(f"Error transitioning memory: {e}")
            raise
    
    def batch_transition(self, limit: int = 10) -> List[str]:
        """批量过渡记忆"""
        try:
            # 获取准备过渡的记忆
            ready_memories = self._get_ready_memories(limit)
            
            transitioned_memories = []
            
            for memory_data in ready_memories:
                try:
                    # 过渡记忆
                    if self.transition_memory(memory_data.memory_id):
                        transitioned_memories.append(memory_data.memory_id)
                except Exception as e:
                    logger.error(f"Error transitioning memory {memory_data.memory_id}: {e}")
                    continue
            
            logger.info(f"Batch transition completed: {len(transitioned_memories)} memories")
            return transitioned_memories
            
        except Exception as e:
            logger.error(f"Error in batch transition: {e}")
            raise
    
    def get_transition_history(self, memory_id: Optional[str] = None,

        """获取过渡历史"""
        try:
            if memory_id:

                history = [h for h in self.transition_history if h["memory_id"] == memory_id]
            else:
                # 获取所有过渡历史
                history = self.transition_history.copy()
            
            # 按时间倒序排序
            history.sort(key=lambda h: h["transition_time"], reverse=True)
            
            # 限制结果数量
            if limit and len(history) > limit:
                history = history[:limit]
            
            return history
            
        except Exception as e:
            logger.error(f"Error getting transition history: {e}")
            raise
    
    def add_transition_rule(self, rule: TransitionRule) -> str:
        """添加过渡规则"""
        try:
            # 生成规则ID
            if not rule.rule_id:
                rule.rule_id = str(uuid.uuid4())

    def update_transition_rule(self, rule_id: str, updates: Dict[str, Any]) -> bool:

            raise
    
    def delete_transition_rule(self, rule_id: str) -> bool:

        try:

            for i, rule in enumerate(self.transition_rules):
                if rule.rule_id == rule_id:
                    rule_index = i

            if rule_index is None:

    def _calculate_transition_score(self, memory_data: MemoryData) -> float:
        """计算过渡分数"""
        # 基础分数 (基于重要性)
        importance_scores = {
            MemoryImportance.LOW: 0.2,
            MemoryImportance.MEDIUM: 0.5,
            MemoryImportance.HIGH: 0.8,
            MemoryImportance.CRITICAL: 1.0
        }
        base_score = importance_scores.get(memory_data.importance, 0.5)
        
        # 访问频率加成
        access_bonus = 0.0
        if hasattr(memory_data, 'access_frequency'):
            access_bonus = min(memory_data.access_frequency / 10, 0.3)  # 最多0.3的加成
        
        # 时间加成 (记忆存在时间越长，分数越高)
        time_bonus = 0.0
        if memory_data.created_at:
            days_alive = (datetime.now() - memory_data.created_at).days
            time_bonus = min(days_alive / 30, 0.2)  # 最多0.2的加成
        
        # 计算总分数
        total_score = base_score + access_bonus + time_bonus
        
        return min(total_score, 1.0)  # 确保分数不超过1.0
    
    def _find_applicable_rules(self, memory_data: MemoryData, 
                              transition_score: float) -> List[TransitionRule]:
        """查找适用的过渡规则"""
        applicable_rules = []
        
        for rule in self.transition_rules:
            if not rule.active:
                continue
            
            # 检查条件
            if self._check_rule_conditions(rule.conditions, memory_data, transition_score):
                applicable_rules.append(rule)
        
        # 按优先级排序
        applicable_rules.sort(key=lambda r: r.actions.get("priority", "medium"))
        
        return applicable_rules
    
    def _check_rule_conditions(self, conditions: Dict[str, Any], 
                              memory_data: MemoryData, 
                              transition_score: float) -> bool:
        """检查规则条件"""
        for key, condition in conditions.items():
            if key == "importance":
                if memory_data.importance != condition:
                    return False
            elif key == "access_frequency":
                if isinstance(condition, dict):
                    operator = condition.get("operator", "==")
                    value = condition.get("value", 0)
                    
                    if not hasattr(memory_data, 'access_frequency'):
                        return False
                    
                    if operator == ">=" and memory_data.access_frequency < value:
                        return False
                    elif operator == ">" and memory_data.access_frequency <= value:
                        return False
                    elif operator == "==" and memory_data.access_frequency != value:
                        return False
                    elif operator == "!=" and memory_data.access_frequency == value:
                        return False
                    elif operator == "<=" and memory_data.access_frequency > value:
                        return False
                    elif operator == "<" and memory_data.access_frequency >= value:
                        return False
                else:
                    if not hasattr(memory_data, 'access_frequency') or memory_data.access_frequency != condition:
                        return False
            elif key == "age_days":
                if isinstance(condition, dict):
                    operator = condition.get("operator", "==")
                    value = condition.get("value", 0)
                    
                    if not memory_data.created_at:
                        return False
                    
                    days_alive = (datetime.now() - memory_data.created_at).days
                    
                    if operator == ">=" and days_alive < value:
                        return False
                    elif operator == ">" and days_alive <= value:
                        return False
                    elif operator == "==" and days_alive != value:
                        return False
                    elif operator == "!=" and days_alive == value:
                        return False
                    elif operator == "<=" and days_alive > value:
                        return False
                    elif operator == "<" and days_alive >= value:
                        return False
                else:
                    if not memory_data.created_at:
                        return False
                    
                    days_alive = (datetime.now() - memory_data.created_at).days
                    if days_alive != condition:
                        return False
            elif key == "transition_score":
                if isinstance(condition, dict):
                    operator = condition.get("operator", "==")
                    value = condition.get("value", 0)
                    
                    if operator == ">=" and transition_score < value:
                        return False
                    elif operator == ">" and transition_score <= value:
                        return False
                    elif operator == "==" and transition_score != value:
                        return False
                    elif operator == "!=" and transition_score == value:
                        return False
                    elif operator == "<=" and transition_score > value:
                        return False
                    elif operator == "<" and transition_score >= value:
                        return False
                else:
                    if transition_score != condition:
                        return False
        
        return True
    
    def _get_ready_memories(self, limit: int) -> List[MemoryData]:
        """获取准备过渡的记忆"""
        try:
            # 从内存存储获取未过渡的记忆
            all_memories = self.memory_storage.list_memories(
                transitioned=False, limit=limit * 2  # 获取更多记忆用于筛选
            )
            
            ready_memories = []

                if evaluation["status"] == TransitionStatus.READY:
                    ready_memories.append(memory)
                
                # 限制结果数量
                if len(ready_memories) >= limit:
                    break
            
            # 按过渡分数排序
            ready_memories.sort(
                key=lambda m: self._calculate_transition_score(m), 
                reverse=True
            )
            
            return ready_memories
            
        except Exception as e:
            logger.error(f"Error getting ready memories: {e}")
            raise
    
    def _execute_transition(self, memory_data: MemoryData, target_system: str) -> bool:
        """执行过渡"""
        try:
            if target_system == "long_term_memory":
                # 这里应该调用长期记忆模块的API

                return True
            else:
                logger.error(f"Unsupported target system: {target_system}")
                return False
                
        except Exception as e:

            return False
    
    def _record_transition(self, memory_id: str, target_system: str,
```

```python
    def consolidate_similar_memories(self, similarity_threshold: float = 0.8) -> int:
        """合并相似记忆"""
        try:
            with self.management_lock:

                            continue
                        
                        # 确定主记忆（重要性更高的）
                        if memories[i].importance.value > memories[j].importance.value:
                            primary_index, secondary_index = i, j
                        elif memories[i].importance.value < memories[j].importance.value:
                            primary_index, secondary_index = j, i
                        else:
                            # 重要性相同，选择访问次数更多的
                            if (hasattr(memories[i], 'access_count') and hasattr(memories[j], 'access_count') and
                                memories[i].access_count >= memories[j].access_count):
                                primary_index, secondary_index = i, j
                            else:
                                primary_index, secondary_index = j, i
                        
                        # 合并记忆
                        primary_memory = memories[primary_index]
                        secondary_memory = memories[secondary_index]
                        
                        # 合并元数据
                        for key, value in secondary_memory.metadata.items():
                            if key not in primary_memory.metadata:
                                primary_memory.metadata[key] = value
                        
                        # 合并标签
                        for tag in secondary_memory.tags:
                            if tag not in primary_memory.tags:
                                primary_memory.tags.append(tag)
                        
                        # 更新主记忆
                        primary_memory.updated_at = datetime.now()
                        primary_memory.metadata["consolidated_from"] = secondary_memory.memory_id
                        primary_memory.metadata["consolidation_similarity"] = similarity
                        
                        self.memory_storage.update_memory(primary_memory)
                        self.persistent_storage.update_memory(primary_memory)
                        
                        # 删除次要记忆
                        self.memory_storage.delete_memory(secondary_memory.memory_id)
                        self.persistent_storage.delete_memory(secondary_memory.memory_id)
                        self.cache.delete(f"memory:{secondary_memory.memory_id}")
                        
                        # 标记为已处理
                        processed_indices.add(secondary_index)
                        
                        consolidated_count += 1
                
                logger.info(f"Consolidated {consolidated_count} similar memories")
                return consolidated_count
                
        except Exception as e:
            logger.error(f"Error consolidating similar memories: {e}")
            raise
    
    def reinforce_important_memories(self) -> int:
        """强化重要记忆"""
        try:
            with self.management_lock:
                # 获取高重要性记忆
                important_memories = self.memory_storage.list_memories(
                    importance=MemoryImportance.HIGH,
                    limit=100
                )
                
                reinforced_count = 0
                
                for memory in important_memories:
                    # 检查是否需要强化
                    if self._should_reinforce_memory(memory):
                        # 强化记忆
                        memory.importance = MemoryImportance.CRITICAL
                        memory.reinforced_at = datetime.now()
                        
                        # 更新存储
                        self.memory_storage.update_memory(memory)
                        self.persistent_storage.update_memory(memory)
                        
                        # 清除缓存
                        self.cache.delete(f"memory:{memory.memory_id}")
                        
                        reinforced_count += 1
                
                logger.info(f"Reinforced {reinforced_count} important memories")
                return reinforced_count
                
        except Exception as e:
            logger.error(f"Error reinforcing important memories: {e}")
            raise
    
    def get_management_statistics(self) -> Dict[str, Any]:
        """获取管理统计信息"""
        try:
            # 获取存储统计
            storage_stats = self.memory_storage.get_statistics()
            
            # 计算管理指标
            total_memories = storage_stats.get("total_memories", 0)
            memory_usage = storage_stats.get("memory_usage", 0)
            
            # 按重要性分组统计
            importance_stats = {}
            for importance in MemoryImportance:
                count = self.memory_storage.count_memories(importance=importance)
                importance_stats[importance.value] = count
            
            # 按类型分组统计
            type_stats = {}
            for memory_type in MemoryType:
                count = self.memory_storage.count_memories(memory_type=memory_type)
                type_stats[memory_type.value] = count
            
            # 计算管理指标
            management_stats = {
                "total_memories": total_memories,
                "memory_usage_mb": memory_usage,
                "memory_usage_percent": (memory_usage / self.config.max_memory_usage) * 100,
                "importance_distribution": importance_stats,
                "type_distribution": type_stats,
                "last_cleanup": getattr(self, '_last_cleanup_time', None),
                "last_optimization": getattr(self, '_last_optimization_time', None),
                "last_consolidation": getattr(self, '_last_consolidation_time', None)
            }
            
            return management_stats
            
        except Exception as e:
            logger.error(f"Error getting management statistics: {e}")
            raise
    
    def _calculate_similarity_matrix(self, memories: List[MemoryData]) -> List[List[float]]:
        """计算记忆相似度矩阵"""
        n = len(memories)
        similarity_matrix = [[0.0 for _ in range(n)] for _ in range(n)]
        
        for i in range(n):
            for j in range(i, n):
                if i == j:
                    similarity_matrix[i][j] = 1.0
                else:
                    # 计算相似度
                    similarity = self._calculate_memory_similarity(memories[i], memories[j])
                    similarity_matrix[i][j] = similarity
                    similarity_matrix[j][i] = similarity
        
        return similarity_matrix
    
    def _calculate_memory_similarity(self, memory1: MemoryData, memory2: MemoryData) -> float:
        """计算两个记忆的相似度"""
        # 内容相似度
        content_similarity = self._calculate_content_similarity(memory1.content, memory2.content)
        
        # 标签相似度
        tags1 = set(memory1.tags)
        tags2 = set(memory2.tags)
        
        if tags1 or tags2:
            intersection = tags1.intersection(tags2)
            union = tags1.union(tags2)
            tags_similarity = len(intersection) / len(union) if union else 0.0
        else:
            tags_similarity = 0.0
        
        # 综合相似度
        overall_similarity = content_similarity * 0.8 + tags_similarity * 0.2
        
        return overall_similarity
    
    def _calculate_content_similarity(self, content1: str, content2: str) -> float:
        """计算内容相似度"""
        # 简单的词汇重叠相似度
        words1 = set(content1.lower().split())
        words2 = set(content2.lower().split())
        
        if not words1 or not words2:
            return 0.0
        
        intersection = words1.intersection(words2)
        union = words1.union(words2)
        
        similarity = len(intersection) / len(union)
        
        return similarity
    
    def _should_reinforce_memory(self, memory: MemoryData) -> bool:
        """判断是否应该强化记忆"""
        # 如果已经是关键重要性，不需要强化
        if memory.importance == MemoryImportance.CRITICAL:
            return False
        
        # 检查访问频率
        if (hasattr(memory, 'access_frequency') and 
            memory.access_frequency >= self.config.reinforce_access_threshold):
            return True
        
        # 检查访问次数
        if (hasattr(memory, 'access_count') and 
            memory.access_count >= self.config.reinforce_access_count):
            return True
        
        # 检查记忆年龄
        if memory.created_at:
            days_alive = (datetime.now() - memory.created_at).days
            if days_alive >= self.config.reinforce_age_threshold:
                return True
        
        return False
    
    def _start_background_tasks(self):
        """启动后台管理任务"""
        # 启动定期清理任务
        cleanup_thread = threading.Thread(target=self._periodic_cleanup, daemon=True)
        cleanup_thread.start()
        
        # 启动定期优化任务
        optimization_thread = threading.Thread(target=self._periodic_optimization, daemon=True)
        optimization_thread.start()
        
        # 启动定期合并任务
        consolidation_thread = threading.Thread(target=self._periodic_consolidation, daemon=True)
        consolidation_thread.start()
        
        logger.info("Background management tasks started")
    
    def _periodic_cleanup(self):
        """定期清理任务"""
        while True:
            try:
                # 等待清理间隔
                time.sleep(self.config.cleanup_interval)
                
                # 执行清理
                deleted_count = self.cleanup_old_memories()
                self._last_cleanup_time = datetime.now()
                
                logger.info(f"Periodic cleanup completed: {deleted_count} memories deleted")
                
            except Exception as e:
                logger.error(f"Error in periodic cleanup: {e}")
    
    def _periodic_optimization(self):
        """定期优化任务"""
        while True:
            try:
                # 等待优化间隔
                time.sleep(self.config.optimization_interval)
                
                # 执行优化
                result = self.optimize_memory_storage()
                self._last_optimization_time = datetime.now()
                
                logger.info(f"Periodic optimization completed: {result}")
                
            except Exception as e:
                logger.error(f"Error in periodic optimization: {e}")
    
    def _periodic_consolidation(self):
        """定期合并任务"""
        while True:
            try:
                # 等待合并间隔
                time.sleep(self.config.consolidation_interval)
                
                # 执行合并
                consolidated_count = self.consolidate_similar_memories()
                self._last_consolidation_time = datetime.now()
                
                logger.info(f"Periodic consolidation completed: {consolidated_count} memories consolidated")
                
            except Exception as e:
                logger.error(f"Error in periodic consolidation: {e}")
```

```python
# utils/model_optimizer.py
import torch
from transformers import AutoModel, AutoTokenizer
import numpy as np
from typing import List, Dict, Any

class ModelOptimizer:
    """模型优化器"""
    
    def __init__(self, model_name: str, device: str = "cpu"):
        self.model_name = model_name
        self.device = device
        self.model = None
        self.tokenizer = None
        self._load_model()
    
    def _load_model(self):
        """加载模型"""
        try:
            # 加载分词器
            self.tokenizer = AutoTokenizer.from_pretrained(self.model_name)
            
            # 加载模型
            self.model = AutoModel.from_pretrained(self.model_name)
            
            # 移动到设备
            self.model.to(self.device)
            
            # 设置为评估模式
            self.model.eval()
            
            # 如果是GPU，使用半精度
            if self.device == "cuda":
                self.model.half()
            
            print(f"Model {self.model_name} loaded on {self.device}")
            
        except Exception as e:
            print(f"Error loading model: {e}")
            raise
    
    def encode(self, texts: List[str], batch_size: int = 32) -> np.ndarray:
        """编码文本"""
        if not self.model or not self.tokenizer:
            raise ValueError("Model not loaded")
        
        all_embeddings = []
        
        # 批量处理
        for i in range(0, len(texts), batch_size):
            batch_texts = texts[i:i+batch_size]
            
            # 分词
            inputs = self.tokenizer(
                batch_texts,
                padding=True,
                truncation=True,
                return_tensors="pt"
            )
            
            # 移动到设备
            inputs = {k: v.to(self.device) for k, v in inputs.items()}
            
            # 获取嵌入
            with torch.no_grad():
                outputs = self.model(**inputs)
                embeddings = outputs.last_hidden_state.mean(dim=1).cpu().numpy()
                all_embeddings.append(embeddings)
        
        # 合并所有嵌入
        return np.vstack(all_embeddings)
    
    def optimize_for_inference(self):
        """优化模型用于推理"""
        try:
            # 融合卷积层
            if hasattr(torch.jit, 'optimize_for_inference'):
                self.model = torch.jit.optimize_for_inference(torch.jit.script(self.model))
            
            # 量化模型（仅CPU）
            if self.device == "cpu":
                self.model = torch.quantization.quantize_dynamic(
                    self.model, {torch.nn.Linear}, dtype=torch.qint8
                )
            
            print("Model optimized for inference")
            
        except Exception as e:
            print(f"Error optimizing model: {e}")
```

```python
# utils/cache_optimizer.py
import redis
import pickle
import json
import hashlib
from typing import Any, Optional, Dict, List
from datetime import datetime, timedelta

class CacheOptimizer:
    """缓存优化器"""
    
    def __init__(self, redis_url: str, default_ttl: int = 3600):
        self.redis_client = redis.from_url(redis_url)
        self.default_ttl = default_ttl
        self.hit_count = 0
        self.miss_count = 0
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存值"""
        try:
            data = self.redis_client.get(key)
            if data:
                self.hit_count += 1
                return pickle.loads(data)
            else:
                self.miss_count += 1
                return None
        except Exception as e:
            print(f"Error getting cache: {e}")
            self.miss_count += 1
            return None
    
    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """设置缓存值"""
        try:
            ttl = ttl or self.default_ttl
            data = pickle.dumps(value)
            return self.redis_client.setex(key, ttl, data)
        except Exception as e:
            print(f"Error setting cache: {e}")
            return False
    
    def delete(self, key: str) -> bool:
        """删除缓存值"""
        try:
            return bool(self.redis_client.delete(key))
        except Exception as e:
            print(f"Error deleting cache: {e}")
            return False
    
    def get_or_set(self, key: str, func, ttl: Optional[int] = None) -> Any:
        """获取缓存值，如果不存在则设置"""
        value = self.get(key)
        if value is not None:
            return value
        
        value = func()
        self.set(key, value, ttl)
        return value
    
    def get_hash(self, data: Any) -> str:
        """获取数据的哈希值"""
        if isinstance(data, str):
            data_str = data
        elif isinstance(data, dict):
            data_str = json.dumps(data, sort_keys=True)
        else:
            data_str = str(data)
        
        return hashlib.md5(data_str.encode()).hexdigest()
    
    def get_query_cache_key(self, query: Dict[str, Any]) -> str:
        """获取查询缓存键"""
        query_hash = self.get_hash(query)
        return f"query:{query_hash}"
    
    def get_memory_cache_key(self, memory_id: str) -> str:
        """获取记忆缓存键"""
        return f"memory:{memory_id}"
    
    def get_association_cache_key(self, memory_id: str, association_type: Optional[str] = None) -> str:
        """获取关联缓存键"""
        if association_type:
            return f"association:{memory_id}:{association_type}"
        else:
            return f"association:{memory_id}"
    
    def invalidate_memory_cache(self, memory_id: str):
        """使记忆缓存失效"""
        memory_key = self.get_memory_cache_key(memory_id)
        self.delete(memory_key)
        
        # 使关联缓存失效
        association_key = self.get_association_cache_key(memory_id)
        self.delete(association_key)
    
    def get_statistics(self) -> Dict[str, Any]:
        """获取缓存统计"""
        total_requests = self.hit_count + self.miss_count
        hit_rate = self.hit_count / total_requests if total_requests > 0 else 0
        
        info = self.redis_client.info()
        
        return {
            "hit_count": self.hit_count,
            "miss_count": self.miss_count,
            "hit_rate": hit_rate,
            "total_requests": total_requests,
            "redis_memory_used": info.get("used_memory", 0),
            "redis_memory_human": info.get("used_memory_human", "0B"),
            "redis_connected_clients": info.get("connected_clients", 0)
        }
    
    def cleanup_expired(self) -> int:
        """清理过期缓存"""
        # Redis会自动清理过期键，这里只是统计
        info = self.redis_client.info()
        expired_keys = info.get("expired_keys", 0)
        return expired_keys
```

```python
# utils/concurrent_processor.py
import asyncio
import concurrent.futures
from typing import List, Dict, Any, Callable, Optional
import time
import threading
from queue import Queue

class ConcurrentProcessor:
    """并发处理器"""
    
    def __init__(self, max_workers: int = 4):
        self.max_workers = max_workers
        self.thread_pool = concurrent.futures.ThreadPoolExecutor(max_workers=max_workers)
        self.process_pool = concurrent.futures.ProcessPoolExecutor(max_workers=max_workers)
        self.task_queue = Queue()
        self.result_queue = Queue()
        self.worker_thread = None
        self.running = False
    
    def start(self):
        """启动处理器"""
        if not self.running:
            self.running = True
            self.worker_thread = threading.Thread(target=self._process_tasks)
            self.worker_thread.start()
    
    def stop(self):
        """停止处理器"""
        if self.running:
            self.running = False
            if self.worker_thread:
                self.worker_thread.join()
    
    def submit_task(self, func: Callable, *args, **kwargs) -> str:
        """提交任务"""
        task_id = str(time.time())
        task = {
            "id": task_id,
            "func": func,
            "args": args,
            "kwargs": kwargs
        }
        self.task_queue.put(task)
        return task_id
    
    def get_result(self, task_id: str, timeout: Optional[float] = None) -> Any:
        """获取任务结果"""
        start_time = time.time()
        
        while True:
            if not self.result_queue.empty():
                result = self.result_queue.get()
                if result["id"] == task_id:
                    if "error" in result:
                        raise result["error"]
                    return result["result"]
            
            # 检查超时
            if timeout and (time.time() - start_time) > timeout:
                raise TimeoutError(f"Task {task_id} timed out")
            
            time.sleep(0.01)
    
    def process_in_parallel(self, func: Callable, items: List[Any], use_processes: bool = False) -> List[Any]:
        """并行处理项目"""
        executor = self.process_pool if use_processes else self.thread_pool
        
        futures = []
        for item in items:
            future = executor.submit(func, item)
            futures.append(future)
        
        results = []
        for future in concurrent.futures.as_completed(futures):
            try:
                result = future.result()
                results.append(result)
            except Exception as e:
                print(f"Error processing item: {e}")
                results.append(None)
        
        return results
    
    def process_batches(self, func: Callable, items: List[Any], batch_size: int = 32) -> List[Any]:
        """批量处理项目"""
        results = []
        
        for i in range(0, len(items), batch_size):
            batch = items[i:i+batch_size]
            batch_results = self.process_in_parallel(func, batch)
            results.extend(batch_results)
        
        return results
    
    def _process_tasks(self):
        """处理任务"""
        while self.running:
            try:
                if not self.task_queue.empty():
                    task = self.task_queue.get(timeout=0.1)
                    
                    try:
                        result = task["func"](*task["args"], **task["kwargs"])
                        self.result_queue.put({
                            "id": task["id"],
                            "result": result
                        })
                    except Exception as e:
                        self.result_queue.put({
                            "id": task["id"],
                            "error": e
                        })
                else:
                    time.sleep(0.01)
            except Exception as e:
                print(f"Error processing task: {e}")
```

```python
# utils/performance_monitor.py
import time
import psutil
import threading
from typing import Dict, Any, List, Optional
from datetime import datetime, timedelta
from collections import deque

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self, history_size: int = 100):
        self.history_size = history_size
        self.cpu_history = deque(maxlen=history_size)
        self.memory_history = deque(maxlen=history_size)
        self.disk_io_history = deque(maxlen=history_size)
        self.network_io_history = deque(maxlen=history_size)
        self.request_times = deque(maxlen=history_size)
        self.error_count = 0
        self.request_count = 0
        self.running = False
        self.monitor_thread = None
    
    def start(self, interval: float = 1.0):
        """启动监控"""
        if not self.running:
            self.running = True
            self.monitor_thread = threading.Thread(
                target=self._monitor_loop,
                args=(interval,),
                daemon=True
            )
            self.monitor_thread.start()
    
    def stop(self):
        """停止监控"""
        if self.running:
            self.running = False
            if self.monitor_thread:
                self.monitor_thread.join()
    
    def record_request(self, duration: float, success: bool = True):
        """记录请求"""
        self.request_times.append(duration)
        self.request_count += 1
        
        if not success:
            self.error_count += 1
    
    def _monitor_loop(self, interval: float):
        """监控循环"""
        while self.running:
            try:
                # CPU使用率
                cpu_percent = psutil.cpu_percent(interval=None)
                self.cpu_history.append({
                    "timestamp": datetime.now(),
                    "value": cpu_percent
                })
                
                # 内存使用
                memory = psutil.virtual_memory()
                self.memory_history.append({
                    "timestamp": datetime.now(),
                    "value": memory.percent,
                    "used": memory.used,
                    "available": memory.available
                })
                
                # 磁盘IO
                disk_io = psutil.disk_io_counters()
                if disk_io:
                    self.disk_io_history.append({
                        "timestamp": datetime.now(),
                        "read_bytes": disk_io.read_bytes,
                        "write_bytes": disk_io.write_bytes,
                        "read_count": disk_io.read_count,
                        "write_count": disk_io.write_count
                    })
                
                # 网络IO
                network_io = psutil.net_io_counters()
                if network_io:
                    self.network_io_history.append({
                        "timestamp": datetime.now(),
                        "bytes_sent": network_io.bytes_sent,
                        "bytes_recv": network_io.bytes_recv,
                        "packets_sent": network_io.packets_sent,
                        "packets_recv": network_io.packets_recv
                    })
                
                time.sleep(interval)
                
            except Exception as e:
                print(f"Error in monitoring loop: {e}")
                time.sleep(interval)
    
    def get_current_stats(self) -> Dict[str, Any]:
        """获取当前统计信息"""
        # CPU
        current_cpu = self.cpu_history[-1]["value"] if self.cpu_history else 0
        avg_cpu = sum(item["value"] for item in self.cpu_history) / len(self.cpu_history) if self.cpu_history else 0
        
        # 内存
        current_memory = self.memory_history[-1]["value"] if self.memory_history else 0
        avg_memory = sum(item["value"] for item in self.memory_history) / len(self.memory_history) if self.memory_history else 0
        
        # 请求时间
        avg_request_time = sum(self.request_times) / len(self.request_times) if self.request_times else 0
        max_request_time = max(self.request_times) if self.request_times else 0
        min_request_time = min(self.request_times) if self.request_times else 0
        
        # 错误率
        error_rate = self.error_count / self.request_count if self.request_count > 0 else 0
        
        return {
            "cpu": {
                "current": current_cpu,
                "average": avg_cpu
            },
            "memory": {
                "current": current_memory,
                "average": avg_memory
            },
            "requests": {
                "count": self.request_count,
                "errors": self.error_count,
                "error_rate": error_rate,
                "avg_time": avg_request_time,
                "max_time": max_request_time,
                "min_time": min_request_time
            },
            "timestamp": datetime.now().isoformat()
        }
    
    def get_historical_stats(self, hours: int = 1) -> Dict[str, Any]:
        """获取历史统计信息"""
        cutoff_time = datetime.now() - timedelta(hours=hours)
        
        # 过滤历史数据
        recent_cpu = [item for item in self.cpu_history if item["timestamp"] > cutoff_time]
        recent_memory = [item for item in self.memory_history if item["timestamp"] > cutoff_time]
        
        # 计算统计
        cpu_values = [item["value"] for item in recent_cpu]
        memory_values = [item["value"] for item in recent_memory]
        
        return {
            "time_range_hours": hours,
            "cpu": {
                "count": len(cpu_values),
                "min": min(cpu_values) if cpu_values else 0,
                "max": max(cpu_values) if cpu_values else 0,
                "avg": sum(cpu_values) / len(cpu_values) if cpu_values else 0
            },
            "memory": {
                "count": len(memory_values),
                "min": min(memory_values) if memory_values else 0,
                "max": max(memory_values) if memory_values else 0,
                "avg": sum(memory_values) / len(memory_values) if memory_values else 0
            },
            "timestamp": datetime.now().isoformat()
        }
```

```python
# utils/logger_config.py
import logging
import logging.handlers
import os
from datetime import datetime
from typing import Dict, Any

def setup_logging(config: Dict[str, Any]):
    """设置日志配置"""
    # 创建日志目录
    log_dir = os.path.dirname(config.get("file", "logs/app.log"))
    os.makedirs(log_dir, exist_ok=True)
    
    # 创建根日志器
    root_logger = logging.getLogger()
    root_logger.setLevel(getattr(logging, config.get("level", "INFO")))
    
    # 清除现有处理器
    for handler in root_logger.handlers[:]:
        root_logger.removeHandler(handler)
    
    # 创建格式化器
    formatter = logging.Formatter(config.get("format", "%(asctime)s - %(name)s - %(levelname)s - %(message)s"))
    
    # 控制台处理器
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.INFO)
    console_handler.setFormatter(formatter)
    root_logger.addHandler(console_handler)
    
    # 文件处理器（按大小轮转）
    file_handler = logging.handlers.RotatingFileHandler(
        config.get("file", "logs/app.log"),
        maxBytes=config.get("max_bytes", 10 * 1024 * 1024),  # 10MB
        backupCount=config.get("backup_count", 5)
    )
    file_handler.setLevel(logging.DEBUG)
    file_handler.setFormatter(formatter)
    root_logger.addHandler(file_handler)
    
    # 错误文件处理器（按时间轮转）
    error_handler = logging.handlers.TimedRotatingFileHandler(
        config.get("error_file", "logs/error.log"),
        when="midnight",
        interval=1,
        backupCount=config.get("backup_count", 30)
    )
    error_handler.setLevel(logging.ERROR)
    error_handler.setFormatter(formatter)
    root_logger.addHandler(error_handler)
    
    # 设置第三方库日志级别
    logging.getLogger("urllib3").setLevel(logging.WARNING)
    logging.getLogger("requests").setLevel(logging.WARNING)
    logging.getLogger("neo4j").setLevel(logging.WARNING)
    
    return root_logger

# 默认日志配置
```
