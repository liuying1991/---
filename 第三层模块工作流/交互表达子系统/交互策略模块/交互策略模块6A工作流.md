# 交互策略模块6A工作流

## 模块概述

### 模块名称
交互策略模块

### 所属子系统
交互表达子系统

### 模块功能描述
交互策略模块负责制定和管理婴儿AI管家的交互策略，根据婴儿的状态、偏好和反馈，动态调整交互方式、内容和节奏。该模块通过分析婴儿的个性化特征和交互历史，生成最适合的交互策略，确保交互的有效性、适应性和个性化。

### 模块职责
1. 策略制定：制定适合婴儿的交互策略
2. 策略管理：管理和维护交互策略库
3. 策略选择：根据情境选择最合适的策略
4. 策略调整：根据反馈调整交互策略
5. 策略优化：优化策略效果和适应性
6. 个性化定制：定制个性化交互策略

### 模块接口
- **输入接口**：婴儿状态数据、交互历史数据、反馈数据、环境数据
- **输出接口**：交互策略、策略参数、调整建议
- **依赖模块**：感知处理子系统、决策模块、反馈处理模块、交互管理模块
- **被依赖模块**：交互控制模块、交互管理模块、反馈处理模块

        try:
            # 分析婴儿状态

class StrategySelector:
    def __init__(self, strategy_repository, selection_model):
        self.strategy_repository = strategy_repository
        self.selection_model = selection_model
        
    async def select_strategy(self, baby_analysis, interaction_analysis, feedback_analysis):
        try:
            # 获取候选策略
            candidate_strategies = await self.strategy_repository.get_candidate_strategies(
                baby_analysis, interaction_analysis, feedback_analysis
            )
            
            # 提取特征
            features = await self._extract_features(
                baby_analysis, interaction_analysis, feedback_analysis
            )
            
            # 预测策略匹配度
            strategy_scores = await self._predict_strategy_scores(
                candidate_strategies, features
            )
            
            # 排序策略
            sorted_strategies = await self._sort_strategies(
                candidate_strategies, strategy_scores
            )
            
            # 选择最适合的策略
            selected_strategy = sorted_strategies[0] if sorted_strategies else None
            
            return {
                "selected_strategy": selected_strategy,
                "candidate_strategies": candidate_strategies,
                "strategy_scores": strategy_scores,
                "sorted_strategies": sorted_strategies
            }
            
        except Exception as e:
            return {
                "error": str(e)
            }
    
    async def _extract_features(self, baby_analysis, interaction_analysis, feedback_analysis):
        # 提取婴儿特征
        baby_features = await self._extract_baby_features(baby_analysis)
        
        # 提取交互特征
        interaction_features = await self._extract_interaction_features(interaction_analysis)
        
        # 提取反馈特征
        feedback_features = await self._extract_feedback_features(feedback_analysis)
        
        # 合并特征
        features = {
            **baby_features,
            **interaction_features,
            **feedback_features
        }
        
        return features
    
    async def _extract_baby_features(self, baby_analysis):
        # 提取年龄特征
        age = baby_analysis.get("age", 0)
        age_group = self._get_age_group(age)
        
        # 提取状态特征
        state = baby_analysis.get("state", "normal")
        state_score = self._get_state_score(state)
        
        # 提取偏好特征
        preferences = baby_analysis.get("preferences", {})
        preference_features = self._extract_preference_features(preferences)
        
        # 提取习惯特征
        habits = baby_analysis.get("habits", {})
        habit_features = self._extract_habit_features(habits)
        
        # 提取个性特征
        personality = baby_analysis.get("personality", {})
        personality_features = self._extract_personality_features(personality)
        
        return {
            "age": age,
            "age_group": age_group,
            "state": state,
            "state_score": state_score,
            **preference_features,
            **habit_features,

        strategy_scores = {}
        
        # 转换特征为模型输入格式
        feature_vector = self._convert_features_to_vector(features)
        
        for strategy in candidate_strategies:
            strategy_id = strategy.get("id")
            
            # 获取策略特征
            strategy_features = await self._extract_strategy_features(strategy)
            
            # 合并特征
            combined_features = {
                **features,
                **strategy_features
            }
            
            # 转换为模型输入格式
            input_vector = self._convert_features_to_vector(combined_features)
            
            # 预测匹配度
            score = self.selection_model.predict_proba([input_vector])[0][1]
            
            strategy_scores[strategy_id] = score
        
        return strategy_scores
    
    async def _sort_strategies(self, candidate_strategies, strategy_scores):
        # 为每个策略添加分数
        scored_strategies = []
        for strategy in candidate_strategies:

            scored_strategy = {

                "score": score
            }
            scored_strategies.append(scored_strategy)
        
        # 按分数排序
        sorted_strategies = sorted(scored_strategies, key=lambda x: x["score"], reverse=True)
        
        return sorted_strategies

class StrategyAdjuster:
    def __init__(self, adjustment_model, adjustment_rules):
        self.adjustment_model = adjustment_model
        self.adjustment_rules = adjustment_rules
        
    async def adjust_strategy(self, current_strategy, feedback_analysis):
        try:
            # 提取反馈特征
            feedback_features = await self._extract_feedback_features(feedback_analysis)
            
            # 提取策略特征
            strategy_features = await self._extract_strategy_features(current_strategy)
            
            # 预测调整方向和幅度
            adjustment_prediction = await self._predict_adjustment(
                feedback_features, strategy_features
            )
            
            # 应用调整规则
            rule_adjustments = await self._apply_adjustment_rules(
                current_strategy, feedback_analysis
            )
            
            # 合并调整建议
            merged_adjustments = await self._merge_adjustments(
                adjustment_prediction, rule_adjustments
            )
            
            # 应用调整
            adjusted_strategy = await self._apply_adjustments(
                current_strategy, merged_adjustments
            )
            
            return {
                "original_strategy": current_strategy,
                "adjusted_strategy": adjusted_strategy,
                "feedback_features": feedback_features,
                "adjustment_prediction": adjustment_prediction,
                "rule_adjustments": rule_adjustments,
                "merged_adjustments": merged_adjustments
            }
            
        except Exception as e:
            return {
                "error": str(e)
            }
    
    async def _extract_feedback_features(self, feedback_analysis):
        # 提取反馈特征
        feedback_summary = feedback_analysis.get("summary", {})
        feedback_trends = feedback_analysis.get("trends", {})
        satisfaction = feedback_analysis.get("satisfaction", {})
        
        # 提取汇总特征
        summary_features = {
            "positive_ratio": feedback_summary.get("positive", 0.0),
            "negative_ratio": feedback_summary.get("negative", 0.0),
            "neutral_ratio": feedback_summary.get("neutral", 0.0)
        }
        
        # 提取趋势特征
        trend_features = {
            "emotion_trend": feedback_trends.get("emotion_trend", 0.0),
            "engagement_trend": feedback_trends.get("engagement_trend", 0.0),
            "attention_trend": feedback_trends.get("attention_trend", 0.0)
        }
        
        # 提取满意度特征
        satisfaction_features = {
            "overall_satisfaction": satisfaction.get("overall", 0.5),
            "content_satisfaction": satisfaction.get("content", 0.5),
            "interaction_satisfaction": satisfaction.get("interaction", 0.5)
        }
        
        return {
            **summary_features,
            **trend_features,
            **satisfaction_features
        }
    
    async def _extract_strategy_features(self, current_strategy):
        # 提取策略特征
        strategy_params = current_strategy.get("parameters", {})
        
        # 提取参数特征
        param_features = {
            "intensity": strategy_params.get("intensity", 0.5),
            "complexity": strategy_params.get("complexity", 0.5),
            "duration": strategy_params.get("duration", 0.5),
            "frequency": strategy_params.get("frequency", 0.5)
        }
        
        # 提取内容特征
        content_features = {
            "educational_ratio": strategy_params.get("educational_ratio", 0.5),
            "entertainment_ratio": strategy_params.get("entertainment_ratio", 0.5),
            "social_ratio": strategy_params.get("social_ratio", 0.5)
        }
        
        return {
            **param_features,
            **content_features
        }
    
    async def _predict_adjustment(self, feedback_features, strategy_features):
        # 合并特征
        combined_features = {
            **feedback_features,
            **strategy_features
        }
        
        # 转换为模型输入格式
        feature_vector = self._convert_features_to_vector(combined_features)
        
        # 预测调整方向和幅度
        adjustment_prediction = self.adjustment_model.predict([feature_vector])[0]
        
        return adjustment_prediction
    
    async def _apply_adjustment_rules(self, current_strategy, feedback_analysis):
        adjustments = []
        
        # 获取反馈数据
        feedback_summary = feedback_analysis.get("summary", {})
        satisfaction = feedback_analysis.get("satisfaction", {})
        
        # 规则1：如果负面反馈比例高，降低强度
        negative_ratio = feedback_summary.get("negative", 0.0)
        if negative_ratio > 0.6:
            adjustments.append({
                "parameter": "intensity",
                "action": "decrease",
                "value": 0.2,
                "reason": "High negative feedback ratio"
            })
        
        # 规则2：如果满意度低，增加娱乐内容比例
        overall_satisfaction = satisfaction.get("overall", 0.5)
        if overall_satisfaction < 0.4:
            adjustments.append({
                "parameter": "entertainment_ratio",
                "action": "increase",
                "value": 0.3,
                "reason": "Low overall satisfaction"
            })
        
        # 规则3：如果参与度下降，增加互动频率
        engagement_trend = feedback_analysis.get("trends", {}).get("engagement_trend", 0.0)
        if engagement_trend < -0.2:
            adjustments.append({
                "parameter": "frequency",
                "action": "increase",
                "value": 0.2,
                "reason": "Decreasing engagement trend"
            })
        
        return adjustments
    
    async def _merge_adjustments(self, adjustment_prediction, rule_adjustments):
        # 合并模型预测和规则调整
        merged_adjustments = []
        
        # 添加规则调整
        merged_adjustments.extend(rule_adjustments)
        
        # 添加模型预测调整
        if isinstance(adjustment_prediction, dict):
            for param, value in adjustment_prediction.items():
                # 确定调整方向
                if value > 0.1:
                    action = "increase"
                elif value < -0.1:
                    action = "decrease"
                else:
                    continue  # 调整幅度太小，忽略
                
                # 检查是否已有相同参数的规则调整
                existing_adjustment = next(
                    (adj for adj in merged_adjustments if adj.get("parameter") == param),
                    None
                )
                
                if existing_adjustment:
                    # 合并调整值
                    existing_value = existing_adjustment.get("value", 0.0)
                    if existing_adjustment.get("action") == action:
                        # 同方向调整，增加幅度
                        existing_adjustment["value"] = min(existing_value + abs(value), 0.5)
                    else:
                        # 反方向调整，取较大值
                        existing_adjustment["value"] = max(existing_value, abs(value))
                        existing_adjustment["action"] = action
                else:
                    # 添加新调整
                    merged_adjustments.append({
                        "parameter": param,
                        "action": action,
                        "value": abs(value),
                        "reason": "Model prediction"
                    })
        
        return merged_adjustments
    
    async def _apply_adjustments(self, current_strategy, adjustments):
        # 复制当前策略
        adjusted_strategy = copy.deepcopy(current_strategy)
        
        # 获取策略参数
        parameters = adjusted_strategy.get("parameters", {})
        
        # 应用调整
        for adjustment in adjustments:
            param = adjustment.get("parameter")
            action = adjustment.get("action")
            value = adjustment.get("value", 0.0)
            
            if param in parameters:
                current_value = parameters[param]
                
                if action == "increase":
                    # 增加参数值
                    new_value = min(current_value + value, 1.0)
                elif action == "decrease":
                    # 减少参数值
                    new_value = max(current_value - value, 0.0)
                else:
                    continue  # 未知调整动作
                
                # 更新参数值
                parameters[param] = new_value
                
                # 记录调整历史
                adjustment_history = adjusted_strategy.get("adjustment_history", [])
                adjustment_history.append({
                    "parameter": param,
                    "from_value": current_value,
                    "to_value": new_value,
                    "reason": adjustment.get("reason", ""),
                    "timestamp": time.time()
                })
                adjusted_strategy["adjustment_history"] = adjustment_history
        
        # 更新策略参数
        adjusted_strategy["parameters"] = parameters
        
        return adjusted_strategy
    
    def _convert_features_to_vector(self, features):
        # 将特征字典转换为向量
        # 这里简化处理，实际应该按照模型训练时的特征顺序进行转换
        feature_names = sorted(features.keys())
        feature_vector = [features[name] for name in feature_names]
        
        return feature_vector
