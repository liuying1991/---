1. **记忆项模型** (src/models/memory.py):
   ```python
   from datetime import datetime
   from enum import Enum
   from typing import Dict, List, Optional, Union, Any
   from pydantic import BaseModel, Field
   import uuid
   
   class MemoryType(str, Enum):
       TEXT = "text"
       IMAGE = "image"
       AUDIO = "audio"
       VIDEO = "video"
       STRUCTURED = "structured"
   
   class MemoryPriority(str, Enum):
       LOW = "low"
       MEDIUM = "medium"
       HIGH = "high"
       CRITICAL = "critical"
   
   class MemoryMetadata(BaseModel):
       created_at: datetime = Field(default_factory=datetime.utcnow)
       updated_at: Optional[datetime] = None
       last_accessed: Optional[datetime] = None
       access_count: int = 0
       priority: MemoryPriority = MemoryPriority.MEDIUM
       tags: List[str] = Field(default_factory=list)
       expires_at: Optional[datetime] = None
       size_bytes: int = 0
   
   class Memory(BaseModel):
       id: str = Field(default_factory=lambda: str(uuid.uuid4()))
       type: MemoryType
       content: Union[str, bytes, Dict[str, Any]]
       metadata: MemoryMetadata
       embedding: Optional[List[float]] = None
       
       class Config:
           json_encoders = {
               datetime: lambda v: v.isoformat()
           }
   
   class MemoryCreate(BaseModel):
       type: MemoryType
       content: Union[str, bytes, Dict[str, Any]]
       priority: MemoryPriority = MemoryPriority.MEDIUM
       tags: List[str] = Field(default_factory=list)
       expires_at: Optional[datetime] = None
   
   class MemoryUpdate(BaseModel):
       content: Optional[Union[str, bytes, Dict[str, Any]]] = None
       priority: Optional[MemoryPriority] = None
       tags: Optional[List[str]] = None
       expires_at: Optional[datetime] = None
   ```

### 存储接口实现

1. **存储接口** (src/storage/storage_interface.py):
   ```python
   from abc import ABC, abstractmethod
   from typing import List, Optional, Dict, Any
   from src.models.memory import Memory
   from src.models.index import SearchQuery
   
   class StorageInterface(ABC):
       @abstractmethod
       async def store_memory(self, memory: Memory) -> bool:
           """存储记忆项"""
           pass
       
       @abstractmethod
       async def get_memory(self, memory_id: str) -> Optional[Memory]:
           """获取记忆项"""
           pass
       
       @abstractmethod
       async def update_memory(self, memory_id: str, memory: Memory) -> bool:
           """更新记忆项"""
           pass
       
       @abstractmethod
       async def delete_memory(self, memory_id: str) -> bool:
           """删除记忆项"""
           pass
       
       @abstractmethod
       async def search_memories(self, query: SearchQuery) -> List[Memory]:
           """搜索记忆项"""
           pass
       
       @abstractmethod
       async def get_all_memories(self) -> List[Memory]:
           """获取所有记忆项"""
           pass
       
       @abstractmethod
       async def clear_all_memories(self) -> bool:
           """清除所有记忆项"""
           pass
   ```

2. **Redis客户端实现** (src/storage/redis_client.py):
   ```python
   import json
   import pickle
   from typing import List, Optional, Dict, Any, Tuple
   import redis.asyncio as redis
   from datetime import datetime, timedelta
   from src.storage.storage_interface import StorageInterface
   from src.models.memory import Memory, MemoryCreate
   from src.models.index import SearchQuery
   from src.core.config import settings
   from src.utils.logger import get_logger
   
   logger = get_logger(__name__)
   
   class RedisClient(StorageInterface):
       def __init__(self):
           self.redis = redis.Redis(
               host=settings.redis_host,
               port=settings.redis_port,
               db=settings.redis_db,
               password=settings.redis_password,
               decode_responses=False
           )
           self.memory_key_prefix = "memory:"
           self.index_key_prefix = "index:"
           self.metadata_key_prefix = "metadata:"
       
       async def store_memory(self, memory: Memory) -> bool:
           """存储记忆项"""
           try:
               # 序列化记忆对象
               memory_data = pickle.dumps(memory.dict())
               
               # 存储记忆数据
               memory_key = f"{self.memory_key_prefix}{memory.id}"
               await self.redis.set(memory_key, memory_data, ex=settings.memory_ttl)
               
               # 存储元数据
               metadata_key = f"{self.metadata_key_prefix}{memory.id}"
               metadata_data = json.dumps(memory.metadata.dict())
               await self.redis.set(metadata_key, metadata_data, ex=settings.memory_ttl)
               
               # 创建索引
               await self._create_indexes(memory)
               
               logger.info(f"Stored memory {memory.id} of type {memory.type}")
               return True
           except Exception as e:
               logger.error(f"Error storing memory {memory.id}: {str(e)}")
               return False
       
       async def get_memory(self, memory_id: str) -> Optional[Memory]:
           """获取记忆项"""
           try:
               memory_key = f"{self.memory_key_prefix}{memory_id}"
               memory_data = await self.redis.get(memory_key)
               
               if memory_data:
                   memory_dict = pickle.loads(memory_data)
                   memory = Memory(**memory_dict)
                   
                   # 更新访问计数和最后访问时间
                   memory.metadata.access_count += 1
                   memory.metadata.last_accessed = datetime.utcnow()
                   await self.update_memory_metadata(memory_id, memory.metadata)
                   
                   return memory
               return None
           except Exception as e:
               logger.error(f"Error getting memory {memory_id}: {str(e)}")
               return None
       
       async def update_memory(self, memory_id: str, memory: Memory) -> bool:
           """更新记忆项"""
           try:
               # 序列化记忆对象
               memory_data = pickle.dumps(memory.dict())
               
               # 更新记忆数据
               memory_key = f"{self.memory_key_prefix}{memory_id}"
               await self.redis.set(memory_key, memory_data, ex=settings.memory_ttl)
               
               # 更新元数据
               metadata_key = f"{self.metadata_key_prefix}{memory_id}"
               metadata_data = json.dumps(memory.metadata.dict())
               await self.redis.set(metadata_key, metadata_data, ex=settings.memory_ttl)
               
               # 重建索引
               await self._delete_indexes(memory_id)
               await self._create_indexes(memory)
               
               logger.info(f"Updated memory {memory_id}")
               return True
           except Exception as e:
               logger.error(f"Error updating memory {memory_id}: {str(e)}")
               return False
       
       async def delete_memory(self, memory_id: str) -> bool:
           """删除记忆项"""
           try:
               # 删除记忆数据
               memory_key = f"{self.memory_key_prefix}{memory_id}"
               await self.redis.delete(memory_key)
               
               # 删除元数据
               metadata_key = f"{self.metadata_key_prefix}{memory_id}"
               await self.redis.delete(metadata_key)
               
               # 删除索引
               await self._delete_indexes(memory_id)
               
               logger.info(f"Deleted memory {memory_id}")
               return True
           except Exception as e:
               logger.error(f"Error deleting memory {memory_id}: {str(e)}")
               return False
       
       async def search_memories(self, query: SearchQuery) -> List[Memory]:
           """搜索记忆项"""
           try:
               if query.type == "keyword":
                   return await self._keyword_search(query)
               elif query.type == "similarity":
                   return await self._similarity_search(query)
               elif query.type == "hybrid":
                   return await self._hybrid_search(query)
               else:
                   raise ValueError(f"Unsupported search type: {query.type}")
           except Exception as e:
               logger.error(f"Error searching memories: {str(e)}")
               return []
       
       async def get_all_memories(self) -> List[Memory]:
           """获取所有记忆项"""
           try:
               memory_keys = await self.redis.keys(f"{self.memory_key_prefix}*")
               memories = []
               
               for key in memory_keys:
                   memory_data = await self.redis.get(key)
                   if memory_data:
                       memory_dict = pickle.loads(memory_data)
                       memory = Memory(**memory_dict)
                       memories.append(memory)
               
               return memories
           except Exception as e:
               logger.error(f"Error getting all memories: {str(e)}")
               return []
       
       async def clear_all_memories(self) -> bool:
           """清除所有记忆项"""
           try:
               # 获取所有相关键
               memory_keys = await self.redis.keys(f"{self.memory_key_prefix}*")
               metadata_keys = await self.redis.keys(f"{self.metadata_key_prefix}*")
               index_keys = await self.redis.keys(f"{self.index_key_prefix}*")
               
               # 删除所有键
               all_keys = memory_keys + metadata_keys + index_keys
               if all_keys:
                   await self.redis.delete(*all_keys)
               
               logger.info("Cleared all memories")
               return True
           except Exception as e:
               logger.error(f"Error clearing all memories: {str(e)}")
               return False
       
       async def update_memory_metadata(self, memory_id: str, metadata) -> bool:
           """更新记忆元数据"""
           try:
               metadata_key = f"{self.metadata_key_prefix}{memory_id}"
               metadata_data = json.dumps(metadata.dict())
               await self.redis.set(metadata_key, metadata_data, ex=settings.memory_ttl)
               return True
           except Exception as e:
               logger.error(f"Error updating metadata for memory {memory_id}: {str(e)}")
               return False
       
       async def _create_indexes(self, memory: Memory):
           """为记忆创建索引"""
           try:
               # 创建关键词索引
               if memory.type == "text":
                   keywords = await self._extract_keywords(memory.content)
                   for keyword in keywords:
                       index_key = f"{self.index_key_prefix}keyword:{keyword}"
                       await self.redis.sadd(index_key, memory.id)
                       await self.redis.expire(index_key, settings.memory_ttl)
               
               # 创建向量索引
               if memory.embedding:
                   vector_key = f"{self.index_key_prefix}vector:{memory.id}"
                   vector_data = pickle.dumps(memory.embedding)
                   await self.redis.set(vector_key, vector_data, ex=settings.memory_ttl)
               
               # 创建时间索引
               time_key = f"{self.index_key_prefix}time:{memory.metadata.created_at.strftime('%Y%m%d%H%M')}"
               await self.redis.sadd(time_key, memory.id)
               await self.redis.expire(time_key, settings.memory_ttl)
               
               # 创建标签索引
               for tag in memory.metadata.tags:
                   tag_key = f"{self.index_key_prefix}tag:{tag}"
                   await self.redis.sadd(tag_key, memory.id)
                   await self.redis.expire(tag_key, settings.memory_ttl)
               
               # 创建类型索引
               type_key = f"{self.index_key_prefix}type:{memory.type}"
               await self.redis.sadd(type_key, memory.id)
               await self.redis.expire(type_key, settings.memory_ttl)
               
               # 创建优先级索引
               priority_key = f"{self.index_key_prefix}priority:{memory.metadata.priority}"
               await self.redis.sadd(priority_key, memory.id)
               await self.redis.expire(priority_key, settings.memory_ttl)
           except Exception as e:
               logger.error(f"Error creating indexes for memory {memory.id}: {str(e)}")
       
       async def _delete_indexes(self, memory_id: str):
           """删除记忆的所有索引"""
           try:
               # 获取所有索引键
               index_keys = await self.redis.keys(f"{self.index_key_prefix}*")
               
               # 从所有索引中删除记忆ID
               for key in index_keys:
                   await self.redis.srem(key, memory_id)
               
               # 删除向量索引
               vector_key = f"{self.index_key_prefix}vector:{memory_id}"
               await self.redis.delete(vector_key)
           except Exception as e:
               logger.error(f"Error deleting indexes for memory {memory_id}: {str(e)}")
       
       async def _keyword_search(self, query: SearchQuery) -> List[Memory]:
           """基于关键词的搜索"""
           try:
               if not query.keywords:
                   return []
               
               # 获取包含所有关键词的记忆ID集合
               result_sets = []
               for keyword in query.keywords:
                   index_key = f"{self.index_key_prefix}keyword:{keyword}"
                   memory_ids = await self.redis.smembers(index_key)
                   result_sets.append(set(memory_ids))
               
               # 计算交集（包含所有关键词的记忆）
               if result_sets:
                   common_ids = set.intersection(*result_sets)
               else:
                   common_ids = set()
               
               # 获取记忆数据
               memories = []
               for memory_id in common_ids:
                   memory = await self.get_memory(memory_id)
                   if memory:
                       memories.append(memory)
               
               # 应用过滤条件
               if query.filters:
                   memories = await self._apply_filters(memories, query.filters)
               
               # 按相关性排序
               memories.sort(key=lambda m: self._calculate_relevance(m, query.keywords), reverse=True)
               
               # 应用分页
               start = (query.page - 1) * query.page_size
               end = start + query.page_size
               return memories[start:end]
           except Exception as e:
               logger.error(f"Error in keyword search: {str(e)}")
               return []
       
       async def _similarity_search(self, query: SearchQuery) -> List[Memory]:
           """基于相似度的搜索"""
           try:
               if not query.content:
                   return []
               
               # 计算查询向量
               query_vector = await self._calculate_vector(query.content)
               
               # 获取所有向量索引
               vector_keys = await self.redis.keys(f"{self.index_key_prefix}vector:*")
               similar_memories = []
               
               for key in vector_keys:
                   memory_id = key.split(":")[-1]
                   vector_data = await self.redis.get(key)
                   if vector_data:
                       memory_vector = pickle.loads(vector_data)
                       similarity = self._calculate_similarity(query_vector, memory_vector)
                       
                       if similarity >= query.threshold:
                           memory = await self.get_memory(memory_id)
                           if memory:
                               memory.similarity = similarity
                               similar_memories.append(memory)
               
               # 应用过滤条件
               if query.filters:
                   similar_memories = await self._apply_filters(similar_memories, query.filters)
               
               # 按相似度排序
               similar_memories.sort(key=lambda m: m.similarity, reverse=True)
               
               # 应用分页
               start = (query.page - 1) * query.page_size
               end = start + query.page_size
               return similar_memories[start:end]
           except Exception as e:
               logger.error(f"Error in similarity search: {str(e)}")
               return []
       
       async def _hybrid_search(self, query: SearchQuery) -> List[Memory]:
           """混合搜索"""
           try:
               # 执行关键词搜索
               keyword_results = await self._keyword_search(query)
               
               # 执行相似度搜索
               similarity_results = await self._similarity_search(query)
               
               # 合并结果
               all_memories = {}
               
               # 添加关键词搜索结果
               for memory in keyword_results:
                   if memory.id not in all_memories:
                       all_memories[memory.id] = memory
                       memory.keyword_score = 1.0
                       memory.similarity_score = 0.0
                   else:
                       all_memories[memory.id].keyword_score = 1.0
               
               # 添加相似度搜索结果
               for memory in similarity_results:
                   if memory.id not in all_memories:
                       all_memories[memory.id] = memory
                       memory.keyword_score = 0.0
                       memory.similarity_score = memory.similarity
                   else:
                       all_memories[memory.id].similarity_score = memory.similarity
               
               # 计算混合分数
               for memory in all_memories.values():
                   memory.hybrid_score = 0.5 * memory.keyword_score + 0.5 * memory.similarity_score
               
               # 应用过滤条件
               if query.filters:
                   filtered_memories = await self._apply_filters(list(all_memories.values()), query.filters)
               else:
                   filtered_memories = list(all_memories.values())
               
               # 按混合分数排序
               filtered_memories.sort(key=lambda m: m.hybrid_score, reverse=True)
               
               # 应用分页
               start = (query.page - 1) * query.page_size
               end = start + query.page_size
               return filtered_memories[start:end]
           except Exception as e:
               logger.error(f"Error in hybrid search: {str(e)}")
               return []
       
       async def _apply_filters(self, memories: List[Memory], filters: Dict[str, Any]) -> List[Memory]:
           """应用过滤条件"""
           try:
               filtered_memories = memories
               
               # 按类型过滤
               if "type" in filters:
                   filtered_memories = [m for m in filtered_memories if m.type == filters["type"]]
               
               # 按优先级过滤
               if "priority" in filters:
                   filtered_memories = [m for m in filtered_memories if m.metadata.priority == filters["priority"]]
               
               # 按标签过滤
               if "tags" in filters:
                   required_tags = set(filters["tags"])
                   filtered_memories = [m for m in filtered_memories if required_tags.issubset(set(m.metadata.tags))]
               
               # 按时间范围过滤
               if "start_time" in filters:
                   start_time = filters["start_time"]
                   filtered_memories = [m for m in filtered_memories if m.metadata.created_at >= start_time]
               
               if "end_time" in filters:
                   end_time = filters["end_time"]
                   filtered_memories = [m for m in filtered_memories if m.metadata.created_at <= end_time]
               
               return filtered_memories
           except Exception as e:
               logger.error(f"Error applying filters: {str(e)}")
               return memories
       
       async def _extract_keywords(self, content: str) -> List[str]:
           """从文本内容中提取关键词"""
           try:
               # 简单的关键词提取，实际应用中可以使用更复杂的NLP技术
               import re
               words = re.findall(r'\b\w+\b', content.lower())
               # 过滤停用词和短词
               stopwords = {'a', 'an', 'the', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'for', 'of', 'with', 'by', 'is', 'are', 'was', 'were', 'be', 'been', 'being', 'have', 'has', 'had', 'do', 'does', 'did', 'will', 'would', 'could', 'should', 'may', 'might', 'must', 'can', 'this', 'that', 'these', 'those', 'i', 'you', 'he', 'she', 'it', 'we', 'they', 'what', 'which', 'who', 'when', 'where', 'why', 'how', 'not', 'no', 'yes'}
               keywords = [word for word in words if len(word) > 2 and word not in stopwords]
               return keywords
           except Exception as e:
               logger.error(f"Error extracting keywords: {str(e)}")
               return []
       
       async def _calculate_vector(self, content: str) -> List[float]:
           """计算内容的向量表示"""
           try:
               # 简单的向量表示，实际应用中可以使用预训练的嵌入模型
               import hashlib
               import numpy as np
               
               # 使用哈希函数生成固定长度的向量
               hash_obj = hashlib.sha256(content.encode())
               hex_dig = hash_obj.hexdigest()
               
               # 将十六进制字符串转换为向量
               vector = []
               for i in range(0, len(hex_dig), 2):
                   byte_val = int(hex_dig[i:i+2], 16)
                   # 归一化到[-1, 1]范围
                   normalized_val = (byte_val - 128) / 128.0
                   vector.append(normalized_val)
               
               return vector
           except Exception as e:
               logger.error(f"Error calculating vector: {str(e)}")
               return [0.0] * 128  # 返回零向量作为默认值
       
       def _calculate_similarity(self, vector1: List[float], vector2: List[float]) -> float:
           """计算两个向量的余弦相似度"""
           try:
               import numpy as np
               
               # 转换为numpy数组
               v1 = np.array(vector1)
               v2 = np.array(vector2)
               
               # 计算余弦相似度
               dot_product = np.dot(v1, v2)
               norm_v1 = np.linalg.norm(v1)
               norm_v2 = np.linalg.norm(v2)
               
               if norm_v1 == 0 or norm_v2 == 0:
                   return 0.0
               
               similarity = dot_product / (norm_v1 * norm_v2)
               return float(similarity)
           except Exception as e:
               logger.error(f"Error calculating similarity: {str(e)}")
               return 0.0
       
       def _calculate_relevance(self, memory: Memory, keywords: List[str]) -> float:
           """计算记忆与关键词的相关性"""
           try:
               if memory.type != "text" or not keywords:
                   return 0.0
               
               content_words = set(memory.content.lower().split())
               keyword_set = set(keywords)
               
               # 计算关键词覆盖率
               coverage = len(content_words.intersection(keyword_set)) / len(keyword_set)
               
               # 考虑记忆优先级
               priority_weight = {
                   "low": 0.5,
                   "medium": 1.0,
                   "high": 1.5,
                   "critical": 2.0
               }
               
               priority_factor = priority_weight.get(memory.metadata.priority, 1.0)
               
               # 考虑访问频率
               access_factor = min(memory.metadata.access_count / 10.0, 1.0)
               
               # 计算综合相关性
               relevance = coverage * priority_factor * (1 + 0.1 * access_factor)
               
               return relevance
           except Exception as e:
               logger.error(f"Error calculating relevance: {str(e)}")
               return 0.0
   ```

   ```python
   from datetime import datetime, timedelta
   from typing import List, Dict, Any, Optional

       async def trigger_conversion(self):
           """触发记忆转换流程"""
           try:
               # 获取所有记忆
               memories = await self.storage_client.get_all_memories()
               
               # 筛选需要转换的记忆
               candidates = await self._identify_conversion_candidates(memories)
               
               # 按优先级排序
               candidates.sort(key=lambda m: self._calculate_conversion_score(m), reverse=True)
               
               # 执行转换
               converted_count = 0
               for memory in candidates:
                   if await self._convert_memory(memory):
                       converted_count += 1
               
               logger.info(f"Converted {converted_count} memories")
               return converted_count
           except Exception as e:
               logger.error(f"Error in conversion process: {str(e)}")
               return 0
       
       async def _identify_conversion_candidates(self, memories: List[Memory]) -> List[Memory]:
           """识别需要转换的记忆候选"""
           try:
               candidates = []
               now = datetime.utcnow()
               
               for memory in memories:
                   # 检查记忆是否过期
                   if memory.metadata.expires_at and memory.metadata.expires_at < now:
                       candidates.append(memory)
                       continue
                   
                   # 检查记忆是否老化
                   age = now - memory.metadata.created_at
                   if age > timedelta(hours=settings.memory_ttl / 3600):
                       # 低优先级且很少被访问的记忆
                       if (memory.metadata.priority in [MemoryPriority.LOW, MemoryPriority.MEDIUM] and
                           memory.metadata.access_count < 3):
                           candidates.append(memory)
                           continue
                   
                   # 检查记忆容量
                   if len(memories) > settings.max_memory_size * 0.8:
                       # 容量接近上限时，考虑转换低优先级记忆
                       if memory.metadata.priority == MemoryPriority.LOW:
                           candidates.append(memory)
               
               return candidates
           except Exception as e:
               logger.error(f"Error identifying conversion candidates: {str(e)}")
               return []
       
       async def _convert_memory(self, memory: Memory) -> bool:
           """转换单个记忆"""
           try:
               # 在实际应用中，这里会将记忆转换为中长期记忆存储
               # 这里简化实现，只是标记为已转换并从短期记忆中删除
               
               # 记录转换日志
               logger.info(f"Converting memory {memory.id} of type {memory.type}")
               
               # 从短期记忆中删除
               success = await self.storage_client.delete_memory(memory.id)
               
               if success:
                   logger.info(f"Successfully converted memory {memory.id}")
               else:
                   logger.error(f"Failed to convert memory {memory.id}")
               
               return success
           except Exception as e:
               logger.error(f"Error converting memory {memory.id}: {str(e)}")
               return False
       
       def _calculate_conversion_score(self, memory: Memory) -> float:
           """计算记忆的转换分数"""
           try:
               now = datetime.utcnow()
               age = (now - memory.metadata.created_at).total_seconds()
               
               # 基础分数基于记忆年龄
               score = age / (settings.memory_ttl * 3600)  # 归一化到[0, 1]
               
               # 优先级因子（低优先级更容易转换）
               priority_factor = {
                   MemoryPriority.LOW: 1.0,
                   MemoryPriority.MEDIUM: 0.7,
                   MemoryPriority.HIGH: 0.3,
                   MemoryPriority.CRITICAL: 0.1
               }
               score *= priority_factor.get(memory.metadata.priority, 0.5)
               
               # 访问频率因子（很少访问的记忆更容易转换）
               access_factor = max(0, 1 - memory.metadata.access_count / 10.0)
               score *= access_factor
               
               # 大小因子（大记忆更容易转换）
               size_factor = min(1.0, memory.metadata.size_bytes / 10240)  # 10KB为基准
               score *= (1 + size_factor)
               
               return score
           except Exception as e:
               logger.error(f"Error calculating conversion score: {str(e)}")
               return 0.0
   ```

   ```python
   from datetime import datetime, timedelta
   from typing import List, Dict, Any
   from src.models.memory import Memory

       async def run_cleanup(self) -> Dict[str, Any]:
           """运行清理流程"""
           try:
               # 获取所有记忆
               memories = await self.storage_client.get_all_memories()
               
               # 清理过期记忆
               expired_count = await self._cleanup_expired_memories(memories)
               
               # 清理低优先级记忆（如果容量接近上限）
               low_priority_count = await self._cleanup_low_priority_memories(memories)
               
               # 清理损坏的记忆
               corrupted_count = await self._cleanup_corrupted_memories(memories)
               
               # 清理孤立索引
               orphaned_index_count = await self._cleanup_orphaned_indexes()
               
               result = {
                   "expired_count": expired_count,
                   "low_priority_count": low_priority_count,
                   "corrupted_count": corrupted_count,
                   "orphaned_index_count": orphaned_index_count,
                   "total_cleaned": expired_count + low_priority_count + corrupted_count
               }
               
               logger.info(f"Cleanup completed: {result}")
               return result
           except Exception as e:
               logger.error(f"Error in cleanup process: {str(e)}")
               return {"error": str(e)}
       
       async def _cleanup_expired_memories(self, memories: List[Memory]) -> int:
           """清理过期记忆"""
           try:
               now = datetime.utcnow()
               expired_count = 0
               
               for memory in memories:
                   if memory.metadata.expires_at and memory.metadata.expires_at < now:
                       if await self.storage_client.delete_memory(memory.id):
                           expired_count += 1
               
               logger.info(f"Cleaned up {expired_count} expired memories")
               return expired_count
           except Exception as e:
               logger.error(f"Error cleaning up expired memories: {str(e)}")
               return 0
       
       async def _cleanup_low_priority_memories(self, memories: List[Memory]) -> int:
           """清理低优先级记忆"""
           try:
               # 检查是否需要清理
               if len(memories) < settings.max_memory_size * 0.9:
                   return 0
               
               # 按优先级和访问时间排序
               memories.sort(key=lambda m: (
                   m.metadata.priority,
                   m.metadata.last_accessed or m.metadata.created_at
               ))
               
               # 计算需要清理的数量
               cleanup_count = len(memories) - int(settings.max_memory_size * 0.8)
               cleanup_count = max(0, cleanup_count)
               
               # 清理低优先级记忆
               cleaned_count = 0
               for i in range(min(cleanup_count, len(memories))):
                   if await self.storage_client.delete_memory(memories[i].id):
                       cleaned_count += 1
               
               logger.info(f"Cleaned up {cleaned_count} low priority memories")
               return cleaned_count
           except Exception as e:
               logger.error(f"Error cleaning up low priority memories: {str(e)}")
               return 0
       
       async def _cleanup_corrupted_memories(self, memories: List[Memory]) -> int:
           """清理损坏的记忆"""
           try:
               corrupted_count = 0
               
               for memory in memories:
                   # 检查记忆数据是否损坏
                   if not self._validate_memory(memory):
                       if await self.storage_client.delete_memory(memory.id):
                           corrupted_count += 1
               
               logger.info(f"Cleaned up {corrupted_count} corrupted memories")
               return corrupted_count
           except Exception as e:
               logger.error(f"Error cleaning up corrupted memories: {str(e)}")
               return 0
       
       async def _cleanup_orphaned_indexes(self) -> int:
           """清理孤立索引"""
           try:
               # 获取所有记忆ID
               memories = await self.storage_client.get_all_memories()
               memory_ids = {memory.id for memory in memories}
               
               # 获取所有索引键
               import redis.asyncio as redis
               redis_client = redis.Redis(
                   host=settings.redis_host,
                   port=settings.redis_port,
                   db=settings.redis_db,
                   password=settings.redis_password,
                   decode_responses=False
               )
               
               index_keys = await redis_client.keys(f"{self.storage_client.index_key_prefix}*")
               orphaned_count = 0
               
               for key in index_keys:
                   # 获取索引中的记忆ID
                   if b":" in key:
                       key_type = key.split(b":")[1]
                       
                       if key_type == b"vector":
                           # 向量索引直接删除
                           memory_id = key.split(b":")[-1].decode()
                           if memory_id not in memory_ids:
                               await redis_client.delete(key)
                               orphaned_count += 1
                       else:
                           # 其他索引类型检查集合成员
                           members = await redis_client.smembers(key)
                           for member in members:
                               memory_id = member.decode()
                               if memory_id not in memory_ids:
                                   await redis_client.srem(key, member)
                                   orphaned_count += 1
               
               await redis_client.close()
               logger.info(f"Cleaned up {orphaned_count} orphaned indexes")
               return orphaned_count
           except Exception as e:
               logger.error(f"Error cleaning up orphaned indexes: {str(e)}")
               return 0
       
       def _validate_memory(self, memory: Memory) -> bool:
           """验证记忆数据是否有效"""
           try:
               # 检查基本字段
               if not memory.id or not memory.type or memory.content is None:
                   return False
               
               # 检查元数据
               if not memory.metadata:
                   return False
               
               # 检查创建时间
               if not memory.metadata.created_at:
                   return False
               
               # 检查大小
               if memory.metadata.size_bytes < 0:
                   return False
               
               # 检查内容类型与类型字段是否匹配
               if memory.type == "text" and not isinstance(memory.content, str):
                   return False
               elif memory.type in ["image", "audio", "video"] and not isinstance(memory.content, bytes):
                   return False
               elif memory.type == "structured" and not isinstance(memory.content, dict):
                   return False
               
               return True
           except Exception as e:
               logger.error(f"Error validating memory {memory.id}: {str(e)}")
               return False
   ```

   ```python
       """处理记忆服务异常"""

       status_code = 500
       detail = "Internal server error"
       
       if isinstance(exc, MemoryNotFoundException):
           status_code = 404
           detail = "Memory not found"
       elif isinstance(exc, MemoryStorageException):
           status_code = 500
           detail = "Memory storage error"
       elif isinstance(exc, MemorySearchException):
           status_code = 500
           detail = "Memory search error"
       elif isinstance(exc, MemoryConversionException):
           status_code = 500

           detail = "Memory capacity exceeded"
       elif isinstance(exc, InvalidMemoryDataException):
           status_code = 400
           detail = "Invalid memory data"
   ```
