# 多模态融合模块6A工作流

## 模块概述

### 模块名称
多模态融合模块

### 所属子系统
交互表达子系统

### 模块功能描述
多模态融合模块负责将交互表达子系统的多种模态信息（语音、表情、动作等）进行有效融合，生成统一的多模态表达。该模块通过分析不同模态之间的关联性和互补性，实现多模态信息的协同表达，提高交互表达的丰富性、准确性和自然度，为婴儿提供更加生动、连贯的交互体验。

### 模块职责
1. 多模态输入：接收和处理多种模态的输入信息
2. 模态分析：分析各模态的特征和表达内容
3. 模态对齐：对齐不同模态的时间和内容
4. 模态融合：融合多种模态信息生成统一表达
5. 冲突解决：解决模态间的冲突和不一致
6. 表达生成：生成融合后的多模态表达
7. 效果评估：评估多模态融合的表达效果

### 模块接口
- **输入接口**：语音表达、表情表达、动作表达、上下文信息
- **输出接口**：融合后的多模态表达、融合策略、融合效果
- **依赖模块**：语音合成模块、表情生成模块、肢体动作生成模块
- **被依赖模块**：交互管理模块、交互控制模块、交互效果评估模块

# 多模态融合管理器实现
class MultimodalFusionManager:
    def __init__(self, modality_inputter, feature_extractor, modality_analyzer,
                 modality_aligner, modality_fuser, conflict_resolver, expression_generator,
                 fusion_evaluator, fusion_manager, strategy_selector):
        self.modality_inputter = modality_inputter
        self.feature_extractor = feature_extractor
        self.modality_analyzer = modality_analyzer
        self.modality_aligner = modality_aligner
        self.modality_fuser = modality_fuser
        self.conflict_resolver = conflict_resolver
        self.expression_generator = expression_generator
        self.fusion_evaluator = fusion_evaluator
        self.fusion_manager = fusion_manager
        self.strategy_selector = strategy_selector
        
        # 初始化模型
        self._init_models()
        
    async def fuse_modalities(self, baby_id, session_id, modality_inputs):
        try:
            # 验证输入数据
            validated_inputs = await self.modality_inputter.validate_inputs(modality_inputs)
            
            # 预处理输入数据
            preprocessed_inputs = await self.modality_inputter.preprocess_inputs(validated_inputs)
            
            # 提取模态特征
            modality_features = await self.feature_extractor.extract_features(preprocessed_inputs)
            
            # 分析模态内容
            modality_analysis = await self.modality_analyzer.analyze_modalities(modality_features)
            
            # 对齐模态信息
            aligned_modalities = await self.modality_aligner.align_modalities(
                modality_features, modality_analysis
            )
            
            # 选择融合策略
            fusion_strategy = await self.strategy_selector.select_strategy(
                aligned_modalities, modality_analysis
            )
            
            # 融合模态信息
            fused_result = await self.modality_fuser.fuse_modalities(
                aligned_modalities, fusion_strategy
            )
            
            # 解决模态冲突
            resolved_result = await self.conflict_resolver.resolve_conflicts(fused_result)
            
            # 生成融合表达
            fused_expression = await self.expression_generator.generate_expression(resolved_result)
            
            # 评估融合效果
            fusion_evaluation = await self.fusion_evaluator.evaluate_fusion(
                fused_expression, aligned_modalities
            )
            
            # 构建融合结果
            fusion_result = {
                "baby_id": baby_id,
                "session_id": session_id,
                "modality_inputs": preprocessed_inputs,
                "modality_features": modality_features,
                "modality_analysis": modality_analysis,
                "aligned_modalities": aligned_modalities,
                "fusion_strategy": fusion_strategy,
                "fused_result": fused_result,
                "resolved_result": resolved_result,
                "fused_expression": fused_expression,
                "fusion_evaluation": fusion_evaluation,
                "timestamp": time.time()
            }
            
            # 保存融合结果
            await self.fusion_manager.save_fusion_result(fusion_result)
            
            return fusion_result
            
        except Exception as e:
            # 处理异常
            error_result = {
                "baby_id": baby_id,
                "session_id": session_id,
                "error": str(e),
                "timestamp": time.time()
            }
            
            return error_result

# 模态融合器实现
class ModalityFuser:
    def __init__(self, early_fusion, late_fusion, hybrid_fusion):
        self.early_fusion = early_fusion
        self.late_fusion = late_fusion
        self.hybrid_fusion = hybrid_fusion
        
    async def fuse_modalities(self, aligned_modalities, fusion_strategy):
        try:
            # 获取融合类型
            fusion_type = fusion_strategy.get("type", "hybrid")
            
            # 根据融合类型执行不同的融合
            if fusion_type == "early":
                fusion_result = await self._early_fuse_modalities(
                    aligned_modalities, fusion_strategy
                )
            elif fusion_type == "late":
                fusion_result = await self._late_fuse_modalities(
                    aligned_modalities, fusion_strategy
                )
            elif fusion_type == "hybrid":
                fusion_result = await self._hybrid_fuse_modalities(
                    aligned_modalities, fusion_strategy
                )
            else:
                fusion_result = await self._hybrid_fuse_modalities(
                    aligned_modalities, fusion_strategy
                )
            
            return {
                "fusion_type": fusion_type,
                "fusion_strategy": fusion_strategy,
                "fusion_result": fusion_result,
                "modalities": list(aligned_modalities.keys())
            }
            
        except Exception as e:
            return {
                "error": str(e),
                "fusion_type": fusion_strategy.get("type", "hybrid"),
                "fusion_strategy": fusion_strategy
            }
    
    async def _early_fuse_modalities(self, aligned_modalities, fusion_strategy):
        # 早期融合：在特征层融合多模态信息
        # 提取各模态的特征向量
        feature_vectors = {}
        for modality, data in aligned_modalities.items():
            features = data.get("features", [])
            if features:
                # 将特征向量转换为统一格式
                feature_vector = await self._normalize_feature_vector(features, modality)
                feature_vectors[modality] = feature_vector
        
        # 融合特征向量
        fused_features = await self.early_fusion.fuse_features(
            feature_vectors, fusion_strategy
        )
        
        # 基于融合特征生成融合结果
        fused_result = await self._generate_result_from_features(fused_features)
        
        return {
            "feature_vectors": feature_vectors,
            "fused_features": fused_features,
            "fused_result": fused_result
        }
    
    async def _late_fuse_modalities(self, aligned_modalities, fusion_strategy):
        # 晚期融合：在决策层融合多模态信息
        # 为每个模态生成独立的决策
        modality_decisions = {}
        for modality, data in aligned_modalities.items():
            # 基于模态数据生成决策
            decision = await self._generate_modality_decision(data, modality)
            modality_decisions[modality] = decision
        
        # 融合决策
        fused_decision = await self.late_fusion.fuse_decisions(
            modality_decisions, fusion_strategy
        )
        
        # 基于融合决策生成融合结果
        fused_result = await self._generate_result_from_decision(fused_decision)
        
        return {
            "modality_decisions": modality_decisions,
            "fused_decision": fused_decision,
            "fused_result": fused_result
        }
    
    async def _hybrid_fuse_modalities(self, aligned_modalities, fusion_strategy):
        # 混合融合：结合早期和晚期融合策略
        # 执行早期融合
        early_fusion_result = await self._early_fuse_modalities(
            aligned_modalities, fusion_strategy
        )
        
        # 执行晚期融合
        late_fusion_result = await self._late_fuse_modalities(
            aligned_modalities, fusion_strategy
        )
        
        # 融合早期和晚期融合结果
        hybrid_fusion_result = await self.hybrid_fusion.fuse_results(
            early_fusion_result, late_fusion_result, fusion_strategy
        )
        
        return {
            "early_fusion_result": early_fusion_result,
            "late_fusion_result": late_fusion_result,
            "hybrid_fusion_result": hybrid_fusion_result
        }
    
    async def _normalize_feature_vector(self, features, modality):
        # 标准化特征向量
        if modality == "audio":
            # 音频特征标准化
            normalized_features = await self._normalize_audio_features(features)
        elif modality == "video":
            # 视频特征标准化
            normalized_features = await self._normalize_video_features(features)
        elif modality == "motion":
            # 动作特征标准化
            normalized_features = await self._normalize_motion_features(features)
        else:
            # 默认标准化
            normalized_features = await self._normalize_default_features(features)
        
        return normalized_features
    
    async def _normalize_audio_features(self, features):
        # 音频特征标准化
        import numpy as np
        
        # 转换为numpy数组
        feature_array = np.array(features)
        
        # 计算均值和标准差
        mean = np.mean(feature_array, axis=0)
        std = np.std(feature_array, axis=0)
        
        # 避免除以0
        std = np.where(std == 0, 1, std)
        
        # 标准化
        normalized_features = (feature_array - mean) / std
        
        # 转换为列表
        return normalized_features.tolist()
    
    async def _normalize_video_features(self, features):
        # 视频特征标准化
        import numpy as np
        
        # 转换为numpy数组
        feature_array = np.array(features)
        
        # 归一化到[0, 1]
        min_val = np.min(feature_array, axis=0)
        max_val = np.max(feature_array, axis=0)
        
        # 避免除以0
        range_val = max_val - min_val
        range_val = np.where(range_val == 0, 1, range_val)
        
        # 归一化
        normalized_features = (feature_array - min_val) / range_val
        
        # 转换为列表
        return normalized_features.tolist()
    
    async def _normalize_motion_features(self, features):
        # 动作特征标准化
        import numpy as np
        
        # 转换为numpy数组
        feature_array = np.array(features)
        
        # 计算L2范数
        norm = np.linalg.norm(feature_array, axis=1, keepdims=True)
        
        # 避免除以0
        norm = np.where(norm == 0, 1, norm)
        
        # L2归一化
        normalized_features = feature_array / norm
        
        # 转换为列表
        return normalized_features.tolist()
    
    async def _normalize_default_features(self, features):
        # 默认特征标准化
        import numpy as np
        
        # 转换为numpy数组
        feature_array = np.array(features)
        
        # Z-score标准化
        mean = np.mean(feature_array)
        std = np.std(feature_array)
        
        # 避免除以0
        std = std if std != 0 else 1
        
        # 标准化
        normalized_features = (feature_array - mean) / std
        
        # 转换为列表
        return normalized_features.tolist()

class ConflictResolver:
    def __init__(self, conflict_detector, resolution_strategy):
        self.conflict_detector = conflict_detector
        self.resolution_strategy = resolution_strategy
        
    async def resolve_conflicts(self, fused_result):
        try:
            # 检测冲突
            conflicts = await self.conflict_detector.detect_conflicts(fused_result)
            
            # 如果没有冲突，直接返回
            if not conflicts:
                return {
                    "has_conflicts": False,
                    "fused_result": fused_result,
                    "resolution_applied": None
                }
            
            # 解决冲突
            resolved_result = await self._resolve_conflicts(conflicts, fused_result)
            
            # 重新检测冲突，确保已解决
            remaining_conflicts = await self.conflict_detector.detect_conflicts(resolved_result)
            
            return {
                "has_conflicts": True,
                "conflicts": conflicts,
                "resolved_result": resolved_result,
                "remaining_conflicts": remaining_conflicts,
                "resolution_applied": True
            }
            
        except Exception as e:
            return {
                "error": str(e),
                "fused_result": fused_result
            }
    
    async def _resolve_conflicts(self, conflicts, fused_result):
        # 解决冲突
        resolved_result = fused_result.copy()
        
        # 按优先级排序冲突
        sorted_conflicts = sorted(conflicts, key=lambda x: x.get("priority", 0), reverse=True)
        
        # 逐个解决冲突
        for conflict in sorted_conflicts:
            conflict_type = conflict.get("type", "unknown")
            
            # 根据冲突类型选择解决策略
            if conflict_type == "emotion_conflict":
                resolution = await self.resolution_strategy.resolve_emotion_conflict(conflict, resolved_result)
            elif conflict_type == "content_conflict":
                resolution = await self.resolution_strategy.resolve_content_conflict(conflict, resolved_result)
            elif conflict_type == "timing_conflict":
                resolution = await self.resolution_strategy.resolve_timing_conflict(conflict, resolved_result)
            else:
                resolution = await self.resolution_strategy.resolve_default_conflict(conflict, resolved_result)
            
            # 应用解决方案
            resolved_result = await self._apply_resolution(resolution, resolved_result)
        
        return resolved_result
    
    async def _apply_resolution(self, resolution, result):
        # 应用解决方案
        resolution_type = resolution.get("type", "replace")
        
        if resolution_type == "replace":
            # 替换解决方案
            target = resolution.get("target", "")
            value = resolution.get("value", None)
            
            if target and value is not None:
                # 设置目标值
                result[target] = value
        
        elif resolution_type == "merge":
            # 合并解决方案
            target = resolution.get("target", "")
            value = resolution.get("value", {})
            
            if target and isinstance(value, dict):
                # 合并字典
                if target in result and isinstance(result[target], dict):
                    result[target].update(value)
                else:
                    result[target] = value
        
        elif resolution_type == "weight":
            # 加权解决方案
            target = resolution.get("target", "")
            weights = resolution.get("weights", {})
            
            if target and isinstance(weights, dict):
                # 计算加权值
                weighted_value = await self._calculate_weighted_value(weights, result)
                result[target] = weighted_value
        
        return result
    
    async def _calculate_weighted_value(self, weights, result):
        # 计算加权值
        total_weight = sum(weights.values())
        if total_weight == 0:
            return None
        
        weighted_sum = 0
        for key, weight in weights.items():
            if key in result:
                value = result[key]
                if isinstance(value, (int, float)):
                    weighted_sum += value * weight
        
        return weighted_sum / total_weight

class FusionEvaluator:
    def __init__(self, quality_assessor, metrics_calculator):
        self.quality_assessor = quality_assessor
        self.metrics_calculator = metrics_calculator
        
    async def evaluate_fusion(self, fused_expression, aligned_modalities):
        try:
            # 计算融合指标
            fusion_metrics = await self.metrics_calculator.calculate_metrics(
                fused_expression, aligned_modalities
            )
            
            # 评估融合质量
            quality_assessment = await self.quality_assessor.assess_quality(
                fused_expression, fusion_metrics
            )
            
            # 构建评估结果
            evaluation_result = {
                "fusion_metrics": fusion_metrics,
                "quality_assessment": quality_assessment,
                "overall_score": await self._calculate_overall_score(fusion_metrics, quality_assessment),
                "recommendations": await self._generate_recommendations(fusion_metrics, quality_assessment)
            }
            
            return evaluation_result
            
        except Exception as e:
            return {
                "error": str(e),
                "fused_expression": fused_expression,
                "aligned_modalities": aligned_modalities
            }
    
    async def _calculate_overall_score(self, metrics, assessment):
        # 计算总体评分
        # 简化示例：基于指标和评估计算加权平均
        metric_scores = metrics.get("scores", {})
        assessment_scores = assessment.get("scores", {})
        
        # 定义权重
        metric_weight = 0.6
        assessment_weight = 0.4
        
        # 计算指标平均分
        metric_avg = sum(metric_scores.values()) / len(metric_scores) if metric_scores else 0
        
        # 计算评估平均分
        assessment_avg = sum(assessment_scores.values()) / len(assessment_scores) if assessment_scores else 0
        
        # 计算加权平均
        overall_score = metric_avg * metric_weight + assessment_avg * assessment_weight
        
        return overall_score
