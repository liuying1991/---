# 数据模型定义
from pydantic import BaseModel, Field
from typing import List, Dict, Optional, Any
from datetime import datetime
from enum import Enum

class ExperienceType(str, Enum):
    """经验类型枚举"""
    INTERACTION = "interaction"
    TASK = "task"
    FEEDBACK = "feedback"
    ERROR = "error"
    PERFORMANCE = "performance"
    RESOURCE = "resource"

class ReflectionType(str, Enum):
    """反思类型枚举"""
    PERFORMANCE = "performance"
    BEHAVIOR = "behavior"
    LEARNING = "learning"
    IMPROVEMENT = "improvement"

class InsightType(str, Enum):
    """洞察类型枚举"""
    STRENGTH = "strength"
    WEAKNESS = "weakness"
    OPPORTUNITY = "opportunity"
    THREAT = "threat"

class ActionType(str, Enum):
    """行动类型枚举"""
    IMPROVEMENT = "improvement"
    LEARNING = "learning"
    OPTIMIZATION = "optimization"
    CORRECTION = "correction"

class Experience(BaseModel):
    """经验模型"""
    experience_id: str = Field(..., description="经验ID")
    experience_type: ExperienceType = Field(..., description="经验类型")
    title: str = Field(..., description="标题")
    description: str = Field(..., description="描述")
    data: Dict[str, Any] = Field(..., description="经验数据")
    context: Dict[str, Any] = Field(default_factory=dict, description="上下文")
    timestamp: datetime = Field(default_factory=datetime.now, description="时间戳")
    metadata: Dict[str, Any] = Field(default_factory=dict, description="元数据")

class Insight(BaseModel):
    """洞察模型"""
    insight_id: str = Field(..., description="洞察ID")
    insight_type: InsightType = Field(..., description="洞察类型")
    title: str = Field(..., description="标题")
    description: str = Field(..., description="描述")
    evidence: List[str] = Field(..., description="证据")
    confidence: float = Field(..., ge=0.0, le=1.0, description="置信度")
    impact: str = Field(..., description="影响")
    related_experiences: List[str] = Field(default_factory=list, description="相关经验")
    created_at: datetime = Field(default_factory=datetime.now, description="创建时间")

class Action(BaseModel):
    """行动模型"""
    action_id: str = Field(..., description="行动ID")
    action_type: ActionType = Field(..., description="行动类型")
    title: str = Field(..., description="标题")
    description: str = Field(..., description="描述")
    steps: List[str] = Field(..., description="步骤")
    priority: int = Field(..., ge=1, le=5, description="优先级(1-5)")
    expected_outcome: str = Field(..., description="预期结果")
    related_insights: List[str] = Field(default_factory=list, description="相关洞察")
    status: str = Field(default="pending", description="状态")
    created_at: datetime = Field(default_factory=datetime.now, description="创建时间")
    due_date: Optional[datetime] = Field(None, description="截止日期")

class Reflection(BaseModel):
    """反思模型"""
    reflection_id: str = Field(..., description="反思ID")
    reflection_type: ReflectionType = Field(..., description="反思类型")
    title: str = Field(..., description="标题")
    description: str = Field(..., description="描述")
    experience_id: str = Field(..., description="经验ID")
    insights: List[str] = Field(default_factory=list, description="洞察ID列表")
    actions: List[str] = Field(default_factory=list, description="行动ID列表")
    summary: str = Field(..., description="总结")
    created_at: datetime = Field(default_factory=datetime.now, description="创建时间")
    updated_at: datetime = Field(default_factory=datetime.now, description="更新时间")

class SelfReflectionConfig(BaseModel):
    """自我反思配置模型"""

    action_db: Dict[str, Any] = Field(default_factory=dict, description="行动数据库配置")

#### 异常处理
```python
# 自定义异常类
class SelfReflectionError(Exception):
    """自我反思错误基类"""
    pass

class ReflectionError(SelfReflectionError):
    """反思错误"""
    pass

class ExperienceError(SelfReflectionError):
    """经验错误"""
    pass

class InsightError(SelfReflectionError):
    """洞察错误"""
    pass

class ActionError(SelfReflectionError):
    """行动错误"""
    pass

# 异常处理示例
def reflect_on_experience(self, experience: Dict[str, Any]) -> Dict[str, Any]:
    try:
        # 验证经验数据
        if not self._validate_experience(experience):
            raise ExperienceError("经验数据无效")
            
        # 分析经验
        analyzed_experience = self.experience_analyzer.analyze(experience)
        
        # 生成洞察
        insights = self.insight_generator.generate(analyzed_experience)
        
        # 规划行动
        actions = self.action_planner.plan(insights)
        
        # 创建反思
        reflection = self._create_reflection(experience, insights, actions)
        
        return {
            'status': 'success',
            'reflection_id': reflection.reflection_id,
            'insights': [insight.dict() for insight in insights],
            'actions': [action.dict() for action in actions]
        }
        
    except ExperienceError as e:
        logger.error(f"经验错误: {str(e)}")
        raise
    except InsightError as e:
        logger.error(f"洞察错误: {str(e)}")
        raise
    except ActionError as e:
        logger.error(f"行动错误: {str(e)}")
        raise
    except Exception as e:
        logger.error(f"未知错误: {str(e)}")
        raise ReflectionError(f"自我反思失败: {str(e)}")
```

# API接口实现
from fastapi import FastAPI, HTTPException, Depends, Query
from pydantic import BaseModel
from typing import List, Dict, Any, Optional

app = FastAPI(title="自我反思API", version="1.0.0")

# API端点
@app.post("/api/reflection/start", response_model=Dict[str, Any])
async def start_reflection(
    service: SelfReflectionManager = Depends(get_self_reflection_service)
):
    """启动自我反思"""
    try:
        result = service.start_reflection()
        return result
    except ReflectionError as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.post("/api/reflection/stop", response_model=Dict[str, Any])
async def stop_reflection(
    service: SelfReflectionManager = Depends(get_self_reflection_service)
):
    """停止自我反思"""
    try:
        result = service.stop_reflection()
        return result
    except ReflectionError as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.post("/api/reflection/reflect", response_model=Dict[str, Any])
async def reflect_on_experience(
    experience: Dict[str, Any],
    service: SelfReflectionManager = Depends(get_self_reflection_service)
):
    """对经验进行反思"""
    try:
        result = service.reflect_on_experience(experience)
        return result
    except ReflectionError as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.get("/api/reflection/reflections", response_model=Dict[str, Any])
async def get_reflections(
    reflection_type: Optional[str] = Query(None, description="反思类型"),
    limit: int = Query(10, ge=1, le=100, description="返回数量限制"),
    service: SelfReflectionManager = Depends(get_self_reflection_service)
):
    """获取反思记录"""
    try:
        # 构建过滤条件
        filters = {}
        if reflection_type:
            filters['reflection_type'] = reflection_type
        
        # 获取反思记录
        result = service.get_reflections(filters, limit)
        return result
    except ReflectionError as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.get("/api/reflection/reflections/{reflection_id}", response_model=Dict[str, Any])
async def get_reflection(
    reflection_id: str,
    service: SelfReflectionManager = Depends(get_self_reflection_service)
):
    """获取特定反思记录"""
    try:
        result = service.get_reflection(reflection_id)
        return result
    except ReflectionError as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.post("/api/reflection/insights", response_model=Dict[str, Any])

    experience: Dict[str, Any],

# 事件驱动接口实现
import asyncio
from typing import Callable, Dict, Any

class SelfReflectionEventHandler:
    """自我反思事件处理器"""
    
    def __init__(self, self_reflection_service: SelfReflectionManager):
        self.self_reflection_service = self_reflection_service
        self.event_bus = EventBus()
        self._register_handlers()
    
    def _register_handlers(self):
        """注册事件处理器"""
        self.event_bus.subscribe("experience_received", self._handle_experience_received)
        self.event_bus.subscribe("reflection_requested", self._handle_reflection_requested)
        self.event_bus.subscribe("insight_generated", self._handle_insight_generated)
        self.event_bus.subscribe("action_planned", self._handle_action_planned)
    
    async def _handle_experience_received(self, event_data: Dict[str, Any]):
        """处理经验接收事件"""
        try:
            # 提取经验数据
            experience_data = event_data.get('experience')
            if not experience_data:
                logger.warning("经验接收事件缺少经验数据")
                return
            
            # 对经验进行反思
            result = await self.self_reflection_service.reflect_on_experience(experience_data)
            
            logger.debug(f"处理经验接收事件: {experience_data.get('type')}")
        except Exception as e:
            logger.error(f"处理经验接收事件失败: {str(e)}")
    
    async def _handle_reflection_requested(self, event_data: Dict[str, Any]):
        """处理反思请求事件"""
        try:
            # 提取反思请求数据
            reflection_request = event_data.get('reflection_request')
            if not reflection_request:
                logger.warning("反思请求事件缺少请求数据")
                return
            
            # 获取经验ID
            experience_id = reflection_request.get('experience_id')
            if not experience_id:
                logger.warning("反思请求缺少经验ID")
                return
            
            # 获取经验数据
            experience = await self.self_reflection_service.get_experience(experience_id)
            if not experience:
                logger.warning(f"经验不存在: {experience_id}")
                return
            
            # 对经验进行反思
            result = await self.self_reflection_service.reflect_on_experience(experience)
            
            logger.debug(f"处理反思请求事件: {experience_id}")
        except Exception as e:
            logger.error(f"处理反思请求事件失败: {str(e)}")
    
    async def _handle_insight_generated(self, event_data: Dict[str, Any]):
        """处理洞察生成事件"""
        try:
            # 提取洞察数据
            insight_data = event_data.get('insight')
            if not insight_data:
                logger.warning("洞察生成事件缺少洞察数据")
                return
            
            # 保存洞察
            await self.self_reflection_service.save_insight(insight_data)
            
            logger.debug(f"处理洞察生成事件: {insight_data.get('insight_id')}")
        except Exception as e:
            logger.error(f"处理洞察生成事件失败: {str(e)}")
    
    async def _handle_action_planned(self, event_data: Dict[str, Any]):
        """处理行动规划事件"""
        try:
            # 提取行动数据
            action_data = event_data.get('action')
            if not action_data:
                logger.warning("行动规划事件缺少行动数据")
                return
            
            # 保存行动
            await self.self_reflection_service.save_action(action_data)
            
            logger.debug(f"处理行动规划事件: {action_data.get('action_id')}")
        except Exception as e:
            logger.error(f"处理行动规划事件失败: {str(e)}")
