# 反馈处理模块6A工作流

## 模块概述

### 模块名称
反馈处理模块

### 所属子系统
交互表达子系统

### 模块功能描述
反馈处理模块负责收集、分析和处理婴儿对交互表达的各种反馈，包括表情、语音、动作等多模态反馈信息，以及环境反馈信息。该模块通过分析反馈信息，评估交互效果，并为交互策略的调整提供依据，从而实现交互系统的自适应优化。

### 模块职责
1. 反馈收集：收集婴儿的多模态反馈信息
2. 反馈分析：分析反馈信息的含义和情感
3. 反馈评估：评估交互效果和满意度
4. 反馈处理：处理反馈信息并生成处理结果
5. 策略建议：根据反馈提供交互策略调整建议
6. 反馈存储：存储反馈信息和分析结果

### 模块接口
- **输入接口**：多模态反馈数据、环境反馈数据、交互记录
- **输出接口**：反馈分析结果、交互效果评估、策略调整建议
- **依赖模块**：感知处理子系统、交互管理模块、交互控制模块
- **被依赖模块**：交互管理模块、交互控制模块、决策模块

class FeedbackProcessor:
    def __init__(self, feedback_collector, data_preprocessor, feature_extractor,
                 emotion_analyzer, intent_recognizer, interaction_evaluator,
                 strategy_advisor, abnormal_detector, feedback_storage, privacy_protector):
        self.feedback_collector = feedback_collector
        self.data_preprocessor = data_preprocessor
        self.feature_extractor = feature_extractor
        self.emotion_analyzer = emotion_analyzer
        self.intent_recognizer = intent_recognizer
        self.interaction_evaluator = interaction_evaluator
        self.strategy_advisor = strategy_advisor
        self.abnormal_detector = abnormal_detector
        self.feedback_storage = feedback_storage
        self.privacy_protector = privacy_protector
        
        # 初始化模型
        self._load_models()
        
    async def process_feedback(self, session_id, interaction_id, feedback_data):
        try:
            # 收集反馈数据
            collected_feedback = await self.feedback_collector.collect_feedback(
                session_id, interaction_id, feedback_data
            )
            
            # 预处理反馈数据
            preprocessed_feedback = await self.data_preprocessor.preprocess_feedback(
                collected_feedback
            )
            
            # 提取特征
            features = await self.feature_extractor.extract_features(preprocessed_feedback)

            # 意图识别
            intent_recognition = await self.intent_recognizer.recognize_intent(features)
            
            # 交互评估
            interaction_evaluation = await self.interaction_evaluator.evaluate_interaction(
                session_id, interaction_id, emotion_analysis, intent_recognition
            )
            
            # 策略建议
            strategy_suggestions = await self.strategy_advisor.advise_strategy(
                session_id, interaction_id, emotion_analysis, intent_recognition, interaction_evaluation
            )
            
            # 异常检测
            abnormal_detection = await self.abnormal_detector.detect_abnormal(
                session_id, interaction_id, features, emotion_analysis, intent_recognition
            )
            
            # 构建反馈处理结果
            feedback_result = {
                "session_id": session_id,
                "interaction_id": interaction_id,
                "emotion_analysis": emotion_analysis,
                "intent_recognition": intent_recognition,
                "interaction_evaluation": interaction_evaluation,
                "strategy_suggestions": strategy_suggestions,
                "abnormal_detection": abnormal_detection,
                "timestamp": time.time()
            }
            
            # 隐私保护
            protected_result = await self.privacy_protector.protect_data(feedback_result)
            
            # 存储反馈结果

            return protected_result

            return error_result

# 情感分析器实现
class EmotionAnalyzer:
    def __init__(self, emotion_model, emotion_labels):
        self.emotion_model = emotion_model
        self.emotion_labels = emotion_labels
        
    async def analyze_emotion(self, features):
        try:
            # 提取多模态特征
            expression_features = features.get("expression", [])
            voice_features = features.get("voice", [])
            action_features = features.get("action", [])
            
            # 分析表情情感
            expression_emotion = await self._analyze_expression_emotion(expression_features)
            
            # 分析语音情感
            voice_emotion = await self._analyze_voice_emotion(voice_features)
            
            # 分析动作情感
            action_emotion = await self._analyze_action_emotion(action_features)
            
            # 融合多模态情感
            fused_emotion = await self._fuse_emotions(
                expression_emotion, voice_emotion, action_emotion
            )
            
            # 分析情感变化
            emotion_change = await self._analyze_emotion_change(fused_emotion)
            
            # 分析情感一致性
            emotion_consistency = await self._analyze_emotion_consistency(
                expression_emotion, voice_emotion, action_emotion
            )
            
            return {
                "expression_emotion": expression_emotion,
                "voice_emotion": voice_emotion,
                "action_emotion": action_emotion,
                "fused_emotion": fused_emotion,
                "emotion_change": emotion_change,
                "emotion_consistency": emotion_consistency,
                "timestamp": time.time()
            }
            
        except Exception as e:
            return {
                "error": str(e),
                "timestamp": time.time()
            }
    
    async def _analyze_expression_emotion(self, expression_features):
        # 预处理表情特征
        if not expression_features:
            return {"emotion": "neutral", "intensity": 0.0, "confidence": 0.0}
        
        # 转换为张量
        features_tensor = torch.tensor(expression_features, dtype=torch.float32)
        
        # 添加批次维度
        if len(features_tensor.shape) == 1:
            features_tensor = features_tensor.unsqueeze(0)
        
        # 使用模型预测情感
        with torch.no_grad():
            predictions = self.emotion_model["expression"](features_tensor)
            probabilities = torch.softmax(predictions, dim=1)
            
            # 获取最高概率的情感
            max_prob, max_idx = torch.max(probabilities, dim=1)
            emotion = self.emotion_labels["expression"][max_idx.item()]
            intensity = max_prob.item()
            confidence = max_prob.item()
        
        return {
            "emotion": emotion,
            "intensity": intensity,
            "confidence": confidence,
            "probabilities": probabilities.squeeze().tolist()
        }
    
    async def _analyze_voice_emotion(self, voice_features):
        # 预处理语音特征
        if not voice_features:
            return {"emotion": "neutral", "intensity": 0.0, "confidence": 0.0}
        
        # 转换为张量
        features_tensor = torch.tensor(voice_features, dtype=torch.float32)
        
        # 添加批次维度
        if len(features_tensor.shape) == 1:
            features_tensor = features_tensor.unsqueeze(0)
        
        # 使用模型预测情感
        with torch.no_grad():
            predictions = self.emotion_model["voice"](features_tensor)
            probabilities = torch.softmax(predictions, dim=1)
            
            # 获取最高概率的情感
            max_prob, max_idx = torch.max(probabilities, dim=1)
            emotion = self.emotion_labels["voice"][max_idx.item()]
            intensity = max_prob.item()
            confidence = max_prob.item()
        
        return {
            "emotion": emotion,
            "intensity": intensity,
            "confidence": confidence,
            "probabilities": probabilities.squeeze().tolist()
        }
    
    async def _analyze_action_emotion(self, action_features):
        # 预处理动作特征
        if not action_features:
            return {"emotion": "neutral", "intensity": 0.0, "confidence": 0.0}
        
        # 转换为张量
        features_tensor = torch.tensor(action_features, dtype=torch.float32)
        
        # 添加批次维度
        if len(features_tensor.shape) == 1:
            features_tensor = features_tensor.unsqueeze(0)
        
        # 使用模型预测情感
        with torch.no_grad():
            predictions = self.emotion_model["action"](features_tensor)
            probabilities = torch.softmax(predictions, dim=1)
            
            # 获取最高概率的情感
            max_prob, max_idx = torch.max(probabilities, dim=1)
            emotion = self.emotion_labels["action"][max_idx.item()]
            intensity = max_prob.item()
            confidence = max_prob.item()
        
        return {
            "emotion": emotion,
            "intensity": intensity,
            "confidence": confidence,
            "probabilities": probabilities.squeeze().tolist()
        }
    
    async def _fuse_emotions(self, expression_emotion, voice_emotion, action_emotion):
        # 获取各模态情感概率
        expression_probs = expression_emotion.get("probabilities", [])
        voice_probs = voice_emotion.get("probabilities", [])
        action_probs = action_emotion.get("probabilities", [])
        
        # 加权融合各模态情感概率
        if expression_probs and voice_probs and action_probs:
            # 转换为numpy数组
            expression_probs = np.array(expression_probs)
            voice_probs = np.array(voice_probs)
            action_probs = np.array(action_probs)
            
            # 设置权重
            weights = [0.4, 0.3, 0.3]  # 表情、语音、动作权重
            
            # 加权平均
            fused_probs = weights[0] * expression_probs + weights[1] * voice_probs + weights[2] * action_probs
            
            # 获取最高概率的情感
            max_idx = np.argmax(fused_probs)
            emotion = self.emotion_labels["fused"][max_idx]
            intensity = fused_probs[max_idx]
            confidence = fused_probs[max_idx]
            
            return {
                "emotion": emotion,
                "intensity": intensity,
                "confidence": confidence,
                "probabilities": fused_probs.tolist()
            }
        else:
            # 如果有模态缺失，使用可用模态的结果
            if expression_probs:
                return expression_emotion
            elif voice_probs:
                return voice_emotion
            elif action_probs:
                return action_emotion
            else:
                return {"emotion": "neutral", "intensity": 0.0, "confidence": 0.0}
    
    async def _analyze_emotion_change(self, fused_emotion):
        # 获取历史情感数据
        # 这里简化处理，实际应该从历史数据中获取
        previous_emotion = "neutral"
        previous_intensity = 0.0
        
        # 分析情感变化
        current_emotion = fused_emotion.get("emotion", "neutral")
        current_intensity = fused_emotion.get("intensity", 0.0)
        
        # 计算情感变化
        emotion_change = {
            "from_emotion": previous_emotion,
            "to_emotion": current_emotion,
            "from_intensity": previous_intensity,
            "to_intensity": current_intensity,
            "emotion_changed": previous_emotion != current_emotion,
            "intensity_changed": abs(current_intensity - previous_intensity) > 0.2,
            "change_magnitude": abs(current_intensity - previous_intensity)
        }
        
        return emotion_change
    
    async def _analyze_emotion_consistency(self, expression_emotion, voice_emotion, action_emotion):
        # 分析多模态情感一致性
        expression_emotion_type = expression_emotion.get("emotion", "neutral")
        voice_emotion_type = voice_emotion.get("emotion", "neutral")
        action_emotion_type = action_emotion.get("emotion", "neutral")
        
        # 计算情感一致性
        emotions = [expression_emotion_type, voice_emotion_type, action_emotion_type]
        emotion_counts = {}
        
        for emotion in emotions:
            emotion_counts[emotion] = emotion_counts.get(emotion, 0) + 1
        
        # 获取最高频情感
        most_common_emotion = max(emotion_counts, key=emotion_counts.get)
        consistency_score = emotion_counts[most_common_emotion] / len(emotions)
        
        return {
            "most_common_emotion": most_common_emotion,
            "consistency_score": consistency_score,
            "is_consistent": consistency_score >= 0.7,
            "emotion_distribution": emotion_counts
        }

class StrategyAdvisor:
    def __init__(self, strategy_repository, adjustment_generator):
        self.strategy_repository = strategy_repository
        self.adjustment_generator = adjustment_generator
        
    async def advise_strategy(self, session_id, interaction_id, emotion_analysis, intent_recognition, interaction_evaluation):
        try:
            # 获取当前策略
            current_strategy = await self.strategy_repository.get_current_strategy(session_id)
            
            # 分析情感反馈
            emotion_feedback = await self._analyze_emotion_feedback(emotion_analysis)
            
            # 分析意图反馈
            intent_feedback = await self._analyze_intent_feedback(intent_recognition)
            
            # 分析交互评估
            evaluation_feedback = await self._analyze_evaluation_feedback(interaction_evaluation)
            
            # 生成策略调整建议
            strategy_adjustments = await self._generate_strategy_adjustments(
                current_strategy, emotion_feedback, intent_feedback, evaluation_feedback
            )
            
            # 优先级排序
            prioritized_adjustments = await self._prioritize_adjustments(strategy_adjustments)
            
            return {
                "session_id": session_id,
                "interaction_id": interaction_id,
                "current_strategy": current_strategy,
                "emotion_feedback": emotion_feedback,
                "intent_feedback": intent_feedback,
                "evaluation_feedback": evaluation_feedback,
                "strategy_adjustments": prioritized_adjustments,
                "timestamp": time.time()
            }
            
        except Exception as e:
            return {
                "session_id": session_id,
                "interaction_id": interaction_id,
                "error": str(e),
                "timestamp": time.time()
            }
    
    async def _analyze_emotion_feedback(self, emotion_analysis):
        # 分析情感反馈
        fused_emotion = emotion_analysis.get("fused_emotion", {})
        emotion_change = emotion_analysis.get("emotion_change", {})
        emotion_consistency = emotion_analysis.get("emotion_consistency", {})
        
        # 获取情感类型和强度
        emotion_type = fused_emotion.get("emotion", "neutral")
        emotion_intensity = fused_emotion.get("intensity", 0.0)
        
        # 分析情感变化
        emotion_changed = emotion_change.get("emotion_changed", False)
        intensity_changed = emotion_change.get("intensity_changed", False)
        change_magnitude = emotion_change.get("change_magnitude", 0.0)
        
        # 分析情感一致性
        is_consistent = emotion_consistency.get("is_consistent", True)
        consistency_score = emotion_consistency.get("consistency_score", 1.0)
        
        # 生成情感反馈
        emotion_feedback = {
            "emotion_type": emotion_type,
            "emotion_intensity": emotion_intensity,
            "emotion_changed": emotion_changed,
            "intensity_changed": intensity_changed,
            "change_magnitude": change_magnitude,
            "is_consistent": is_consistent,
            "consistency_score": consistency_score
        }
        
        # 根据情感反馈生成建议
        if emotion_type == "negative" and emotion_intensity > 0.7:
            emotion_feedback["suggestion"] = "reduce_intensity"
            emotion_feedback["priority"] = "high"
        elif emotion_type == "positive" and emotion_intensity > 0.7:
            emotion_feedback["suggestion"] = "maintain_current"
            emotion_feedback["priority"] = "low"
        elif not is_consistent:
            emotion_feedback["suggestion"] = "increase_consistency"
            emotion_feedback["priority"] = "medium"
        else:
            emotion_feedback["suggestion"] = "no_change"
            emotion_feedback["priority"] = "low"
        
        return emotion_feedback
    
    async def _analyze_intent_feedback(self, intent_recognition):
        # 分析意图反馈
        intent_type = intent_recognition.get("intent", "unknown")
        intent_confidence = intent_recognition.get("confidence", 0.0)
        
        # 生成意图反馈
        intent_feedback = {
            "intent_type": intent_type,
            "intent_confidence": intent_confidence
        }
        
        # 根据意图反馈生成建议
        if intent_type == "disengagement" and intent_confidence > 0.7:
            intent_feedback["suggestion"] = "increase_engagement"
            intent_feedback["priority"] = "high"
        elif intent_type == "engagement" and intent_confidence > 0.7:
            intent_feedback["suggestion"] = "maintain_engagement"
            intent_feedback["priority"] = "low"
        elif intent_confidence < 0.5:
            intent_feedback["suggestion"] = "clarify_intent"
            intent_feedback["priority"] = "medium"
        else:
            intent_feedback["suggestion"] = "no_change"
            intent_feedback["priority"] = "low"
        
        return intent_feedback
    
    async def _analyze_evaluation_feedback(self, interaction_evaluation):
        # 分析交互评估反馈
        satisfaction_score = interaction_evaluation.get("satisfaction_score", 0.5)
        engagement_score = interaction_evaluation.get("engagement_score", 0.5)
        effectiveness_score = interaction_evaluation.get("effectiveness_score", 0.5)
        
        # 计算综合评分
        overall_score = (satisfaction_score + engagement_score + effectiveness_score) / 3
        
        # 生成评估反馈
        evaluation_feedback = {
            "satisfaction_score": satisfaction_score,
            "engagement_score": engagement_score,
            "effectiveness_score": effectiveness_score,
            "overall_score": overall_score
        }
        
        # 根据评估反馈生成建议
        if overall_score < 0.4:
            evaluation_feedback["suggestion"] = "major_adjustment"
            evaluation_feedback["priority"] = "high"
        elif overall_score < 0.6:
            evaluation_feedback["suggestion"] = "minor_adjustment"
            evaluation_feedback["priority"] = "medium"
        elif overall_score > 0.8:
            evaluation_feedback["suggestion"] = "maintain_current"
            evaluation_feedback["priority"] = "low"
        else:
            evaluation_feedback["suggestion"] = "fine_tune"
            evaluation_feedback["priority"] = "low"
        
        return evaluation_feedback
    
    async def _generate_strategy_adjustments(self, current_strategy, emotion_feedback, intent_feedback, evaluation_feedback):
        # 生成策略调整建议
        adjustments = []
        
        # 基于情感反馈的调整
        emotion_suggestion = emotion_feedback.get("suggestion", "no_change")
        if emotion_suggestion == "reduce_intensity":
            adjustments.append({
                "type": "intensity",
                "action": "decrease",
                "value": 0.2,
                "reason": "Negative emotion detected with high intensity"
            })
        elif emotion_suggestion == "increase_consistency":
            adjustments.append({
                "type": "consistency",
                "action": "increase",
                "value": 0.3,
                "reason": "Inconsistent emotions across modalities"
            })
        
        # 基于意图反馈的调整
        intent_suggestion = intent_feedback.get("suggestion", "no_change")
        if intent_suggestion == "increase_engagement":
            adjustments.append({
                "type": "engagement",
                "action": "increase",
                "value": 0.3,
                "reason": "Disengagement intent detected"
            })
        elif intent_suggestion == "clarify_intent":
            adjustments.append({
                "type": "clarity",
                "action": "increase",
                "value": 0.2,
                "reason": "Low confidence in intent recognition"
            })
        
        # 基于评估反馈的调整
        evaluation_suggestion = evaluation_feedback.get("suggestion", "no_change")
        if evaluation_suggestion == "major_adjustment":
            adjustments.append({
                "type": "strategy",
                "action": "change",
                "value": "alternative_strategy",
                "reason": "Low overall interaction evaluation score"
            })
        elif evaluation_suggestion == "minor_adjustment":
            adjustments.append({
                "type": "parameters",
                "action": "tune",
                "value": "fine_tune_parameters",
                "reason": "Moderate overall interaction evaluation score"
            })
        
        return adjustments
    
    async def _prioritize_adjustments(self, adjustments):
        # 优先级排序
        high_priority = [adj for adj in adjustments if adj.get("priority") == "high"]
        medium_priority = [adj for adj in adjustments if adj.get("priority") == "medium"]
        low_priority = [adj for adj in adjustments if adj.get("priority") == "low"]
        
        # 合并并排序
        prioritized_adjustments = high_priority + medium_priority + low_priority
        
        return prioritized_adjustments
