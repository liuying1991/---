# 交互管理模块6A工作流

## 模块概述

### 模块名称
交互管理模块

### 所属子系统
交互表达子系统

### 模块功能描述
交互管理模块负责协调和管理交互表达子系统的各个组件，确保表情生成、语音合成和肢体动作生成等模块能够协同工作，为婴儿提供一致、自然、富有表现力的交互体验。该模块还负责交互流程的控制、状态管理和交互策略的优化。

### 模块职责
1. 交互协调：协调各表达模块的工作，确保输出一致性
2. 状态管理：管理交互状态和上下文信息
3. 流程控制：控制交互流程和节奏
4. 策略优化：根据反馈优化交互策略
5. 资源管理：管理交互过程中的资源分配
6. 质量保证：确保交互质量和适宜性

### 模块接口
- **输入接口**：交互请求、婴儿状态、环境信息
- **输出接口**：协调后的表达指令、交互状态、反馈信息
- **依赖模块**：表情生成模块、语音合成模块、肢体动作生成模块
- **被依赖模块**：交互控制模块、用户界面模块

# 交互管理器实现
class InteractionManager:
    def __init__(self, expression_coordinator, voice_coordinator, action_coordinator,
                 state_manager, flow_controller, strategy_optimizer, resource_manager,
                 quality_monitor, exception_handler, history_recorder, feedback_processor):
        self.expression_coordinator = expression_coordinator
        self.voice_coordinator = voice_coordinator
        self.action_coordinator = action_coordinator
        self.state_manager = state_manager
        self.flow_controller = flow_controller
        self.strategy_optimizer = strategy_optimizer
        self.resource_manager = resource_manager
        self.quality_monitor = quality_monitor
        self.exception_handler = exception_handler
        self.history_recorder = history_recorder
        self.feedback_processor = feedback_processor
        
        # 初始化异步任务
        self.background_tasks = set()
        self._lock = asyncio.Lock()
        
    async def process_interaction(self, interaction_request):
        try:
            # 获取交互会话ID
            session_id = interaction_request.get("session_id", str(uuid.uuid4()))
            
            # 获取或创建交互状态
            async with self._lock:
                state = await self.state_manager.get_or_create_state(session_id)
            
            # 更新交互状态
            await self.state_manager.update_state(session_id, {
                "last_request": interaction_request,
                "timestamp": time.time()
            })
            
            # 处理交互请求
            result = await self._process_interaction_request(session_id, interaction_request, state)
            
            # 记录交互历史
            await self.history_recorder.record_interaction(session_id, interaction_request, result)
            
            # 监控交互质量
            quality_score = await self.quality_monitor.assess_interaction(session_id, interaction_request, result)
            
            # 更新结果
            result["quality_score"] = quality_score
            
            # 异步优化策略
            task = asyncio.create_task(self._async_optimize_strategy(session_id, interaction_request, result, quality_score))
            self.background_tasks.add(task)
            task.add_done_callback(self.background_tasks.discard)
            
            return result
            
        except Exception as e:
            # 处理异常
            return await self.exception_handler.handle_exception(session_id, e, interaction_request)
    
    async def _process_interaction_request(self, session_id, interaction_request, state):
        # 分析交互请求
        request_analysis = await self._analyze_interaction_request(interaction_request, state)
        
        # 选择交互策略
        strategy = await self.strategy_optimizer.select_strategy(session_id, request_analysis, state)
        
        # 分配资源
        resources = await self.resource_manager.allocate_resources(session_id, strategy, request_analysis)
        
        # 控制交互流程
        flow_control = await self.flow_controller.control_flow(session_id, strategy, request_analysis, state)
        
        # 协调多模态表达
        expression_tasks = []
        
        # 表情生成任务
        if strategy.get("use_expression", True):
            expression_task = asyncio.create_task(
                self.expression_coordinator.coordinate_expression(
                    session_id, interaction_request, strategy, resources, flow_control
                )
            )
            expression_tasks.append(("expression", expression_task))
        
        # 语音合成任务
        if strategy.get("use_voice", True):
            voice_task = asyncio.create_task(
                self.voice_coordinator.coordinate_voice(
                    session_id, interaction_request, strategy, resources, flow_control
                )
            )
            expression_tasks.append(("voice", voice_task))
        
        # 动作生成任务
        if strategy.get("use_action", True):
            action_task = asyncio.create_task(
                self.action_coordinator.coordinate_action(
                    session_id, interaction_request, strategy, resources, flow_control
                )
            )
            expression_tasks.append(("action", action_task))
        
        # 等待所有任务完成
        results = {}
        for task_type, task in expression_tasks:
            try:
                results[task_type] = await task
            except Exception as e:
                results[task_type] = {"error": str(e)}
                await self.exception_handler.handle_expression_error(session_id, task_type, e)
        
        # 同步多模态表达
        synchronized_results = await self._synchronize_expressions(results, strategy, flow_control)
        
        # 构建交互结果
        interaction_result = {
            "session_id": session_id,
            "request_id": interaction_request.get("request_id", str(uuid.uuid4())),
            "strategy": strategy,
            "expressions": synchronized_results,
            "flow_control": flow_control,
            "timestamp": time.time()
        }
        
        return interaction_result
    
    async def _analyze_interaction_request(self, interaction_request, state):
        # 分析交互请求的内容和意图
        content_analysis = await self._analyze_content(interaction_request.get("content", ""))
        
        # 分析婴儿状态
        baby_state_analysis = await self._analyze_baby_state(interaction_request.get("baby_state", {}))
        
        # 分析环境信息
        environment_analysis = await self._analyze_environment(interaction_request.get("environment", {}))
        
        # 分析历史交互
        history_analysis = await self._analyze_interaction_history(state.get("history", []))
        
        # 综合分析结果
        request_analysis = {
            "content": content_analysis,
            "baby_state": baby_state_analysis,
            "environment": environment_analysis,
            "history": history_analysis
        }
        
        return request_analysis
    
    async def _synchronize_expressions(self, results, strategy, flow_control):
        # 同步多模态表达
        synchronized_results = {}
        
        # 获取同步参数
        sync_params = strategy.get("sync_params", {})
        time_tolerance = sync_params.get("time_tolerance", 50)  # 50ms时间容差
        
        # 计算基准时间
        base_time = None
        for expression_type, result in results.items():
            if "error" not in result and "timestamp" in result:
                if base_time is None or result["timestamp"] < base_time:
                    base_time = result["timestamp"]
        
        if base_time is None:
            base_time = time.time() * 1000  # 当前时间（毫秒）
        
        # 调整各表达的时间戳
        for expression_type, result in results.items():
            if "error" in result:
                synchronized_results[expression_type] = result
                continue
            
            # 计算时间偏移
            timestamp = result.get("timestamp", base_time)
            time_offset = timestamp - base_time
            
            # 如果时间偏移在容差范围内，调整到基准时间
            if abs(time_offset) <= time_tolerance:
                result["timestamp"] = base_time
                result["adjusted"] = True
            
            synchronized_results[expression_type] = result
        
        return synchronized_results
    
    async def _async_optimize_strategy(self, session_id, interaction_request, result, quality_score):
        try:
            # 获取当前状态
            state = await self.state_manager.get_state(session_id)
            
            # 分析交互结果
            result_analysis = await self._analyze_interaction_result(result)
            
            # 优化策略
            optimized_strategy = await self.strategy_optimizer.optimize_strategy(
                session_id, state.get("strategy", {}), interaction_request, result, quality_score
            )
            
            # 更新状态中的策略
            await self.state_manager.update_state(session_id, {
                "strategy": optimized_strategy,
                "last_optimization": time.time()
            })
            
        except Exception as e:
            await self.exception_handler.handle_optimization_error(session_id, e)

# 交互协调器实现
class ExpressionCoordinator:
    def __init__(self, expression_generator, multimodal_synchronizer):
        self.expression_generator = expression_generator
        self.multimodal_synchronizer = multimodal_synchronizer
        
    async def coordinate_expression(self, session_id, interaction_request, strategy, resources, flow_control):
        try:
            # 提取表情生成参数
            expression_params = self._extract_expression_params(interaction_request, strategy, resources, flow_control)
            
            # 生成表情
            expression_result = await self.expression_generator.generate(
                text=interaction_request.get("content", ""),
                generation_params=expression_params,
                baby_state=interaction_request.get("baby_state", {})
            )
            
            # 添加时间戳
            expression_result["timestamp"] = time.time() * 1000  # 毫秒
            
            return expression_result
            
        except Exception as e:
            return {"error": str(e)}
    
    def _extract_expression_params(self, interaction_request, strategy, resources, flow_control):
        # 提取表情生成参数
        expression_params = {
            "apply_emotion": strategy.get("apply_emotion", True),
            "emotion_intensity": strategy.get("emotion_intensity", 0.8),
            "generate_dynamic": strategy.get("generate_dynamic", True),
            "optimize_expression": strategy.get("optimize_expression", True),
            "animation_params": {
                "duration": flow_control.get("duration", 3000),  # 3秒
                "intensity": flow_control.get("intensity", 0.8)
            }
        }
        
        # 根据资源限制调整参数
        if resources.get("gpu_memory", 0) < 1024:  # GPU内存小于1GB
            expression_params["optimize_expression"] = False
            expression_params["animation_params"]["duration"] = 2000  # 缩短到2秒
        
        return expression_params

# 策略优化器实现
class StrategyOptimizer:
    def __init__(self, strategy_repository, performance_analyzer):
        self.strategy_repository = strategy_repository
        self.performance_analyzer = performance_analyzer
        
    async def select_strategy(self, session_id, request_analysis, state):
        # 获取候选策略
        candidate_strategies = await self.strategy_repository.get_candidate_strategies(request_analysis)
        
        # 评估候选策略
        strategy_scores = []
        for strategy in candidate_strategies:
            score = await self._evaluate_strategy(session_id, strategy, request_analysis, state)
            strategy_scores.append((strategy, score))
        
        # 选择最佳策略
        best_strategy = max(strategy_scores, key=lambda x: x[1])[0]
        
        return best_strategy
    
    async def optimize_strategy(self, session_id, current_strategy, interaction_request, result, quality_score):
        # 分析当前策略表现
        strategy_performance = await self.performance_analyzer.analyze_strategy_performance(
            session_id, current_strategy, interaction_request, result, quality_score
        )
        
        # 获取优化建议
        optimization_suggestions = await self._get_optimization_suggestions(
            session_id, current_strategy, strategy_performance
        )
        
        # 应用优化建议
        optimized_strategy = await self._apply_optimization_suggestions(
            current_strategy, optimization_suggestions
        )
        
        return optimized_strategy
    
    async def _evaluate_strategy(self, session_id, strategy, request_analysis, state):
        # 评估策略的适用性
        content_suitability = await self._evaluate_content_suitability(strategy, request_analysis["content"])
        baby_state_suitability = await self._evaluate_baby_state_suitability(strategy, request_analysis["baby_state"])
        environment_suitability = await self._evaluate_environment_suitability(strategy, request_analysis["environment"])
        history_suitability = await self._evaluate_history_suitability(strategy, request_analysis["history"])
        
        # 计算综合评分
        score = (
            content_suitability * 0.3 +
            baby_state_suitability * 0.3 +
            environment_suitability * 0.2 +
            history_suitability * 0.2
        )
        
        return score
    
    async def _evaluate_content_suitability(self, strategy, content_analysis):
        # 评估策略对内容的适用性
        content_type = content_analysis.get("type", "general")
        emotion = content_analysis.get("emotion", "neutral")
        
        # 根据策略配置评估适用性
        suitability = 0.5  # 默认中等适用性
        
        # 内容类型匹配
        if "content_types" in strategy:
            if content_type in strategy["content_types"]:
                suitability += 0.3
            else:
                suitability -= 0.2
        
        # 情感匹配
        if "emotions" in strategy:
            if emotion in strategy["emotions"]:
                suitability += 0.2
            else:
                suitability -= 0.1
        
        return max(0.0, min(1.0, suitability))
    
    async def _get_optimization_suggestions(self, session_id, current_strategy, strategy_performance):
        # 获取优化建议
        suggestions = []
        
        # 基于性能分析的优化建议
        if strategy_performance.get("response_time", 0) > 300:  # 响应时间超过300ms
            suggestions.append({
                "type": "performance",
                "action": "reduce_complexity",
                "reason": "响应时间过长"
            })
        
        if strategy_performance.get("quality_score", 0) < 4.0:  # 质量评分低于4.0
            suggestions.append({
                "type": "quality",
                "action": "increase_optimization",
                "reason": "质量评分过低"
            })
        
        if strategy_performance.get("resource_usage", 0) > 0.8:  # 资源使用率超过80%
            suggestions.append({
                "type": "resource",
                "action": "reduce_resource_usage",
                "reason": "资源使用率过高"
            })
        
        return suggestions
    
    async def _apply_optimization_suggestions(self, current_strategy, suggestions):
        # 应用优化建议
        optimized_strategy = current_strategy.copy()
        
        for suggestion in suggestions:
            if suggestion["type"] == "performance":
                if suggestion["action"] == "reduce_complexity":
                    # 降低复杂度
                    optimized_strategy["optimize_expression"] = False
                    optimized_strategy["optimize_action"] = False
                    optimized_strategy["animation_params"]["duration"] = 2000
            
            elif suggestion["type"] == "quality":
                if suggestion["action"] == "increase_optimization":
                    # 提高质量
                    optimized_strategy["optimize_expression"] = True
                    optimized_strategy["optimize_action"] = True
                    optimized_strategy["emotion_intensity"] = min(1.0, optimized_strategy.get("emotion_intensity", 0.8) + 0.1)
            
            elif suggestion["type"] == "resource":
                if suggestion["action"] == "reduce_resource_usage":
                    # 减少资源使用
                    optimized_strategy["generate_dynamic"] = False
                    optimized_strategy["animation_params"]["duration"] = 1500
        
        return optimized_strategy
