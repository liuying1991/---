# 语音合成模块实现工作流

## 1. 模块概述

### 1.1 模块功能与职责

语音合成模块是婴儿AI管家系统交互表达子系统的核心组件之一，负责将文本信息转换为自然、流畅的语音输出。该模块具有以下主要功能：

1. **文本预处理**：对输入文本进行规范化、分词、韵律预测等预处理
2. **语音合成**：使用先进的深度学习模型生成高质量语音波形
3. **声音定制**：支持多种声音风格和情感表达
4. **语音优化**：对生成的语音进行后处理，提高自然度和清晰度
5. **多语言支持**：支持多种语言的语音合成

### 1.2 模块在系统中的位置

语音合成模块位于交互表达子系统中，与自然语言生成模块、表情生成模块等紧密协作：

```
交互表达子系统
├── 自然语言生成模块 (生成文本内容)
├── 语音合成模块 (将文本转换为语音)  ← 当前实现
├── 表情生成模块 (生成面部表情)
├── 肢体动作生成模块 (生成肢体动作)
└── 多模态融合模块 (融合多种表达方式)
```

### 1.3 关键技术选型

1. **TTS引擎**：基于Tacotron 2 + WaveGlow或FastSpeech 2 + HiFi-GAN的端到端语音合成
2. **深度学习框架**：PyTorch或TensorFlow
3. **声音转换**：基于Voice Conversion技术实现多种声音风格
4. **情感合成**：支持情感语音合成，表达不同情绪状态
5. **实时处理**：优化模型推理速度，实现实时语音合成

## 3. 核心组件实现

### 3.1 文本预处理组件

```python
# core/text_processor.py
import re
import unicodedata
from typing import List, Dict, Any, Optional
import logging

from models.synthesis_request import SynthesisRequest
from utils.text_utils import TextNormalizer, Phonemizer
from config.tts_config import TTSConfig

logger = logging.getLogger(__name__)

class TextProcessor:
    """文本预处理组件"""
    
    def __init__(self, config: TTSConfig):
        self.config = config
        self.normalizer = TextNormalizer(config.normalizer)
        self.phonemizer = Phonemizer(config.phonemizer)
        
        logger.info("TextProcessor initialized")
    
    def process(self, request: SynthesisRequest) -> Dict[str, Any]:
        """处理文本输入"""
        try:
            # 1. 文本规范化
            normalized_text = self._normalize_text(request.text)
            
            # 2. 文本分词
            tokens = self._tokenize(normalized_text)
            
            # 3. 韵律预测
            prosody = self._predict_prosody(tokens, request.emotion)
            
            # 4. 音素转换
            phonemes = self._text_to_phonemes(tokens, request.language)
            
            # 5. 生成输入特征
            input_features = self._generate_input_features(
                tokens, phonemes, prosody, request
            )
            
            return {
                "normalized_text": normalized_text,
                "tokens": tokens,
                "phonemes": phonemes,
                "prosody": prosody,
                "input_features": input_features
            }
            
        except Exception as e:
            logger.error(f"Error processing text: {e}")
            raise
    
    def _normalize_text(self, text: str) -> str:
        """文本规范化"""
        # 1. Unicode规范化
        text = unicodedata.normalize("NFKC", text)
        
        # 2. 移除特殊字符
        text = re.sub(r"[^\w\s\.,!?;:'\"\-]", "", text)
        
        # 3. 数字转换
        text = self.normalizer.normalize_numbers(text)
        
        # 4. 缩写展开
        text = self.normalizer.expand_abbreviations(text)
        
        return text
    
    def _tokenize(self, text: str) -> List[str]:
        """文本分词"""
        # 使用简单的空格分词，实际应用中可使用更复杂的分词器
        tokens = text.split()
        return tokens
    
    def _predict_prosody(self, tokens: List[str], emotion: str) -> Dict[str, Any]:
        """预测韵律特征"""
        # 简化的韵律预测，实际应用中应使用专门的模型
        prosody = {
            "duration": [0.2] * len(tokens),  # 每个token的持续时间
            "pitch": [200.0] * len(tokens),   # 基频
            "energy": [0.8] * len(tokens),    # 能量
            "emotion": emotion
        }
        
        # 根据情感调整韵律
        if emotion == "happy":
            prosody["pitch"] = [p * 1.2 for p in prosody["pitch"]]
            prosody["energy"] = [e * 1.1 for e in prosody["energy"]]
        elif emotion == "sad":
            prosody["pitch"] = [p * 0.8 for p in prosody["pitch"]]
            prosody["energy"] = [e * 0.9 for e in prosody["energy"]]
        elif emotion == "angry":
            prosody["pitch"] = [p * 1.3 for p in prosody["pitch"]]
            prosody["energy"] = [e * 1.2 for e in prosody["energy"]]
        
        return prosody
    
    def _text_to_phonemes(self, tokens: List[str], language: str) -> List[str]:
        """文本转音素"""
        phonemes = []
        for token in tokens:
            token_phonemes = self.phonemizer.phonemize(token, language)
            phonemes.extend(token_phonemes)
        return phonemes
    
    def _generate_input_features(
        self, 
        tokens: List[str], 
        phonemes: List[str], 
        prosody: Dict[str, Any],
        request: SynthesisRequest
    ) -> Dict[str, Any]:
        """生成模型输入特征"""
        # 将音素转换为ID
        phoneme_ids = [self.phonemizer.phoneme_to_id(p) for p in phonemes]
        
        # 创建输入序列
        input_sequence = {
            "phoneme_ids": phoneme_ids,
            "speaker_id": request.speaker_id,
            "speed": request.speed,
            "pitch": request.pitch_adjustment,
            "energy": request.energy_adjustment,
            "prosody": prosody
        }
        
        return input_sequence
```

### 3.2 声学模型组件

```python
# core/acoustic_model.py
import torch
import torch.nn as nn
import torch.nn.functional as F
from typing import Dict, Any, Optional, Tuple
import logging
import math

from models.synthesis_request import SynthesisRequest
from storage.model_storage import ModelStorage
from config.tts_config import TTSConfig

logger = logging.getLogger(__name__)

class AcousticModel(nn.Module):
    """声学模型组件"""
    
    def __init__(self, config: TTSConfig):
        super(AcousticModel, self).__init__()
        self.config = config
        self.model_storage = ModelStorage(config.model_storage)
        
        # 模型架构参数
        self.vocab_size = config.acoustic_model.vocab_size
        self.embedding_dim = config.acoustic_model.embedding_dim
        self.hidden_dim = config.acoustic_model.hidden_dim
        self.num_mel_bins = config.acoustic_model.num_mel_bins
        
        # 构建模型层
        self._build_model()
        
        # 加载预训练模型
        self._load_pretrained_model()
        
        logger.info("AcousticModel initialized")
    
    def _build_model(self):
        """构建模型架构"""
        # 1. 音素嵌入层
        self.phoneme_embedding = nn.Embedding(
            self.vocab_size, self.embedding_dim
        )
        
        # 2. 说话人嵌入层
        self.speaker_embedding = nn.Embedding(
            self.config.acoustic_model.num_speakers, self.embedding_dim
        )
        
        # 3. 位置编码
        self.position_encoding = PositionalEncoding(self.embedding_dim)
        
        # 4. 编码器 (Transformer)
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=self.embedding_dim,
            nhead=self.config.acoustic_model.num_heads,
            dim_feedforward=self.hidden_dim,
            dropout=self.config.acoustic_model.dropout
        )
        self.encoder = nn.TransformerEncoder(
            encoder_layer, 
            num_layers=self.config.acoustic_model.num_encoder_layers
        )
        
        # 5. 韵律预测器
        self.duration_predictor = DurationPredictor(self.embedding_dim)
        self.pitch_predictor = PitchPredictor(self.embedding_dim)
        self.energy_predictor = EnergyPredictor(self.embedding_dim)
        
        # 6. 解码器 (基于注意力的解码器)
        self.decoder = AttentionDecoder(
            embedding_dim=self.embedding_dim,
            hidden_dim=self.hidden_dim,
            num_mel_bins=self.num_mel_bins
        )
        
        # 7. 后置网络
        self.postnet = Postnet(self.num_mel_bins)
    
    def _load_pretrained_model(self):
        """加载预训练模型"""
        try:
            model_path = self.model_storage.get_model_path("acoustic_model")
            if model_path:
                checkpoint = torch.load(model_path, map_location='cpu')
                self.load_state_dict(checkpoint['model_state_dict'])
                logger.info(f"Loaded pretrained acoustic model from {model_path}")
            else:
                logger.warning("No pretrained acoustic model found")
        except Exception as e:
            logger.error(f"Error loading pretrained model: {e}")
    
    def forward(
        self, 
        phoneme_ids: torch.Tensor, 
        speaker_id: torch.Tensor,
        input_lengths: torch.Tensor
    ) -> Dict[str, torch.Tensor]:
        """前向传播"""
        batch_size = phoneme_ids.size(0)
        
        # 1. 音素嵌入
        phoneme_embeds = self.phoneme_embedding(phoneme_ids)
        
        # 2. 说话人嵌入
        speaker_embeds = self.speaker_embedding(speaker_id).unsqueeze(1)
        speaker_embeds = speaker_embeds.expand(-1, phoneme_embeds.size(1), -1)
        
        # 3. 位置编码
        phoneme_embeds = self.position_encoding(phoneme_embeds)
        
        # 4. 合并嵌入
        embeddings = phoneme_embeds + speaker_embeds
        
        # 5. 编码器
        encoder_outputs = self.encoder(embeddings.transpose(0, 1)).transpose(0, 1)
        
        # 6. 韵律预测
        log_durations = self.duration_predictor(encoder_outputs)
        pitch = self.pitch_predictor(encoder_outputs)
        energy = self.energy_predictor(encoder_outputs)
        
        # 7. 长度调节器
        mel_outputs, mel_lengths = self._length_regulator(
            encoder_outputs, log_durations, input_lengths
        )
        
        # 8. 解码器
        mel_outputs = self.decoder(mel_outputs)
        
        # 9. 后置网络
        mel_outputs_postnet = self.postnet(mel_outputs)
        mel_outputs = mel_outputs + mel_outputs_postnet
        
        return {
            "mel_outputs": mel_outputs,
            "mel_outputs_postnet": mel_outputs_postnet,
            "log_durations": log_durations,
            "pitch": pitch,
            "energy": energy,
            "mel_lengths": mel_lengths
        }
    
    def inference(
        self, 
        input_features: Dict[str, Any]
    ) -> torch.Tensor:
        """推理模式"""
        self.eval()
        with torch.no_grad():
            # 准备输入
            phoneme_ids = torch.LongTensor(input_features["phoneme_ids"]).unsqueeze(0)
            speaker_id = torch.LongTensor([input_features["speaker_id"]])
            input_lengths = torch.LongTensor([len(input_features["phoneme_ids"])])
            
            # 前向传播
            outputs = self.forward(phoneme_ids, speaker_id, input_lengths)
            
            return outputs["mel_outputs_postnet"]
    
    def _length_regulator(
        self, 
        encoder_outputs: torch.Tensor, 
        log_durations: torch.Tensor,
        input_lengths: torch.Tensor
    ) -> Tuple[torch.Tensor, torch.Tensor]:
        """长度调节器"""
        # 将对数持续时间转换为实际持续时间
        durations = torch.exp(log_durations) - 1
        durations = torch.round(durations).long()
        
        # 创建输出张量
        max_len = torch.max(durations.sum(dim=1))
        batch_size = encoder_outputs.size(0)
        hidden_dim = encoder_outputs.size(2)
        
        mel_outputs = torch.zeros(
            batch_size, max_len, hidden_dim, 
            device=encoder_outputs.device
        )
        mel_lengths = torch.zeros(batch_size, dtype=torch.long)
        
        for b in range(batch_size):
            # 获取当前样本的编码器输出和持续时间
            enc_out = encoder_outputs[b, :input_lengths[b]]
            dur = durations[b, :input_lengths[b]]
            
            # 根据持续时间扩展特征
            expanded = []
            for i, d in enumerate(dur):
                if d > 0:
                    expanded.append(enc_out[i].unsqueeze(0).expand(d, -1))
            
            if expanded:
                expanded = torch.cat(expanded, dim=0)
                mel_outputs[b, :expanded.size(0)] = expanded
                mel_lengths[b] = expanded.size(0)
        
        return mel_outputs, mel_lengths


class PositionalEncoding(nn.Module):
    """位置编码"""
    
    def __init__(self, d_model: int, max_len: int = 5000):
        super(PositionalEncoding, self).__init__()
        
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(
            torch.arange(0, d_model, 2).float() * 
            (-math.log(10000.0) / d_model)
        )
        
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0).transpose(0, 1)
        
        self.register_buffer('pe', pe)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return x + self.pe[:x.size(0), :]


class DurationPredictor(nn.Module):
    """持续时间预测器"""
    
    def __init__(self, hidden_dim: int):
        super(DurationPredictor, self).__init__()
        self.conv1 = nn.Conv1d(hidden_dim, hidden_dim, kernel_size=3, padding=1)
        self.conv2 = nn.Conv1d(hidden_dim, hidden_dim, kernel_size=3, padding=1)
        self.linear = nn.Linear(hidden_dim, 1)
        self.dropout = nn.Dropout(0.1)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x.transpose(1, 2)
        x = F.relu(self.conv1(x))
        x = self.dropout(x)
        x = F.relu(self.conv2(x))
        x = self.dropout(x)
        x = x.transpose(1, 2)
        x = self.linear(x)
        return x.squeeze(-1)


class PitchPredictor(nn.Module):

class AttentionDecoder(nn.Module):
    """基于注意力的解码器"""
    
    def __init__(self, embedding_dim: int, hidden_dim: int, num_mel_bins: int):
        super(AttentionDecoder, self).__init__()
        self.attention = Attention(hidden_dim)
        self.prenet = Prenet(num_mel_bins, hidden_dim)
        self.decoder_rnn = nn.LSTM(
            hidden_dim * 2, hidden_dim, batch_first=True
        )
        self.proj = nn.Linear(hidden_dim, num_mel_bins)
        self.stop_token = nn.Linear(hidden_dim, 1)
    
    def forward(self, encoder_outputs: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = encoder_outputs.size()
        
        # 初始化解码器状态
        decoder_input = torch.zeros(
            batch_size, 1, encoder_outputs.size(2),
            device=encoder_outputs.device
        )
        hidden_state = None
        
        outputs = []
        
        for i in range(seq_len):
            # 预网络
            prenet_out = self.prenet(decoder_input)
            
            # 注意力
            context, _ = self.attention(hidden_state[0], encoder_outputs)
            
            # 合并预网络输出和上下文
            rnn_input = torch.cat([prenet_out, context], dim=-1)
            
            # 解码器RNN
            output, hidden_state = self.decoder_rnn(rnn_input, hidden_state)
            
            # 投影到梅尔频谱
            mel_output = self.proj(output)
            outputs.append(mel_output)
            
            # 下一步输入
            decoder_input = mel_output
        
        return torch.cat(outputs, dim=1)


class Attention(nn.Module):
    """注意力机制"""
    
    def __init__(self, hidden_dim: int):
        super(Attention, self).__init__()
        self.query_layer = nn.Linear(hidden_dim, hidden_dim)
        self.key_layer = nn.Linear(hidden_dim, hidden_dim)
        self.value_layer = nn.Linear(hidden_dim, hidden_dim)
        self.v = nn.Parameter(torch.rand(hidden_dim))
    
    def forward(
        self, 
        query: torch.Tensor, 
        encoder_outputs: torch.Tensor
    ) -> Tuple[torch.Tensor, torch.Tensor]:
        # 计算注意力分数
        q = self.query_layer(query)
        k = self.key_layer(encoder_outputs)
        v = self.value_layer(encoder_outputs)
        
        # 注意力权重
        scores = torch.matmul(q, k.transpose(1, 2)) / math.sqrt(q.size(-1))
        attn_weights = F.softmax(scores, dim=-1)
        
        # 上下文向量
        context = torch.matmul(attn_weights, v)
        
        return context, attn_weights


class Prenet(nn.Module):
    """预网络"""
    
    def __init__(self, num_mel_bins: int, hidden_dim: int):
        super(Prenet, self).__init__()
        self.prenet = nn.Sequential(
            nn.Linear(num_mel_bins, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.5)
        )
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.prenet(x)


class Postnet(nn.Module):
    """后置网络"""
    
    def __init__(self, num_mel_bins: int):
        super(Postnet, self).__init__()
        self.convolutions = nn.ModuleList([
            nn.Conv1d(num_mel_bins, 512, kernel_size=5, padding=2),
            nn.BatchNorm1d(512),
            nn.Tanh(),
            nn.Dropout(0.5),
            
            nn.Conv1d(512, 512, kernel_size=5, padding=2),
            nn.BatchNorm1d(512),
            nn.Tanh(),
            nn.Dropout(0.5),
            
            nn.Conv1d(512, 512, kernel_size=5, padding=2),
            nn.BatchNorm1d(512),
            nn.Tanh(),
            nn.Dropout(0.5),
            
            nn.Conv1d(512, 512, kernel_size=5, padding=2),
            nn.BatchNorm1d(512),
            nn.Tanh(),
            nn.Dropout(0.5),
            
            nn.Conv1d(512, num_mel_bins, kernel_size=5, padding=2),
            nn.BatchNorm1d(num_mel_bins)
        ])
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x.transpose(1, 2)
        
        for layer in self.convolutions:
            x = layer(x)
        
        return x.transpose(1, 2)
```

```python
        self._load_pretrained_model()
        
        logger.info("Vocoder initialized")
    
    def _build_model(self):
        """构建模型架构"""
        # 1. 初始卷积层
        self.initial_conv = nn.Conv1d(
            self.num_mel_bins, self.hidden_dim, 
            kernel_size=7, padding=3
        )
        
        # 2. 残差块
        self.res_blocks = nn.ModuleList([
            ResidualBlock(self.hidden_dim) for _ in range(self.num_res_blocks)
        ])
        
        # 3. 上采样层
        self.upsample_layers = nn.ModuleList([
            nn.ConvTranspose1d(
                self.hidden_dim, self.hidden_dim,
                kernel_size=16, stride=8, padding=4
            ),
            nn.ConvTranspose1d(
                self.hidden_dim, self.hidden_dim,
                kernel_size=16, stride=8, padding=4
            )
        ])
        
        # 4. 最终卷积层
        self.final_conv = nn.Conv1d(
            self.hidden_dim, 1, kernel_size=7, padding=3
        )
        
        # 5. 激活函数
        self.leaky_relu = nn.LeakyReLU(0.2)
        self.tanh = nn.Tanh()
    
    def _load_pretrained_model(self):
        """加载预训练模型"""
        try:
            model_path = self.model_storage.get_model_path("vocoder")
            if model_path:
                checkpoint = torch.load(model_path, map_location='cpu')
                self.load_state_dict(checkpoint['model_state_dict'])
                logger.info(f"Loaded pretrained vocoder from {model_path}")
            else:
                logger.warning("No pretrained vocoder found")
        except Exception as e:
            logger.error(f"Error loading pretrained model: {e}")
    
    def forward(self, mel_spectrogram: torch.Tensor) -> torch.Tensor:
        """前向传播"""
        # 1. 初始卷积
        x = self.initial_conv(mel_spectrogram)
        x = self.leaky_relu(x)
        
        # 2. 残差块
        for res_block in self.res_blocks:
            x = res_block(x)
        
        # 3. 上采样
        for upsample in self.upsample_layers:
            x = upsample(x)
            x = self.leaky_relu(x)
        
        # 4. 最终卷积
        x = self.final_conv(x)
        x = self.tanh(x)
        
        return x.squeeze(1)
    
    def inference(self, mel_spectrogram: torch.Tensor) -> np.ndarray:
        """推理模式"""
        self.eval()
        with torch.no_grad():
            # 前向传播
            waveform = self.forward(mel_spectrogram)
            
            # 转换为numpy数组
            waveform = waveform.cpu().numpy()
            
            # 后处理
            waveform = self.audio_processor.post_process(waveform)
            
            return waveform


class ResidualBlock(nn.Module):
    """残差块"""
    
    def __init__(self, hidden_dim: int):
        super(ResidualBlock, self).__init__()
        self.conv1 = nn.Conv1d(hidden_dim, hidden_dim, kernel_size=3, padding=1)
        self.conv2 = nn.Conv1d(hidden_dim, hidden_dim, kernel_size=3, padding=1)
        self.conv3 = nn.Conv1d(hidden_dim, hidden_dim, kernel_size=1)
        self.leaky_relu = nn.LeakyReLU(0.2)
        self.dropout = nn.Dropout(0.1)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        residual = x
        
        # 第一个卷积
        out = self.conv1(x)
        out = self.leaky_relu(out)
        out = self.dropout(out)
        
        # 第二个卷积
        out = self.conv2(out)
        out = self.leaky_relu(out)
        out = self.dropout(out)
        
        # 第三个卷积
        out = self.conv3(out)
        
        # 残差连接
        out = out + residual
        
        return out
```

```python
@router.post("/synthesize", response_model=SynthesisResponse)
async def synthesize(
    request: SynthesisRequest,
    background_tasks: BackgroundTasks,
    text_processor: TextProcessor = Depends(get_text_processor),
    acoustic_model: AcousticModel = Depends(get_acoustic_model),
    vocoder: Vocoder = Depends(get_vocoder),
    voice_converter: VoiceConverter = Depends(get_voice_converter),
    emotion_synthesizer: EmotionSynthesizer = Depends(get_emotion_synthesizer),
    audio_processor: AudioProcessor = Depends(get_audio_processor)
):
    """语音合成API"""
    import time
    start_time = time.time()
    
    try:
        # 1. 文本预处理
        processed_text = text_processor.process(request)
        
        # 2. 声学模型推理
        mel_spectrogram = acoustic_model.inference(processed_text["input_features"])
        
        # 3. 情感合成（如果需要）
        if request.emotion.value != "neutral":
            from models.emotion_profile import EmotionProfile
            emotion_profile = EmotionProfile(
                emotion_id=_get_emotion_id(request.emotion.value),
                name=request.emotion.value,
                intensity=1.0,
                pitch_scale=1.0,
                energy_scale=1.0,
                speed_scale=1.0
            )
            mel_spectrogram = emotion_synthesizer.synthesize_emotion(
                mel_spectrogram, emotion_profile
            )
        
        # 4. 声码器生成波形
        waveform = vocoder.inference(mel_spectrogram)
        
        # 5. 音频后处理
        audio_data = audio_processor.post_process(waveform)
        
        # 6. 编码音频数据
        audio_bytes = audio_processor.encode(audio_data, request.output_format, request.sample_rate)
        audio_base64 = base64.b64encode(audio_bytes).decode('utf-8')
        
        # 计算处理时间
        processing_time = time.time() - start_time
        
        # 计算音频时长
        duration = len(audio_data) / request.sample_rate
        
        return SynthesisResponse(
            success=True,
            audio_data=audio_base64,
            duration=duration,
            sample_rate=request.sample_rate,
            processing_time=processing_time
        )
        
    except Exception as e:
        logger.error(f"Error in synthesis: {e}")
        processing_time = time.time() - start_time
        
        return SynthesisResponse(
            success=False,
            error_message=str(e),
            sample_rate=request.sample_rate,
            processing_time=processing_time
        )
```
