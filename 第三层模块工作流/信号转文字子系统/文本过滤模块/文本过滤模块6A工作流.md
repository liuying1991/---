# 文本过滤模块6A工作流

## 模块概述

### 模块名称
文本过滤模块

### 所属子系统
信号转文字子系统

### 模块功能描述
文本过滤模块负责对生成和编辑后的文本内容进行安全性和适宜性过滤，确保所有输出给婴儿的文本内容符合儿童安全和教育标准。该模块能够识别和过滤不适宜的内容，包括暴力、恐怖、不当语言、成人内容等，同时还能根据婴儿的年龄和认知水平，调整过滤标准和严格程度。

### 模块职责
1. 安全内容过滤：识别和过滤不适宜的安全内容
2. 教育适宜性过滤：确保内容符合教育标准和价值观
3. 年龄适宜性过滤：根据婴儿年龄调整过滤标准
4. 情感适宜性过滤：过滤可能引起负面情绪的内容
5. 文化适宜性过滤：确保内容符合文化背景和价值观
6. 过滤结果评估：评估过滤后内容的质量和适宜性

### 模块接口
- **输入接口**：待过滤文本、过滤参数、婴儿状态
- **输出接口**：过滤后文本、过滤标签、安全评分
- **依赖模块**：文本生成模块、文本编辑模块
- **被依赖模块**：语音合成模块、交互表达子系统

class TextFilter:
    def __init__(self, safety_filter, education_filter, age_filter, 
                 emotion_filter, culture_filter, filter_optimizer, safety_assessor):
        self.safety_filter = safety_filter
        self.education_filter = education_filter
        self.age_filter = age_filter
        self.emotion_filter = emotion_filter
        self.culture_filter = culture_filter
        self.filter_optimizer = filter_optimizer
        self.safety_assessor = safety_assessor
        self.text_analyzer = TextAnalyzer()
        
    def filter(self, text, filter_params, baby_state):
        # 输入处理
        processed_input = self._process_input(text, filter_params, baby_state)
        
        # 文本分析
        text_analysis = self.text_analyzer.analyze(processed_input["text"])
        
        # 过滤操作
        filtered_text = processed_input["text"]
        filter_history = []
        
        # 安全过滤
        if filter_params.get("filter_safety", True):
            safety_result = self.safety_filter.filter(filtered_text)
            if safety_result["is_filtered"]:
                filtered_text = safety_result["filtered_text"]
                filter_history.append({
                    "filter_type": "safety",
                    "original": processed_input["text"],
                    "filtered": filtered_text,
                    "detected_issues": safety_result["detected_issues"]
                })
        
        # 教育过滤
        if filter_params.get("filter_education", True):
            education_result = self.education_filter.filter(filtered_text)
            if education_result["is_filtered"]:
                filtered_text = education_result["filtered_text"]
                filter_history.append({
                    "filter_type": "education",
                    "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                    "filtered": filtered_text,
                    "detected_issues": education_result["detected_issues"]
                })
        
        # 年龄过滤
        if "age" in baby_state:
            age_result = self.age_filter.filter(filtered_text, baby_state["age"])
            if age_result["is_filtered"]:
                filtered_text = age_result["filtered_text"]
                filter_history.append({
                    "filter_type": "age",
                    "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                    "filtered": filtered_text,
                    "detected_issues": age_result["detected_issues"]
                })
        
        # 情感过滤
        if filter_params.get("filter_emotion", True):
            emotion_result = self.emotion_filter.filter(filtered_text)
            if emotion_result["is_filtered"]:
                filtered_text = emotion_result["filtered_text"]
                filter_history.append({
                    "filter_type": "emotion",
                    "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                    "filtered": filtered_text,
                    "detected_issues": emotion_result["detected_issues"]
                })
        
        # 文化过滤
        if filter_params.get("filter_culture", True) and "culture" in baby_state:
            culture_result = self.culture_filter.filter(filtered_text, baby_state["culture"])
            if culture_result["is_filtered"]:
                filtered_text = culture_result["filtered_text"]
                filter_history.append({
                    "filter_type": "culture",
                    "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                    "filtered": filtered_text,
                    "detected_issues": culture_result["detected_issues"]
                })
        
        # 过滤优化
        if filter_params.get("optimize_filter", True):
            optimization_result = self.filter_optimizer.optimize(filtered_text, filter_history)
            filtered_text = optimization_result["optimized_text"]
            filter_history.append({
                "filter_type": "optimization",
                "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                "filtered": filtered_text,
                "optimizations": optimization_result["optimizations"]
            })
        
        # 安全评估
        safety_score = self.safety_assessor.assess(filtered_text, baby_state)
        
        # 如果安全评分不达标，尝试重新过滤
        if safety_score < 0.8:
            # 尝试调整过滤参数并重新过滤
            adjusted_params = self._adjust_filter_params(filter_params, safety_score)
            return self.filter(text, adjusted_params, baby_state)
        
        return {
            "original_text": text,
            "filtered_text": filtered_text,
            "filter_history": filter_history,
            "safety_score": safety_score,
            "timestamp": time.time()
        }

# 安全过滤器实现
class SafetyFilter:
    def __init__(self):
        self.safety_classifier = self._load_safety_classifier()
        self.safety_rules = self._load_safety_rules()
        self.inappropriate_keywords = self._load_inappropriate_keywords()
        
    def filter(self, text):
        # 检测不适宜内容
        detected_issues = []
        filtered_text = text
        
        # 基于关键词的过滤
        keyword_issues = self._check_keywords(text)
        detected_issues.extend(keyword_issues)
        
        # 基于规则的过滤
        rule_issues = self._check_rules(text)
        detected_issues.extend(rule_issues)
        
        # 基于模型的过滤
        model_result = self.safety_classifier.predict(text)
        if model_result["is_inappropriate"]:
            detected_issues.append({
                "type": "model_detection",
                "category": model_result["category"],
                "confidence": model_result["confidence"],
                "description": model_result["description"]
            })
        
        # 如果检测到问题，进行过滤
        if detected_issues:
            filtered_text = self._filter_content(text, detected_issues)
        
        return {
            "original_text": text,
            "filtered_text": filtered_text,
            "is_filtered": text != filtered_text,
            "detected_issues": detected_issues
        }
    
    def _check_keywords(self, text):
        issues = []
        words = nltk.word_tokenize(text.lower())
        
        for word in words:
            if word in self.inappropriate_keywords:
                issues.append({
                    "type": "keyword",
                    "word": word,
                    "category": self.inappropriate_keywords[word]["category"],
                    "severity": self.inappropriate_keywords[word]["severity"]
                })
        
        return issues
    
    def _check_rules(self, text):
        issues = []
        
        for rule in self.safety_rules:
            if rule["type"] == "pattern":
                if re.search(rule["pattern"], text, re.IGNORECASE):
                    issues.append({
                        "type": "rule_pattern",
                        "rule_id": rule["id"],
                        "category": rule["category"],
                        "severity": rule["severity"],
                        "description": rule["description"]
                    })
            elif rule["type"] == "semantic":
                # 使用语义相似度检查
                if self._semantic_similarity(text, rule["target"]) > rule["threshold"]:
                    issues.append({
                        "type": "rule_semantic",
                        "rule_id": rule["id"],
                        "category": rule["category"],
                        "severity": rule["severity"],
                        "description": rule["description"],
                        "similarity": self._semantic_similarity(text, rule["target"])
                    })
        
        return issues
    
    def _filter_content(self, text, issues):
        filtered_text = text
        
        # 根据问题严重程度决定过滤策略
        high_severity_issues = [issue for issue in issues if issue.get("severity") == "high"]
        medium_severity_issues = [issue for issue in issues if issue.get("severity") == "medium"]
        low_severity_issues = [issue for issue in issues if issue.get("severity") == "low"]
        
        # 高严重性问题：替换整个内容
        if high_severity_issues:
            filtered_text = self._replace_with_safe_content(text)
        
        # 中等严重性问题：替换问题部分
        elif medium_severity_issues:
            for issue in medium_severity_issues:
                if issue["type"] == "keyword":
                    filtered_text = filtered_text.replace(issue["word"], self._get_safe_replacement(issue["word"]))
                elif issue["type"] in ["rule_pattern", "rule_semantic"]:
                    filtered_text = self._replace_problematic_phrase(filtered_text, issue)
        
        # 低严重性问题：修改问题部分
        elif low_severity_issues:
            for issue in low_severity_issues:
                if issue["type"] == "keyword":
                    filtered_text = filtered_text.replace(issue["word"], self._get_mild_replacement(issue["word"]))
                elif issue["type"] in ["rule_pattern", "rule_semantic"]:
                    filtered_text = self._modify_problematic_phrase(filtered_text, issue)
        
        return filtered_text
