# 交互效果评估模块6A工作流

## 模块概述

### 模块名称
交互效果评估模块

### 所属子系统
交互表达子系统

### 模块功能描述
交互效果评估模块负责评估交互表达子系统的交互效果，包括婴儿的参与度、满意度、学习效果和情感反应。该模块通过分析婴儿的反馈数据、行为数据和环境数据，评估交互的有效性和质量，为交互策略和内容生成提供优化依据，确保交互的持续改进和适应性。

### 模块职责
1. 效果评估：评估交互效果和质量
2. 反馈分析：分析婴儿反馈和行为数据
3. 效果预测：预测交互效果和趋势
4. 效果优化：提供交互效果优化建议
5. 效果报告：生成交互效果报告
6. 效果学习：学习效果评估模式和标准

### 模块接口
- **输入接口**：婴儿反馈数据、行为数据、环境数据、交互数据
- **输出接口**：效果评估结果、效果预测结果、优化建议、效果报告
- **依赖模块**：感知处理子系统、交互管理模块、反馈处理模块
- **被依赖模块**：交互策略模块、交互内容生成模块

class InteractionEffectManager:
    def __init__(self, feedback_analyzer, behavior_analyzer, environment_analyzer,
                 interaction_analyzer, effect_evaluator, engagement_evaluator,
                 satisfaction_evaluator, learning_evaluator, emotion_evaluator,
                 effect_predictor, trend_predictor, outcome_predictor,
                 effect_optimizer, strategy_optimizer, content_optimizer,
                 effect_reporter, visualization_generator, summary_generator,
                 effect_manager, data_manager, model_manager, effect_learner,
                 pattern_learner, standard_learner):
        self.feedback_analyzer = feedback_analyzer
        self.behavior_analyzer = behavior_analyzer
        self.environment_analyzer = environment_analyzer
        self.interaction_analyzer = interaction_analyzer
        self.effect_evaluator = effect_evaluator
        self.engagement_evaluator = engagement_evaluator
        self.satisfaction_evaluator = satisfaction_evaluator
        self.learning_evaluator = learning_evaluator
        self.emotion_evaluator = emotion_evaluator
        self.effect_predictor = effect_predictor
        self.trend_predictor = trend_predictor
        self.outcome_predictor = outcome_predictor
        self.effect_optimizer = effect_optimizer
        self.strategy_optimizer = strategy_optimizer
        self.content_optimizer = content_optimizer
        self.effect_reporter = effect_reporter
        self.visualization_generator = visualization_generator
        self.summary_generator = summary_generator
        self.effect_manager = effect_manager
        self.data_manager = data_manager
        self.model_manager = model_manager
        self.effect_learner = effect_learner
        self.pattern_learner = pattern_learner
        self.standard_learner = standard_learner
        
        # 初始化模型
        self._load_models()
        
    async def evaluate_effect(self, baby_id, session_id, interaction_data):
        try:
            # 分析反馈数据
            feedback_analysis = await self.feedback_analyzer.analyze_feedback(
                baby_id, session_id, interaction_data
            )
            
            # 分析行为数据
            behavior_analysis = await self.behavior_analyzer.analyze_behavior(
                baby_id, session_id, interaction_data
            )
            
            # 分析环境数据
            environment_analysis = await self.environment_analyzer.analyze_environment(
                baby_id, session_id, interaction_data
            )
            
            # 分析交互数据
            interaction_analysis = await self.interaction_analyzer.analyze_interaction(
                baby_id, session_id, interaction_data
            )
            
            # 评估参与度
            engagement_evaluation = await self.engagement_evaluator.evaluate_engagement(
                feedback_analysis, behavior_analysis, interaction_analysis
            )
            
            # 评估满意度
            satisfaction_evaluation = await self.satisfaction_evaluator.evaluate_satisfaction(
                feedback_analysis, behavior_analysis, interaction_analysis
            )
            
            # 评估学习效果
            learning_evaluation = await self.learning_evaluator.evaluate_learning(
                feedback_analysis, behavior_analysis, interaction_analysis
            )
            
            # 评估情感反应
            emotion_evaluation = await self.emotion_evaluator.evaluate_emotion(
                feedback_analysis, behavior_analysis, interaction_analysis
            )
            
            # 综合评估效果
            overall_evaluation = await self.effect_evaluator.evaluate_overall_effect(
                engagement_evaluation, satisfaction_evaluation,
                learning_evaluation, emotion_evaluation
            )
            
            # 构建评估结果
            evaluation_result = {
                "baby_id": baby_id,
                "session_id": session_id,
                "feedback_analysis": feedback_analysis,
                "behavior_analysis": behavior_analysis,
                "environment_analysis": environment_analysis,
                "interaction_analysis": interaction_analysis,
                "engagement_evaluation": engagement_evaluation,
                "satisfaction_evaluation": satisfaction_evaluation,
                "learning_evaluation": learning_evaluation,
                "emotion_evaluation": emotion_evaluation,
                "overall_evaluation": overall_evaluation,
                "timestamp": time.time()
            }
            
            # 保存评估结果
            await self.effect_manager.save_evaluation_result(evaluation_result)
            
            return evaluation_result
            
        except Exception as e:
            # 处理异常
            error_result = {
                "baby_id": baby_id,
                "session_id": session_id,
                "error": str(e),
                "timestamp": time.time()
            }
            
            return error_result
    
    async def predict_effect(self, baby_id, session_id, interaction_data):
        try:
            # 获取历史评估数据
            historical_evaluations = await self.effect_manager.get_historical_evaluations(
                baby_id, session_id
            )
            
            # 分析当前交互数据
            current_analysis = await self._analyze_current_interaction(
                baby_id, session_id, interaction_data
            )
            
            # 预测短期效果
            short_term_prediction = await self.effect_predictor.predict_short_term_effect(
                historical_evaluations, current_analysis
            )
            
            # 预测长期效果
            long_term_prediction = await self.effect_predictor.predict_long_term_effect(
                historical_evaluations, current_analysis
            )
            
            # 预测效果趋势
            trend_prediction = await self.trend_predictor.predict_effect_trend(

            )
            
            # 预测交互结果
            outcome_prediction = await self.outcome_predictor.predict_interaction_outcome(
                historical_evaluations, current_analysis
            )
            
            # 构建预测结果
            prediction_result = {

                "trend_prediction": trend_prediction,
                "outcome_prediction": outcome_prediction,
                "timestamp": time.time()
            }

            await self.effect_manager.save_prediction_result(prediction_result)
            
            return prediction_result
            
        except Exception as e:
            # 处理异常
            error_result = {
                "baby_id": baby_id,
                "session_id": session_id,
                "error": str(e),
                "timestamp": time.time()
            }
            
            return error_result
    
    async def optimize_effect(self, baby_id, session_id, evaluation_result):

            content_optimization = await self.content_optimizer.optimize_content(

            interaction_optimization = await self.effect_optimizer.optimize_interaction(

                strategy_optimization, content_optimization,
                interaction_optimization, environment_optimization

            # 构建优化结果

                "strategy_optimization": strategy_optimization,

                "environment_optimization": environment_optimization,

            # 保存优化结果

class EffectEvaluator:
    def __init__(self, evaluation_model, evaluation_standards):
        self.evaluation_model = evaluation_model
        self.evaluation_standards = evaluation_standards
        
    async def evaluate_overall_effect(self, engagement_evaluation, satisfaction_evaluation,
                                     learning_evaluation, emotion_evaluation):
        try:
            # 提取评估特征
            evaluation_features = await self._extract_evaluation_features(
                engagement_evaluation, satisfaction_evaluation,
                learning_evaluation, emotion_evaluation
            )
            
            # 模型预测整体效果
            model_prediction = await self._predict_overall_effect(evaluation_features)
            
            # 应用评估标准
            standard_results = await self._apply_evaluation_standards(
                engagement_evaluation, satisfaction_evaluation,
                learning_evaluation, emotion_evaluation
            )
            
            # 合并评估结果
            overall_effect = await self._merge_evaluation_results(
                model_prediction, standard_results
            )
            
            return {
                "engagement_evaluation": engagement_evaluation,
                "satisfaction_evaluation": satisfaction_evaluation,
                "learning_evaluation": learning_evaluation,
                "emotion_evaluation": emotion_evaluation,
                "evaluation_features": evaluation_features,
                "model_prediction": model_prediction,
                "standard_results": standard_results,
                "overall_effect": overall_effect,
                "effect_score": overall_effect.get("score", 0.0),
                "effect_level": overall_effect.get("level", "medium"),
                "effect_description": overall_effect.get("description", "Average effect")
            }
            
        except Exception as e:
            return {
                "error": str(e),
                "effect_score": 0.0,
                "effect_level": "low",
                "effect_description": "Evaluation failed"
            }
    
    async def _extract_evaluation_features(self, engagement_evaluation, satisfaction_evaluation,
                                          learning_evaluation, emotion_evaluation):
        # 提取参与度特征
        engagement_features = await self._extract_engagement_features(engagement_evaluation)
        
        # 提取满意度特征
        satisfaction_features = await self._extract_satisfaction_features(satisfaction_evaluation)
        
        # 提取学习效果特征
        learning_features = await self._extract_learning_features(learning_evaluation)
        
        # 提取情感反应特征
        emotion_features = await self._extract_emotion_features(emotion_evaluation)
        
        # 合并特征
        evaluation_features = {
            **engagement_features,
            **satisfaction_features,
            **learning_features,
            **emotion_features
        }
        
        return evaluation_features
    
    async def _extract_engagement_features(self, engagement_evaluation):
        # 提取参与度分数
        engagement_score = engagement_evaluation.get("score", 0.5)
        
        # 提取参与度持续时间
        engagement_duration = engagement_evaluation.get("duration", 0)
        
        # 提取参与度强度
        engagement_intensity = engagement_evaluation.get("intensity", 0.5)
        
        # 提取参与度稳定性
        engagement_stability = engagement_evaluation.get("stability", 0.5)
        
        return {
            "engagement_score": engagement_score,
            "engagement_duration": engagement_duration,
            "engagement_intensity": engagement_intensity,
            "engagement_stability": engagement_stability
        }
    
    async def _extract_satisfaction_features(self, satisfaction_evaluation):
        # 提取满意度分数
        satisfaction_score = satisfaction_evaluation.get("score", 0.5)
        
        # 提取满意度持续性
        satisfaction_continuity = satisfaction_evaluation.get("continuity", 0.5)
        
        # 提取满意度强度
        satisfaction_intensity = satisfaction_evaluation.get("intensity", 0.5)
        
        # 提取满意度稳定性
        satisfaction_stability = satisfaction_evaluation.get("stability", 0.5)
        
        return {
            "satisfaction_score": satisfaction_score,
            "satisfaction_continuity": satisfaction_continuity,
            "satisfaction_intensity": satisfaction_intensity,
            "satisfaction_stability": satisfaction_stability
        }
    
    async def _extract_learning_features(self, learning_evaluation):
        # 提取学习效果分数
        learning_score = learning_evaluation.get("score", 0.5)
        
        # 提取学习速度
        learning_speed = learning_evaluation.get("speed", 0.5)
        
        # 提取学习深度
        learning_depth = learning_evaluation.get("depth", 0.5)
        
        # 提取学习持久性
        learning_persistence = learning_evaluation.get("persistence", 0.5)
        
        return {
            "learning_score": learning_score,
            "learning_speed": learning_speed,
            "learning_depth": learning_depth,
            "learning_persistence": learning_persistence
        }
    
    async def _extract_emotion_features(self, emotion_evaluation):
        # 提取情感反应分数
        emotion_score = emotion_evaluation.get("score", 0.5)
        
        # 提取情感多样性
        emotion_diversity = emotion_evaluation.get("diversity", 0.5)
        
        # 提取情感强度
        emotion_intensity = emotion_evaluation.get("intensity", 0.5)
        
        # 提取情感稳定性
        emotion_stability = emotion_evaluation.get("stability", 0.5)
        
        return {
            "emotion_score": emotion_score,
            "emotion_diversity": emotion_diversity,
            "emotion_intensity": emotion_intensity,
            "emotion_stability": emotion_stability
        }
    
    async def _predict_overall_effect(self, evaluation_features):
        # 转换特征为模型输入格式
        feature_vector = self._convert_features_to_vector(evaluation_features)
        
        # 模型预测
        effect_score = self.evaluation_model.predict_proba([feature_vector])[0][1]
        
        # 确定效果等级
        effect_level = self._determine_effect_level(effect_score)
        
        # 生成效果描述
        effect_description = self._generate_effect_description(effect_score, effect_level)
        
        return {
            "score": effect_score,
            "level": effect_level,
            "description": effect_description
        }
    
    async def _apply_evaluation_standards(self, engagement_evaluation, satisfaction_evaluation,
                                         learning_evaluation, emotion_evaluation):
        # 获取评估标准
        standards = self.evaluation_standards.get("overall", {})
        
        # 计算加权分数
        engagement_weight = standards.get("engagement_weight", 0.25)
        satisfaction_weight = standards.get("satisfaction_weight", 0.25)
        learning_weight = standards.get("learning_weight", 0.25)
        emotion_weight = standards.get("emotion_weight", 0.25)
        
        engagement_score = engagement_evaluation.get("score", 0.5)
        satisfaction_score = satisfaction_evaluation.get("score", 0.5)
        learning_score = learning_evaluation.get("score", 0.5)
        emotion_score = emotion_evaluation.get("score", 0.5)
        
        weighted_score = (
            engagement_score * engagement_weight +
            satisfaction_score * satisfaction_weight +
            learning_score * learning_weight +
            emotion_score * emotion_weight
        )
        
        # 确定效果等级
        effect_level = self._determine_effect_level(weighted_score)
        
        # 生成效果描述
        effect_description = self._generate_effect_description(weighted_score, effect_level)
        
        return {
            "score": weighted_score,
            "level": effect_level,
            "description": effect_description,
            "weights": {
                "engagement_weight": engagement_weight,
                "satisfaction_weight": satisfaction_weight,
                "learning_weight": learning_weight,
                "emotion_weight": emotion_weight
            }
        }
    
    async def _merge_evaluation_results(self, model_prediction, standard_results):
        # 获取模型预测分数
        model_score = model_prediction.get("score", 0.0)
        
        # 获取标准评估分数
        standard_score = standard_results.get("score", 0.0)
        
        # 计算加权平均分数
        model_weight = 0.6  # 模型权重
        standard_weight = 0.4  # 标准权重
        
        combined_score = model_score * model_weight + standard_score * standard_weight
        
        # 确定效果等级
        effect_level = self._determine_effect_level(combined_score)
        
        # 生成效果描述
        effect_description = self._generate_effect_description(combined_score, effect_level)
        
        return {
            "score": combined_score,
            "level": effect_level,
            "description": effect_description,
            "model_score": model_score,
            "standard_score": standard_score,
            "weights": {
                "model_weight": model_weight,
                "standard_weight": standard_weight
            }
        }
    
    def _convert_features_to_vector(self, features):
        # 将特征字典转换为向量
        # 这里简化处理，实际应该按照模型训练时的特征顺序进行转换
        feature_names = sorted(features.keys())
        feature_vector = [features[name] for name in feature_names]
        
        return feature_vector
    
    def _determine_effect_level(self, score):
        # 根据分数确定效果等级
        if score >= 0.8:
            return "high"
        elif score >= 0.6:
            return "medium"
        else:
            return "low"
    
    def _generate_effect_description(self, score, level):
        # 根据分数和等级生成效果描述
        if level == "high":
            return "Excellent interaction effect with high engagement, satisfaction, learning and positive emotional response"
        elif level == "medium":
            return "Average interaction effect with moderate engagement, satisfaction, learning and emotional response"
        else:
            return "Poor interaction effect with low engagement, satisfaction, learning and negative emotional response"

        if len(effect_scores) >= 2:
            recent_scores = effect_scores[-3:]  # 最近3次评估
            older_scores = effect_scores[-6:-3] if len(effect_scores) >= 6 else effect_scores[:-3]
            avg_recent = sum(recent_scores) / len(recent_scores)

            trend = avg_recent - avg_older
        else:
            trend = 0.0
