```python
import asyncio
import logging
from typing import Dict, List, Optional, Any, Tuple
from dataclasses import dataclass
from enum import Enum

from multimodal_fusion.models.config_models import FusionConfig
from multimodal_fusion.models.modality_models import ModalityInput, ModalityFeatures, ModalityAnalysis

from multimodal_fusion.services.evaluation_service import EvaluationService

logger = logging.getLogger(__name__)

class FusionStrategyType(Enum):
    """融合策略类型"""
    EARLY_FUSION = "early_fusion"
    LATE_FUSION = "late_fusion"
    HYBRID_FUSION = "hybrid_fusion"
    ATTENTION_FUSION = "attention_fusion"

@dataclass
class ModalityInput:
    """模态输入数据"""
    modality_type: str  # "audio", "video", "motion"
    data: Any
    timestamp: float
    metadata: Dict[str, Any] = None

@dataclass
class ModalityFeatures:
    """模态特征"""
    modality_type: str
    features: Any
    confidence: float
    extraction_time: float

@dataclass
class FusionResult:
    """融合结果"""
    fused_expression: Dict[str, Any]
    fusion_strategy: FusionStrategyType
    confidence: float
    processing_time: float
    alignment_quality: float
    conflict_resolution: List[Dict[str, Any]]

class MultimodalFusionManager:
    """多模态融合管理器"""
    
    def __init__(self, config: FusionConfig):
        """初始化多模态融合管理器"""
        self.config = config
        self._initialize_components()
    
    async def fuse_modalities(
        self, 
        baby_id: str, 
        session_id: str, 
        modality_inputs: List[ModalityInput],
        context: Optional[Dict[str, Any]] = None
    ) -> FusionResult:
        """
        融合多种模态信息
        
        Args:
            baby_id: 婴儿ID
            session_id: 会话ID
            modality_inputs: 模态输入列表
            context: 上下文信息
            
        Returns:
            FusionResult: 融合结果
            
        Raises:
            FusionError: 融合过程中的错误
        """
        try:
            # 验证输入
            self._validate_inputs(modality_inputs)
            
            # 预处理输入
            preprocessed_inputs = await self.input_service.preprocess_inputs(modality_inputs)
            
            # 提取模态特征
            modality_features = await self.feature_service.extract_features(preprocessed_inputs)
            
            # 分析模态内容
            modality_analysis = await self.feature_service.analyze_modalities(modality_features)
            
            # 对齐模态信息
            aligned_modalities = await self.alignment_engine.align_modalities(
                modality_features, modality_analysis
            )
            
            # 选择融合策略
            fusion_strategy = await self.fusion_service.select_strategy(
                aligned_modalities, modality_analysis, context
            )
            
            # 融合模态信息
            fused_result = await self.fusion_service.fuse_modalities(
                aligned_modalities, fusion_strategy
            )
            
            # 解决模态冲突
            resolved_result = await self.fusion_service.resolve_conflicts(fused_result)
            
            # 生成融合表达
            fused_expression = await self.fusion_service.generate_expression(resolved_result)
            
            # 评估融合效果
            fusion_evaluation = await self.evaluation_service.evaluate_fusion(
                fused_expression, aligned_modalities
            )
            
            # 构建融合结果
            result = FusionResult(
                fused_expression=fused_expression,
                fusion_strategy=fusion_strategy,
                confidence=fusion_evaluation.confidence,
                processing_time=fusion_evaluation.processing_time,
                alignment_quality=fusion_evaluation.alignment_quality,
                conflict_resolution=resolved_result.conflict_resolutions
            )
            
            # 记录融合结果
            await self._log_fusion_result(baby_id, session_id, result)
            
            return result
            
        except Exception as e:
            logger.error(f"多模态融合失败: {str(e)}")
            raise FusionError(f"多模态融合失败: {str(e)}")
    
    def _validate_inputs(self, modality_inputs: List[ModalityInput]):
        """验证输入数据"""
        if not modality_inputs:
            raise ValueError("模态输入不能为空")
        
        modality_types = set(input.modality_type for input in modality_inputs)
        if len(modality_types) < 2:
            raise ValueError("至少需要两种不同类型的模态输入")
    
    async def _log_fusion_result(self, baby_id: str, session_id: str, result: FusionResult):
        """记录融合结果"""
        # 实现融合结果记录逻辑
        pass
    
    def _initialize_components(self):
        """初始化组件"""
        self.input_service = InputService(self.config)
        self.feature_service = FeatureService(self.config)
        self.alignment_engine = AlignmentEngine(self.config)
        self.fusion_service = FusionService(self.config)
        self.evaluation_service = EvaluationService(self.config)
```

```python
# 数据模型定义
from pydantic import BaseModel, Field
from typing import List, Dict, Optional, Any, Union
from enum import Enum
from datetime import datetime

class ModalityType(str, Enum):
    """模态类型枚举"""
    AUDIO = "audio"
    VIDEO = "video"
    MOTION = "motion"

class FusionStrategyType(str, Enum):
    """融合策略类型枚举"""
    EARLY_FUSION = "early_fusion"
    LATE_FUSION = "late_fusion"
    HYBRID_FUSION = "hybrid_fusion"
    ATTENTION_FUSION = "attention_fusion"

class ModalityInput(BaseModel):
    """模态输入数据模型"""
    modality_type: ModalityType = Field(..., description="模态类型")
    data: Union[Dict[str, Any], bytes, str] = Field(..., description="模态数据")
    timestamp: float = Field(..., description="时间戳")
    metadata: Optional[Dict[str, Any]] = Field(default=None, description="元数据")

class ModalityFeatures(BaseModel):
    """模态特征模型"""
    modality_type: ModalityType = Field(..., description="模态类型")
    features: List[float] = Field(..., description="特征向量")
    confidence: float = Field(..., ge=0.0, le=1.0, description="置信度")
    extraction_time: float = Field(..., description="特征提取时间")

class ModalityAnalysis(BaseModel):
    """模态分析模型"""
    modality_type: ModalityType = Field(..., description="模态类型")
    content: Dict[str, Any] = Field(..., description="内容分析")
    emotion: Dict[str, float] = Field(..., description="情感分析")
    intent: Optional[str] = Field(default=None, description="意图分析")
    analysis_time: float = Field(..., description="分析时间")

class AlignedModality(BaseModel):
    """对齐后的模态模型"""
    modality_type: ModalityType = Field(..., description="模态类型")
    features: List[float] = Field(..., description="对齐后的特征")
    analysis: ModalityAnalysis = Field(..., description="模态分析")
    alignment_score: float = Field(..., ge=0.0, le=1.0, description="对齐分数")

class FusionStrategy(BaseModel):
    """融合策略模型"""
    strategy_type: FusionStrategyType = Field(..., description="策略类型")
    weights: Dict[str, float] = Field(..., description="模态权重")
    parameters: Dict[str, Any] = Field(default={}, description="策略参数")

class ConflictResolution(BaseModel):
    """冲突解决模型"""
    conflict_type: str = Field(..., description="冲突类型")
    resolution_method: str = Field(..., description="解决方法")
    resolution_result: Dict[str, Any] = Field(..., description="解决结果")

class FusionEvaluation(BaseModel):
    """融合评估模型"""
    confidence: float = Field(..., ge=0.0, le=1.0, description="融合置信度")
    naturalness: float = Field(..., ge=0.0, le=5.0, description="自然度评分")
    consistency: float = Field(..., ge=0.0, le=1.0, description="一致性评分")
    processing_time: float = Field(..., description="处理时间")
    alignment_quality: float = Field(..., ge=0.0, le=1.0, description="对齐质量")

class FusionResult(BaseModel):
    """融合结果模型"""
    fused_expression: Dict[str, Any] = Field(..., description="融合后的表达")
    fusion_strategy: FusionStrategy = Field(..., description="使用的融合策略")
    confidence: float = Field(..., ge=0.0, le=1.0, description="融合置信度")
    processing_time: float = Field(..., description="处理时间")
    alignment_quality: float = Field(..., ge=0.0, le=1.0, description="对齐质量")
    conflict_resolutions: List[ConflictResolution] = Field(default=[], description="冲突解决列表")
    evaluation: Optional[FusionEvaluation] = Field(default=None, description="融合评估")

class FusionConfig(BaseModel):
    """融合配置模型"""

    max_modalities: int = Field(default=5, description="最大模态数")
    default_strategy: FusionStrategyType = Field(default=FusionStrategyType.HYBRID_FUSION, description="默认融合策略")
    alignment_threshold: float = Field(default=0.7, description="对齐阈值")
    fusion_threshold: float = Field(default=0.8, description="融合阈值")
    cache_enabled: bool = Field(default=True, description="是否启用缓存")
    cache_ttl: int = Field(default=3600, description="缓存生存时间(秒)")
```

```python
# 自定义异常类
class MultimodalFusionError(Exception):
    """多模态融合错误基类"""
    def __init__(self, message: str, error_code: str = None, details: Dict[str, Any] = None):
        super().__init__(message)
        self.error_code = error_code
        self.details = details or {}

class InputValidationError(MultimodalFusionError):
    """输入验证错误"""
    def __init__(self, message: str, invalid_inputs: List[str] = None):
        super().__init__(message, "INPUT_VALIDATION_ERROR")
        self.invalid_inputs = invalid_inputs or []

class FeatureExtractionError(MultimodalFusionError):
    """特征提取错误"""
    def __init__(self, message: str, modality_type: str = None):
        super().__init__(message, "FEATURE_EXTRACTION_ERROR")
        self.modality_type = modality_type

class AlignmentError(MultimodalFusionError):
    """对齐错误"""
    def __init__(self, message: str, alignment_score: float = None):
        super().__init__(message, "ALIGNMENT_ERROR")
        self.alignment_score = alignment_score

class FusionError(MultimodalFusionError):
    """融合错误"""
    def __init__(self, message: str, strategy_type: str = None):
        super().__init__(message, "FUSION_ERROR")
        self.strategy_type = strategy_type

class ConflictResolutionError(MultimodalFusionError):
    """冲突解决错误"""
    def __init__(self, message: str, conflict_type: str = None):
        super().__init__(message, "CONFLICT_RESOLUTION_ERROR")
        self.conflict_type = conflict_type

# 异常处理示例
async def fuse_modalities_with_error_handling(
    self, 
    baby_id: str, 
    session_id: str, 
    modality_inputs: List[ModalityInput],
    context: Optional[Dict[str, Any]] = None
) -> FusionResult:
    """带错误处理的多模态融合"""
    try:
        # 验证输入
        try:
            self._validate_inputs(modality_inputs)
        except ValueError as e:
            raise InputValidationError(
                f"输入验证失败: {str(e)}",
                invalid_inputs=[inp.modality_type for inp in modality_inputs]
            )
        
        # 预处理输入
        try:
            preprocessed_inputs = await self.input_service.preprocess_inputs(modality_inputs)
        except Exception as e:
            raise InputValidationError(f"输入预处理失败: {str(e)}")
        
        # 提取模态特征
        try:
            modality_features = await self.feature_service.extract_features(preprocessed_inputs)
        except Exception as e:
            raise FeatureExtractionError(f"特征提取失败: {str(e)}")
        
        # 分析模态内容
        try:
            modality_analysis = await self.feature_service.analyze_modalities(modality_features)
        except Exception as e:
            raise FeatureExtractionError(f"模态分析失败: {str(e)}")
        
        # 对齐模态信息
        try:
            aligned_modalities = await self.alignment_engine.align_modalities(
                modality_features, modality_analysis
            )
            # 检查对齐质量
            alignment_quality = min(almod.alignment_score for almod in aligned_modalities)
            if alignment_quality < self.config.alignment_threshold:
                raise AlignmentError(
                    f"模态对齐质量过低: {alignment_quality}",
                    alignment_score=alignment_quality
                )
        except Exception as e:
            if isinstance(e, AlignmentError):
                raise
            raise AlignmentError(f"模态对齐失败: {str(e)}")
        
        # 选择融合策略
        try:
            fusion_strategy = await self.fusion_service.select_strategy(
                aligned_modalities, modality_analysis, context
            )
        except Exception as e:
            raise FusionError(f"融合策略选择失败: {str(e)}")
        
        # 融合模态信息
        try:
            fused_result = await self.fusion_service.fuse_modalities(
                aligned_modalities, fusion_strategy
            )
        except Exception as e:
            raise FusionError(
                f"模态融合失败: {str(e)}",
                strategy_type=fusion_strategy.strategy_type
            )
        
        # 解决模态冲突
        try:
            resolved_result = await self.fusion_service.resolve_conflicts(fused_result)
        except Exception as e:
            raise ConflictResolutionError(f"冲突解决失败: {str(e)}")
        
        # 生成融合表达
        try:
            fused_expression = await self.fusion_service.generate_expression(resolved_result)
        except Exception as e:
            raise FusionError(f"融合表达生成失败: {str(e)}")
        
        # 评估融合效果
        try:
            fusion_evaluation = await self.evaluation_service.evaluate_fusion(
                fused_expression, aligned_modalities
            )
        except Exception as e:
            logger.warning(f"融合效果评估失败: {str(e)}")
            fusion_evaluation = None
        
        # 构建融合结果
        result = FusionResult(
            fused_expression=fused_expression,
            fusion_strategy=fusion_strategy,
            confidence=fusion_evaluation.confidence if fusion_evaluation else 0.5,
            processing_time=fusion_evaluation.processing_time if fusion_evaluation else 0.0,
            alignment_quality=fusion_evaluation.alignment_quality if fusion_evaluation else 0.5,
            conflict_resolutions=resolved_result.conflict_resolutions
        )
        
        # 记录融合结果
        await self._log_fusion_result(baby_id, session_id, result)
        
        return result
        
    except MultimodalFusionError as e:
        logger.error(f"多模态融合错误 [{e.error_code}]: {str(e)}")
        if e.details:
            logger.error(f"错误详情: {e.details}")
        raise
    except Exception as e:
        logger.error(f"未知错误: {str(e)}")
        raise MultimodalFusionError(f"多模态融合操作失败: {str(e)}", "UNKNOWN_ERROR")
```

```python
from fastapi import FastAPI, HTTPException, Depends, BackgroundTasks
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from typing import List, Dict, Any, Optional
import asyncio

from multimodal_fusion.core.fusion_manager import MultimodalFusionManager
from multimodal_fusion.models.modality_models import (
    ModalityInput, FusionResult, FusionRequest, FusionResponse
)
from multimodal_fusion.models.config_models import FusionConfig
from multimodal_fusion.utils.auth import get_current_user
from multimodal_fusion.utils.metrics import record_fusion_metrics

app = FastAPI(title="多模态融合API", version="1.0.0")
security = HTTPBearer()

# 依赖注入
async def get_fusion_manager() -> MultimodalFusionManager:
    """获取融合管理器实例"""
    config = FusionConfig()
    return MultimodalFusionManager(config)

# API端点
@app.post("/api/fusion/fuse", response_model=FusionResponse)
async def fuse_modalities(
    request: FusionRequest,
    background_tasks: BackgroundTasks,
    credentials: HTTPAuthorizationCredentials = Depends(security),
    fusion_manager: MultimodalFusionManager = Depends(get_fusion_manager)
):
    """
    融合多种模态信息
    
    Args:
        request: 融合请求，包含婴儿ID、会话ID和模态输入列表
        background_tasks: 后台任务
        credentials: 认证凭据
        fusion_manager: 融合管理器
        
    Returns:
        FusionResponse: 融合响应，包含融合结果
        
    Raises:
        HTTPException: 请求处理失败
    """
    try:
        # 认证用户
        user = await get_current_user(credentials.credentials)
        
        # 记录请求
        logger.info(f"用户 {user.id} 请求融合婴儿 {request.baby_id} 的多模态数据")
        
        # 执行融合
        result = await fusion_manager.fuse_modalities(
            baby_id=request.baby_id,
            session_id=request.session_id,
            modality_inputs=request.modality_inputs,
            context=request.context
        )
        
        # 添加后台任务记录指标
        background_tasks.add_task(
            record_fusion_metrics,
            user_id=user.id,
            baby_id=request.baby_id,
            session_id=request.session_id,
            result=result
        )
        
        # 返回响应
        return FusionResponse(
            success=True,
            message="多模态融合成功",
            data=result
        )
        
    except InputValidationError as e:
        logger.error(f"输入验证错误: {str(e)}")
        raise HTTPException(status_code=400, detail={
            "error_code": e.error_code,
            "message": str(e),
            "invalid_inputs": e.invalid_inputs
        })
    except AlignmentError as e:
        logger.error(f"模态对齐错误: {str(e)}")
        raise HTTPException(status_code=422, detail={
            "error_code": e.error_code,
            "message": str(e),
            "alignment_score": e.alignment_score
        })
    except FusionError as e:
        logger.error(f"模态融合错误: {str(e)}")
        raise HTTPException(status_code=422, detail={
            "error_code": e.error_code,
            "message": str(e),
            "strategy_type": e.strategy_type
        })
    except MultimodalFusionError as e:
        logger.error(f"多模态融合错误: {str(e)}")
        raise HTTPException(status_code=500, detail={
            "error_code": e.error_code,
            "message": str(e),
            "details": e.details
        })
    except Exception as e:
        logger.error(f"未知错误: {str(e)}")
        raise HTTPException(status_code=500, detail={
            "error_code": "UNKNOWN_ERROR",
            "message": "服务器内部错误"
```

```python
# 事件驱动接口实现
import asyncio
from typing import Callable, Dict, Any
from multimodal_fusion.core.event_bus import EventBus
from multimodal_fusion.models.event_models import (
    ModalityInputEvent, FusionRequestEvent, FusionCompletedEvent
)

class MultimodalFusionEventHandler:
    """多模态融合事件处理器"""
    
    def __init__(self, fusion_manager: MultimodalFusionManager):
        self.fusion_manager = fusion_manager
        self.event_bus = EventBus()
        self._register_handlers()
    
    def _register_handlers(self):
        """注册事件处理器"""
        self.event_bus.subscribe("modality_input_received", self._handle_modality_input)
        self.event_bus.subscribe("fusion_request_received", self._handle_fusion_request)
        self.event_bus.subscribe("fusion_completed", self._handle_fusion_completed)
    
    async def _handle_modality_input(self, event_data: Dict[str, Any]):
        """处理模态输入事件"""
        try:
            # 解析事件数据
            modality_input = ModalityInput.parse_obj(event_data.get("modality_input"))
            baby_id = event_data.get("baby_id")
            session_id = event_data.get("session_id")
            
            # 存储模态输入
            await self.fusion_manager.input_service.store_modality_input(
                baby_id, session_id, modality_input
            )
            
            # 检查是否已收集足够模态进行融合
            modalities = await self.fusion_manager.input_service.get_collected_modalities(
                baby_id, session_id
            )
            
            if len(modalities) >= 2:  # 至少需要两种模态
                # 发布融合请求事件
                await self.event_bus.publish("fusion_request", {
                    "baby_id": baby_id,
                    "session_id": session_id,
                    "modalities": modalities
                })
            
        except Exception as e:
            logger.error(f"处理模态输入事件失败: {str(e)}")
    
    async def _handle_fusion_request(self, event_data: Dict[str, Any]):
        """处理融合请求事件"""
        try:
            # 解析事件数据
            baby_id = event_data.get("baby_id")
            session_id = event_data.get("session_id")
            modalities = event_data.get("modalities")
            context = event_data.get("context", {})
            
            # 执行融合
            fusion_result = await self.fusion_manager.fuse_modalities(
                baby_id=baby_id,
                session_id=session_id,
                modality_inputs=modalities,
                context=context
            )
            
            # 发布融合完成事件
            await self.event_bus.publish("fusion_completed", {
                "baby_id": baby_id,
                "session_id": session_id,
                "fusion_result": fusion_result.dict()
            })
            
        except Exception as e:
            logger.error(f"处理融合请求事件失败: {str(e)}")
            
            # 发布融合失败事件
            await self.event_bus.publish("fusion_failed", {
                "baby_id": baby_id,
                "session_id": session_id,
                "error": str(e)
            })
    
    async def _handle_fusion_completed(self, event_data: Dict[str, Any]):
        """处理融合完成事件"""
        try:
            # 解析事件数据
            baby_id = event_data.get("baby_id")
            session_id = event_data.get("session_id")
            fusion_result = event_data.get("fusion_result")
            
            # 存储融合结果
            await self.fusion_manager.storage_service.store_fusion_result(
                baby_id, session_id, fusion_result
            )
            
            # 通知相关模块
            await self._notify_interested_modules(baby_id, session_id, fusion_result)
            
        except Exception as e:
            logger.error(f"处理融合完成事件失败: {str(e)}")
    
    async def _notify_interested_modules(
        self, 
        baby_id: str, 
        session_id: str, 
        fusion_result: Dict[str, Any]
    ):
        """通知相关模块"""
        # 通知交互管理模块
        await self.event_bus.publish("fusion_result_available", {
            "baby_id": baby_id,
            "session_id": session_id,
            "fusion_result": fusion_result,
            "target_module": "interaction_management"
        })
        
        # 通知交互效果评估模块
        await self.event_bus.publish("fusion_result_available", {
            "baby_id": baby_id,
            "session_id": session_id,
            "fusion_result": fusion_result,
            "target_module": "interaction_evaluation"
        })
```

## 功能说明

### 模态输入
多模态融合模块支持以下模态类型的输入：
- **音频**：包括语音、声音等音频数据
- **视频**：包括面部表情、肢体动作等视频数据
- **动作**：包括关节位置、动作类型等动作数据

每种模态输入需要包含以下信息：
- 模态类型（modality_type）
- 模态数据（data）
- 时间戳（timestamp）
- 元数据（metadata，可选）

### 融合策略
多模态融合模块支持以下融合策略：
- **早期融合**：在特征层融合多模态信息，计算效率高，特征交互充分
- **晚期融合**：在决策层融合多模态信息，模态独立处理，灵活性高
- **混合融合**：结合早期和晚期融合策略，平衡性能和灵活性
- **注意力融合**：使用注意力机制融合多模态信息，自适应权重，性能优异

### 融合结果
融合结果包含以下信息：
- **融合表达**：融合后的多模态表达，包括各模态的表达内容
- **融合策略**：使用的融合策略及其参数
- **置信度**：融合结果的置信度，范围0-1
- **处理时间**：融合处理所需的时间（秒）
- **对齐质量**：模态对齐的质量评分，范围0-1
- **冲突解决**：模态冲突的解决方法和结果

## 常见问题

### Q1: 如何选择合适的融合策略？
A1: 融合策略的选择取决于具体的应用场景和数据特点：
- 如果模态间同步性高，特征对齐好，可以选择早期融合
- 如果模态间差异性大，独立性高，可以选择晚期融合
- 如果需要平衡性能和灵活性，可以选择混合融合
- 如果计算资源充足，追求最佳性能，可以选择注意力融合

### Q2: 如何提高融合结果的置信度？
A2: 提高融合结果置信度的方法包括：
- 提高输入数据的质量
- 确保模态间的时间同步性
- 选择适合当前场景的融合策略
- 提供丰富的上下文信息

### Q3: 如何处理模态冲突？
A3: 多模态融合模块内置了冲突解决机制：
- 基于置信度的冲突解决：优先选择置信度高的模态
- 基于上下文的冲突解决：结合上下文信息选择最合适的模态
- 基于历史行为的冲突解决：参考婴儿的历史行为模式

## 故障排除

### 问题1：融合请求失败
**症状**：API返回400错误
**可能原因**：
- 输入数据格式不正确
- 缺少必要的参数
- 模态数据质量过低

**解决方案**：
- 检查输入数据格式是否符合API文档要求
- 确保所有必要参数都已提供
- 提高质量更高的模态数据

### 问题2：融合结果置信度低
**症状**：融合结果的置信度低于0.5
**可能原因**：
- 模态间时间不同步
- 模态数据质量低
- 融合策略不适合当前场景

**解决方案**：
- 确保模态数据的时间戳相近
- 提高质量更高的模态数据
- 尝试不同的融合策略

### 问题3：处理时间过长
**症状**：融合请求处理时间超过1秒
**可能原因**：
- 模态数据过大
- 计算资源不足
- 融合策略计算复杂度高

**解决方案**：
- 压缩模态数据大小
- 增加计算资源
- 选择计算复杂度更低的融合策略
