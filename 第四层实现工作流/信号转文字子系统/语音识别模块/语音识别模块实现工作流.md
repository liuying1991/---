# 语音识别模块实现工作流

## 模块信息

### 模块名称
语音识别模块

### 所属子系统
信号转文字子系统

### 模块优先级
高

### 模块功能描述
语音识别模块负责将婴儿的语音信号转换为文字信息，为后续的情感分析和意图理解提供基础数据。该模块能够准确识别婴儿的各种声音表达，包括哭声、笑声、咿呀学语等，并将其转换为可理解的文字描述，同时识别语音中的情感色彩和情绪状态。

### 模块职责
1. 语音信号预处理：对原始语音信号进行降噪、增强和分段处理
2. 特征提取：提取语音信号的声学特征，包括MFCC、音高、能量等
3. 声学模型应用：使用深度学习模型进行语音到文本的转换
4. 语言模型应用：结合语言模型提高识别准确率
5. 情感特征提取：识别语音中的情感色彩和情绪状态
6. 后处理优化：对识别结果进行后处理和优化

### 模块接口
- **输入接口**：音频流、音频文件、语音识别参数
- **输出接口**：识别文本、置信度分数、情感标签、时间戳
- **依赖模块**：音频处理模块
- **被依赖模块**：自然语言理解模块、情感分析模块

#### AudioPreprocessor 类
```python
import librosa
import numpy as np
from typing import Tuple, Optional

class AudioPreprocessor:
    """音频预处理器，负责音频信号的降噪、增强和分段"""
    
    def __init__(self, sample_rate: int = 16000, n_fft: int = 512):
        self.sample_rate = sample_rate
        self.n_fft = n_fft
        self.hop_length = n_fft // 4
        self.window = np.hanning(n_fft)
        
    def load_audio(self, audio_path: str) -> Tuple[np.ndarray, int]:
        """加载音频文件"""
        try:
            audio, sr = librosa.load(audio_path, sr=self.sample_rate)
            return audio, sr
        except Exception as e:
            raise AudioProcessingError(f"Failed to load audio file: {e}")
    
    def resample(self, audio: np.ndarray, original_sr: int) -> np.ndarray:
        """重采样音频到目标采样率"""
        if original_sr == self.sample_rate:
            return audio
        return librosa.resample(audio, orig_sr=original_sr, target_sr=self.sample_rate)
    
    def normalize(self, audio: np.ndarray) -> np.ndarray:
        """音频归一化"""
        max_val = np.max(np.abs(audio))
        if max_val > 0:
            return audio / max_val
        return audio
    
    def reduce_noise(self, audio: np.ndarray, method: str = "spectral_subtraction") -> np.ndarray:
        """降噪处理"""
        if method == "spectral_subtraction":
            return self._spectral_subtraction(audio)
        elif method == "wiener":
            return self._wiener_filter(audio)
        else:
            raise ValueError(f"Unsupported noise reduction method: {method}")
    
    def _spectral_subtraction(self, audio: np.ndarray, alpha: float = 2.0, beta: float = 0.01) -> np.ndarray:
        """谱减法降噪"""
        # 计算短时傅里叶变换
        stft = librosa.stft(audio, n_fft=self.n_fft, hop_length=self.hop_length, window=self.window)
        magnitude = np.abs(stft)
        phase = np.angle(stft)
        
        # 估计噪声谱（使用前几帧）
        noise_frames = 5
        noise_spectrum = np.mean(magnitude[:, :noise_frames], axis=1, keepdims=True)
        
        # 谱减法
        enhanced_magnitude = magnitude - alpha * noise_spectrum
        enhanced_magnitude = np.maximum(enhanced_magnitude, beta * magnitude)
        
        # 重建音频
        enhanced_stft = enhanced_magnitude * np.exp(1j * phase)
        enhanced_audio = librosa.istft(enhanced_stft, hop_length=self.hop_length, window=self.window)
        
        return enhanced_audio
    
    def _wiener_filter(self, audio: np.ndarray) -> np.ndarray:
        """维纳滤波降噪"""
        # 简化的维纳滤波实现
        stft = librosa.stft(audio, n_fft=self.n_fft, hop_length=self.hop_length, window=self.window)
        magnitude = np.abs(stft)
        phase = np.angle(stft)
        
        # 估计信噪比
        noise_frames = 5
        noise_spectrum = np.mean(magnitude[:, :noise_frames], axis=1, keepdims=True)
        snr = magnitude / (noise_spectrum + 1e-10)
        
        # 维纳滤波器
        wiener_gain = snr / (snr + 1)
        filtered_magnitude = magnitude * wiener_gain
        
        # 重建音频
        filtered_stft = filtered_magnitude * np.exp(1j * phase)
        filtered_audio = librosa.istft(filtered_stft, hop_length=self.hop_length, window=self.window)
        
        return filtered_audio
    
    def preemphasis(self, audio: np.ndarray, coeff: float = 0.97) -> np.ndarray:
        """预加重滤波"""
        return np.append(audio[0], audio[1:] - coeff * audio[:-1])
    
    def apply_preemphasis(self, audio: np.ndarray, coeff: float = 0.97) -> np.ndarray:
        """应用预加重滤波"""
        return self.preemphasis(audio, coeff)
    
    def remove_preemphasis(self, audio: np.ndarray, coeff: float = 0.97) -> np.ndarray:
        """移除预加重滤波"""
        if len(audio) == 0:
            return audio
        
        result = np.zeros_like(audio)
        result[0] = audio[0]
        for i in range(1, len(audio)):
            result[i] = audio[i] + coeff * audio[i-1]
        return result
    
    def segment_audio(self, audio: np.ndarray, segment_length: float = 10.0, 
                     overlap: float = 0.5) -> list:
        """音频分段"""
        segment_samples = int(segment_length * self.sample_rate)
        overlap_samples = int(overlap * segment_samples)
        step_samples = segment_samples - overlap_samples
        
        segments = []
        start = 0
        while start + segment_samples <= len(audio):
            segments.append(audio[start:start + segment_samples])
            start += step_samples
        
        # 处理最后一段不足segment_samples的情况
        if start < len(audio):
            last_segment = np.zeros(segment_samples)
            last_segment[:len(audio) - start] = audio[start:]
            segments.append(last_segment)
        
        return segments
    
    def preprocess(self, audio: np.ndarray, original_sr: Optional[int] = None) -> np.ndarray:
        """完整的预处理流程"""
        # 重采样
        if original_sr is not None and original_sr != self.sample_rate:
            audio = self.resample(audio, original_sr)
        
        # 归一化
        audio = self.normalize(audio)
        
        # 预加重
        audio = self.apply_preemphasis(audio)
        
        # 降噪
        audio = self.reduce_noise(audio)
        
        return audio


class AudioProcessingError(Exception):
    """音频处理异常"""
    pass
```

#### VoiceActivityDetector 类
```python
import numpy as np
import librosa
from typing import List, Tuple

class VoiceActivityDetector:
    """语音活动检测器，用于检测音频中的语音片段"""
    
    def __init__(self, sample_rate: int = 16000, frame_duration: float = 0.03,
                 energy_threshold: float = 0.01, zcr_threshold: float = 0.1):
        self.sample_rate = sample_rate
        self.frame_length = int(frame_duration * sample_rate)
        self.hop_length = self.frame_length // 2
        self.energy_threshold = energy_threshold
        self.zcr_threshold = zcr_threshold
    
    def compute_energy(self, audio: np.ndarray) -> np.ndarray:
        """计算短时能量"""
        frames = librosa.util.frame(audio, frame_length=self.frame_length, 
                                   hop_length=self.hop_length)
        return np.sum(frames ** 2, axis=0)
    
    def compute_zero_crossing_rate(self, audio: np.ndarray) -> np.ndarray:
        """计算过零率"""
        return librosa.feature.zero_crossing_rate(audio, frame_length=self.frame_length,
                                                 hop_length=self.hop_length)[0]
    
    def detect_vad_energy(self, audio: np.ndarray) -> np.ndarray:
        """基于能量的语音活动检测"""
        energy = self.compute_energy(audio)
        # 归一化能量
        energy = energy / np.max(energy)
        return energy > self.energy_threshold
    
    def detect_vad_zcr(self, audio: np.ndarray) -> np.ndarray:
        """基于过零率的语音活动检测"""
        zcr = self.compute_zero_crossing_rate(audio)
        return zcr < self.zcr_threshold
    
    def detect_vad_combined(self, audio: np.ndarray) -> np.ndarray:
        """结合能量和过零率的语音活动检测"""
        energy_vad = self.detect_vad_energy(audio)
        zcr_vad = self.detect_vad_zcr(audio)
        return energy_vad & zcr_vad
    
    def smooth_vad(self, vad: np.ndarray, window_size: int = 5) -> np.ndarray:
        """平滑VAD结果"""
        # 中值滤波
        smoothed = np.zeros_like(vad, dtype=bool)
        half_window = window_size // 2
        
        for i in range(len(vad)):
            start = max(0, i - half_window)
            end = min(len(vad), i + half_window + 1)
            smoothed[i] = np.median(vad[start:end])
        
        return smoothed
    
    def remove_short_segments(self, vad: np.ndarray, min_duration: float = 0.3) -> np.ndarray:
        """移除过短的语音片段"""
        min_frames = int(min_duration * self.sample_rate / self.hop_length)
        result = vad.copy()
        
        # 找到语音片段的开始和结束
        speech_segments = self._get_speech_segments(vad)
        
        # 移除过短的片段
        for start, end in speech_segments:
            if end - start < min_frames:
                result[start:end] = False
        
        return result
    
    def _get_speech_segments(self, vad: np.ndarray) -> List[Tuple[int, int]]:
        """获取语音片段的开始和结束帧索引"""
        segments = []
        in_speech = False
        start = 0
        
        for i, is_speech in enumerate(vad):
            if is_speech and not in_speech:
                # 语音开始
                start = i
                in_speech = True
            elif not is_speech and in_speech:
                # 语音结束
                segments.append((start, i))
                in_speech = False
        
        # 处理最后一个片段
        if in_speech:
            segments.append((start, len(vad)))
        
        return segments
    
    def get_speech_segments(self, audio: np.ndarray) -> List[Tuple[float, float]]:
        """获取语音片段的时间范围（秒）"""
        vad = self.detect_vad_combined(audio)
        vad = self.smooth_vad(vad)
        vad = self.remove_short_segments(vad)
        
        segments = self._get_speech_segments(vad)
        
        # 转换为时间（秒）
        time_segments = []
        for start, end in segments:
            start_time = start * self.hop_length / self.sample_rate
            end_time = end * self.hop_length / self.sample_rate
            time_segments.append((start_time, end_time))
        
        return time_segments
    
    def extract_speech_segments(self, audio: np.ndarray) -> List[np.ndarray]:
        """提取语音片段"""
        segments = self.get_speech_segments(audio)
        speech_segments = []
        
        for start, end in segments:
            start_sample = int(start * self.sample_rate)
            end_sample = int(end * self.sample_rate)
            speech_segments.append(audio[start_sample:end_sample])
        
        return speech_segments
```

```python
import numpy as np
import librosa
from typing import Optional, Tuple

class FeatureExtractor:
    """特征提取器，负责提取语音信号的声学特征"""
    
    def __init__(self, sample_rate: int = 16000, n_fft: int = 512, 
                 n_mels: int = 80, n_mfcc: int = 13):
        self.sample_rate = sample_rate
        self.n_fft = n_fft
        self.hop_length = n_fft // 4
        self.n_mels = n_mels
        self.n_mfcc = n_mfcc
        self.window = np.hanning(n_fft)
        
        # 初始化MFCC提取器
        self.mfcc_extractor = MFCCExtractor(sample_rate, n_fft, n_mels, n_mfcc)
        
        # 初始化音高提取器
        self.pitch_extractor = PitchExtractor(sample_rate, n_fft, hop_length=self.hop_length)
        
        # 初始化能量提取器
        self.energy_extractor = EnergyExtractor(sample_rate, n_fft, hop_length=self.hop_length)
    
    def extract(self, audio: np.ndarray) -> np.ndarray:
        """提取综合特征"""
        # 提取MFCC特征
        mfcc_features = self.mfcc_extractor.extract(audio)
        
        # 提取音高特征
        pitch_features = self.pitch_extractor.extract(audio)
        
        # 提取能量特征
        energy_features = self.energy_extractor.extract(audio)
        
        # 特征融合
        combined_features = np.concatenate([
            mfcc_features,
            pitch_features.reshape(-1, 1),
            energy_features.reshape(-1, 1)
        ], axis=1)
        
        return combined_features
    
    def extract_mfcc(self, audio: np.ndarray) -> np.ndarray:
        """提取MFCC特征"""
        return self.mfcc_extractor.extract(audio)
    
    def extract_pitch(self, audio: np.ndarray) -> np.ndarray:
        """提取音高特征"""
        return self.pitch_extractor.extract(audio)
    
    def extract_energy(self, audio: np.ndarray) -> np.ndarray:
        """提取能量特征"""
        return self.energy_extractor.extract(audio)


class MFCCExtractor:
    """MFCC特征提取器"""
    
    def __init__(self, sample_rate: int = 16000, n_fft: int = 512, 
                 n_mels: int = 80, n_mfcc: int = 13):
        self.sample_rate = sample_rate
        self.n_fft = n_fft
        self.hop_length = n_fft // 4
        self.n_mels = n_mels
        self.n_mfcc = n_mfcc
        self.window = np.hanning(n_fft)
        
        # 创建梅尔滤波器组
        self.mel_filters = librosa.filters.mel(
            sr=sample_rate, n_fft=n_fft, n_mels=n_mels
        )
    
    def extract(self, audio: np.ndarray) -> np.ndarray:
        """提取MFCC特征"""
        # 计算短时傅里叶变换
        stft = librosa.stft(audio, n_fft=self.n_fft, hop_length=self.hop_length, window=self.window)
        magnitude = np.abs(stft) ** 2
        
        # 应用梅尔滤波器组
        mel_spectrogram = np.dot(self.mel_filters, magnitude)
        
        # 对数变换
        log_mel_spectrogram = np.log(mel_spectrogram + 1e-10)
        
        # 离散余弦变换
        mfcc = librosa.feature.mfcc(
            S=log_mel_spectrogram, n_mfcc=self.n_mfcc, dct_type=2
        )
        
        return mfcc.T  # 转置，使时间维度在第一维


class PitchExtractor:
    """音高特征提取器"""
    
    def __init__(self, sample_rate: int = 16000, n_fft: int = 512, hop_length: int = 128):
        self.sample_rate = sample_rate
        self.n_fft = n_fft
        self.hop_length = hop_length
        self.window = np.hanning(n_fft)
        
        # 音高范围（Hz）
        self.min_f0 = 50.0
        self.max_f0 = 500.0
    
    def extract(self, audio: np.ndarray) -> np.ndarray:
        """提取音高特征"""
        # 使用librosa的piptrack方法提取音高
        pitches, magnitudes = librosa.piptrack(
            y=audio, sr=self.sample_rate, n_fft=self.n_fft, 
            hop_length=self.hop_length, fmin=self.min_f0, fmax=self.max_f0
        )
        
        # 提取每个帧的音高
        pitch_values = np.zeros(pitches.shape[1])
        for t in range(pitches.shape[1]):
            index = magnitudes[:, t].argmax()
            pitch_values[t] = pitches[index, t]
        
        # 处理无声帧（音高为0）
        pitch_values[pitch_values < self.min_f0] = 0
        
        return pitch_values
    
    def extract_yin(self, audio: np.ndarray) -> np.ndarray:
        """使用YIN算法提取音高"""
        frame_length = self.n_fft
        hop_length = self.hop_length
        
        # 分帧
        frames = librosa.util.frame(audio, frame_length=frame_length, hop_length=hop_length)
        
        pitch_values = np.zeros(frames.shape[1])
        
        for i, frame in enumerate(frames.T):
            # 应用窗函数
            windowed_frame = frame * self.window
            
            # YIN算法
            pitch_values[i] = self._yin_pitch_tracking(windowed_frame)
        
        return pitch_values
    
    def _yin_pitch_tracking(self, frame: np.ndarray) -> float:
        """YIN音高跟踪算法"""
        W = len(frame)
        tau_max = W // 2
        tau_min = self.sample_rate // self.max_f0
        
        # 计算差分函数
        diff_function = np.zeros(tau_max)
        for tau in range(tau_min, tau_max):
            for j in range(W - tau):
                diff_function[tau] += (frame[j] - frame[j + tau]) ** 2
        
        # 计算累积均值归一化差分函数
        cmndf = np.zeros(tau_max)
        cmndf[tau_min] = 1
        for tau in range(tau_min + 1, tau_max):
            cmndf[tau] = diff_function[tau] / (1 / tau * np.sum(diff_function[1:tau + 1]))
        
        # 寻找最小值
        tau = tau_min + np.argmin(cmndf[tau_min:])
        
        # 抛物线插值提高精度
        if tau_min < tau < tau_max - 1:
            s0 = cmndf[tau - 1]
            s1 = cmndf[tau]
            s2 = cmndf[tau + 1]
            
            a = (s0 - 2 * s1 + s2) / 2
            b = (s2 - s0) / 2
            
            if a != 0:
                tau = tau - b / (2 * a)
        
        # 转换为频率
        if tau > 0:
            pitch = self.sample_rate / tau
            if pitch < self.min_f0 or pitch > self.max_f0:
                return 0
            return pitch
        else:
            return 0


class EnergyExtractor:
    """能量特征提取器"""
    
    def __init__(self, sample_rate: int = 16000, n_fft: int = 512, hop_length: int = 128):
        self.sample_rate = sample_rate
        self.n_fft = n_fft
        self.hop_length = hop_length
        self.window = np.hanning(n_fft)
    
    def extract(self, audio: np.ndarray) -> np.ndarray:
        """提取短时能量特征"""
        # 分帧
        frames = librosa.util.frame(audio, frame_length=self.n_fft, hop_length=self.hop_length)
        
        # 应用窗函数并计算能量
        windowed_frames = frames * self.window.reshape(-1, 1)
        energy = np.sum(windowed_frames ** 2, axis=0)
        
        return energy
    
    def extract_log_energy(self, audio: np.ndarray) -> np.ndarray:
        """提取对数能量特征"""
        energy = self.extract(audio)
        return np.log(energy + 1e-10)
```

```python
import torch
import time
from typing import Dict, Any, Optional, Tuple
from .preprocessing.audio_preprocessor import AudioPreprocessor
from .preprocessing.voice_activity_detector import VoiceActivityDetector
from .feature_extraction.feature_extractor import FeatureExtractor

from .models.language_model import LanguageModel
from .models.emotion_model import EmotionModel
from .decoding.decoder import Decoder
from .decoding.postprocessor import Postprocessor

class SpeechRecognizer:
    """语音识别器，负责将语音信号转换为文字"""
    
    def __init__(self, 
                 acoustic_model: AcousticModel,
                 language_model: LanguageModel,
                 emotion_model: EmotionModel,
                 device: str = "cpu"):
        self.acoustic_model = acoustic_model
        self.language_model = language_model
        self.emotion_model = emotion_model
        self.device = device
        
        # 初始化组件
        self.audio_preprocessor = AudioPreprocessor()
        self.vad = VoiceActivityDetector()
        self.feature_extractor = FeatureExtractor()
        self.decoder = Decoder(language_model)
        self.postprocessor = Postprocessor()
        
        # 将模型移动到指定设备
        self.acoustic_model.to(device)
        self.language_model.to(device)
        self.emotion_model.to(device)
        
        # 设置为评估模式
        self.acoustic_model.eval()
        self.language_model.eval()
        self.emotion_model.eval()
    
    def recognize(self, audio_data: np.ndarray, sample_rate: int) -> Dict[str, Any]:
        """识别语音并返回结果"""
        start_time = time.time()
        
        try:
            # 预处理
            preprocessed_audio = self.audio_preprocessor.preprocess(audio_data, sample_rate)
            
            # 语音活动检测
            speech_segments = self.vad.extract_speech_segments(preprocessed_audio)
            
            # 如果没有检测到语音，返回空结果
            if not speech_segments:
                return {
                    "text": "",
                    "emotion": "neutral",
                    "confidence": 0.0,
                    "timestamp": time.time(),
                    "processing_time": time.time() - start_time,
                    "error": "No speech detected"
                }
            
            # 合并所有语音片段
            combined_speech = np.concatenate(speech_segments)
            
            # 特征提取
            features = self.feature_extractor.extract(combined_speech)
            
            # 声学模型推理
            with torch.no_grad():
                features_tensor = torch.FloatTensor(features).unsqueeze(0).to(self.device)
                acoustic_output = self.acoustic_model(features_tensor)
                
                # 解码
                text = self.decoder.decode(acoustic_output)
                
                # 情感识别
                emotion = self._recognize_emotion(features)
                
                # 计算置信度
                confidence = self._calculate_confidence(acoustic_output)
            
            # 后处理
            result = self.postprocessor.process({
                "text": text,
                "emotion": emotion,
                "confidence": confidence,
                "timestamp": time.time(),
                "processing_time": time.time() - start_time
            })
            
            return result
            
        except Exception as e:
            return {
                "text": "",
                "emotion": "neutral",
                "confidence": 0.0,
                "timestamp": time.time(),
                "processing_time": time.time() - start_time,
                "error": str(e)
            }
    
    def recognize_file(self, audio_path: str) -> Dict[str, Any]:
        """识别音频文件"""
        try:
            # 加载音频
            audio_data, sample_rate = self.audio_preprocessor.load_audio(audio_path)
            
            # 识别
            return self.recognize(audio_data, sample_rate)
            
        except Exception as e:
            return {
                "text": "",
                "emotion": "neutral",
                "confidence": 0.0,
                "timestamp": time.time(),
                "processing_time": 0.0,
                "error": f"Failed to load audio file: {str(e)}"
            }
    
    def recognize_stream(self, audio_stream: np.ndarray, sample_rate: int) -> Dict[str, Any]:
        """识别音频流"""
        # 实现流式识别逻辑
        # 这里简化为直接调用recognize方法
        return self.recognize(audio_stream, sample_rate)
    
    def _recognize_emotion(self, features: np.ndarray) -> str:
        """识别语音情感"""
        with torch.no_grad():
            features_tensor = torch.FloatTensor(features).unsqueeze(0).to(self.device)
            emotion_output = self.emotion_model(features_tensor)
            
            # 获取最高概率的情感
            _, predicted = torch.max(emotion_output, dim=1)
            emotion_index = predicted.item()
            
            # 将索引映射到情感标签
            emotion_labels = ["happy", "sad", "angry", "neutral", "surprised"]
            return emotion_labels[emotion_index] if emotion_index < len(emotion_labels) else "neutral"
    
    def _calculate_confidence(self, acoustic_output: torch.Tensor) -> float:
        """计算识别置信度"""
        # 简化的置信度计算，实际实现可能更复杂
        # 这里使用输出概率的最大值作为置信度
        if hasattr(acoustic_output, 'softmax'):
            probs = torch.softmax(acoustic_output, dim=-1)
            confidence = torch.max(probs).item()
        else:
            # 如果没有softmax输出，使用其他方法计算置信度
            confidence = 0.8  # 默认置信度
        
        return confidence
    
    def update_models(self, 
                     acoustic_model: Optional[AcousticModel] = None,
                     language_model: Optional[LanguageModel] = None,
                     emotion_model: Optional[EmotionModel] = None):
        """更新模型"""
        if acoustic_model is not None:
            self.acoustic_model = acoustic_model
            self.acoustic_model.to(self.device)
            self.acoustic_model.eval()
        
        if language_model is not None:
            self.language_model = language_model
            self.language_model.to(self.device)
            self.language_model.eval()
            self.decoder = Decoder(language_model)
        
        if emotion_model is not None:
            self.emotion_model = emotion_model
            self.emotion_model.to(self.device)
            self.emotion_model.eval()
    
    def get_model_info(self) -> Dict[str, Any]:
        """获取模型信息"""
        return {
            "acoustic_model": {

            },
            "language_model": {

            },
            "emotion_model": {

            }
        }
```

```python
import torch
import torch.nn as nn
from typing import Tuple, Optional

class AcousticModel(nn.Module):
    """声学模型，负责将声学特征映射到音素序列"""
    
    def __init__(self, 
                 input_dim: int = 80,  # MFCC特征维度
                 hidden_dim: int = 256,
                 num_layers: int = 4,
                 num_classes: int = 40,  # 音素类别数
                 dropout: float = 0.1,
                 model_type: str = "transformer"):
        super(AcousticModel, self).__init__()
        
        self.model_type = model_type
        self.input_dim = input_dim
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers
        self.num_classes = num_classes
        self.dropout = dropout
        
        if model_type == "transformer":
            self._build_transformer_model()
        elif model_type == "lstm":
            self._build_lstm_model()
        elif model_type == "wav2vec2":
            self._build_wav2vec2_model()
        else:
            raise ValueError(f"Unsupported model type: {model_type}")
    
    def _build_transformer_model(self):
        """构建Transformer模型"""
        # 输入投影层
        self.input_projection = nn.Linear(self.input_dim, self.hidden_dim)
        
        # 位置编码
        self.pos_encoding = PositionalEncoding(self.hidden_dim)
        
        # Transformer编码器层
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=self.hidden_dim,
            nhead=8,
            dim_feedforward=self.hidden_dim * 4,
            dropout=self.dropout,
            activation="relu"
        )
        self.transformer_encoder = nn.TransformerEncoder(
            encoder_layer, 
            num_layers=self.num_layers
        )
        
        # 输出层
        self.output_layer = nn.Linear(self.hidden_dim, self.num_classes)
        
        # 初始化权重
        self._init_weights()
    
    def _build_lstm_model(self):
        """构建LSTM模型"""
        # LSTM层
        self.lstm = nn.LSTM(
            input_size=self.input_dim,
            hidden_size=self.hidden_dim,
            num_layers=self.num_layers,
            dropout=self.dropout if self.num_layers > 1 else 0,
            bidirectional=True,
            batch_first=True
        )
        
        # 输出层
        self.output_layer = nn.Linear(self.hidden_dim * 2, self.num_classes)
        
        # 初始化权重
        self._init_weights()
    
    def _build_wav2vec2_model(self):
        """构建Wav2Vec2模型"""
        # 这里简化实现，实际应该使用预训练的Wav2Vec2模型
        from transformers import Wav2Vec2Model
        
        self.wav2vec2 = Wav2Vec2Model.from_pretrained("facebook/wav2vec2-base")
        
        # 冻结部分层
        for param in self.wav2vec2.parameters():
            param.requires_grad = False
        
        # 解冻最后几层
        for param in self.wav2vec2.encoder.layers[-2:].parameters():
            param.requires_grad = True
        
        # 输出层
        self.output_layer = nn.Linear(self.wav2vec2.config.hidden_size, self.num_classes)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """前向传播"""
        if self.model_type == "transformer":
            return self._forward_transformer(x)
        elif self.model_type == "lstm":
            return self._forward_lstm(x)
        elif self.model_type == "wav2vec2":
            return self._forward_wav2vec2(x)
    
    def _forward_transformer(self, x: torch.Tensor) -> torch.Tensor:
        """Transformer前向传播"""
        # 输入投影
        x = self.input_projection(x)
        
        # 添加位置编码
        x = self.pos_encoding(x)
        
        # Transformer编码器
        x = self.transformer_encoder(x)
        
        # 输出层
        logits = self.output_layer(x)
        
        return logits
    
    def _forward_lstm(self, x: torch.Tensor) -> torch.Tensor:
        """LSTM前向传播"""
        # LSTM层
        lstm_out, _ = self.lstm(x)
        
        # 输出层
        logits = self.output_layer(lstm_out)
        
        return logits
    
    def _forward_wav2vec2(self, x: torch.Tensor) -> torch.Tensor:
        """Wav2Vec2前向传播"""
        # Wav2Vec2模型
        outputs = self.wav2vec2(x)
        
        # 输出层
        logits = self.output_layer(outputs.last_hidden_state)
        
        return logits
    
    def _init_weights(self):
        """初始化权重"""
        for name, param in self.named_parameters():
            if 'weight' in name:
                if 'lstm' in name:
                    # LSTM权重初始化
                    nn.init.orthogonal_(param.data)
                else:
                    # 其他层权重初始化
                    nn.init.xavier_uniform_(param.data)
            elif 'bias' in name:
                nn.init.constant_(param.data, 0)


class PositionalEncoding(nn.Module):
    """位置编码"""
    
    def __init__(self, d_model: int, max_len: int = 5000):
        super(PositionalEncoding, self).__init__()
        
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * 
                            (-math.log(10000.0) / d_model))
        
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0).transpose(0, 1)
        
        self.register_buffer('pe', pe)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return x + self.pe[:x.size(0), :]
```

```python
import functools
import logging
from typing import Callable, Any

logger = logging.getLogger(__name__)

def handle_speech_recognition_errors(func: Callable) -> Callable:
    """语音识别异常处理装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs) -> Any:
        try:
            return func(*args, **kwargs)
        except Exception as e:
            logger.error(f"Error in {func.__name__}: {str(e)}", exc_info=True)
            
            # 根据异常类型返回相应的错误信息
            if "audio" in str(e).lower():
                raise AudioProcessingError(f"Audio processing failed: {str(e)}")
            elif "model" in str(e).lower():
                raise ModelLoadError(f"Model operation failed: {str(e)}")
            elif "feature" in str(e).lower():
                raise FeatureExtractionError(f"Feature extraction failed: {str(e)}")
            elif "decode" in str(e).lower():
                raise DecodingError(f"Decoding failed: {str(e)}")
            elif "emotion" in str(e).lower():
                raise EmotionRecognitionError(f"Emotion recognition failed: {str(e)}")
            else:
                raise SpeechRecognitionError(f"Speech recognition failed: {str(e)}")
    
    return wrapper
```

```python
            threads.append(thread)
            thread.start()
        
        # 等待所有线程完成
        for thread in threads:
            thread.join()
        
        end_time = time.time()
        
        # 验证所有识别都成功
        for result in results:
            self.assertIsNotNone(result)
            self.assertIn("text", result)
            self.assertIn("emotion", result)
            self.assertIn("confidence", result)
        
        # 验证并发识别时间不超过单次识别时间的5倍
        single_recognition_time = results[0]["processing_time"]
        concurrent_time = end_time - start_time
        self.assertLess(concurrent_time, single_recognition_time * 5, 
                       f"Concurrent recognition time {concurrent_time:.3f}s exceeds 5x single recognition time {single_recognition_time:.3f}s")
    
    def test_memory_usage(self):
        """测试内存使用"""
        import psutil
        import os
        
        # 获取当前进程
        process = psutil.Process(os.getpid())
        
        # 记录初始内存使用
        initial_memory = process.memory_info().rss / 1024 / 1024  # MB
        
        # 创建1秒的音频
        sample_rate = 16000
        duration = 1.0
        audio = np.random.randn(int(sample_rate * duration))
        
        # 执行多次识别
        for _ in range(100):
            result = self.recognizer.recognize(audio, sample_rate)
        
        # 记录最终内存使用
        final_memory = process.memory_info().rss / 1024 / 1024  # MB
        
        # 计算内存增长
        memory_increase = final_memory - initial_memory
        
        # 验证内存增长不超过100MB
        self.assertLess(memory_increase, 100, f"Memory increase {memory_increase:.2f}MB exceeds 100MB")

if __name__ == "__main__":
    unittest.main()
```

1. **识别结果为空**

   - 考虑使用GPU加速

   - 检查音频质量，确保无过多噪声
   - 考虑使用更适合的模型
   - 检查模型是否正确加载

4. **情感识别不准确**
   - 检查情感模型是否正确加载
   - 确认音频包含足够的情感信息
   - 考虑使用更适合的情感模型
