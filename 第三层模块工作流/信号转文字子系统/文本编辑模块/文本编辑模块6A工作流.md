# 文本编辑模块6A工作流

## 模块概述

### 模块名称
文本编辑模块

### 所属子系统
信号转文字子系统

### 模块功能描述
文本编辑模块负责对生成的文本内容进行编辑和优化，确保文本内容符合婴儿互动的需求。该模块能够对文本进行语法检查、风格调整、长度控制、内容简化等操作，使文本更适合婴儿理解和互动。同时，该模块还能根据婴儿的年龄、情绪状态和认知水平，对文本内容进行个性化调整。

### 模块职责
1. 语法检查：检查文本的语法正确性和表达清晰度
2. 风格调整：调整文本风格，使其更适合婴儿互动
3. 长度控制：控制文本长度，使其适合婴儿注意力范围
4. 内容简化：简化复杂词汇和句子结构
5. 个性化调整：根据婴儿特点调整文本内容
6. 质量评估：评估编辑后文本的质量和适宜性

### 模块接口
- **输入接口**：原始文本、编辑参数、婴儿状态
- **输出接口**：编辑后文本、编辑标签、质量评分
- **依赖模块**：文本生成模块
- **被依赖模块**：语音合成模块、交互表达子系统

# 文本编辑器实现
class TextEditor:
    def __init__(self, grammar_checker, style_editor, length_controller, 
                 content_simplifier, personalization_editor, quality_assessor):
        self.grammar_checker = grammar_checker
        self.style_editor = style_editor
        self.length_controller = length_controller
        self.content_simplifier = content_simplifier
        self.personalization_editor = personalization_editor
        self.quality_assessor = quality_assessor
        self.text_analyzer = TextAnalyzer()
        
    def edit(self, text, edit_params, baby_state):
        # 输入处理
        processed_input = self._process_input(text, edit_params, baby_state)
        
        # 文本分析
        text_analysis = self.text_analyzer.analyze(processed_input["text"])
        
        # 编辑操作
        edited_text = processed_input["text"]
        edit_history = []
        
        # 语法检查
        if edit_params.get("check_grammar", True):
            grammar_result = self.grammar_checker.check(edited_text)
            if grammar_result["has_errors"]:
                edited_text = grammar_result["corrected_text"]
                edit_history.append({
                    "operation": "grammar_check",
                    "original": processed_input["text"],
                    "edited": edited_text,
                    "changes": grammar_result["corrections"]
                })
        
        # 风格调整
        if "style" in edit_params:
            style_result = self.style_editor.adjust(edited_text, edit_params["style"])
            edited_text = style_result["adjusted_text"]
            edit_history.append({
                "operation": "style_adjustment",
                "original": edit_history[-1]["edited"] if edit_history else processed_input["text"],
                "edited": edited_text,
                "style": edit_params["style"]
            })
        
        # 长度控制
        if "max_length" in edit_params:
            length_result = self.length_controller.control(edited_text, edit_params["max_length"])
            edited_text = length_result["controlled_text"]
            edit_history.append({
                "operation": "length_control",
                "original": edit_history[-1]["edited"] if edit_history else processed_input["text"],
                "edited": edited_text,
                "max_length": edit_params["max_length"]
            })
        
        # 内容简化
        if edit_params.get("simplify", False):
            simplify_result = self.content_simplifier.simplify(edited_text, baby_state["age"])
            edited_text = simplify_result["simplified_text"]
            edit_history.append({
                "operation": "content_simplification",
                "original": edit_history[-1]["edited"] if edit_history else processed_input["text"],
                "edited": edited_text,
                "age": baby_state["age"]
            })
        
        # 个性化编辑
        if edit_params.get("personalize", False):
            personalize_result = self.personalization_editor.personalize(edited_text, baby_state)
            edited_text = personalize_result["personalized_text"]
            edit_history.append({
                "operation": "personalization",
                "original": edit_history[-1]["edited"] if edit_history else processed_input["text"],
                "edited": edited_text,
                "baby_state": baby_state
            })
        
        # 质量评估
        quality_score = self.quality_assessor.assess(edited_text, baby_state)
        
        # 如果质量不达标，尝试重新编辑
        if quality_score < 0.8:
            # 尝试调整编辑参数并重新编辑
            adjusted_params = self._adjust_edit_params(edit_params, quality_score)
            return self.edit(text, adjusted_params, baby_state)
        
        return {
            "original_text": text,
            "edited_text": edited_text,
            "edit_history": edit_history,
            "quality_score": quality_score,
            "timestamp": time.time()
        }

# 内容简化器实现
class ContentSimplifier:
    def __init__(self):
        self.complex_words = self._load_complex_words()
        self.simplification_rules = self._load_simplification_rules()
        
    def simplify(self, text, age):
        # 根据年龄确定简化程度
        if age < 1:
            simplification_level = "high"
        elif age < 2:
            simplification_level = "medium"
        else:
            simplification_level = "low"
        
        # 简化词汇
        simplified_text = self._simplify_vocabulary(text, simplification_level)
        
        # 简化句子结构
        simplified_text = self._simplify_sentences(simplified_text, simplification_level)
        
        # 应用简化规则
        simplified_text = self._apply_simplification_rules(simplified_text, simplification_level)
        
        return {
            "original_text": text,
            "simplified_text": simplified_text,
            "simplification_level": simplification_level,
            "changes": self._get_changes(text, simplified_text)
        }
    
    def _simplify_vocabulary(self, text, level):
        words = nltk.word_tokenize(text)
        simplified_words = []
        
        for word in words:
            if word.lower() in self.complex_words:
                simplified_word = self._get_simpler_word(word, level)
                simplified_words.append(simplified_word)
            else:
                simplified_words.append(word)
        
        return " ".join(simplified_words)
    
    def _simplify_sentences(self, text, level):
        sentences = nltk.sent_tokenize(text)
        simplified_sentences = []
        
        for sentence in sentences:
            # 根据简化程度处理句子
            if level == "high":
                # 分解长句为短句
                simplified_sentence = self._break_long_sentence(sentence)
            elif level == "medium":
                # 简化复杂句子结构
                simplified_sentence = self._simplify_sentence_structure(sentence)
            else:
                # 保持原句结构，只做微调
                simplified_sentence = sentence
            
            simplified_sentences.append(simplified_sentence)
        
        return " ".join(simplified_sentences)

   - 启动容器
   - 配置负载均衡

3. **部署脚本**：
   ```bash
   # Dockerfile
   FROM python:3.8-slim
   WORKDIR /app
   COPY requirements.txt .
   RUN pip install -r requirements.txt
   COPY src/ ./src/
   COPY data/ ./data/
   CMD ["uvicorn", "src.api.text_edit_api:app", "--host", "0.0.0.0", "--port", "8007"]
   ```
