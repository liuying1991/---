```python
# 响应时间测量工具
import time
import statistics
from typing import List, Dict

class ResponseTimeMonitor:
    def __init__(self, module_name: str):
        self.module_name = module_name
        self.response_times = []
        self.percentiles = [50, 90, 95, 99]
    
    def measure_response_time(self, func, *args, **kwargs):
        """测量函数响应时间"""
        start_time = time.perf_counter()
        result = func(*args, **kwargs)
        end_time = time.perf_counter()
        
        response_time = (end_time - start_time) * 1000  # 转换为毫秒
        self.response_times.append(response_time)
        
        return result, response_time
    
    def get_statistics(self) -> Dict:
        """获取响应时间统计信息"""
        if not self.response_times:
            return {}
        
        sorted_times = sorted(self.response_times)
        
        stats = {
            "module": self.module_name,
            "count": len(self.response_times),
            "min": min(self.response_times),
            "max": max(self.response_times),
            "mean": statistics.mean(self.response_times),
            "median": statistics.median(self.response_times),
            "std_dev": statistics.stdev(self.response_times) if len(self.response_times) > 1 else 0
        }
        
        # 计算百分位数
        for percentile in self.percentiles:
            index = int(len(sorted_times) * percentile / 100)
            if index >= len(sorted_times):
                index = len(sorted_times) - 1
            stats[f"p{percentile}"] = sorted_times[index]
        
        return stats
    
    def export_report(self, filename: str):
        """导出测试报告"""
        stats = self.get_statistics()
        
        with open(filename, 'w') as f:
            f.write(f"响应时间测试报告 - {self.module_name}\n")
            f.write("=" * 50 + "\n")
            f.write(f"测试样本数: {stats['count']}\n")
            f.write(f"最小响应时间: {stats['min']:.2f}ms\n")
            f.write(f"最大响应时间: {stats['max']:.2f}ms\n")
            f.write(f"平均响应时间: {stats['mean']:.2f}ms\n")
            f.write(f"中位数响应时间: {stats['median']:.2f}ms\n")
            f.write(f"标准差: {stats['std_dev']:.2f}ms\n")
            
            for percentile in self.percentiles:
                f.write(f"P{percentile}响应时间: {stats[f'p{percentile}']:.2f}ms\n")
```

```python
# 吞吐量测量工具
import time
import threading
from concurrent.futures import ThreadPoolExecutor, as_completed
from typing import Callable, List, Dict

class ThroughputMonitor:
    def __init__(self, module_name: str):
        self.module_name = module_name
        self.request_count = 0
        self.success_count = 0
        self.error_count = 0
        self.start_time = None
        self.end_time = None
        self.lock = threading.Lock()
    
    def start_monitoring(self):
        """开始监控"""
        self.start_time = time.perf_counter()
        self.request_count = 0
        self.success_count = 0
        self.error_count = 0
    
    def stop_monitoring(self):
        """停止监控"""
        self.end_time = time.perf_counter()
    
    def record_request(self, success: bool = True):
        """记录请求"""
        with self.lock:
            self.request_count += 1
            if success:
                self.success_count += 1
            else:
                self.error_count += 1
    
    def get_throughput_stats(self) -> Dict:
        """获取吞吐量统计"""
        if not self.start_time or not self.end_time:
            return {}
        
        duration = self.end_time - self.start_time
        
        return {
            "module": self.module_name,
            "duration": duration,
            "total_requests": self.request_count,
            "successful_requests": self.success_count,
            "failed_requests": self.error_count,
            "success_rate": self.success_count / self.request_count if self.request_count > 0 else 0,
            "requests_per_second": self.request_count / duration if duration > 0 else 0,
            "successful_requests_per_second": self.success_count / duration if duration > 0 else 0
        }
    
    def load_test(self, test_function: Callable, concurrent_users: int = 10, 
                  test_duration: int = 60, requests_per_user: int = 100):
        """执行负载测试"""
        self.start_monitoring()
        
        def user_task(user_id):
            """用户任务"""
            local_success = 0
            local_error = 0
            
            for i in range(requests_per_user):
                try:
                    result = test_function(user_id, i)
                    local_success += 1
                except Exception as e:
                    local_error += 1
                
                # 控制请求频率
                time.sleep(test_duration / requests_per_user)
            
            return local_success, local_error
        
        # 执行并发测试
        with ThreadPoolExecutor(max_workers=concurrent_users) as executor:
            futures = [executor.submit(user_task, i) for i in range(concurrent_users)]
            
            for future in as_completed(futures):
                success, error = future.result()
                self.record_request(True)
                for _ in range(success - 1):
                    self.record_request(True)
                for _ in range(error):
                    self.record_request(False)
        
        self.stop_monitoring()
        return self.get_throughput_stats()
```

#### 3.3.2 资源利用率测量方法
```python
# 资源利用率监控工具
import psutil
import GPUtil
import time
import threading
from typing import Dict, List

class ResourceMonitor:
    def __init__(self, monitoring_interval: float = 1.0):
        self.monitoring_interval = monitoring_interval
        self.monitoring = False
        self.data = {
            "cpu": [],
            "memory": [],
            "gpu": [],
            "disk_io": [],
            "network": []
        }
        self.monitor_thread = None
    
    def start_monitoring(self):
        """开始资源监控"""
        self.monitoring = True
        self.monitor_thread = threading.Thread(target=self._monitor_loop)
        self.monitor_thread.start()
    
    def stop_monitoring(self):
        """停止资源监控"""
        self.monitoring = False
        if self.monitor_thread:
            self.monitor_thread.join()
    
    def _monitor_loop(self):
        """监控循环"""
        while self.monitoring:
            # CPU利用率
            cpu_percent = psutil.cpu_percent(interval=None)
            self.data["cpu"].append({
                "timestamp": time.time(),
                "percent": cpu_percent,
                "per_cpu": psutil.cpu_percent(percpu=True)
            })
            
            # 内存利用率
            memory = psutil.virtual_memory()
            self.data["memory"].append({
                "timestamp": time.time(),
                "percent": memory.percent,
                "available": memory.available,
                "used": memory.used,
                "total": memory.total
            })
            
            # GPU利用率
            gpus = GPUtil.getGPUs()
            gpu_data = []
            for gpu in gpus:
                gpu_data.append({
                    "id": gpu.id,
                    "name": gpu.name,
                    "percent": gpu.load * 100,
                    "memory_percent": gpu.memoryUtil * 100,
                    "memory_used": gpu.memoryUsed,
                    "memory_total": gpu.memoryTotal
                })
            
            self.data["gpu"].append({
                "timestamp": time.time(),
                "gpus": gpu_data
            })
            
            # 磁盘I/O
            disk_io = psutil.disk_io_counters()
            self.data["disk_io"].append({
                "timestamp": time.time(),
                "read_bytes": disk_io.read_bytes,
                "write_bytes": disk_io.write_bytes,
                "read_count": disk_io.read_count,
                "write_count": disk_io.write_count,
                "read_time": disk_io.read_time,
                "write_time": disk_io.write_time
            })
            
            # 网络I/O
            network_io = psutil.net_io_counters()
            self.data["network"].append({
                "timestamp": time.time(),
                "bytes_sent": network_io.bytes_sent,
                "bytes_recv": network_io.bytes_recv,
                "packets_sent": network_io.packets_sent,
                "packets_recv": network_io.packets_recv,
                "errin": network_io.errin,
                "errout": network_io.errout,
                "dropin": network_io.dropin,
                "dropout": network_io.dropout
            })
            
            time.sleep(self.monitoring_interval)
    
    def get_resource_statistics(self) -> Dict:
        """获取资源利用率统计"""
        stats = {}
        
        # CPU统计
        if self.data["cpu"]:
            cpu_percentages = [d["percent"] for d in self.data["cpu"]]
            stats["cpu"] = {
                "avg_percent": sum(cpu_percentages) / len(cpu_percentages),
                "max_percent": max(cpu_percentages),
                "min_percent": min(cpu_percentages),
                "std_dev": self._calculate_std_dev(cpu_percentages)
            }
        
        # 内存统计
        if self.data["memory"]:
            memory_percentages = [d["percent"] for d in self.data["memory"]]
            stats["memory"] = {
                "avg_percent": sum(memory_percentages) / len(memory_percentages),
                "max_percent": max(memory_percentages),
                "min_percent": min(memory_percentages),
                "std_dev": self._calculate_std_dev(memory_percentages)
            }
        
        # GPU统计
        if self.data["gpu"] and self.data["gpu"][0]["gpus"]:
            gpu_stats = {}
            for gpu_id in range(len(self.data["gpu"][0]["gpus"])):
                gpu_percentages = [d["gpus"][gpu_id]["percent"] for d in self.data["gpu"] if len(d["gpus"]) > gpu_id]
                gpu_stats[f"gpu_{gpu_id}"] = {
                    "avg_percent": sum(gpu_percentages) / len(gpu_percentages),
                    "max_percent": max(gpu_percentages),
                    "min_percent": min(gpu_percentages),
                    "std_dev": self._calculate_std_dev(gpu_percentages)
                }
            stats["gpu"] = gpu_stats
        
        return stats
    
    def _calculate_std_dev(self, values: List[float]) -> float:
        """计算标准差"""
        if len(values) < 2:
            return 0.0
        
        mean = sum(values) / len(values)
        variance = sum((x - mean) ** 2 for x in values) / (len(values) - 1)
        return variance ** 0.5
    
    def detect_memory_leaks(self) -> Dict:
        """检测内存泄漏"""
        if len(self.data["memory"]) < 10:
            return {"error": "Insufficient data for memory leak detection"}
        
        # 分析内存使用趋势
        memory_usage = [d["percent"] for d in self.data["memory"]]
        
        # 计算线性回归
        n = len(memory_usage)
        x_values = list(range(n))
        y_values = memory_usage
        
        # 计算斜率
        x_mean = sum(x_values) / n
        y_mean = sum(y_values) / n
        
        numerator = sum((x - x_mean) * (y - y_mean) for x, y in zip(x_values, y_values))
        denominator = sum((x - x_mean) ** 2 for x in x_values)
        
        if denominator == 0:
            slope = 0
        else:
            slope = numerator / denominator
        
        # 计算内存泄漏率 (百分比/小时)
        monitoring_duration_hours = (self.data["memory"][-1]["timestamp"] - 
                                   self.data["memory"][0]["timestamp"]) / 3600
        
        memory_leak_rate = slope / monitoring_duration_hours if monitoring_duration_hours > 0 else 0
        
        return {
            "memory_leak_rate_percent_per_hour": memory_leak_rate,
            "trend": "increasing" if slope > 0.1 else "stable" if slope > -0.1 else "decreasing",
            "severity": "high" if memory_leak_rate > 1.0 else "medium" if memory_leak_rate > 0.1 else "low"
        }
```

```python
# 稳定性测试框架
import time
import random
import logging
from datetime import datetime, timedelta
from typing import Dict, List, Callable

class StabilityTestFramework:
    def __init__(self, test_duration_hours: int = 24):
        self.test_duration = timedelta(hours=test_duration_hours)
        self.start_time = None
        self.end_time = None
        self.test_results = {
            "errors": [],
            "performance_degradation": [],
            "resource_leaks": [],
            "availability_events": []
        }

        file_handler.setLevel(logging.INFO)
        
        # 控制台处理器
        console_handler = logging.StreamHandler()
        console_handler.setLevel(logging.INFO)
        
        # 格式化器
        formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
        file_handler.setFormatter(formatter)
        console_handler.setFormatter(formatter)
        
        logger.addHandler(file_handler)
        logger.addHandler(console_handler)
        
        return logger
    
    def run_stability_test(self, test_functions: Dict[str, Callable]):
        """运行稳定性测试"""
        self.start_time = datetime.now()
        self.end_time = self.start_time + self.test_duration
        
        self.logger.info(f"开始稳定性测试，持续时间: {self.test_duration}")
        
        try:
            while datetime.now() < self.end_time:
                # 执行功能测试
                self._run_functional_tests(test_functions)
                
                # 执行性能测试
                self._run_performance_tests()
                
                # 执行资源监控
                self._run_resource_monitoring()
                
                # 执行可用性测试
                self._run_availability_tests()
                
                # 等待下一次测试周期
                time.sleep(60)  # 每分钟执行一次测试
                
        except KeyboardInterrupt:
            self.logger.info("稳定性测试被用户中断")
        except Exception as e:
            self.logger.error(f"稳定性测试发生错误: {str(e)}")
            self.test_results["errors"].append({
                "timestamp": datetime.now(),
                "error": str(e),
                "type": "system_error"
            })
        finally:
            self._generate_stability_report()
    
    def _run_functional_tests(self, test_functions: Dict[str, Callable]):
        """运行功能测试"""
        self.logger.info("执行功能测试")
        
        for test_name, test_func in test_functions.items():
            try:
                result = test_func()
                if not result.get("success", False):
                    self.test_results["errors"].append({
                        "timestamp": datetime.now(),
                        "test_name": test_name,
                        "error": result.get("error", "Unknown error"),
                        "type": "functional_error"
                    })
                    self.logger.warning(f"功能测试失败: {test_name}")
            except Exception as e:
                self.test_results["errors"].append({
                    "timestamp": datetime.now(),
                    "test_name": test_name,
                    "error": str(e),
                    "type": "test_execution_error"
                })
                self.logger.error(f"功能测试执行错误: {test_name} - {str(e)}")

        self.logger.info("执行资源监控")

            })

        if not service_availability["available"]:

                "details": service_availability
            })
            self.logger.warning("检测到服务不可用")
    
    def _measure_current_performance(self) -> Dict:

            "memory_usage": random.uniform(50, 85),
            "response_time": random.uniform(100, 1000),
            "throughput": random.uniform(50, 200)
        }
    
    def _detect_performance_degradation(self, current_performance: Dict) -> bool:
        """检测性能下降"""
        # 简化的性能下降检测逻辑
        return current_performance["response_time"] > 2000 or current_performance["throughput"] < 50
    
    def _calculate_degradation(self, current_performance: Dict) -> float:
        """计算性能下降百分比"""
        # 简化的性能下降计算
        return max(0, (current_performance["response_time"] - 1000) / 1000 * 100)
    
    def _detect_memory_leak(self) -> Dict:
        """检测内存泄漏"""
        # 简化的内存泄漏检测
        if random.random() < 0.01:  # 1%概率检测到内存泄漏
            return {
                "leak_rate": random.uniform(0.1, 5.0),  # MB/小时
                "severity": random.choice(["low", "medium", "high"])
            }
        return None
    
    def _detect_resource_exhaustion(self) -> Dict:
        """检测资源耗尽"""
        # 简化的资源耗尽检测
        if random.random() < 0.005:  # 0.5%概率检测到资源耗尽
            return {
                "resource_type": random.choice(["cpu", "memory", "disk", "network"]),
                "exhaustion_percentage": random.uniform(90, 100)
            }

        """检查服务可用性"""
        # 简化的服务可用性检查
        available = random.random() > 0.01  # 99%可用性
        return {
            "available": available,
            "response_time": random.uniform(10, 1000) if available else -1,
            "error_code": None if available else "SERVICE_UNAVAILABLE"
        }
    
    def _generate_stability_report(self):
        """生成稳定性测试报告"""
        test_duration = datetime.now() - self.start_time
        
        report = {
            "test_duration": str(test_duration),
            "total_errors": len(self.test_results["errors"]),
            "performance_degradation_events": len(self.test_results["performance_degradation"]),
            "resource_leak_events": len(self.test_results["resource_leaks"]),
            "availability_events": len(self.test_results["availability_events"]),
            "stability_score": self._calculate_stability_score(),
            "recommendations": self._generate_recommendations()
        }
        
        # 保存报告到文件
        with open(f"stability_report_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json", 'w') as f:
            import json
            json.dump(report, f, indent=2, default=str)
        
        self.logger.info("稳定性测试报告已生成")
    
    def _calculate_stability_score(self) -> float:
        """计算稳定性评分"""
        # 基于错误数量和类型计算稳定性评分
        base_score = 100.0
        
        # 功能错误扣分
        functional_errors = len([e for e in self.test_results["errors"] if e["type"] == "functional_error"])
        base_score -= functional_errors * 5.0
        
        # 性能下降扣分
        base_score -= len(self.test_results["performance_degradation"]) * 3.0
        
        # 资源泄漏扣分
        base_score -= len(self.test_results["resource_leaks"]) * 4.0
        
        # 可用性问题扣分
        base_score -= len(self.test_results["availability_events"]) * 10.0
        
        return max(0.0, base_score)
    
    def _generate_recommendations(self) -> List[str]:
        """生成改进建议"""
        recommendations = []
        
        if len(self.test_results["errors"]) > 0:
            recommendations.append("修复功能错误，提高系统可靠性")
        
        if len(self.test_results["performance_degradation"]) > 0:
            recommendations.append("优化性能，减少性能下降事件")
        
        if len(self.test_results["resource_leaks"]) > 0:
            recommendations.append("修复资源泄漏，提高资源利用率")
        
        if len(self.test_results["availability_events"]) > 0:
            recommendations.append("提高系统可用性，减少服务中断")
        
        return recommendations
```

```python
    async def run_async_load_test(self, test_config: Dict, duration_seconds: int = 300):
        """运行异步负载测试"""
        print(f"开始异步负载测试: {test_config['name']}")
        
        start_time = time.time()
        end_time = start_time + duration_seconds
        
        async with aiohttp.ClientSession() as session:
            tasks = []
            
            while time.time() < end_time:

                    task = asyncio.create_task(
                        self._execute_async_request(session, test_config, i)
                    )
                    tasks.append(task)
                
                # 等待部分任务完成
                if len(tasks) >= self.max_workers:
                    completed, pending = await asyncio.wait(
                        tasks, return_when=asyncio.FIRST_COMPLETED
                    )
                    tasks = list(pending)
                
                # 控制请求频率
                await asyncio.sleep(0.1)
            
            # 等待剩余任务完成
            if tasks:
                await asyncio.gather(*tasks, return_exceptions=True)

        else:
            return await self._send_audio_request(session, test_config, user_id)
    
    def run_sync_load_test(self, test_config: Dict, duration_seconds: int = 300):
        """运行同步负载测试"""
        print(f"开始同步负载测试: {test_config['name']}")
        
        start_time = time.time()
        end_time = start_time + duration_seconds
        
        def worker_task(user_id):
            """工作线程任务"""
            local_results = []
            
            while time.time() < end_time:
                result = self._execute_sync_request(test_config, user_id)
                local_results.append(result)
                
                # 控制请求频率
                time.sleep(0.1)
            
            return local_results
        
        # 执行并发测试
        all_results = []
        with ThreadPoolExecutor(max_workers=test_config["concurrent_users"]) as executor:
            futures = [executor.submit(worker_task, i) 
                      for i in range(test_config["concurrent_users"])]
            
            for future in as_completed(futures):
                results = future.result()
                all_results.extend(results)
        
        print(f"同步负载测试完成，总请求数: {len(all_results)}")
        return self._calculate_performance_metrics()
    
    def _execute_sync_request(self, test_config: Dict, user_id: int) -> Dict:
        """执行同步请求"""
        import requests
        
        request_start = time.time()
        
        try:
            # 根据测试配置发送请求
            if test_config["request_type"] == "text":
                text_data = random.choice(test_config["test_data"]["text_inputs"])
                response = requests.post(
                    f"{self.base_url}/api/text/process",
                    json={"text": text_data["text"], "user_id": user_id},
                    headers={"Content-Type": "application/json"},
                    timeout=30
                )
            else:
                # 其他类型请求
                response = requests.get(f"{self.base_url}/api/health", timeout=10)
            
            request_end = time.time()
            response_time = (request_end - request_start) * 1000
            
            return {
                "success": response.status_code == 200,
                "response_time": response_time,
                "status_code": response.status_code
            }
            
        except Exception as e:
            request_end = time.time()
            response_time = (request_end - request_start) * 1000
            
            return {
                "success": False,
                "response_time": response_time,
                "error": str(e)
            }
    
    def _calculate_performance_metrics(self) -> Dict:
        """计算性能指标"""
        if not self.results["response_times"]:
            return {"error": "No test results available"}
        
        response_times = [r["response_time"] for r in self.results["response_times"]]
        
        # 计算响应时间统计
        response_stats = {
            "min": min(response_times),
            "max": max(response_times),
            "mean": statistics.mean(response_times),
            "median": statistics.median(response_times),
            "p90": statistics.quantiles(response_times, n=10)[8],
            "p95": statistics.quantiles(response_times, n=20)[18],
            "p99": statistics.quantiles(response_times, n=100)[98]
        }
        
        # 计算吞吐量
        if self.results["response_times"]:
            duration = (self.results["response_times"][-1]["timestamp"] - 
                       self.results["response_times"][0]["timestamp"])
            throughput = len(self.results["response_times"]) / duration if duration > 0 else 0
        else:
            throughput = 0
        
        # 计算错误率
        total_requests = len(self.results["response_times"]) + len(self.results["errors"])
        error_rate = len(self.results["errors"]) / total_requests if total_requests > 0 else 0
        
        return {
            "response_time_stats": response_stats,
            "throughput": throughput,
            "error_rate": error_rate,
            "total_requests": total_requests,
            "successful_requests": len(self.results["response_times"]),
            "failed_requests": len(self.results["errors"])
        }
```

```python
# 性能测试结果分析器
import pandas as pd

        # 检查响应时间

        # 检查吞吐量

            assessment["performance_grade"] = "B"

    def _analyze_response_times(self, test_results: Dict) -> Dict:
        """分析响应时间"""
        if "response_time_stats" not in test_results:
            return {"error": "No response time data available"}
        
        stats = test_results["response_time_stats"]
        response_times = [r["response_time"] for r in test_results.get("raw_response_times", [])]
        
        if not response_times:
            return {"error": "No raw response time data available"}
        
        # 计算响应时间分布
        response_time_df = pd.DataFrame(response_times, columns=["response_time"])
        
        analysis = {

            "trend": self._calculate_trend(response_times),
            "outliers": self._detect_outliers(response_times),
            "comparison_with_baseline": self._compare_with_baseline(stats, "response_time")
        }
        
        return analysis
    
    def _calculate_distribution(self, values: List[float]) -> Dict:
        """计算数据分布"""
        if not values:
            return {}
        
        # 计算分位数
        percentiles = [10, 25, 50, 75, 90, 95, 99]
        distribution = {}
        
        for p in percentiles:
            distribution[f"p{p}"] = np.percentile(values, p)
        
        # 计算分布形状
        distribution["skewness"] = stats.skew(values)
        distribution["kurtosis"] = stats.kurtosis(values)
        
        return distribution
    
    def _calculate_trend(self, values: List[float]) -> Dict:
        """计算趋势"""
        if len(values) < 10:
            return {"error": "Insufficient data for trend analysis"}
        
        # 计算线性回归
        x = np.arange(len(values))
        y = np.array(values)
        
        slope, intercept, r_value, p_value, std_err = stats.linregress(x, y)
        
        return {
            "slope": slope,
            "intercept": intercept,
            "r_squared": r_value ** 2,
            "p_value": p_value,
            "trend": "increasing" if slope > 0 else "decreasing" if slope < 0 else "stable",
            "significance": "significant" if p_value < 0.05 else "not significant"
        }
    
    def _detect_outliers(self, values: List[float]) -> Dict:
        """检测异常值"""
        if not values:
            return {}
        
        # 使用IQR方法检测异常值
        q1 = np.percentile(values, 25)
        q3 = np.percentile(values, 75)
        iqr = q3 - q1
        
        lower_bound = q1 - 1.5 * iqr
        upper_bound = q3 + 1.5 * iqr
        
        outliers = [v for v in values if v < lower_bound or v > upper_bound]
        
        return {
            "count": len(outliers),
            "percentage": len(outliers) / len(values) * 100,
            "values": outliers[:10],  # 只返回前10个异常值
            "lower_bound": lower_bound,
            "upper_bound": upper_bound
        }
    
    def _compare_with_baseline(self, stats: Dict, metric_type: str) -> Dict:
        """与基线比较"""
        if metric_type not in self.performance_baselines:
            return {"error": f"Unknown metric type: {metric_type}"}
        
        baseline = self.performance_baselines[metric_type]
        comparison = {}
        
        # 比较平均值
        if "mean" in stats:
            mean_value = stats["mean"]
            comparison["mean_vs_target"] = mean_value / baseline["target"]
            comparison["mean_vs_max"] = mean_value / baseline["max"]
        
        # 比较百分位数
        for key in stats:
            if key.startswith("p") and key.replace("p", "").isdigit():
                percentile_value = stats[key]
                comparison[f"{key}_vs_target"] = percentile_value / baseline["target"]
                comparison[f"{key}_vs_max"] = percentile_value / baseline["max"]
        
        return comparison
```

```python
# 响应时间测量工具
import time
import statistics
from typing import List, Dict

class ResponseTimeMonitor:
    def __init__(self, module_name: str):
        self.module_name = module_name
        self.response_times = []
        self.percentiles = [50, 90, 95, 99]
    
    def measure_response_time(self, func, *args, **kwargs):
        """测量函数响应时间"""
        start_time = time.perf_counter()
        result = func(*args, **kwargs)
        end_time = time.perf_counter()
        
        response_time = (end_time - start_time) * 1000  # 转换为毫秒
        self.response_times.append(response_time)
        
        return result, response_time
    
    def get_statistics(self) -> Dict:
        """获取响应时间统计信息"""
        if not self.response_times:
            return {}
        
        sorted_times = sorted(self.response_times)
        
        stats = {
            "module": self.module_name,
            "count": len(self.response_times),
            "min": min(self.response_times),
            "max": max(self.response_times),
            "mean": statistics.mean(self.response_times),
            "median": statistics.median(self.response_times),
            "std_dev": statistics.stdev(self.response_times) if len(self.response_times) > 1 else 0
        }
        
        # 计算百分位数
        for percentile in self.percentiles:
            index = int(len(sorted_times) * percentile / 100)
            if index >= len(sorted_times):
                index = len(sorted_times) - 1
            stats[f"p{percentile}"] = sorted_times[index]
        
        return stats
    
    def export_report(self, filename: str):
        """导出测试报告"""
        stats = self.get_statistics()
        
        with open(filename, 'w') as f:
            f.write(f"响应时间测试报告 - {self.module_name}\n")
            f.write("=" * 50 + "\n")
            f.write(f"测试样本数: {stats['count']}\n")
            f.write(f"最小响应时间: {stats['min']:.2f}ms\n")
            f.write(f"最大响应时间: {stats['max']:.2f}ms\n")
            f.write(f"平均响应时间: {stats['mean']:.2f}ms\n")
            f.write(f"中位数响应时间: {stats['median']:.2f}ms\n")
            f.write(f"标准差: {stats['std_dev']:.2f}ms\n")
            
            for percentile in self.percentiles:
                f.write(f"P{percentile}响应时间: {stats[f'p{percentile}']:.2f}ms\n")
```

```python
import time
import threading
from concurrent.futures import ThreadPoolExecutor, as_completed
from typing import Callable, List, Dict

class ThroughputMonitor:
    def __init__(self, module_name: str):
        self.module_name = module_name
        self.request_count = 0
        self.success_count = 0
        self.error_count = 0
        self.start_time = None
        self.end_time = None
        self.lock = threading.Lock()
    
    def start_monitoring(self):
        """开始监控"""
        self.start_time = time.perf_counter()
        self.request_count = 0
        self.success_count = 0
        self.error_count = 0
    
    def stop_monitoring(self):
        """停止监控"""
        self.end_time = time.perf_counter()
    
    def record_request(self, success: bool = True):
        """记录请求"""
        with self.lock:
            self.request_count += 1
            if success:
                self.success_count += 1
            else:
                self.error_count += 1
    
    def get_throughput_stats(self) -> Dict:
        """获取吞吐量统计"""
        if not self.start_time or not self.end_time:
            return {}
        
        duration = self.end_time - self.start_time
        
        return {
            "module": self.module_name,
            "duration": duration,
            "total_requests": self.request_count,
            "successful_requests": self.success_count,
            "failed_requests": self.error_count,
            "success_rate": self.success_count / self.request_count if self.request_count > 0 else 0,
            "requests_per_second": self.request_count / duration if duration > 0 else 0,
            "successful_requests_per_second": self.success_count / duration if duration > 0 else 0
        }
    
    def load_test(self, test_function: Callable, concurrent_users: int = 10, 
                  test_duration: int = 60, requests_per_user: int = 100):
        """执行负载测试"""
        self.start_monitoring()
        
        def user_task(user_id):
            """用户任务"""
            local_success = 0
            local_error = 0
            
            for i in range(requests_per_user):
                try:
                    result = test_function(user_id, i)
                    local_success += 1
                except Exception as e:
                    local_error += 1
                
                # 控制请求频率
                time.sleep(test_duration / requests_per_user)
            
            return local_success, local_error
        
        # 执行并发测试
        with ThreadPoolExecutor(max_workers=concurrent_users) as executor:
            futures = [executor.submit(user_task, i) for i in range(concurrent_users)]
            
            for future in as_completed(futures):
                success, error = future.result()
                self.record_request(True)
                for _ in range(success - 1):
                    self.record_request(True)
                for _ in range(error):
                    self.record_request(False)
        
        self.stop_monitoring()
        return self.get_throughput_stats()
```

```python
# 资源利用率监控工具
import psutil
import GPUtil
import time
import threading
from typing import Dict, List

class ResourceMonitor:
    def __init__(self, monitoring_interval: float = 1.0):
        self.monitoring_interval = monitoring_interval
        self.monitoring = False
        self.data = {
            "cpu": [],
            "memory": [],
            "gpu": [],
            "disk_io": [],
            "network": []
        }
        self.monitor_thread = None
    
    def start_monitoring(self):
        """开始资源监控"""
        self.monitoring = True
        self.monitor_thread = threading.Thread(target=self._monitor_loop)
        self.monitor_thread.start()
    
    def stop_monitoring(self):
        """停止资源监控"""
        self.monitoring = False
        if self.monitor_thread:
            self.monitor_thread.join()
    
    def _monitor_loop(self):
        """监控循环"""
        while self.monitoring:
            # CPU利用率
            cpu_percent = psutil.cpu_percent(interval=None)
            self.data["cpu"].append({
                "timestamp": time.time(),
                "percent": cpu_percent,
                "per_cpu": psutil.cpu_percent(percpu=True)
            })
            
            # 内存利用率
            memory = psutil.virtual_memory()
            self.data["memory"].append({
                "timestamp": time.time(),
                "percent": memory.percent,
                "available": memory.available,
                "used": memory.used,
                "total": memory.total
            })
            
            # GPU利用率
            gpus = GPUtil.getGPUs()
            gpu_data = []
            for gpu in gpus:
                gpu_data.append({
                    "id": gpu.id,
                    "name": gpu.name,
                    "percent": gpu.load * 100,
                    "memory_percent": gpu.memoryUtil * 100,
                    "memory_used": gpu.memoryUsed,
                    "memory_total": gpu.memoryTotal
                })
            
            self.data["gpu"].append({
                "timestamp": time.time(),
                "gpus": gpu_data
            })
            
            # 磁盘I/O
            disk_io = psutil.disk_io_counters()
            self.data["disk_io"].append({
                "timestamp": time.time(),
                "read_bytes": disk_io.read_bytes,
                "write_bytes": disk_io.write_bytes,
                "read_count": disk_io.read_count,
                "write_count": disk_io.write_count,
                "read_time": disk_io.read_time,
                "write_time": disk_io.write_time
            })
            
            # 网络I/O
            network_io = psutil.net_io_counters()
            self.data["network"].append({
                "timestamp": time.time(),
                "bytes_sent": network_io.bytes_sent,
                "bytes_recv": network_io.bytes_recv,
                "packets_sent": network_io.packets_sent,
                "packets_recv": network_io.packets_recv,
                "errin": network_io.errin,
                "errout": network_io.errout,
                "dropin": network_io.dropin,
                "dropout": network_io.dropout
            })
            
            time.sleep(self.monitoring_interval)
    
    def get_resource_statistics(self) -> Dict:
        """获取资源利用率统计"""
        stats = {}
        
        # CPU统计
        if self.data["cpu"]:
            cpu_percentages = [d["percent"] for d in self.data["cpu"]]
            stats["cpu"] = {
                "avg_percent": sum(cpu_percentages) / len(cpu_percentages),
                "max_percent": max(cpu_percentages),
                "min_percent": min(cpu_percentages),
                "std_dev": self._calculate_std_dev(cpu_percentages)
            }
        
        # 内存统计
        if self.data["memory"]:
            memory_percentages = [d["percent"] for d in self.data["memory"]]
            stats["memory"] = {
                "avg_percent": sum(memory_percentages) / len(memory_percentages),
                "max_percent": max(memory_percentages),
                "min_percent": min(memory_percentages),
                "std_dev": self._calculate_std_dev(memory_percentages)
            }
        
        # GPU统计
        if self.data["gpu"] and self.data["gpu"][0]["gpus"]:
            gpu_stats = {}
            for gpu_id in range(len(self.data["gpu"][0]["gpus"])):
                gpu_percentages = [d["gpus"][gpu_id]["percent"] for d in self.data["gpu"] if len(d["gpus"]) > gpu_id]
                gpu_stats[f"gpu_{gpu_id}"] = {
                    "avg_percent": sum(gpu_percentages) / len(gpu_percentages),
                    "max_percent": max(gpu_percentages),
                    "min_percent": min(gpu_percentages),
                    "std_dev": self._calculate_std_dev(gpu_percentages)
                }
            stats["gpu"] = gpu_stats
        
        return stats
    
    def _calculate_std_dev(self, values: List[float]) -> float:
        """计算标准差"""
        if len(values) < 2:
            return 0.0
        
        mean = sum(values) / len(values)
        variance = sum((x - mean) ** 2 for x in values) / (len(values) - 1)
        return variance ** 0.5
    
    def detect_memory_leaks(self) -> Dict:
        """检测内存泄漏"""
        if len(self.data["memory"]) < 10:
            return {"error": "Insufficient data for memory leak detection"}
        
        # 分析内存使用趋势
        memory_usage = [d["percent"] for d in self.data["memory"]]
        
        # 计算线性回归
        n = len(memory_usage)
        x_values = list(range(n))
        y_values = memory_usage
        
        # 计算斜率
        x_mean = sum(x_values) / n
        y_mean = sum(y_values) / n
        
        numerator = sum((x - x_mean) * (y - y_mean) for x, y in zip(x_values, y_values))
        denominator = sum((x - x_mean) ** 2 for x in x_values)
        
        if denominator == 0:
            slope = 0
        else:
            slope = numerator / denominator
        
        # 计算内存泄漏率 (百分比/小时)
        monitoring_duration_hours = (self.data["memory"][-1]["timestamp"] - 
                                   self.data["memory"][0]["timestamp"]) / 3600
        
        memory_leak_rate = slope / monitoring_duration_hours if monitoring_duration_hours > 0 else 0
        
        return {
            "memory_leak_rate_percent_per_hour": memory_leak_rate,
            "trend": "increasing" if slope > 0.1 else "stable" if slope > -0.1 else "decreasing",
            "severity": "high" if memory_leak_rate > 1.0 else "medium" if memory_leak_rate > 0.1 else "low"
        }
```

```python
import time
import random
import logging
from datetime import datetime, timedelta
from typing import Dict, List, Callable

class StabilityTestFramework:
    def __init__(self, test_duration_hours: int = 24):
```

```python
import time
import asyncio
import aiohttp

from concurrent.futures import ThreadPoolExecutor, as_completed
from typing import Dict, List, Any

class PerformanceTestExecutor:
    def __init__(self, base_url: str, max_workers: int = 50):
        self.base_url = base_url

        self.results = {

    async def run_async_load_test(self, test_config: Dict, duration_seconds: int = 300):
        """运行异步负载测试"""
        print(f"开始异步负载测试: {test_config['name']}")
        
        start_time = time.time()
        end_time = start_time + duration_seconds
        
        async with aiohttp.ClientSession() as session:
            tasks = []
            
            while time.time() < end_time:
                # 创建测试任务
                for i in range(test_config["concurrent_users"]):

                        self._execute_async_request(session, test_config, i)
                    )

            # 等待剩余任务完成
            if tasks:
                await asyncio.gather(*tasks, return_exceptions=True)
        
        print(f"负载测试完成，耗时: {time.time() - start_time:.2f}秒")
        return self._calculate_performance_metrics()

                "response_time": response_time,
                "status": response.status,

        """运行同步负载测试"""
        print(f"开始同步负载测试: {test_config['name']}")
        
        start_time = time.time()
        end_time = start_time + duration_seconds
        
        def worker_task(user_id):
            """工作线程任务"""
            local_results = []
            
            while time.time() < end_time:
                result = self._execute_sync_request(test_config, user_id)
                local_results.append(result)
                
                # 控制请求频率
                time.sleep(0.1)
            
            return local_results
        
        # 执行并发测试
        all_results = []
        with ThreadPoolExecutor(max_workers=test_config["concurrent_users"]) as executor:
            futures = [executor.submit(worker_task, i) 
                      for i in range(test_config["concurrent_users"])]
            
            for future in as_completed(futures):
                results = future.result()
                all_results.extend(results)
        
        print(f"同步负载测试完成，总请求数: {len(all_results)}")
        return self._calculate_performance_metrics()

        """执行同步请求"""

        request_start = time.time()
        
        try:
            # 根据测试配置发送请求

                response = requests.post(

                    headers={"Content-Type": "application/json"},
                    timeout=30
                )

                # 其他类型请求
                response = requests.get(f"{self.base_url}/api/health", timeout=10)
            
            request_end = time.time()
            response_time = (request_end - request_start) * 1000
            
            return {
                "success": response.status_code == 200,
                "response_time": response_time,
                "status_code": response.status_code
            }
            
        except Exception as e:
            request_end = time.time()
            response_time = (request_end - request_start) * 1000
            
            return {
                "success": False,
                "response_time": response_time,
                "error": str(e)
            }
    
    def _calculate_performance_metrics(self) -> Dict:
        """计算性能指标"""
        if not self.results["response_times"]:
            return {"error": "No test results available"}
        
        response_times = [r["response_time"] for r in self.results["response_times"]]
        
        # 计算响应时间统计
        response_stats = {
            "min": min(response_times),
            "max": max(response_times),
            "mean": statistics.mean(response_times),
            "median": statistics.median(response_times),
            "p90": statistics.quantiles(response_times, n=10)[8],
            "p95": statistics.quantiles(response_times, n=20)[18],
            "p99": statistics.quantiles(response_times, n=100)[98]
        }
        
        # 计算吞吐量
        if self.results["response_times"]:
            duration = (self.results["response_times"][-1]["timestamp"] - 
                       self.results["response_times"][0]["timestamp"])
            throughput = len(self.results["response_times"]) / duration if duration > 0 else 0
        else:
            throughput = 0
        
        # 计算错误率
        total_requests = len(self.results["response_times"]) + len(self.results["errors"])
        error_rate = len(self.results["errors"]) / total_requests if total_requests > 0 else 0
        
        return {
            "response_time_stats": response_stats,
            "throughput": throughput,
            "error_rate": error_rate,
            "total_requests": total_requests,
            "successful_requests": len(self.results["response_times"]),
            "failed_requests": len(self.results["errors"])
        }
```

import pandas as pd

            "meets_requirements": True,
            "performance_grade": "A",
            "critical_issues": [],
            "warnings": []
        }
        
        # 检查响应时间
        if "response_time_stats" in test_results:
            p95_time = test_results["response_time_stats"].get("p95", 0)
            if p95_time > self.performance_baselines["response_time"]["max"]:
                assessment["meets_requirements"] = False
                assessment["critical_issues"].append(f"P95响应时间超标: {p95_time}ms")
            elif p95_time > self.performance_baselines["response_time"]["target"]:
                assessment["warnings"].append(f"P95响应时间接近上限: {p95_time}ms")
        
        # 检查错误率
        if "error_rate" in test_results:
            error_rate = test_results["error_rate"]
            if error_rate > self.performance_baselines["error_rate"]["max"]:
                assessment["meets_requirements"] = False
                assessment["critical_issues"].append(f"错误率超标: {error_rate:.2%}")
            elif error_rate > self.performance_baselines["error_rate"]["target"]:
                assessment["warnings"].append(f"错误率偏高: {error_rate:.2%}")
        
        # 检查吞吐量
        if "throughput" in test_results:
            throughput = test_results["throughput"]
            if throughput < self.performance_baselines["throughput"]["min"]:
                assessment["meets_requirements"] = False
                assessment["critical_issues"].append(f"吞吐量不足: {throughput:.2f} req/s")
        
        # 确定性能等级
        if assessment["meets_requirements"] and len(assessment["warnings"]) == 0:
            assessment["performance_grade"] = "A"
        elif assessment["meets_requirements"]:
            assessment["performance_grade"] = "B"
        elif len(assessment["critical_issues"]) <= 2:
            assessment["performance_grade"] = "C"

            "distribution": self._calculate_distribution(response_times),
            "trend": self._calculate_trend(response_times),
