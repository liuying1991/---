# 肢体动作生成模块实现工作流

## 模块信息

### 模块名称
肢体动作生成模块

### 所属子系统
交互表达子系统

### 模块功能描述
肢体动作生成模块负责根据文本内容、情感色彩和交互场景，生成适合婴儿认知和情感需求的虚拟角色肢体动作。该模块能够创建生动、友好、富有表现力的肢体动作，增强与婴儿的交互体验，促进婴儿的情感发展和认知能力提升。

### 模块职责
1. 动作生成：根据内容和情感生成相应的肢体动作
2. 情感表达：在肢体动作中融入适当的情感色彩
3. 年龄适配：根据婴儿年龄调整动作特征
4. 动态动作：生成动态变化的动作序列
5. 个性化动作：根据婴儿偏好生成个性化动作
6. 动作优化：优化动作质量和自然度

### 模块接口
- **输入接口**：文本内容、情感参数、婴儿状态
- **输出接口**：动作数据、动作元数据
- **依赖模块**：文本分析模块、情感识别模块
- **被依赖模块**：显示输出模块、交互管理模块

### 优先级

```python
# src/action_generation/action_generation_manager.py
import time
from typing import Dict, Any, List, Optional, Tuple
from dataclasses import dataclass

from ..content_analysis.content_analyzer import ContentAnalyzer
from ..action_generation.base_action_generator import BaseActionGenerator
from ..action_generation.emotion_action_generator import EmotionActionGenerator
from ..action_generation.dynamic_action_generator import DynamicActionGenerator
from ..age_adaptation.age_adapter import AgeAdapter
from ..personalization.personalization_adapter import PersonalizationAdapter
from ..action_optimization.action_optimizer import ActionOptimizer
from ..quality_assessment.quality_assessor import QualityAssessor
from ..utils.config import ActionGenerationConfig

@dataclass
class GenerationRequest:
    """动作生成请求"""
    text: str
    generation_params: Dict[str, Any]
    baby_state: Dict[str, Any]
    request_id: str
    timestamp: float

@dataclass
class GenerationResult:
    """动作生成结果"""
    text: str
    action: Dict[str, Any]
    metadata: Dict[str, Any]
    quality_score: float
    generation_time: float
    request_id: str
    timestamp: float

class ActionGenerationManager:
    """动作生成管理器"""
    
    def __init__(self, config: ActionGenerationConfig):
        self.config = config
        self.content_analyzer = ContentAnalyzer()
        self.base_action_generator = BaseActionGenerator(config)
        self.emotion_action_generator = EmotionActionGenerator(config)
        self.dynamic_action_generator = DynamicActionGenerator(config)
        self.age_adapter = AgeAdapter(config)
        self.personalization_adapter = PersonalizationAdapter(config)
        self.action_optimizer = ActionOptimizer(config)
        self.quality_assessor = QualityAssessor(config)
        
        # 初始化模型
        self._initialize_models()
    
    def _initialize_models(self):
        """初始化模型"""
        self.base_action_generator.load_model(self.config.action_model_path)
        self.emotion_action_generator.load_model(self.config.emotion_model_path)
        self.quality_assessor.load_model(self.config.quality_model_path)
    
    def generate_action(self, request: GenerationRequest) -> GenerationResult:
        """生成动作"""
        start_time = time.time()
        
        # 1. 内容分析
        content_analysis = self.content_analyzer.analyze(request.text)
        
        # 2. 基础动作生成
        base_action = self.base_action_generator.generate(
            request.text,
            content_analysis,
            request.generation_params
        )
        
        # 3. 情感动作生成
        if request.generation_params.get("apply_emotion", True):
            emotion_action = self.emotion_action_generator.add_emotion(
                base_action,
                content_analysis["emotion"],
                request.generation_params.get("emotion_intensity", 0.8)
            )
        else:
            emotion_action = base_action
        
        # 4. 动态动作生成
        if request.generation_params.get("generate_dynamic", False):
            dynamic_action = self.dynamic_action_generator.generate_dynamic(
                emotion_action,
                content_analysis,
                request.generation_params.get("animation_params", {})
            )
        else:
            dynamic_action = emotion_action
        
        # 5. 年龄适配
        age_group = self._determine_age_group(request.baby_state.get("age", None))
        age_adapted_action = self.age_adapter.adapt_action(
            dynamic_action,
            age_group,
            self.config.age_groups[age_group]
        )
        
        # 6. 个性化适配
        if self.config.personalization_enabled:
            personalized_action = self.personalization_adapter.adapt_action(
                age_adapted_action,
                request.baby_state.get("user_id", None),
                content_analysis
            )
        else:
            personalized_action = age_adapted_action
        
        # 7. 动作优化
        optimized_action = self.action_optimizer.optimize(
            personalized_action,
            request.generation_params
        )
        
        # 8. 质量评估
        quality_score = self.quality_assessor.assess_quality(
            optimized_action,
            content_analysis
        )
        
        # 9. 质量检查和重试
        if quality_score < self.config.min_quality_score:
            # 尝试调整生成参数并重新生成
            adjusted_params = self._adjust_generation_params(
                request.generation_params,
                quality_score
            )
            
            # 创建新的请求
            retry_request = GenerationRequest(
                text=request.text,
                generation_params=adjusted_params,
                baby_state=request.baby_state,
                request_id=f"{request.request_id}_retry",
                timestamp=time.time()
            )
            
            # 递归调用，但限制重试次数
            if adjusted_params.get("retry_count", 0) < self.config.max_generation_attempts:
                adjusted_params["retry_count"] = adjusted_params.get("retry_count", 0) + 1
                return self.generate_action(retry_request)
        
        generation_time = time.time() - start_time
        
        return GenerationResult(
            text=request.text,
            action=optimized_action,
            metadata={
                "emotion": content_analysis["emotion"],
                "age_group": age_group,
                "user_id": request.baby_state.get("user_id", None),
                "is_dynamic": request.generation_params.get("generate_dynamic", False),
                "quality_score": quality_score
            },
            quality_score=quality_score,
            generation_time=generation_time,
            request_id=request.request_id,
            timestamp=time.time()
        )
    
    def _determine_age_group(self, age_months: Optional[int]) -> str:
        """确定年龄组"""
        if age_months is None:
            return "12-24_months"  # 默认年龄组
        
        if age_months <= 6:
            return "0-6_months"
        elif age_months <= 12:
            return "6-12_months"
        else:
            return "12-24_months"
    
    def _adjust_generation_params(self, params: Dict[str, Any], quality_score: float) -> Dict[str, Any]:
        """调整生成参数"""
        adjusted_params = params.copy()
        
        # 根据质量分数调整参数
        if quality_score < 3.0:
            # 质量很低，大幅调整参数
            adjusted_params["action_amplitude"] = min(1.0, adjusted_params.get("action_amplitude", 0.8) + 0.2)
            adjusted_params["action_speed"] = max(0.5, adjusted_params.get("action_speed", 0.8) - 0.2)
            adjusted_params["transition_smoothness"] = min(1.0, adjusted_params.get("transition_smoothness", 0.8) + 0.2)
        elif quality_score < 3.5:
            # 质量较低，小幅调整参数
            adjusted_params["action_amplitude"] = min(1.0, adjusted_params.get("action_amplitude", 0.8) + 0.1)
            adjusted_params["transition_smoothness"] = min(1.0, adjusted_params.get("transition_smoothness", 0.8) + 0.1)
        
        return adjusted_params
```

```python
# src/action_generation/base_action_generator.py
import os
import json
import torch
import numpy as np
from typing import Dict, Any, List, Optional

from ..utils.config import ActionGenerationConfig
from ..utils.motion_utils import MotionUtils

class BaseActionGenerator:
    """基础动作生成器"""
    
    def __init__(self, config: ActionGenerationConfig):
        self.config = config
        self.motion_utils = MotionUtils()
        self.action_model = None
        self.action_templates = {}
        
        # 加载动作模板
        self._load_action_templates()
    
    def load_model(self, model_path: str):
        """加载动作生成模型"""
        if os.path.exists(model_path):
            self.action_model = torch.load(model_path)
            self.action_model.eval()
        else:
            raise FileNotFoundError(f"Action model not found at {model_path}")
    
    def _load_action_templates(self):
        """加载动作模板"""
        basic_action_path = self.config.basic_action_library_path
        
        # 加载基础动作模板
        for file_name in os.listdir(basic_action_path):
            if file_name.endswith('.json'):
                action_name = file_name[:-5]  # 去掉.json后缀
                with open(os.path.join(basic_action_path, file_name), 'r') as f:
                    self.action_templates[action_name] = json.load(f)
    
    def generate(self, text: str, content_analysis: Dict[str, Any], 
                generation_params: Dict[str, Any]) -> Dict[str, Any]:
        """生成基础动作"""
        # 确定基础动作类型
        action_type = self._determine_action_type(content_analysis)
        
        # 获取动作模板
        template = self.action_templates.get(action_type, self.action_templates["wave"])
        
        # 生成基础动作
        base_action = self._generate_base_action(template, text, generation_params)
        
        return base_action
    
    def _determine_action_type(self, content_analysis: Dict[str, Any]) -> str:
        """根据内容分析确定基础动作类型"""
        # 根据内容类型确定动作类型
        content_type = content_analysis.get("content_type", [])
        
        if "greeting" in content_type:
            return "wave"
        elif "farewell" in content_type:
            return "wave"
        elif "question" in content_type:
            return "nod"
        elif "affirmation" in content_type:
            return "thumbs_up"
        elif "celebration" in content_type:
            return "clap"
        elif "attention" in content_type:
            return "point"
        else:
            return "wave"  # 默认动作
    
    def _generate_base_action(self, template: Dict[str, Any], text: str, 
                             generation_params: Dict[str, Any]) -> Dict[str, Any]:
        """生成基础动作"""
        # 获取动作参数
        action_amplitude = generation_params.get("action_amplitude", 0.8)
        action_speed = generation_params.get("action_speed", 0.8)
        
        # 创建动作数据
        frames = []
        num_frames = int(template.get("duration", 2.0) * 30)  # 30fps
        
        # 生成动作帧
        for i in range(num_frames):
            # 计算当前帧的时间点
            t = i / num_frames
            
            # 根据模板生成关节位置
            joint_positions = self._generate_joint_positions(template, t, action_amplitude)
            
            # 创建帧数据
            frame = {
                "timestamp": t,
                "joint_positions": joint_positions,
                "joint_rotations": self._generate_joint_rotations(template, t, action_amplitude)
            }
            
            frames.append(frame)
        
        # 创建动作数据
        action_data = {
            "action_type": template["action_type"],
            "frames": frames,
            "metadata": {
                "duration": template.get("duration", 2.0),
                "amplitude": action_amplitude,
                "speed": action_speed,
                "description": template.get("description", ""),
                "source": "base_generator"
            }
        }
        
        return action_data
    
    def _generate_joint_positions(self, template: Dict[str, Any], t: float, 
                                 amplitude: float) -> List[List[float]]:
        """生成关节位置"""
        # 获取关节位置模板
        position_template = template.get("joint_positions", {})
        
        # 生成关节位置
        positions = []
        
        for joint_name, position_data in position_template.items():
            # 获取基础位置
            base_position = position_data.get("base", [0.0, 0.0, 0.0])
            
            # 获取运动数据
            motion_data = position_data.get("motion", {})
            
            # 计算当前位置
            position = self._calculate_joint_position(
                base_position, 
                motion_data, 
                t, 
                amplitude
            )
            
            positions.append(position)
        
        return positions
    
    def _generate_joint_rotations(self, template: Dict[str, Any], t: float, 
                                 amplitude: float) -> List[List[float]]:
        """生成关节旋转"""
        # 获取关节旋转模板
        rotation_template = template.get("joint_rotations", {})
        
        # 生成关节旋转
        rotations = []
        
        for joint_name, rotation_data in rotation_template.items():
            # 获取基础旋转
            base_rotation = rotation_data.get("base", [0.0, 0.0, 0.0, 1.0])  # 四元数
            
            # 获取运动数据
            motion_data = rotation_data.get("motion", {})
            
            # 计算当前旋转
            rotation = self._calculate_joint_rotation(
                base_rotation, 
                motion_data, 
                t, 
                amplitude
            )
            
            rotations.append(rotation)
        
        return rotations
    
    def _calculate_joint_position(self, base_position: List[float], 
                                  motion_data: Dict[str, Any], 
                                  t: float, amplitude: float) -> List[float]:
        """计算关节位置"""
        # 初始化位置
        position = base_position.copy()
        
        # 处理不同类型的运动
        for motion_type, motion_params in motion_data.items():
            if motion_type == "sinusoidal":
                # 正弦运动
                frequency = motion_params.get("frequency", 1.0)
                phase = motion_params.get("phase", 0.0)
                axis = motion_params.get("axis", [1.0, 0.0, 0.0])
                
                # 计算位移
                displacement = amplitude * np.sin(2 * np.pi * frequency * t + phase)
                
                # 应用到位移
                for i in range(3):
                    position[i] += displacement * axis[i]
            
            elif motion_type == "linear":
                # 线性运动
                start_pos = motion_params.get("start", [0.0, 0.0, 0.0])
                end_pos = motion_params.get("end", [0.0, 0.0, 0.0])
                
                # 计算插值
                for i in range(3):
                    position[i] += amplitude * (start_pos[i] + (end_pos[i] - start_pos[i]) * t)
        
        return position
    
    def _calculate_joint_rotation(self, base_rotation: List[float], 
                                 motion_data: Dict[str, Any], 
                                 t: float, amplitude: float) -> List[float]:
        """计算关节旋转"""
        # 初始化旋转
        rotation = base_rotation.copy()
        
        # 处理不同类型的运动
        for motion_type, motion_params in motion_data.items():
            if motion_type == "sinusoidal":
                # 正弦运动
                frequency = motion_params.get("frequency", 1.0)
                phase = motion_params.get("phase", 0.0)
                axis = motion_params.get("axis", [1.0, 0.0, 0.0])
                
                # 计算旋转角度
                angle = amplitude * np.sin(2 * np.pi * frequency * t + phase)
                
                # 创建旋转四元数
                rotation_quat = self.motion_utils.axis_angle_to_quaternion(axis, angle)
                
                # 应用旋转
                rotation = self.motion_utils.multiply_quaternions(rotation, rotation_quat)
        
        return rotation
```

```python
import requests

# 认证令牌

def generate_action(text, baby_age, baby_id):
```
