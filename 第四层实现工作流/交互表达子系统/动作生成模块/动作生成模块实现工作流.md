# src/core/manager.py
from typing import List, Dict, Optional

logger = logging.getLogger(__name__)

class ActionGenerationManager:
    """动作生成管理器"""
    
    def __init__(self, config: Dict):
        """初始化动作生成管理器"""
        self.config = config
        
        # 初始化策略层组件
        self.strategy_parser = StrategyParser(config.get('strategy', {}))
        
        # 初始化规划层组件
        self.sequence_planner = SequencePlanner(config.get('planner', {}))
        
        # 初始化生成层组件
        self.base_generator = BaseGenerator(config.get('generator', {}))
        self.detail_generator = DetailGenerator(config.get('generator', {}))
        self.action_optimizer = ActionOptimizer(config.get('generator', {}))
        
        # 初始化执行层组件
        self.command_converter = CommandConverter(config.get('executor', {}))
        self.action_coordinator = ActionCoordinator(config.get('executor', {}))
        self.execution_monitor = ExecutionMonitor(config.get('executor', {}))
        
        # 初始化机器学习模型
        self.generation_model = ActionGenerationModel(config.get('models', {}).get('generation', {}))
        self.optimization_model = ActionOptimizationModel(config.get('models', {}).get('optimization', {}))
        self.style_model = StyleAdaptationModel(config.get('models', {}).get('style', {}))
        
        logger.info("ActionGenerationManager initialized")
    
    def generate_action(self, request: Dict) -> Dict:
        """生成动作"""
        try:
            # 解析请求
            strategy_id = request.get('strategy_id')
            context = request.get('context', {})
            profile_id = request.get('profile_id')
            
            # 解析策略
            strategy = self.strategy_parser.parse_strategy(strategy_id)
            
            # 获取动作风格
            style = None
            if profile_id:
                style = self.style_model.get_style(profile_id)
            
            # 规划动作序列
            sequence = self.sequence_planner.plan_sequence(strategy, context)
            
            # 生成基础动作
            actions = []
            for action_plan in sequence.actions:
                base_action = self.base_generator.generate_base_action(
                    action_plan.action_type, 
                    action_plan.parameters
                )
                
                # 生成动作细节
                if style:
                    detailed_action = self.detail_generator.generate_details(
                        base_action, 
                        style
                    )
                else:
                    detailed_action = base_action
                
                # 优化动作
                optimized_action = self.action_optimizer.optimize_action(detailed_action)
                
                actions.append(optimized_action)
            
            # 转换为执行指令
            commands = self.command_converter.convert_to_commands(actions)
            
            # 协调动作
            coordinated_commands = self.action_coordinator.coordinate_actions(commands)
            
            # 返回结果
            result = {
                'action_sequence_id': sequence.sequence_id,
                'actions': [action.to_dict() for action in actions],
                'commands': [command.to_dict() for command in coordinated_commands],
                'style': style.to_dict() if style else None,
                'timestamp': datetime.now().isoformat()
            }
            
            logger.info(f"Generated action sequence: {sequence.sequence_id}")
            return result
            
        except Exception as e:
            logger.error(f"Error generating action: {str(e)}")
            raise
    
    def optimize_existing_action(self, action_id: str, optimization_params: Dict) -> Dict:
        """优化现有动作"""
        try:
            # 获取动作
            action = self.get_action_by_id(action_id)
            if not action:
                raise ValueError(f"Action not found: {action_id}")
            
            # 应用优化参数
            optimized_action = self.action_optimizer.optimize_action_with_params(
                action, 
                optimization_params
            )
            
            # 更新动作
            self.update_action(action_id, optimized_action)
            
            return {
                'action_id': action_id,
                'optimized_action': optimized_action.to_dict(),
                'timestamp': datetime.now().isoformat()
            }
            
        except Exception as e:
            logger.error(f"Error optimizing action {action_id}: {str(e)}")

#### 2. 动作序列规划器实现
```python
# src/core/planner/sequence_planner.py
from typing import List, Dict, Optional, Tuple
import numpy as np
from datetime import datetime, timedelta
import logging

from src.models.action_models import ActionSequence, ActionPlan, Action
from src.models.sequence_models import Transition
from src.core.strategy.strategy_parser import StrategyParser
from src.core.planner.action_selector import ActionSelector
from src.core.planner.param_calculator import ParamCalculator

logger = logging.getLogger(__name__)

class SequencePlanner:
    """动作序列规划器"""
    
    def __init__(self, config: Dict):
        """初始化序列规划器"""
        self.config = config
        self.action_selector = ActionSelector(config.get('selector', {}))
        self.param_calculator = ParamCalculator(config.get('calculator', {}))
        
        # 加载配置参数
        self.max_sequence_length = config.get('max_sequence_length', 10)
        self.default_action_duration = config.get('default_action_duration', 2.0)
        self.transition_duration = config.get('transition_duration', 0.5)
        
        logger.info("SequencePlanner initialized")
    
    def plan_sequence(self, strategy, context: Dict) -> ActionSequence:
        """规划动作序列"""
        try:
            # 创建序列ID
            sequence_id = f"seq_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
            
            # 解析策略和上下文
            strategy_info = self._parse_strategy(strategy)
            context_info = self._parse_context(context)
            
            # 确定序列长度
            sequence_length = min(
                strategy_info.get('preferred_length', 3),
                self.max_sequence_length
            )
            
            # 规划动作列表
            actions = []
            total_duration = 0
            
            for i in range(sequence_length):
                # 选择动作类型
                action_type = self.action_selector.select_action_type(
                    strategy_info, 
                    context_info, 
                    actions
                )
                
                # 计算动作参数
                action_params = self.param_calculator.calculate_parameters(
                    action_type, 
                    strategy_info, 
                    context_info,
                    i,
                    sequence_length
                )
                
                # 创建动作计划
                action_plan = ActionPlan(
                    action_type=action_type,
                    parameters=action_params,
                    start_time=total_duration,
                    duration=self._calculate_action_duration(action_type, action_params)
                )
                
                actions.append(action_plan)
                total_duration += action_plan.duration
                
                # 添加转场时间
                if i < sequence_length - 1:
                    total_duration += self.transition_duration
            
            # 创建转场列表
            transitions = self._create_transitions(actions)
            
            # 创建动作序列
            sequence = ActionSequence(
                sequence_id=sequence_id,
                sequence_name=f"Generated_{sequence_id}",
                actions=actions,
                duration=total_duration,
                transitions=transitions,
                metadata={
                    'strategy_id': strategy.strategy_id,
                    'context': context_info,
                    'created_at': datetime.now().isoformat()
                }
            )
            
            logger.info(f"Planned action sequence: {sequence_id} with {len(actions)} actions")
            return sequence
            
        except Exception as e:
            logger.error(f"Error planning sequence: {str(e)}")
            raise
    
    def _parse_strategy(self, strategy) -> Dict:
        """解析策略信息"""
        return {
            'strategy_id': strategy.strategy_id,
            'strategy_type': strategy.strategy_type,
            'action_types': strategy.action_types,
            'action_params': strategy.action_params,
            'constraints': strategy.constraints,
            'priority': strategy.priority,
            'preferred_length': strategy.action_params.get('sequence_length', 3)
        }
    
    def _parse_context(self, context: Dict) -> Dict:
        """解析上下文信息"""
        return {
            'baby_state': context.get('baby_state', {}),
            'environment': context.get('environment', {}),
            'interaction_history': context.get('interaction_history', []),
            'time_of_day': context.get('time_of_day', 'day'),
            'interaction_type': context.get('interaction_type', 'general')
        }
    
    def _calculate_action_duration(self, action_type: str, params: Dict) -> float:
        """计算动作持续时间"""
        # 根据动作类型和参数计算持续时间
        base_duration = self.default_action_duration
        
        # 根据动作类型调整
        if action_type in ['wave', 'nod']:
            base_duration *= 0.8
        elif action_type in ['hug', 'pat']:
            base_duration *= 1.2
        
        # 根据参数调整
        intensity = params.get('intensity', 1.0)
        base_duration *= (1.0 + (intensity - 1.0) * 0.3)
        
        return base_duration
    
    def _create_transitions(self, actions: List[ActionPlan]) -> List[Transition]:
        """创建转场列表"""
        transitions = []
        
        for i in range(len(actions) - 1):
            transition = Transition(
                from_action=actions[i].action_type,
                to_action=actions[i+1].action_type,
                duration=self.transition_duration,
                transition_type='smooth',
                parameters={}
            )
            transitions.append(transition)
        
        return transitions

# src/core/generator/base_generator.py
from typing import Dict, List, Optional
import numpy as np
import logging

from src.models.action_models import Action, Keyframe, Pose, Velocity, Acceleration
from src.ml.action_generation_model import ActionGenerationModel

logger = logging.getLogger(__name__)

class BaseGenerator:
    """基础动作生成器"""
    
    def __init__(self, config: Dict):
        """初始化基础动作生成器"""
        self.config = config
        
        # 初始化动作生成模型
        self.generation_model = ActionGenerationModel(config.get('model', {}))
        
        # 加载动作模板
        self.action_templates = self._load_action_templates()
        
        # 加载配置参数
        self.default_fps = config.get('default_fps', 30)
        self.default_keyframe_count = config.get('default_keyframe_count', 5)
        
        logger.info("BaseGenerator initialized")
    
    def generate_base_action(self, action_type: str, params: Dict) -> Action:
        """生成基础动作"""
        try:
            # 获取动作模板
            template = self.action_templates.get(action_type)
            if not template:
                raise ValueError(f"Unknown action type: {action_type}")
            
            # 创建动作ID
            action_id = f"action_{action_type}_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
            
            # 计算动作参数
            duration = params.get('duration', template['default_duration'])
            intensity = params.get('intensity', 1.0)
            speed = params.get('speed', 1.0)
            
            # 生成关键帧
            keyframes = self._generate_keyframes(
                action_type, 
                duration, 
                intensity, 
                speed,
                params
            )
            
            # 创建动作
            action = Action(
                action_id=action_id,
                action_name=f"Generated_{action_type}",
                action_type=action_type,
                body_parts=template['body_parts'],
                parameters=params,
                keyframes=keyframes,
                constraints=template['constraints'],
                style=None,  # 将在detail_generator中设置
                created_at=datetime.now(),
                updated_at=datetime.now()
            )
            
            logger.info(f"Generated base action: {action_id} of type {action_type}")
            return action
            
        except Exception as e:
            logger.error(f"Error generating base action: {str(e)}")
            raise
    
    def _load_action_templates(self) -> Dict:
        """加载动作模板"""
        # 这里简化实现，实际应从配置文件或数据库加载
        return {
            'wave': {
                'default_duration': 2.0,
                'body_parts': ['right_arm', 'left_arm'],
                'constraints': {
                    'max_angle': 120,
                    'min_angle': -120,
                    'max_speed': 180
                }
            },
            'nod': {
                'default_duration': 1.5,
                'body_parts': ['head'],
                'constraints': {
                    'max_angle': 30,
                    'min_angle': -30,
                    'max_speed': 90
                }
            },
            'hug': {
                'default_duration': 3.0,
                'body_parts': ['right_arm', 'left_arm', 'torso'],
                'constraints': {
                    'max_distance': 50,
                    'min_distance': 0,
                    'max_speed': 30
                }
            },
            'pat': {
                'default_duration': 1.0,
                'body_parts': ['right_hand', 'left_hand'],
                'constraints': {
                    'max_force': 5,
                    'min_force': 0.5,
                    'max_speed': 60
                }
            }
        }
    
    def _generate_keyframes(
        self, 
        action_type: str, 
        duration: float, 
        intensity: float, 
        speed: float,
        params: Dict
    ) -> List[Keyframe]:
        """生成关键帧"""
        # 计算关键帧数量
        keyframe_count = max(
            int(duration * self.default_fps / 6),  # 每6帧一个关键帧
            3  # 至少3个关键帧
        )
        
        # 生成时间戳
        timestamps = np.linspace(0, duration, keyframe_count)
        
        # 根据动作类型生成关键帧
        if action_type == 'wave':
            return self._generate_wave_keyframes(timestamps, intensity, speed, params)
        elif action_type == 'nod':
            return self._generate_nod_keyframes(timestamps, intensity, speed, params)
        elif action_type == 'hug':
            return self._generate_hug_keyframes(timestamps, intensity, speed, params)
        elif action_type == 'pat':
            return self._generate_pat_keyframes(timestamps, intensity, speed, params)
        else:
            # 使用模型生成通用动作
            return self._generate_model_keyframes(action_type, timestamps, intensity, speed, params)
    
    def _generate_wave_keyframes(
        self, 
        timestamps: np.ndarray, 
        intensity: float, 
        speed: float,
        params: Dict
    ) -> List[Keyframe]:
        """生成挥手动作关键帧"""
        keyframes = []
        
        # 挥手参数
        arm = params.get('arm', 'right')
        wave_amplitude = 45 * intensity  # 挥手幅度
        wave_frequency = 2 * speed      # 挥手频率
        
        for i, t in enumerate(timestamps):
            # 计算挥手角度
            angle = wave_amplitude * np.sin(2 * np.pi * wave_frequency * t)
            
            # 创建姿态
            pose = Pose()
            if arm == 'right':
                pose.right_arm = {'angle': angle, 'position': None}
            else:
                pose.left_arm = {'angle': angle, 'position': None}
            
            # 计算速度和加速度
            velocity = Velocity()
            acceleration = Acceleration()
            
            # 创建关键帧
            keyframe = Keyframe(
                timestamp=t,
                pose=pose,
                velocity=velocity,
                acceleration=acceleration,
                smoothness=0.8
            )
            
            keyframes.append(keyframe)
        
        return keyframes
    
    def _generate_nod_keyframes(
        self, 
        timestamps: np.ndarray, 
        intensity: float, 
        speed: float,
        params: Dict
    ) -> List[Keyframe]:
        """生成点头动作关键帧"""
        keyframes = []
        
        # 点头参数
        nod_amplitude = 15 * intensity  # 点头幅度
        nod_frequency = 1.5 * speed     # 点头频率
        
        for i, t in enumerate(timestamps):
            # 计算点头角度
            angle = nod_amplitude * np.sin(2 * np.pi * nod_frequency * t)
            
            # 创建姿态
            pose = Pose()
            pose.head = {'angle': angle, 'position': None}
            
            # 计算速度和加速度
            velocity = Velocity()
            acceleration = Acceleration()
            
            # 创建关键帧
            keyframe = Keyframe(
                timestamp=t,
                pose=pose,
                velocity=velocity,
                acceleration=acceleration,
                smoothness=0.9
            )
            
            keyframes.append(keyframe)
        
        return keyframes
    
    def _generate_hug_keyframes(
        self, 
        timestamps: np.ndarray, 
        intensity: float, 
        speed: float,
        params: Dict
    ) -> List[Keyframe]:
        """生成拥抱动作关键帧"""
        keyframes = []
        
        # 拥抱参数
        hug_distance = 30 * intensity  # 拥抱距离
        hug_speed = 10 * speed         # 拥抱速度
        
        # 计算拥抱时间点
        mid_time = timestamps[len(timestamps) // 2]
        
        for i, t in enumerate(timestamps):
            # 计算拥抱距离
            if t < mid_time:
                # 拥抱阶段
                distance = hug_distance * (t / mid_time)
            else:
                # 保持阶段
                distance = hug_distance
            
            # 创建姿态
            pose = Pose()
            pose.right_arm = {'position': {'x': -distance, 'y': 0, 'z': 0}}
            pose.left_arm = {'position': {'x': distance, 'y': 0, 'z': 0}}
            pose.torso = {'angle': 5}  # 轻微前倾
            
            # 计算速度和加速度
            velocity = Velocity()
            acceleration = Acceleration()
            
            # 创建关键帧
            keyframe = Keyframe(
                timestamp=t,
                pose=pose,
                velocity=velocity,
                acceleration=acceleration,
                smoothness=0.7
            )
            
            keyframes.append(keyframe)
        
        return keyframes
    
    def _generate_pat_keyframes(
        self, 
        timestamps: np.ndarray, 
        intensity: float, 
        speed: float,
        params: Dict
    ) -> List[Keyframe]:
        """生成拍打动作关键帧"""
        keyframes = []
        
        # 拍打参数
        pat_height = 10 * intensity  # 拍打高度
        pat_speed = 20 * speed       # 拍打速度
        pat_force = 2 * intensity    # 拍打力度
        
        # 计算拍打时间点
        pat_time = timestamps[len(timestamps) // 2]
        
        for i, t in enumerate(timestamps):
            # 计算拍打高度
            if t < pat_time:
                # 举起阶段
                height = pat_height * (t / pat_time)
            else:
                # 拍打阶段
                height = pat_height * (1 - (t - pat_time) / (timestamps[-1] - pat_time))
            
            # 创建姿态
            pose = Pose()
            pose.right_hand = {'position': {'x': 0, 'y': height, 'z': 0}, 'force': pat_force}
            
            # 计算速度和加速度
            velocity = Velocity()
            acceleration = Acceleration()
            
            # 创建关键帧
            keyframe = Keyframe(
                timestamp=t,
                pose=pose,
                velocity=velocity,
                acceleration=acceleration,
                smoothness=0.6
            )
            
            keyframes.append(keyframe)
        
        return keyframes
    
    def _generate_model_keyframes(
        self, 
        action_type: str, 
        timestamps: np.ndarray, 
        intensity: float, 
        speed: float,
        params: Dict
    ) -> List[Keyframe]:
        """使用模型生成通用动作关键帧"""
        # 准备模型输入
        model_input = {
            'action_type': action_type,
            'timestamps': timestamps.tolist(),
            'intensity': intensity,
            'speed': speed,
            'params': params
        }
        
        # 调用模型生成关键帧
        model_output = self.generation_model.generate(model_input)
        
        # 转换为关键帧对象
        keyframes = []
        for i, t in enumerate(timestamps):
            # 从模型输出提取姿态、速度、加速度
            pose_data = model_output['poses'][i]
            velocity_data = model_output['velocities'][i]
            acceleration_data = model_output['accelerations'][i]
            
            # 创建姿态
            pose = Pose()
            for part, data in pose_data.items():
                setattr(pose, part, data)
            
            # 创建速度
            velocity = Velocity()
            for part, data in velocity_data.items():
                setattr(velocity, part, data)
            
            # 创建加速度
            acceleration = Acceleration()
            for part, data in acceleration_data.items():
                setattr(acceleration, part, data)
            
            # 创建关键帧
            keyframe = Keyframe(
                timestamp=t,
                pose=pose,
                velocity=velocity,
                acceleration=acceleration,
                smoothness=model_output['smoothness'][i]
            )
            
            keyframes.append(keyframe)
        
        return keyframes
```

        self.assertLess(avg_latency, 100, "平均动作生成延迟应小于100ms")
        self.assertLess(max_latency, 200, "最大动作生成延迟应小于200ms")
    
    def test_concurrent_action_generation(self):
        """测试并发动作生成"""
        # 准备测试数据
        request = {
            'strategy_id': 'test_strategy',
            'context': {'baby_state': 'happy'},
            'profile_id': 'test_profile'
        }
        
        # 并发执行测试
        thread_count = 10
        latencies = []
        errors = []
        
        def generate_action():
            start_time = time.time()
            try:
                self.manager.generate_action(request)
                end_time = time.time()
                latencies.append((end_time - start_time) * 1000)  # 转换为毫秒
            except Exception as e:
                errors.append(str(e))
        
        # 启动线程
        threads = []
        start_time = time.time()
        
        for _ in range(thread_count):
            thread = threading.Thread(target=generate_action)
            threads.append(thread)
            thread.start()
        
        # 等待所有线程完成
        for thread in threads:
            thread.join()
        
        end_time = time.time()
        total_time = (end_time - start_time) * 1000  # 转换为毫秒
        
        # 计算统计信息
        avg_latency = statistics.mean(latencies) if latencies else 0
        throughput = thread_count / (total_time / 1000)  # 每秒请求数
        
        # 输出结果
        print(f"总时间: {total_time:.2f}ms")
        print(f"平均延迟: {avg_latency:.2f}ms")

        print(f"错误数: {len(errors)}")
        
        # 验证性能要求
        self.assertGreater(throughput, 10, "并发处理能力应不低于10个请求/秒")
        self.assertLess(avg_latency, 200, "并发平均延迟应小于200ms")

## 常见问题

### Q: 如何调整动作强度？

A: 可以通过以下方式调整动作强度：

A: 可以尝试以下方法：
1. 调整smoothness和naturalness参数
2. 选择更适合的动作风格
3. 检查上下文信息是否准确

### Q: 如何保存自定义动作风格？

A: 可以通过以下步骤：
1. 创建风格配置文件
2. 训练风格适应模型
3. 在个人档案中关联风格
