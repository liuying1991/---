```python
# 数据模型定义
from pydantic import BaseModel, Field
from typing import List, Dict, Optional, Any
from datetime import datetime
from enum import Enum

class MetricType(str, Enum):
    """指标类型枚举"""
    COUNTER = "counter"
    GAUGE = "gauge"
    HISTOGRAM = "histogram"
    SUMMARY = "summary"

class Severity(str, Enum):
    """严重程度枚举"""
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"
    CRITICAL = "critical"

class MonitoringMetric(BaseModel):
    """监控指标模型"""
    metric_id: str = Field(..., description="指标ID")
    metric_name: str = Field(..., description="指标名称")
    metric_type: MetricType = Field(..., description="指标类型")
    metric_value: float = Field(..., description="指标值")
    metric_unit: str = Field(..., description="指标单位")
    source: str = Field(..., description="数据源")
    tags: Dict[str, str] = Field(default_factory=dict, description="标签")
    timestamp: datetime = Field(default_factory=datetime.now, description="时间戳")

class AnomalyEvent(BaseModel):
    """异常事件模型"""
    event_id: str = Field(..., description="事件ID")
    event_type: str = Field(..., description="事件类型")
    severity: Severity = Field(..., description="严重程度")
    description: str = Field(..., description="事件描述")
    metrics: List[MonitoringMetric] = Field(..., description="相关指标")
    source: str = Field(..., description="数据源")
    detection_method: str = Field(..., description="检测方法")
    confidence: float = Field(..., ge=0.0, le=1.0, description="置信度")
    status: str = Field(default="active", description="状态")
    timestamp: datetime = Field(default_factory=datetime.now, description="时间戳")

class AlertEvent(BaseModel):
    """预警事件模型"""
    alert_id: str = Field(..., description="预警ID")
    alert_type: str = Field(..., description="预警类型")
    severity: Severity = Field(..., description="严重程度")
    title: str = Field(..., description="预警标题")
    message: str = Field(..., description="预警消息")
    metrics: List[MonitoringMetric] = Field(..., description="相关指标")
    source: str = Field(..., description="数据源")
    trigger_rule: str = Field(..., description="触发规则")
    status: str = Field(default="active", description="状态")
    notifications: List[Dict[str, Any]] = Field(default_factory=list, description="通知记录")
    timestamp: datetime = Field(default_factory=datetime.now, description="时间戳")

class SelfMonitoringConfig(BaseModel):
    """自我监控配置模型"""
    data_collector: Dict[str, Any] = Field(default_factory=dict, description="数据采集器配置")
    data_processor: Dict[str, Any] = Field(default_factory=dict, description="数据处理器配置")
    anomaly_detector: Dict[str, Any] = Field(default_factory=dict, description="异常检测器配置")
    alert_manager: Dict[str, Any] = Field(default_factory=dict, description="预警管理器配置")
    dashboard: Dict[str, Any] = Field(default_factory=dict, description="仪表板配置")
    anomaly_detection_models: Dict[str, Any] = Field(default_factory=dict, description="异常检测模型配置")
    metrics_db: Dict[str, Any] = Field(default_factory=dict, description="指标数据库配置")
    alerts_db: Dict[str, Any] = Field(default_factory=dict, description="预警数据库配置")
    config_db: Dict[str, Any] = Field(default_factory=dict, description="配置数据库配置")
    collection_interval: int = Field(default=60, description="采集间隔(秒)")
```

```python
# 事件驱动接口实现
import asyncio
from typing import Callable, Dict, Any

class SelfMonitoringEventHandler:
    """自我监控事件处理器"""
    
    def __init__(self, self_monitoring_service: SelfMonitoringManager):
        self.self_monitoring_service = self_monitoring_service
        self.event_bus = EventBus()
        self._register_handlers()
    
    def _register_handlers(self):
        """注册事件处理器"""
        self.event_bus.subscribe("system_metric_collected", self._handle_system_metric)
        self.event_bus.subscribe("anomaly_detected", self._handle_anomaly_detected)
        self.event_bus.subscribe("alert_triggered", self._handle_alert_triggered)
        self.event_bus.subscribe("config_updated", self._handle_config_updated)
    
    async def _handle_system_metric(self, event_data: Dict[str, Any]):
        """处理系统指标事件"""
        try:
            # 提取指标数据
            metric_data = event_data.get('metric')
            if not metric_data:
                logger.warning("系统指标事件缺少指标数据")
                return
            
            # 转换为监控指标模型
            metric = MonitoringMetric(**metric_data)
            
            # 存储指标
            await self.self_monitoring_service.store_metric(metric)
            
            logger.debug(f"处理系统指标事件: {metric.metric_name}={metric.metric_value}")
        except Exception as e:
            logger.error(f"处理系统指标事件失败: {str(e)}")
    
    async def _handle_anomaly_detected(self, event_data: Dict[str, Any]):
        """处理异常检测事件"""
        try:
            # 提取异常数据
            anomaly_data = event_data.get('anomaly')
            if not anomaly_data:
                logger.warning("异常检测事件缺少异常数据")
                return
            
            # 转换为异常事件模型
            anomaly = AnomalyEvent(**anomaly_data)
            
            # 存储异常
            await self.self_monitoring_service.store_anomaly(anomaly)
            
            # 触发预警检查
            await self.self_monitoring_service.check_alert_rules(anomaly)
            
            logger.debug(f"处理异常检测事件: {anomaly.event_type}, 严重程度: {anomaly.severity}")
        except Exception as e:
            logger.error(f"处理异常检测事件失败: {str(e)}")
    
    async def _handle_alert_triggered(self, event_data: Dict[str, Any]):
        """处理预警触发事件"""
        try:
            # 提取预警数据
            alert_data = event_data.get('alert')
            if not alert_data:
                logger.warning("预警触发事件缺少预警数据")
                return
            
            # 转换为预警事件模型
            alert = AlertEvent(**alert_data)
            
            # 存储预警
            await self.self_monitoring_service.store_alert(alert)
            
            # 发送通知
            await self.self_monitoring_service.send_notifications(alert)
            
            logger.debug(f"处理预警触发事件: {alert.alert_type}, 严重程度: {alert.severity}")
        except Exception as e:
            logger.error(f"处理预警触发事件失败: {str(e)}")
    
    async def _handle_config_updated(self, event_data: Dict[str, Any]):
        """处理配置更新事件"""
        try:
            # 提取配置数据
            config_data = event_data.get('config')
            config_type = event_data.get('config_type')
            
            if not config_data or not config_type:
                logger.warning("配置更新事件缺少必要数据")
                return
            
            # 更新配置
            await self.self_monitoring_service.update_config(config_type, config_data)
            
            logger.debug(f"处理配置更新事件: {config_type}")
        except Exception as e:
            logger.error(f"处理配置更新事件失败: {str(e)}")
```

```markdown
## 简介
自我监控模块是真实婴儿AI管家系统的自我意识子系统组件，负责实时监测和评估系统的运行状态、性能指标和行为模式。通过自我监控，系统能够及时发现异常情况并采取相应措施，确保系统稳定运行。

## 功能说明

### 系统状态监控
系统状态监控功能实时收集系统的运行状态信息，包括CPU使用率、内存使用率、磁盘使用率等关键指标。用户可以通过监控面板查看系统当前状态和历史趋势。

### 性能指标监控
性能指标监控功能收集和分析系统的性能数据，包括响应时间、吞吐量、错误率等。通过性能指标监控，用户可以了解系统的性能表现，及时发现性能瓶颈。

### 异常检测
异常检测功能使用机器学习算法自动检测系统中的异常情况。当检测到异常时，系统会自动记录异常信息，并根据配置的规则触发预警。

### 预警机制
预警机制在检测到异常或系统指标超过阈值时，通过多种方式通知用户，包括邮件、短信、系统通知等。用户可以配置不同的预警规则和通知方式。

## 常见问题

### Q1: 如何启动监控？
A1: 通过监控管理界面的"启动监控"按钮，或调用API的`/api/monitoring/start`接口。

### Q2: 如何创建预警规则？
A2: 进入"预警规则"页面，点击"新建规则"按钮，填写规则信息并保存。

### Q3: 如何查看历史监控数据？
A3: 在"监控数据"页面，选择时间范围和指标类型，点击"查询"按钮。
```
