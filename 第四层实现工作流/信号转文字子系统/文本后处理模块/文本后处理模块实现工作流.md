# 文本后处理模块实现工作流

## 模块信息

| 属性 | 内容 |
|------|------|
| 模块名称 | 文本后处理模块 |
| 所属子系统 | 信号转文字子系统 |
| 模块类型 | 数据处理模块 |
| 优先级 | 高 |
| 预估工期 | 4周 |
| 负责人 | 待分配 |
| 创建日期 | 2023-11-15 |
| 最后更新 | 2023-11-15 |

### 模块职责

1. 对多模态融合后的文本进行精细化处理和优化
2. 提高文本的可读性、准确性和语义连贯性
3. 实现文本纠错、语义补全、上下文优化、情感增强和格式规范
4. 适配婴儿语言特点，提供个性化文本处理

### 模块接口

| 接口名称 | 接口类型 | 输入 | 输出 | 描述 |
|----------|----------|------|------|------|
| process_text | REST API | 原始文本、处理选项 | 处理后文本、质量评分 | 综合文本处理 |
| correct_text | REST API | 原始文本、纠错选项 | 纠错后文本、错误列表 | 文本纠错 |
| complete_text | REST API | 不完整文本、补全选项 | 补全后文本、补全列表 | 语义补全 |
| enhance_sentiment | REST API | 原始文本、情感选项 | 增强后文本、情感信息 | 情感增强 |
| batch_process | REST API | 文本列表、处理选项 | 处理结果列表 | 批量处理 |
| stream_process | WebSocket | 文本流、处理选项 | 处理后文本流 | 流式处理 |

```python
# src/text_post_processing/core/manager.py
import time
import logging
from typing import Dict, List, Optional, Iterator
from dataclasses import dataclass

from ..models.text_models import TextProcessing, ProcessingOptions
from ..models.result_models import ProcessingResult, BatchProcessingResult
from ..services.text_service import TextService
from ..utils.metrics_utils import MetricsCollector
from ..utils.cache_utils import CacheManager

logger = logging.getLogger(__name__)

@dataclass
class ProcessingStats:
    """处理统计信息"""
    total_processed: int = 0
    total_errors: int = 0
    avg_processing_time: float = 0.0
    avg_quality_score: float = 0.0

class TextPostProcessingManager:
    """文本后处理管理器"""
    
    def __init__(self, config: Dict):
        """初始化文本后处理管理器"""
        self.config = config
        self.text_service = TextService(config)
        self.cache_manager = CacheManager(config.get('cache', {}))
        self.metrics_collector = MetricsCollector(config.get('monitoring', {}))
        self.stats = ProcessingStats()
        
        logger.info("TextPostProcessingManager initialized")
    
    def process_text(self, text: str, options: Optional[Dict] = None) -> ProcessingResult:
        """处理单个文本"""
        start_time = time.time()
        
        try:
            # 检查缓存
            cache_key = self._generate_cache_key(text, options)
            cached_result = self.cache_manager.get(cache_key)
            if cached_result:
                logger.debug(f"Cache hit for text: {text[:50]}...")
                return cached_result
            
            # 解析选项
            processing_options = ProcessingOptions.from_dict(options or {})
            
            # 创建文本处理对象
            text_processing = TextProcessing(
                text_id=self._generate_text_id(),
                original_text=text,
                options=processing_options
            )
            
            # 处理文本
            processed_text = self.text_service.process_text(text_processing)
            
            # 计算处理时间
            processing_time = time.time() - start_time
            
            # 创建结果对象
            result = ProcessingResult(
                status="success",
                text_id=text_processing.text_id,
                original_text=text,
                processed_text=processed_text.text,
                corrections=processed_text.corrections,
                completions=processed_text.completions,
                optimizations=processed_text.optimizations,
                enhancements=processed_text.enhancements,
                quality_score=processed_text.quality_score,
                processing_time=processing_time
            )
            
            # 缓存结果
            self.cache_manager.set(cache_key, result, ttl=3600)
            
            # 更新统计信息
            self._update_stats(result)
            
            # 记录指标
            self.metrics_collector.record_processing_time(processing_time)
            self.metrics_collector.record_quality_score(result.quality_score)
            
            return result
            
        except Exception as e:
            processing_time = time.time() - start_time
            logger.error(f"Error processing text: {str(e)}")
            
            # 更新错误统计
            self.stats.total_errors += 1
            
            # 返回错误结果
            return ProcessingResult(
                status="error",
                text_id=text_processing.text_id if 'text_processing' in locals() else None,
                original_text=text,
                error_message=str(e),
                processing_time=processing_time
            )
    
    def batch_process(self, text_list: List[str], options: Optional[Dict] = None) -> BatchProcessingResult:
        """批量处理文本"""
        start_time = time.time()
        results = []
        
        for i, text in enumerate(text_list):
            try:
                result = self.process_text(text, options)
                result.batch_index = i
                results.append(result)
            except Exception as e:
                logger.error(f"Error processing text {i}: {str(e)}")
                results.append(ProcessingResult(
                    status="error",
                    batch_index=i,
                    original_text=text,
                    error_message=str(e)
                ))
        
        total_time = time.time() - start_time
        
        return BatchProcessingResult(
            status="completed",
            batch_size=len(text_list),
            results=results,
            total_time=total_time,
            avg_time_per_text=total_time / len(text_list)
        )
    
    def stream_process(self, text_stream: Iterator[str], options: Optional[Dict] = None) -> Iterator[ProcessingResult]:
        """流式处理文本"""
        for text in text_stream:
            try:
                result = self.process_text(text, options)
                yield result
            except Exception as e:
                logger.error(f"Error processing stream text: {str(e)}")
                yield ProcessingResult(
                    status="error",
                    original_text=text,
                    error_message=str(e)
                )
    
    def get_stats(self) -> ProcessingStats:
        """获取处理统计信息"""
        return self.stats
    
    def _generate_text_id(self) -> str:
        """生成文本ID"""
        import uuid
        return str(uuid.uuid4())
    
    def _generate_cache_key(self, text: str, options: Optional[Dict]) -> str:
        """生成缓存键"""
        import hashlib
        options_str = str(sorted(options.items())) if options else ""
        content = f"{text}:{options_str}"
        return hashlib.md5(content.encode()).hexdigest()
    
    def _update_stats(self, result: ProcessingResult):
        """更新统计信息"""
        self.stats.total_processed += 1
        
        if result.status == "success":
            # 更新平均处理时间
            total_time = self.stats.avg_processing_time * (self.stats.total_processed - 1) + result.processing_time
            self.stats.avg_processing_time = total_time / self.stats.total_processed
            
            # 更新平均质量分数
            total_score = self.stats.avg_quality_score * (self.stats.total_processed - 1) + result.quality_score
            self.stats.avg_quality_score = total_score / self.stats.total_processed
```

```python
# src/text_post_processing/services/text_service.py
import logging
from typing import Dict, List, Optional

from ..models.text_models import TextProcessing, ProcessingOptions
from ..models.result_models import ProcessedText
from ..core.preprocessor import TextPreprocessor
from ..core.corrector import TextCorrector
from ..core.completer import TextCompleter
from ..core.optimizer import TextOptimizer
from ..core.enhancer import SentimentEnhancer
from ..core.formatter import TextFormatter

logger = logging.getLogger(__name__)

class TextService:
    """文本服务"""
    
    def __init__(self, config: Dict):
        """初始化文本服务"""
        self.config = config
        
        # 初始化处理器
        self.preprocessor = TextPreprocessor(config.get('preprocessor', {}))
        self.corrector = TextCorrector(config.get('corrector', {}))
        self.completer = TextCompleter(config.get('completer', {}))
        self.optimizer = TextOptimizer(config.get('optimizer', {}))
        self.enhancer = SentimentEnhancer(config.get('enhancer', {}))
        self.formatter = TextFormatter(config.get('formatter', {}))
        
        logger.info("TextService initialized")
    
    def process_text(self, text_processing: TextProcessing) -> ProcessedText:
        """处理文本"""
        try:
            # 预处理
            preprocessed_text = self.preprocessor.preprocess(
                text_processing.original_text,
                text_processing.options
            )
            
            # 纠错
            corrected_text = None
            corrections = []
            if text_processing.options.enable_correction:
                correction_result = self.corrector.correct(
                    preprocessed_text,
                    text_processing.options
                )
                corrected_text = correction_result.text
                corrections = correction_result.corrections
            else:
                corrected_text = preprocessed_text
            
            # 补全
            completed_text = None
            completions = []
            if text_processing.options.enable_completion:
                completion_result = self.completer.complete(
                    corrected_text,
                    text_processing.options
                )
                completed_text = completion_result.text
                completions = completion_result.completions
            else:
                completed_text = corrected_text
            
            # 优化
            optimized_text = None
            optimizations = []
            if text_processing.options.enable_optimization:
                optimization_result = self.optimizer.optimize(
                    completed_text,
                    text_processing.options
                )
                optimized_text = optimization_result.text
                optimizations = optimization_result.optimizations
            else:
                optimized_text = completed_text
            
            # 情感增强
            enhanced_text = None
            enhancements = []
            if text_processing.options.enable_sentiment:
                enhancement_result = self.enhancer.enhance(
                    optimized_text,
                    text_processing.options
                )
                enhanced_text = enhancement_result.text
                enhancements = enhancement_result.enhancements
            else:
                enhanced_text = optimized_text
            
            # 格式化
            formatted_text = None
            if text_processing.options.enable_formatting:
                formatted_text = self.formatter.format(
                    enhanced_text,
                    text_processing.options
                )
            else:
                formatted_text = enhanced_text
            
            # 计算质量分数
            quality_score = self._calculate_quality_score(
                corrections, completions, optimizations, enhancements
            )
            
            # 创建处理结果
            result = ProcessedText(
                text=formatted_text,
                corrections=corrections,
                completions=completions,
                optimizations=optimizations,
                enhancements=enhancements,
                quality_score=quality_score
            )
            
            return result
            
        except Exception as e:
            logger.error(f"Error in text processing: {str(e)}")
            raise
    
    def _calculate_quality_score(
        self,
        corrections: List[Dict],
        completions: List[Dict],
        optimizations: List[Dict],
        enhancements: List[Dict]
    ) -> float:
        """计算文本质量分数"""
        # 基础分数
        base_score = 0.8
        
        # 根据纠错调整分数
        if corrections:
            avg_confidence = sum(c.get('confidence', 0) for c in corrections) / len(corrections)
            base_score += 0.1 * avg_confidence
        
        # 根据补全调整分数
        if completions:
            avg_confidence = sum(c.get('confidence', 0) for c in completions) / len(completions)
            base_score += 0.05 * avg_confidence
        
        # 根据优化调整分数
        if optimizations:
            avg_confidence = sum(o.get('confidence', 0) for o in optimizations) / len(optimizations)
            base_score += 0.03 * avg_confidence
        
        # 根据情感增强调整分数
        if enhancements:
            avg_confidence = sum(e.get('confidence', 0) for e in enhancements) / len(enhancements)
            base_score += 0.02 * avg_confidence
        
        # 确保分数在0-1范围内
        return min(max(base_score, 0.0), 1.0)
```

```python
# src/text_post_processing/core/corrector.py
import logging
from typing import Dict, List, Tuple, Optional

from ..models.text_models import TextProcessing, ProcessingOptions
from ..models.result_models import CorrectionResult, TextCorrection
from ..algorithms.correction.grammar import GrammarCorrector
from ..algorithms.correction.spelling import SpellingCorrector
from ..algorithms.correction.speech import SpeechCorrector
from ..resources.dictionaries import DictionaryManager
from ..resources.rules import RuleManager

logger = logging.getLogger(__name__)

class TextCorrector:
    """文本纠错器"""
    
    def __init__(self, config: Dict):
        """初始化文本纠错器"""
        self.config = config
        
        # 初始化纠错器
        self.grammar_corrector = GrammarCorrector(config.get('grammar', {}))
        self.spelling_corrector = SpellingCorrector(config.get('spelling', {}))
        self.speech_corrector = SpeechCorrector(config.get('speech', {}))
        
        # 初始化资源管理器
        self.dictionary_manager = DictionaryManager(config.get('dictionaries', {}))
        self.rule_manager = RuleManager(config.get('rules', {}))
        
        logger.info("TextCorrector initialized")
    
    def correct(self, text: str, options: ProcessingOptions) -> CorrectionResult:
        """纠错文本"""
        try:
            corrections = []
            corrected_text = text
            
            # 语法纠错
            if options.enable_grammar_correction:
                grammar_result = self.grammar_corrector.correct(corrected_text)
                for correction in grammar_result.corrections:
                    corrections.append(correction)
                corrected_text = grammar_result.text
            
            # 拼写纠错
            if options.enable_spelling_correction:
                spelling_result = self.spelling_corrector.correct(corrected_text)
                for correction in spelling_result.corrections:
                    corrections.append(correction)
                corrected_text = spelling_result.text
            
            # 语音识别纠错
            if options.enable_speech_correction:
                speech_result = self.speech_corrector.correct(corrected_text)
                for correction in speech_result.corrections:
                    corrections.append(correction)
                corrected_text = speech_result.text
            
            # 应用规则纠错
            rule_corrections = self.rule_manager.apply_correction_rules(corrected_text)
            for correction in rule_corrections:
                corrections.append(correction)
                corrected_text = correction['corrected_text']
            
            # 去重和排序纠错
            corrections = self._deduplicate_corrections(corrections)
            corrections = self._sort_corrections(corrections)
            
            return CorrectionResult(
                text=corrected_text,
                corrections=corrections
            )
            
        except Exception as e:
            logger.error(f"Error in text correction: {str(e)}")
            raise
    
    def _deduplicate_corrections(self, corrections: List[Dict]) -> List[Dict]:
        """去重纠错"""
        seen = set()
        unique_corrections = []
        
        for correction in corrections:
            key = (
                correction.get('start_pos', 0),
                correction.get('end_pos', 0),
                correction.get('original_text', ''),
                correction.get('corrected_text', '')
            )
            if key not in seen:
                seen.add(key)
                unique_corrections.append(correction)
        
        return unique_corrections
    
    def _sort_corrections(self, corrections: List[Dict]) -> List[Dict]:
        """排序纠错"""
        return sorted(
            corrections,
            key=lambda x: (
                x.get('start_pos', 0),
                -x.get('confidence', 0)
            )
        )
```

```python
@router.post("/stream")
async def stream_process(
    request: TextProcessingRequest,
    manager: TextPostProcessingManager = Depends(get_text_processing_manager)
):
    """流式处理文本"""
    async def text_stream():
        try:
            # 模拟文本流
            text_chunks = request.text.split('\n')
            for chunk in text_chunks:
                if chunk.strip():  # 跳过空行
                    result = manager.process_text(
                        text=chunk,
                        options=request.options.dict() if request.options else None
                    )
                    yield {
                        "event": "text_processed",
                        "data": result.dict()
                    }
        except Exception as e:
            logger.error(f"Error in stream processing: {str(e)}")
            yield {
                "event": "error",
                "data": {"error": str(e)}
            }
    
    return EventSourceResponse(text_stream())
```
