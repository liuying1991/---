# 自然语言生成模块实现工作流

## 模块信息

### 模块名称
自然语言生成模块

### 所属子系统
交互表达子系统

### 模块功能描述
自然语言生成模块负责将系统的意图、情感和信息转化为自然、流畅、适合婴儿理解的语言表达。该模块能够生成符合婴儿认知水平的语言内容，调整语言复杂度，融入适当的情感色彩，并确保语言的安全性和适宜性。

### 模块职责
1. 语言策略解析：解析语言表达策略和规则
2. 语言结构规划：规划语言结构和逻辑
3. 基础语言生成：生成基础语言模板和内容
4. 语言细节生成：生成语言细节和变化
5. 语言优化：优化语言的自然性和流畅性
6. 语言表达：将语言内容转换为语音输出
7. 多模态协调：协调语言与其他模态的表达
8. 个性化适配：根据婴儿特点调整语言风格

### 模块接口
- **输入接口**：交互意图、情感状态、上下文信息、婴儿状态
- **输出接口**：语言文本、语音数据、语言元数据
- **依赖模块**：交互策略模块、情感识别模块、上下文管理模块
- **被依赖模块**：语音合成模块、显示输出模块、交互管理模块

### 优先级
高

```python
# src/core/nlg_manager.py
import time
import logging
from typing import Dict, Any, List, Optional, Tuple
from dataclasses import dataclass
from datetime import datetime

from ..models.language_models import (
    LanguageStrategy, 
    LanguageExpression, 
    LanguageStyle,
    LanguageStructure
)
from ..core.strategy.strategy_parser import StrategyParser
from ..core.strategy.style_manager import StyleManager
from ..core.strategy.safety_checker import SafetyChecker
from ..core.planner.structure_planner import StructurePlanner
from ..core.planner.vocabulary_selector import VocabularySelector
from ..core.planner.complexity_calculator import ComplexityCalculator
from ..core.generator.base_generator import BaseGenerator
from ..core.generator.detail_generator import DetailGenerator
from ..core.generator.language_optimizer import LanguageOptimizer
from ..core.expressor.speech_synthesizer import SpeechSynthesizer
from ..core.expressor.multimodal_coordinator import MultimodalCoordinator
from ..ml.language_generation_model import LanguageGenerationModel
from ..ml.language_optimization_model import LanguageOptimizationModel
from ..ml.style_adaptation_model import StyleAdaptationModel
from ..utils.config import NLGConfig

logger = logging.getLogger(__name__)

@dataclass
class GenerationRequest:
    """语言生成请求"""
    intent: str
    context: Dict[str, Any]
    baby_state: Dict[str, Any]
    request_id: str
    timestamp: float

@dataclass
class GenerationResult:
    """语言生成结果"""
    text: str
    metadata: Dict[str, Any]
    quality_score: float
    generation_time: float
    request_id: str
    timestamp: float

class NaturalLanguageGenerationManager:
    """自然语言生成管理器"""
    
    def __init__(self, config: NLGConfig):
        """初始化自然语言生成管理器"""
        self.config = config
        
        # 初始化策略层组件
        self.strategy_parser = StrategyParser(config)
        self.style_manager = StyleManager(config)
        self.safety_checker = SafetyChecker(config)
        
        # 初始化规划层组件
        self.structure_planner = StructurePlanner(config)
        self.vocabulary_selector = VocabularySelector(config)
        self.complexity_calculator = ComplexityCalculator(config)
        
        # 初始化生成层组件
        self.base_generator = BaseGenerator(config)
        self.detail_generator = DetailGenerator(config)
        self.language_optimizer = LanguageOptimizer(config)
        
        # 初始化表达层组件
        self.speech_synthesizer = SpeechSynthesizer(config)
        self.multimodal_coordinator = MultimodalCoordinator(config)
        
        # 初始化机器学习模型
        self.generation_model = LanguageGenerationModel(config)
        self.optimization_model = LanguageOptimizationModel(config)
        self.style_model = StyleAdaptationModel(config)
        
        # 初始化模型
        self._initialize_models()
        
        logger.info("NaturalLanguageGenerationManager initialized")
    
    def _initialize_models(self):
        """初始化模型"""
        try:
            self.generation_model.load_model(self.config.generation_model_path)
            self.optimization_model.load_model(self.config.optimization_model_path)
            self.style_model.load_model(self.config.style_adaptation_model_path)
            
            if self.config.content_safety_enabled:
                self.safety_checker.load_model(self.config.safety_model_path)
                
            logger.info("Models loaded successfully")
        except Exception as e:
            logger.error(f"Error loading models: {str(e)}")
            raise
    
    def generate_language(self, request: GenerationRequest) -> GenerationResult:
        """生成语言"""
        start_time = time.time()
        
        try:
            # 1. 解析策略
            strategy = self.strategy_parser.parse_strategy(request.intent)
            
            # 2. 获取语言风格
            style = None
            if self.config.personalization_enabled and "user_id" in request.baby_state:
                style = self.style_model.get_style(request.baby_state["user_id"])
            
            # 3. 规划语言结构
            language_structure = self.structure_planner.plan_structure(
                strategy, 
                request.context, 
                request.baby_state
            )
            
            # 4. 选择词汇
            age_group = self._determine_age_group(request.baby_state.get("age", None))
            vocabulary = self.vocabulary_selector.select_vocabulary(
                language_structure,
                age_group,
                style
            )
            
            # 5. 生成基础语言
            base_content = self.base_generator.generate_base_language(
                language_structure,
                vocabulary
            )
            
            # 6. 生成语言细节
            if style:
                detailed_content = self.detail_generator.generate_details(
                    base_content, 
                    style
                )
            else:
                detailed_content = base_content
            
            # 7. 优化语言
            optimized_content = self.language_optimizer.optimize_language(
                detailed_content,
                language_structure
            )
            
            # 8. 安全检查
            if self.config.content_safety_enabled:
                is_safe = self.safety_checker.check_safety(optimized_content)
                if not is_safe:
                    # 使用安全内容替换
                    optimized_content = self.safety_checker.get_safe_alternative(
                        optimized_content,
                        language_structure
                    )
            
            # 9. 计算质量分数
            quality_score = self._calculate_quality_score(
                optimized_content,
                language_structure,
                style
            )
            
            # 10. 质量检查和重试
            if quality_score < self.config.min_quality_score:
                # 尝试调整生成参数并重新生成
                adjusted_request = self._adjust_generation_params(
                    request,
                    quality_score
                )
                
                # 递归调用，但限制重试次数
                if adjusted_request.metadata.get("retry_count", 0) < self.config.max_generation_attempts:
                    adjusted_request.metadata["retry_count"] = adjusted_request.metadata.get("retry_count", 0) + 1
                    return self.generate_language(adjusted_request)
            
            generation_time = time.time() - start_time
            
            # 创建语言表达对象
            expression = LanguageExpression(
                expression_id=f"expr_{datetime.now().strftime('%Y%m%d_%H%M%S')}",
                expression_type=strategy.strategy_type,
                content=optimized_content,
                language=request.context.get('language', 'zh-CN'),
                complexity_score=self.complexity_calculator.calculate_complexity(optimized_content),
                emotion_tags=request.context.get('emotions', []),
                style_features=style.style_features if style else {},
                multimodal_refs=request.context.get('multimodal_refs', []),
                created_at=datetime.now()
            )
            
            # 返回结果
            return GenerationResult(
                text=expression.content,
                metadata={
                    "expression_id": expression.expression_id,
                    "language": expression.language,
                    "complexity_score": expression.complexity_score,
                    "emotion_tags": expression.emotion_tags,
                    "style_features": expression.style_features,
                    "multimodal_refs": expression.multimodal_refs,
                    "age_group": age_group,
                    "user_id": request.baby_state.get("user_id", None),
                    "is_safe": self.config.content_safety_enabled
                },
                quality_score=quality_score,
                generation_time=generation_time,
                request_id=request.request_id,
                timestamp=time.time()
            )
            
        except Exception as e:
            logger.error(f"Error generating language: {str(e)}")
            raise
    
    def _determine_age_group(self, age_months: Optional[int]) -> str:
        """确定年龄组"""
        if age_months is None:
            return "12-24_months"  # 默认年龄组
        
        if age_months <= 6:
            return "0-6_months"
        elif age_months <= 12:
            return "6-12_months"
        else:
            return "12-24_months"
    
    def _calculate_quality_score(self, content: str, structure: LanguageStructure, 
                                style: Optional[LanguageStyle]) -> float:
        """计算语言质量分数"""
        # 使用优化模型计算质量分数
        quality_metrics = self.optimization_model.assess_quality(content, structure, style)
        
        # 综合各项指标计算总分
        naturalness = quality_metrics.get("naturalness", 0.0)
        fluency = quality_metrics.get("fluency", 0.0)
        appropriateness = quality_metrics.get("appropriateness", 0.0)
        complexity = quality_metrics.get("complexity", 0.0)
        
        # 加权平均
        total_score = (
            naturalness * 0.3 +
            fluency * 0.3 +
            appropriateness * 0.3 +
            complexity * 0.1
        )
        
        return total_score
    
    def _adjust_generation_params(self, request: GenerationRequest, 
                                  quality_score: float) -> GenerationRequest:
        """调整生成参数"""
        # 创建新的请求对象
        adjusted_request = GenerationRequest(
            intent=request.intent,
            context=request.context.copy(),
            baby_state=request.baby_state.copy(),
            request_id=f"{request.request_id}_retry",
            timestamp=time.time()
        )
        
        # 根据质量分数调整参数
        if quality_score < 3.0:
            # 质量很低，大幅调整参数
            adjusted_request.context["max_sentence_length"] = max(
                5, 
                request.context.get("max_sentence_length", 10) - 3
            )
            adjusted_request.context["simplify_grammar"] = True
            adjusted_request.context["increase_repetition"] = True
        elif quality_score < 3.5:
            # 质量较低，小幅调整参数
            adjusted_request.context["max_sentence_length"] = max(
                7, 
                request.context.get("max_sentence_length", 10) - 1
            )
            adjusted_request.context["simplify_grammar"] = request.context.get("simplify_grammar", False)
        
        # 添加重试次数
        adjusted_request.metadata = {"retry_count": request.metadata.get("retry_count", 0) + 1}
        
        return adjusted_request
```

### 2. 基础语言生成器

```python
# src/core/generator/base_generator.py
import os
import json
import random
import re
from typing import Dict, Any, List, Optional, Tuple

from ..models.language_models import LanguageStructure, Vocabulary
from ..utils.config import NLGConfig
from ..utils.text_utils import TextUtils

class BaseGenerator:
    """基础语言生成器"""
    
    def __init__(self, config: NLGConfig):
        self.config = config
        self.text_utils = TextUtils()
        self.language_templates = {}
        
        # 加载语言模板
        self._load_language_templates()
    
    def _load_language_templates(self):
        """加载语言模板"""
        template_path = self.config.template_library_path
        
        # 加载基础模板
        basic_template_path = os.path.join(template_path, "basic_templates")
        for file_name in os.listdir(basic_template_path):
            if file_name.endswith('.json'):
                template_name = file_name[:-5]  # 去掉.json后缀
                with open(os.path.join(basic_template_path, file_name), 'r', encoding='utf-8') as f:
                    self.language_templates[template_name] = json.load(f)
        
        # 加载情感模板
        emotion_template_path = os.path.join(template_path, "emotion_templates")
        for file_name in os.listdir(emotion_template_path):
            if file_name.endswith('.json'):
                template_name = file_name[:-5]  # 去掉.json后缀
                with open(os.path.join(emotion_template_path, file_name), 'r', encoding='utf-8') as f:
                    emotion_templates = json.load(f)
                    # 合并到主模板字典，添加emotion_前缀
                    for emotion, templates in emotion_templates.items():
                        self.language_templates[f"emotion_{emotion}_{template_name}"] = templates
    
    def generate_base_language(self, structure: LanguageStructure, 
                               vocabulary: Vocabulary) -> str:
        """生成基础语言"""
        # 获取模板
        template_id = structure.template_id
        emotion = structure.emotion
        
        # 如果有情感，尝试使用情感模板
        if emotion:
            emotion_template_id = f"emotion_{emotion}_{template_id}"
            templates = self.language_templates.get(emotion_template_id)
        
        # 如果没有情感模板或情感模板不存在，使用基础模板
        if not emotion or not templates:
            templates = self.language_templates.get(template_id, 
                                                 self.language_templates.get("greeting", []))
        
        # 随机选择一个模板
        if isinstance(templates, list):
            template = random.choice(templates)
        else:
            template = templates
        
        # 填充模板变量
        filled_template = self._fill_template(template, structure.variables, vocabulary)
        
        # 后处理
        processed_text = self._post_process_text(filled_template, structure)
        
        return processed_text
    
    def _fill_template(self, template: str, variables: Dict[str, Any], 
                       vocabulary: Vocabulary) -> str:
        """填充模板变量"""
        # 替换变量占位符
        for var_name, var_value in variables.items():
            # 查找所有匹配的占位符
            pattern = re.compile(r'\{' + var_name + r'(\|[^}]*)?\}')
            matches = pattern.findall(template)
            
            # 替换每个匹配
            for match in matches:
                # 解析修饰符
                modifiers = match[1:].split('|') if match else []
                
                # 应用修饰符
                value = var_value
                for modifier in modifiers:
                    if modifier == 'upper':
                        value = str(value).upper()
                    elif modifier == 'lower':
                        value = str(value).lower()
                    elif modifier == 'title':
                        value = str(value).title()
                    elif modifier.startswith('replace:'):
                        # 格式: replace:old:new
                        replace_parts = modifier[8:].split(':')
                        if len(replace_parts) == 2:
                            old, new = replace_parts
                            value = str(value).replace(old, new)
                
                # 替换占位符
                placeholder = '{' + var_name + match + '}'
                template = template.replace(placeholder, str(value))
        
        return template
    
    def _post_process_text(self, text: str, structure: LanguageStructure) -> str:
        """后处理文本"""
        # 应用年龄适配规则
        age_group = structure.age_group
        max_sentence_length = self.config.age_groups[age_group]["max_sentence_length"]
        max_words_per_sentence = self.config.age_groups[age_group]["max_words_per_sentence"]
        simple_grammar = self.config.age_groups[age_group]["simple_grammar"]
        
        # 分割句子
        sentences = self.text_utils.split_sentences(text)
        
        # 处理每个句子
        processed_sentences = []
        for sentence in sentences:
            # 检查句子长度
            words = self.text_utils.tokenize(sentence)
            if len(words) > max_words_per_sentence:
                # 分割长句子
                shortened_sentence = self._shorten_sentence(sentence, max_words_per_sentence)
                processed_sentences.append(shortened_sentence)
            else:
                processed_sentences.append(sentence)
        
        # 合并处理后的句子
        processed_text = ' '.join(processed_sentences)
        
        # 应用简单语法规则
        if simple_grammar:
            processed_text = self._apply_simple_grammar(processed_text)
        
        return processed_text
    
    def _shorten_sentence(self, sentence: str, max_words: int) -> str:
        """缩短句子"""
        words = self.text_utils.tokenize(sentence)
        
        # 如果句子已经足够短，直接返回
        if len(words) <= max_words:
            return sentence
        
        # 尝试保留最重要的部分
        # 简单策略：保留前max_words个词，并确保句子以标点符号结尾
        shortened_words = words[:max_words]
        
        # 如果最后一个词不是标点符号，添加省略号
        if shortened_words and not self.text_utils.is_punctuation(shortened_words[-1]):
            shortened_words.append('...')
        
        return ' '.join(shortened_words)
    
    def _apply_simple_grammar(self, text: str) -> str:
        """应用简单语法规则"""
        # 简化复杂语法结构
        # 这里是一些示例规则，实际应用中可能需要更复杂的规则
        
        # 1. 简化复杂时态
        text = re.sub(r'正在(.+?)ing', r'\1ing', text)  # 简化进行时
        
        # 2. 简化复合句
        text = re.sub(r'(.+?)，但是(.+?)', r'\1。但是\2', text)  # 将复合句拆分为简单句
        
        # 3. 简化被动语态
        text = re.sub(r'(.+?)被(.+?)', r'\2\1', text)  # 简单的被动转主动
        
        return text
```
