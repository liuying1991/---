# 长期记忆模块实现工作流

## 1. 模块信息

### 1.1 模块名称
长期记忆模块 (Long-Term Memory Module)

### 1.2 模块功能
长期记忆模块是记忆存储子系统的核心组件，负责存储和管理婴儿AI管家系统的长期记忆信息。该模块能够持久化存储各种类型的记忆数据，包括概念知识、经验事件、技能模式、情感记忆等，并提供高效的检索、更新和遗忘机制。

### 1.3 模块职责
- 持久化存储各类长期记忆数据
- 实现记忆的编码、存储和检索机制
- 管理记忆的组织结构和关联关系
- 实现记忆的巩固和遗忘过程
- 提供记忆查询和推理接口
- 维护记忆的一致性和完整性

```python
# core/memory_store.py
from typing import Dict, List, Optional, Union, Any
from datetime import datetime
import uuid
import numpy as np
from abc import ABC, abstractmethod

from models.memory_data import MemoryData, MemoryType, MemoryImportance
from storage.vector_storage import VectorStorage
from storage.relational_storage import RelationalStorage
from storage.graph_storage import GraphStorage
from encoders.base_encoder import BaseEncoder
from config.long_term_memory_config import LongTermMemoryConfig

class MemoryStore:
    """记忆存储组件，负责记忆数据的存储和管理"""
    
    def __init__(self, config: LongTermMemoryConfig):
        self.config = config
        self.vector_storage = VectorStorage(config.vector_db_config)
        self.relational_storage = RelationalStorage(config.database_url)
        self.graph_storage = GraphStorage(
            config.graph_db_url, 
            config.graph_db_user, 
            config.graph_db_password
        )
        self.encoder = self._initialize_encoder()
        
    def _initialize_encoder(self) -> BaseEncoder:
        """初始化编码器"""
        if self.config.encoder_type == "transformer":
            from encoders.transformer_encoder import TransformerEncoder
            return TransformerEncoder(self.config.encoder_model)
        elif self.config.encoder_type == "multimodal":
            from encoders.multimodal_encoder import MultimodalEncoder
            return MultimodalEncoder(self.config.encoder_model)
        else:
            raise ValueError(f"Unsupported encoder type: {self.config.encoder_type}")
    
    def store_memory(self, memory_data: MemoryData) -> str:
        """存储记忆数据"""
        try:
            # 生成记忆ID
            memory_id = str(uuid.uuid4())
            memory_data.memory_id = memory_id
            
            # 编码记忆内容
            if memory_data.content_vector is None:
                memory_data.content_vector = self.encoder.encode(memory_data.content)
            
            # 设置创建时间
            memory_data.created_at = datetime.now()
            
            # 计算记忆重要性
            if memory_data.importance == MemoryImportance.UNKNOWN:
                memory_data.importance = self._calculate_importance(memory_data)
            
            # 存储到向量数据库
            self.vector_storage.add(memory_id, memory_data.content_vector)
            
            # 存储到关系数据库
            self.relational_storage.add_memory(memory_data)
            
            # 存储到图数据库
            self.graph_storage.add_memory_node(memory_id, memory_data)
            
            return memory_id
            
        except Exception as e:
            print(f"Error storing memory: {e}")
            return None
    
    def retrieve_memory(self, query: str, memory_type: Optional[MemoryType] = None, 
                       limit: int = 10) -> List[MemoryData]:
        """检索记忆数据"""
        try:
            # 编码查询
            query_vector = self.encoder.encode(query)
            
            # 向量检索
            vector_results = self.vector_storage.search(query_vector, limit)
            
            # 获取记忆ID列表
            memory_ids = [result[0] for result in vector_results]
            
            # 从关系数据库获取详细记忆数据
            memories = self.relational_storage.get_memories(memory_ids)
            
            # 过滤记忆类型
            if memory_type:
                memories = [m for m in memories if m.memory_type == memory_type]
            
            return memories
            
        except Exception as e:
            print(f"Error retrieving memory: {e}")
            return []
    
    def update_memory(self, memory_id: str, updates: Dict) -> bool:
        """更新记忆数据"""
        try:
            # 获取当前记忆数据
            current_memory = self.relational_storage.get_memory(memory_id)
            if not current_memory:
                return False
            
            # 更新字段
            for field, value in updates.items():
                if hasattr(current_memory, field):
                    setattr(current_memory, field, value)
            
            # 如果内容更新，重新编码向量
            if 'content' in updates:
                current_memory.content_vector = self.encoder.encode(current_memory.content)
                self.vector_storage.update(memory_id, current_memory.content_vector)
            
            # 更新关系数据库
            self.relational_storage.update_memory(memory_id, updates)
            
            # 更新图数据库
            self.graph_storage.update_memory_node(memory_id, current_memory)
            
            return True
            
        except Exception as e:
            print(f"Error updating memory: {e}")
            return False
    
    def delete_memory(self, memory_id: str) -> bool:
        """删除记忆数据"""
        try:
            # 从向量数据库删除
            self.vector_storage.delete(memory_id)
            
            # 从关系数据库删除
            self.relational_storage.delete_memory(memory_id)
            
            # 从图数据库删除
            self.graph_storage.delete_memory_node(memory_id)
            
            return True
            
        except Exception as e:
            print(f"Error deleting memory: {e}")
            return False
    
    def _calculate_importance(self, memory_data: MemoryData) -> MemoryImportance:
        """计算记忆重要性"""
        # 基于内容长度、情感强度、重复次数等因素计算重要性
        content_length = len(memory_data.content)
        emotional_intensity = getattr(memory_data, 'emotional_intensity', 0.5)
        access_count = getattr(memory_data, 'access_count', 0)
        
        # 简单的重要性计算公式
        importance_score = (content_length / 1000) * 0.3 + \
                          emotional_intensity * 0.4 + \
                          (access_count / 10) * 0.3
        
        if importance_score >= 0.8:
            return MemoryImportance.HIGH
        elif importance_score >= 0.5:
            return MemoryImportance.MEDIUM
        elif importance_score >= 0.2:
            return MemoryImportance.LOW
        else:
            return MemoryImportance.MINIMAL
```

```python
@router.post("/memories", response_model=StoreMemoryResponse)
async def store_memory(
    request: StoreMemoryRequest,
    memory_store: MemoryStore = Depends(get_memory_store)
):
    """存储记忆"""
    try:
        memory_data = MemoryData(
            content=request.content,
            memory_type=request.memory_type,
            importance=request.importance,
            metadata=request.metadata,
            tags=request.tags
        )
        
        memory_id = memory_store.store_memory(memory_data)
        
        if memory_id:
            return StoreMemoryResponse(
                memory_id=memory_id,
                status="success",
                message="Memory stored successfully"
            )
        else:
            raise HTTPException(status_code=500, detail="Failed to store memory")
            
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

```python
# examples/python_client.py
import requests
import json

class LongTermMemoryClient:
    """长期记忆模块Python客户端"""
    
    def __init__(self, base_url="http://localhost:8000"):
        self.base_url = base_url
        self.session = requests.Session()
    
    def store_memory(self, content, memory_type, importance="MEDIUM", metadata=None, tags=None):
        """存储记忆"""
        url = f"{self.base_url}/long-term-memory/memories"
        data = {
            "content": content,
            "memory_type": memory_type,
            "importance": importance,
            "metadata": metadata or {},
            "tags": tags or []
        }
        response = self.session.post(url, json=data)
        return response.json()
    
    def retrieve_memories(self, text=None, query_type="SEMANTIC", memory_type=None, 
                         filters=None, limit=10):
        """检索记忆"""
        url = f"{self.base_url}/long-term-memory/memories/retrieve"
        data = {
            "text": text,
            "query_type": query_type,
            "memory_type": memory_type,
            "filters": filters or {},
            "limit": limit
        }
        response = self.session.post(url, json=data)
        return response.json()
    
    def update_memory(self, memory_id, updates):
        """更新记忆"""
        url = f"{self.base_url}/long-term-memory/memories/{memory_id}"
        data = {"updates": updates}
        response = self.session.put(url, json=data)
        return response.json()
    
    def delete_memory(self, memory_id):
        """删除记忆"""
        url = f"{self.base_url}/long-term-memory/memories/{memory_id}"
        response = self.session.delete(url)
        return response.json()
    
    def consolidate_memories(self, memory_ids=None):
        """巩固记忆"""
        url = f"{self.base_url}/long-term-memory/memories/consolidate"
        data = {"memory_ids": memory_ids}
        response = self.session.post(url, json=data)
        return response.json()
    
    def forget_memories(self, memory_ids=None):
        """遗忘记忆"""
        url = f"{self.base_url}/long-term-memory/memories/forget"
        data = {"memory_ids": memory_ids}
        response = self.session.post(url, json=data)
        return response.json()
    
    def associate_memories(self, source_memory_id, target_memory_id, 
                          association_type, strength=0.5, metadata=None):
        """关联记忆"""
        url = f"{self.base_url}/long-term-memory/memories/associate"
        data = {
            "source_memory_id": source_memory_id,
            "target_memory_id": target_memory_id,
            "association_type": association_type,
            "strength": strength,
            "metadata": metadata or {}
        }
        response = self.session.post(url, json=data)
        return response.json()
    
    def get_memory(self, memory_id):
        """获取单个记忆"""
        url = f"{self.base_url}/long-term-memory/memories/{memory_id}"
        response = self.session.get(url)
        return response.json()
    
    def get_memory_associations(self, memory_id, association_type=None):
        """获取记忆关联"""
        url = f"{self.base_url}/long-term-memory/memories/{memory_id}/associations"
        params = {"association_type": association_type} if association_type else {}
        response = self.session.get(url, params=params)
        return response.json()
```

```python
# utils/model_optimizer.py
import torch
import numpy as np
from typing import List, Dict, Any, Optional
from transformers import AutoModel, AutoTokenizer
import logging

logger = logging.getLogger(__name__)

class ModelOptimizer:
    """模型优化工具"""
    
    def __init__(self, model_name: str):
        self.model_name = model_name
        self.model = None
        self.tokenizer = None
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        
    def load_model(self):
        """加载模型"""
        try:
            self.tokenizer = AutoTokenizer.from_pretrained(self.model_name)
            self.model = AutoModel.from_pretrained(self.model_name)
            self.model.to(self.device)
            self.model.eval()
            
            # 启用模型优化
            if self.device.type == "cuda":
                self.model = self._optimize_for_gpu(self.model)
            
            return True
        except Exception as e:
            logger.error(f"Error loading model: {e}")
            return False
    
    def _optimize_for_gpu(self, model):
        """为GPU优化模型"""
        # 启用混合精度
        if hasattr(torch.cuda, "amp") and torch.cuda.is_available():
            model = model.half()
        
        # 启用编译优化 (PyTorch 2.0+)
        if hasattr(torch, "compile"):
            try:
                model = torch.compile(model)
            except Exception as e:
                logger.warning(f"Model compilation failed: {e}")
        
        return model
    
    def encode_batch(self, texts: List[str], batch_size: int = 32) -> List[List[float]]:
        """批量编码文本"""
        if not self.model or not self.tokenizer:
            raise ValueError("Model not loaded")
        
        all_embeddings = []
        
        for i in range(0, len(texts), batch_size):
            batch_texts = texts[i:i+batch_size]
            
            # 分词
            inputs = self.tokenizer(
                batch_texts,
                padding=True,
                truncation=True,
                max_length=512,
                return_tensors="pt"
            )
            
            # 移动到设备
            inputs = {k: v.to(self.device) for k, v in inputs.items()}
            
            # 推理
            with torch.no_grad():
                outputs = self.model(**inputs)
                # 使用平均池化获取句子嵌入
                embeddings = self._mean_pooling(outputs, inputs["attention_mask"])
                embeddings = embeddings.cpu().numpy()
                all_embeddings.extend(embeddings.tolist())
        
        return all_embeddings
    
    def _mean_pooling(self, model_output, attention_mask):
        """平均池化"""
        token_embeddings = model_output[0]
        input_mask_expanded = attention_mask.unsqueeze(-1).expand(token_embeddings.size()).float()
        return torch.sum(token_embeddings * input_mask_expanded, 1) / torch.clamp(input_mask_expanded.sum(1), min=1e-9)
    
    def unload_model(self):
        """卸载模型"""
        if self.model:
            del self.model
            self.model = None
        
        if self.tokenizer:
            del self.tokenizer
            self.tokenizer = None
        
        # 清理GPU内存
        if self.device.type == "cuda":
            torch.cuda.empty_cache()
```

```python
import redis
import pickle
import json
import hashlib
from typing import Any, Optional, Dict, List
import logging

logger = logging.getLogger(__name__)

class CacheManager:
    """缓存管理器"""
    
    def __init__(self, redis_url: str = "redis://localhost:6379/0"):
        try:
            self.redis_client = redis.from_url(redis_url)
            self.redis_client.ping()  # 测试连接
            self.available = True
        except Exception as e:
            logger.warning(f"Redis not available: {e}")
            self.redis_client = None
            self.available = False
        
        # 本地缓存作为后备
        self.local_cache = {}
        self.max_local_size = 1000
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存值"""
        if self.available:
            try:
                value = self.redis_client.get(key)
                if value:
                    return pickle.loads(value)
            except Exception as e:
                logger.error(f"Error getting from Redis: {e}")
        
        # 从本地缓存获取
        return self.local_cache.get(key)
    
    def set(self, key: str, value: Any, expire: int = 3600) -> bool:
        """设置缓存值"""
        if self.available:
            try:
                serialized_value = pickle.dumps(value)
                self.redis_client.setex(key, expire, serialized_value)
                return True
            except Exception as e:
                logger.error(f"Error setting to Redis: {e}")
        
        # 设置本地缓存
        if len(self.local_cache) >= self.max_local_size:
            # 简单的LRU：删除第一个元素
            oldest_key = next(iter(self.local_cache))
            del self.local_cache[oldest_key]
        
        self.local_cache[key] = value
        return True
    
    def delete(self, key: str) -> bool:
        """删除缓存值"""
        if self.available:
            try:
                self.redis_client.delete(key)
            except Exception as e:
                logger.error(f"Error deleting from Redis: {e}")
        
        if key in self.local_cache:
            del self.local_cache[key]
        
        return True
    
    def exists(self, key: str) -> bool:
        """检查缓存是否存在"""
        if self.available:
            try:
                return self.redis_client.exists(key)
            except Exception as e:
                logger.error(f"Error checking existence in Redis: {e}")
        
        return key in self.local_cache
    
    def clear(self) -> bool:
        """清空缓存"""
        if self.available:
            try:
                self.redis_client.flushdb()
            except Exception as e:
                logger.error(f"Error clearing Redis: {e}")
        
        self.local_cache.clear()
        return True
    
    def get_memory_cache_key(self, text: str, query_type: str) -> str:
```

```python
# utils/concurrent_processor.py
import asyncio
import concurrent.futures
from typing import List, Callable, Any, Dict
import logging

logger = logging.getLogger(__name__)

class ConcurrentProcessor:
    """并发处理器"""
    
    def __init__(self, max_workers: int = 4):
        self.max_workers = max_workers
        self.thread_pool = concurrent.futures.ThreadPoolExecutor(max_workers=max_workers)
        self.process_pool = concurrent.futures.ProcessPoolExecutor(max_workers=max_workers)
    
    def process_batch(self, items: List[Any], process_func: Callable, 
                     use_thread: bool = True) -> List[Any]:
        """批量处理项目"""
        if use_thread:
            with concurrent.futures.ThreadPoolExecutor(max_workers=self.max_workers) as executor:
                futures = [executor.submit(process_func, item) for item in items]
                results = [future.result() for future in concurrent.futures.as_completed(futures)]
        else:
            with concurrent.futures.ProcessPoolExecutor(max_workers=self.max_workers) as executor:
                futures = [executor.submit(process_func, item) for item in items]
                results = [future.result() for future in concurrent.futures.as_completed(futures)]
        
        return results
    
    async def process_batch_async(self, items: List[Any], process_func: Callable) -> List[Any]:
        """异步批量处理项目"""
        loop = asyncio.get_event_loop()
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [loop.run_in_executor(executor, process_func, item) for item in items]
            results = await asyncio.gather(*futures)
        
        return results
    
    def map_reduce(self, items: List[Any], map_func: Callable, 
                  reduce_func: Callable) -> Any:
        """Map-Reduce处理"""
        # Map阶段
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            mapped_items = list(executor.map(map_func, items))
        
        # Reduce阶段
        result = reduce_func(mapped_items)
        
        return result
    
    def parallel_search(self, queries: List[Dict[str, Any]], 
                      search_func: Callable) -> List[Dict[str, Any]]:
        """并行搜索"""
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [executor.submit(search_func, query) for query in queries]
            results = [future.result() for future in concurrent.futures.as_completed(futures)]
        
        return results
    
    def shutdown(self):
        """关闭处理器"""
        self.thread_pool.shutdown(wait=True)
        self.process_pool.shutdown(wait=True)
```

```python
# utils/performance_monitor.py
import time
import psutil
import threading
from typing import Dict, List, Any, Optional
from collections import deque
import logging

logger = logging.getLogger(__name__)

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self, max_history: int = 1000):
        self.max_history = max_history
        self.metrics_history = {
            "response_time": deque(maxlen=max_history),
            "memory_usage": deque(maxlen=max_history),
            "cpu_usage": deque(maxlen=max_history),
            "request_count": deque(maxlen=max_history),
            "error_count": deque(maxlen=max_history)
        }
        self.request_count = 0
        self.error_count = 0
        self.start_time = time.time()
        self.lock = threading.Lock()
        
        # 启动系统监控线程
        self.monitor_thread = threading.Thread(target=self._monitor_system, daemon=True)
        self.monitor_thread.start()
    
    def _monitor_system(self):
        """监控系统资源"""
        while True:
            try:
                # 记录内存使用
                memory_info = psutil.virtual_memory()
                self.metrics_history["memory_usage"].append({
                    "timestamp": time.time(),
                    "value": memory_info.percent
                })
                
                # 记录CPU使用
                cpu_percent = psutil.cpu_percent(interval=1)
                self.metrics_history["cpu_usage"].append({
                    "timestamp": time.time(),
                    "value": cpu_percent
                })
                
                time.sleep(5)  # 每5秒记录一次
            except Exception as e:
                logger.error(f"Error monitoring system: {e}")
                time.sleep(10)  # 出错时等待更长时间
    
    def record_request(self, response_time: float, success: bool = True):
        """记录请求"""
        with self.lock:
            self.request_count += 1
            if not success:
                self.error_count += 1
            
            self.metrics_history["response_time"].append({
                "timestamp": time.time(),
                "value": response_time
            })
            
            self.metrics_history["request_count"].append({
                "timestamp": time.time(),
                "value": self.request_count
            })
            
            self.metrics_history["error_count"].append({
                "timestamp": time.time(),
                "value": self.error_count
            })
    
    def get_metrics(self) -> Dict[str, Any]:
        """获取性能指标"""
        with self.lock:
            uptime = time.time() - self.start_time
            
            # 计算平均响应时间
            response_times = [m["value"] for m in self.metrics_history["response_time"]]
            avg_response_time = sum(response_times) / len(response_times) if response_times else 0
            
            # 计算错误率
            error_rate = (self.error_count / self.request_count * 100) if self.request_count > 0 else 0
            
            # 获取最新的系统资源使用情况
```

            elif query.query_type == QueryType.KEYWORD:

            elif query.query_type == QueryType.ASSOCIATIVE:

            elif query.query_type == QueryType.HYBRID:
                return self._hybrid_retrieval(query)
            else:
                return []
                
        except Exception as e:

    def _semantic_retrieval(self, query: MemoryQuery) -> List[MemoryData]:
        """语义检索"""
        # 编码查询
        query_vector = self.encoder.encode(query.text)
        
        # 向量检索
        vector_results = self.vector_storage.search(
            query_vector, 
            limit=query.limit or self.config.max_retrieval_count
        )
        
        # 获取记忆ID和相似度分数
        memory_ids = [result[0] for result in vector_results]
        similarity_scores = [result[1] for result in vector_results]
        
        # 从关系数据库获取详细记忆数据
        memories = self.relational_storage.get_memories(memory_ids)
        
        # 添加相似度分数
        for memory, score in zip(memories, similarity_scores):
            memory.similarity_score = score
        
        # 应用过滤器
        memories = self._apply_filters(memories, query.filters)
        
        # 排序
        memories.sort(key=lambda x: x.similarity_score, reverse=True)
        
        return memories
    
    def _keyword_retrieval(self, query: MemoryQuery) -> List[MemoryData]:
        """关键词检索"""
        # 使用Elasticsearch进行关键词检索
        search_results = self.search_storage.search(
            query.text,
            filters=query.filters,
            limit=query.limit or self.config.max_retrieval_count
        )
        
        # 获取记忆ID和相关性分数
        memory_ids = [result[0] for result in search_results]
        relevance_scores = [result[1] for result in search_results]
        
        # 从关系数据库获取详细记忆数据
        memories = self.relational_storage.get_memories(memory_ids)
        
        # 添加相关性分数
        for memory, score in zip(memories, relevance_scores):
            memory.relevance_score = score
        
        # 排序
        memories.sort(key=lambda x: x.relevance_score, reverse=True)
        
        return memories
    
    def _temporal_retrieval(self, query: MemoryQuery) -> List[MemoryData]:
        """时间检索"""
        # 从关系数据库按时间范围检索
        memories = self.relational_storage.get_memories_by_time_range(
            start_time=query.start_time,
            end_time=query.end_time,
            memory_type=query.memory_type,
            limit=query.limit or self.config.max_retrieval_count
        )
        
        # 如果有文本查询，计算语义相似度并重新排序
        if query.text:
            query_vector = self.encoder.encode(query.text)
            for memory in memories:
                memory.similarity_score = np.dot(
                    query_vector, 
                    memory.content_vector

                    np.linalg.norm(query_vector) * np.linalg.norm(memory.content_vector)

        return memories

        if query.retrieval_strategy == RetrievalStrategy.FUSION:

        elif query.retrieval_strategy == RetrievalStrategy.CASCADE:
            return self._cascade_retrieval(query)
        elif query.retrieval_strategy == RetrievalStrategy.INTERLEAVING:
            return self._interleaving_retrieval(query)
        else:
            # 默认使用融合检索
            return self._fusion_retrieval(query)
    
    def _fusion_retrieval(self, query: MemoryQuery) -> List[MemoryData]:
        """融合检索"""
        # 执行多种检索
        semantic_results = self._semantic_retrieval(query) if query.text else []
        keyword_results = self._keyword_retrieval(query) if query.text else []
        temporal_results = self._temporal_retrieval(query) if query.start_time or query.end_time else []
        associative_results = self._associative_retrieval(query) if query.seed_memory_ids else []
        
        # 合并结果并去重
        all_memories = {}
        
        # 添加语义检索结果
        for memory in semantic_results:
            if memory.memory_id not in all_memories:
                all_memories[memory.memory_id] = memory
            else:
                # 合并分数
                existing = all_memories[memory.memory_id]
                existing.similarity_score = max(
                    existing.similarity_score or 0,
                    memory.similarity_score or 0
                )
        
        # 添加关键词检索结果
        for memory in keyword_results:
            if memory.memory_id not in all_memories:
                all_memories[memory.memory_id] = memory
            else:
                # 合并分数
                existing = all_memories[memory.memory_id]
                existing.relevance_score = max(
                    existing.relevance_score or 0,
                    memory.relevance_score or 0
                )
        
        # 添加时间检索结果
        for memory in temporal_results:
            if memory.memory_id not in all_memories:
                all_memories[memory.memory_id] = memory
        
        # 添加关联检索结果
        for memory in associative_results:
            if memory.memory_id not in all_memories:
                all_memories[memory.memory_id] = memory
        
        # 计算综合分数并排序
        result_memories = list(all_memories.values())
        for memory in result_memories:
            # 综合分数计算
            semantic_score = memory.similarity_score or 0
            relevance_score = memory.relevance_score or 0
            importance_score = self._importance_to_score(memory.importance)
            
            memory.composite_score = (
                semantic_score * 0.5 + 
                relevance_score * 0.3 + 
                importance_score * 0.2
            )
        
        # 排序
        result_memories.sort(key=lambda x: x.composite_score, reverse=True)
        
        # 限制结果数量
        limit = query.limit or self.config.max_retrieval_count
        return result_memories[:limit]
    
    def _cascade_retrieval(self, query: MemoryQuery) -> List[MemoryData]:
        """级联检索"""
        # 先执行一种检索，然后基于结果执行另一种检索
        if query.text:
            # 先语义检索
            semantic_results = self._semantic_retrieval(query)
            
            # 如果结果不足，再执行关键词检索
            if len(semantic_results) < (query.limit or self.config.max_retrieval_count) // 2:
                keyword_results = self._keyword_retrieval(query)
                
                # 合并结果
                memory_ids = {m.memory_id for m in semantic_results}
                for memory in keyword_results:
                    if memory.memory_id not in memory_ids:
                        semantic_results.append(memory)
                
                # 重新排序
                for memory in semantic_results:
                    memory.composite_score = (
                        (memory.similarity_score or 0) * 0.7 + 
                        (memory.relevance_score or 0) * 0.3
                    )
                
                semantic_results.sort(key=lambda x: x.composite_score, reverse=True)
            
            return semantic_results[:query.limit or self.config.max_retrieval_count]
        else:
            # 没有文本查询，使用时间检索或关联检索
            if query.start_time or query.end_time:
                return self._temporal_retrieval(query)
            elif query.seed_memory_ids:
                return self._associative_retrieval(query)
            else:
                return []
    
    def _interleaving_retrieval(self, query: MemoryQuery) -> List[MemoryData]:
        """交错检索"""
        # 执行多种检索
        semantic_results = self._semantic_retrieval(query) if query.text else []
        keyword_results = self._keyword_retrieval(query) if query.text else []
        temporal_results = self._temporal_retrieval(query) if query.start_time or query.end_time else []
        associative_results = self._associative_retrieval(query) if query.seed_memory_ids else []
        
        # 交错合并结果
        result_memories = []
        max_length = max(len(semantic_results), len(keyword_results), 
                         len(temporal_results), len(associative_results))
        
        for i in range(max_length):
            if i < len(semantic_results):
                result_memories.append(semantic_results[i])
            if i < len(keyword_results):
                result_memories.append(keyword_results[i])
            if i < len(temporal_results):
                result_memories.append(temporal_results[i])
            if i < len(associative_results):
                result_memories.append(associative_results[i])
        
        # 去重
        seen_ids = set()
        unique_memories = []
        for memory in result_memories:
            if memory.memory_id not in seen_ids:
                seen_ids.add(memory.memory_id)
                unique_memories.append(memory)
        
        # 限制结果数量
        limit = query.limit or self.config.max_retrieval_count
        return unique_memories[:limit]
    
    def _apply_filters(self, memories: List[MemoryData], filters: Dict) -> List[MemoryData]:
        """应用过滤器"""
        if not filters:
            return memories
        
        filtered_memories = []
        for memory in memories:
            include = True
            
            # 记忆类型过滤
            if 'memory_type' in filters and memory.memory_type != filters['memory_type']:
                include = False
            
            # 重要性过滤
            if 'importance' in filters and memory.importance != filters['importance']:
                include = False
            
            # 情感类型过滤
            if 'emotion_type' in filters and getattr(memory, 'emotion_type', None) != filters['emotion_type']:
                include = False
            
            # 时间范围过滤
            if 'start_time' in filters and memory.created_at < filters['start_time']:
                include = False
            if 'end_time' in filters and memory.created_at > filters['end_time']:
                include = False
            
            # 标签过滤
            if 'tags' in filters:
                memory_tags = set(getattr(memory, 'tags', []))
                filter_tags = set(filters['tags'])
                if not memory_tags.intersection(filter_tags):
                    include = False
            
            if include:
                filtered_memories.append(memory)
        
        return filtered_memories
    
    def _importance_to_score(self, importance) -> float:
        """将重要性转换为分数"""
        if importance == MemoryImportance.HIGH:
            return 1.0
        elif importance == MemoryImportance.MEDIUM:
            return 0.7
        elif importance == MemoryImportance.LOW:
            return 0.4
        elif importance == MemoryImportance.MINIMAL:
            return 0.1
        else:
            return 0.5

```python
from typing import Dict, List, Optional, Tuple, Any

class MemoryConsolidation:

    def consolidate_memories(self, memory_ids: List[str] = None) -> Dict[str, Any]:
        """巩固记忆"""
        try:
            # 如果没有指定记忆ID，获取需要巩固的记忆
            if memory_ids is None:
                memory_ids = self._get_memories_for_consolidation()
            
            if not memory_ids:
                return {"status": "success", "message": "No memories to consolidate"}
            
            # 获取记忆数据
            memories = self.relational_storage.get_memories(memory_ids)
            
            # 按类型分组
            memories_by_type = self._group_memories_by_type(memories)
            
            # 对每种类型的记忆执行巩固
            consolidation_results = {}
            for memory_type, type_memories in memories_by_type.items():
                if memory_type == MemoryType.EPISODIC:
                    result = self._consolidate_episodic_memories(type_memories)
                elif memory_type == MemoryType.SEMANTIC:
                    result = self._consolidate_semantic_memories(type_memories)
                elif memory_type == MemoryType.PROCEDURAL:
                    result = self._consolidate_procedural_memories(type_memories)
                elif memory_type == MemoryType.EMOTIONAL:
                    result = self._consolidate_emotional_memories(type_memories)
                else:
                    result = {"status": "skipped", "message": f"Unsupported memory type: {memory_type}"}
                
                consolidation_results[memory_type.value] = result
            
            # 更新记忆状态
            self._update_memory_consolidation_status(memory_ids)
            
            return {
                "status": "success",
                "consolidated_count": len(memory_ids),
                "results": consolidation_results
            }
            
        except Exception as e:
            logger.error(f"Error consolidating memories: {e}")
            return {"status": "error", "message": str(e)}
    
    def _get_memories_for_consolidation(self) -> List[str]:
        """获取需要巩固的记忆ID"""
        # 获取最近创建但未巩固的记忆
        cutoff_time = datetime.now() - timedelta(seconds=self.config.consolidation_interval)
        memories = self.relational_storage.get_unconsolidated_memories(cutoff_time)
        
        # 按重要性排序，优先巩固重要记忆
        memories.sort(key=lambda x: self._importance_to_score(x.importance), reverse=True)
        
        # 限制数量
        max_consolidation_count = 100  # 每次最多巩固100个记忆
        return [memory.memory_id for memory in memories[:max_consolidation_count]]
    
    def _group_memories_by_type(self, memories: List[MemoryData]) -> Dict[MemoryType, List[MemoryData]]:
        """按类型分组记忆"""
        memories_by_type = {}
        for memory in memories:
            if memory.memory_type not in memories_by_type:
                memories_by_type[memory.memory_type] = []
            memories_by_type[memory.memory_type].append(memory)
        return memories_by_type
    
    def _consolidate_episodic_memories(self, memories: List[MemoryData]) -> Dict[str, Any]:
        """巩固情景记忆"""
        try:
            # 识别相关记忆群组
            memory_clusters = self._cluster_episodic_memories(memories)
            
            consolidated_memories = []
            for cluster in memory_clusters:
                if len(cluster) == 1:
                    # 单个记忆，只需更新状态
                    consolidated_memories.append(cluster[0])
                else:
                    # 多个相关记忆，创建摘要记忆
                    summary_memory = self._create_episodic_summary(cluster)
                    consolidated_memories.append(summary_memory)
                    
                    # 创建记忆间的关联
                    self._create_memory_associations(cluster, summary_memory.memory_id)
            
            # 更新记忆向量
            for memory in consolidated_memories:
                if memory.content_vector is None:
                    memory.content_vector = self.encoder.encode(memory.content)
                    self.vector_storage.update(memory.memory_id, memory.content_vector)
            
            return {
                "status": "success",
                "original_count": len(memories),
                "consolidated_count": len(consolidated_memories),
                "clusters_created": len(memory_clusters)
            }
            
        except Exception as e:
            logger.error(f"Error consolidating episodic memories: {e}")
            return {"status": "error", "message": str(e)}
    
    def _consolidate_semantic_memories(self, memories: List[MemoryData]) -> Dict[str, Any]:
        """巩固语义记忆"""
        try:
            # 识别相似概念
            concept_clusters = self._cluster_semantic_memories(memories)
            
            consolidated_memories = []
            for cluster in concept_clusters:
                if len(cluster) == 1:
                    # 单个记忆，只需更新状态
                    consolidated_memories.append(cluster[0])
                else:
                    # 多个相似概念，创建更抽象的概念
                    abstract_concept = self._create_abstract_concept(cluster)
                    consolidated_memories.append(abstract_concept)
                    
                    # 创建概念间的层次关系
                    self._create_concept_hierarchy(cluster, abstract_concept.memory_id)
            
            # 更新记忆向量
            for memory in consolidated_memories:
                if memory.content_vector is None:
                    memory.content_vector = self.encoder.encode(memory.content)
                    self.vector_storage.update(memory.memory_id, memory.content_vector)
            
            return {
                "status": "success",
                "original_count": len(memories),
                "consolidated_count": len(consolidated_memories),
                "concepts_created": len(concept_clusters)
            }
            
        except Exception as e:
            logger.error(f"Error consolidating semantic memories: {e}")
            return {"status": "error", "message": str(e)}
    
    def _consolidate_procedural_memories(self, memories: List[MemoryData]) -> Dict[str, Any]:

    def _cluster_episodic_memories(self, memories: List[MemoryData]) -> List[List[MemoryData]]:
        """聚类情景记忆"""
        # 基于时间、地点和参与者的相似性聚类
        clusters = []
        unclustered = memories.copy()
        
        while unclustered:
            # 取第一个记忆作为新簇的中心
            center_memory = unclustered.pop(0)
            cluster = [center_memory]
            
            # 查找相似的记忆
            i = 0
            while i < len(unclustered):
                memory = unclustered[i]
                if self._episodic_similarity(center_memory, memory) > 0.7:
                    cluster.append(memory)
                    unclustered.pop(i)
                else:
                    i += 1
            
            clusters.append(cluster)
        
        return clusters

    def _episodic_similarity(self, memory1: MemoryData, memory2: MemoryData) -> float:
        """计算情景记忆相似度"""
        # 获取记忆元数据
        metadata1 = getattr(memory1, 'metadata', {})
        metadata2 = getattr(memory2, 'metadata', {})
        
        # 时间相似度（同一天内）
        time_diff = abs((memory1.created_at - memory2.created_at).total_seconds())
        time_similarity = 1.0 - min(time_diff / 86400, 1.0)  # 一天内的相似度
        
        # 地点相似度
        location1 = metadata1.get('location', '')
        location2 = metadata2.get('location', '')
        location_similarity = 1.0 if location1 == location2 and location1 else 0.0
        
        # 参与者相似度
        participants1 = set(metadata1.get('participants', []))
        participants2 = set(metadata2.get('participants', []))
        if participants1 and participants2:
            intersection = participants1.intersection(participants2)
            union = participants1.union(participants2)
            participant_similarity = len(intersection) / len(union)
        else:
            participant_similarity = 0.0
        
        # 语义相似度
        if memory1.content_vector is not None and memory2.content_vector is not None:
            semantic_similarity = np.dot(memory1.content_vector, memory2.content_vector) / (
                np.linalg.norm(memory1.content_vector) * np.linalg.norm(memory2.content_vector)
            )
        else:
            semantic_similarity = 0.0
        
        # 综合相似度
        return (
            time_similarity * 0.3 +
            location_similarity * 0.3 +
            participant_similarity * 0.2 +
            semantic_similarity * 0.2
        )
    
    def _semantic_similarity(self, memory1: MemoryData, memory2: MemoryData) -> float:
        """计算语义记忆相似度"""
        if memory1.content_vector is not None and memory2.content_vector is not None:
            return np.dot(memory1.content_vector, memory2.content_vector) / (
                np.linalg.norm(memory1.content_vector) * np.linalg.norm(memory2.content_vector)
            )
        return 0.0
    
    def _procedural_similarity(self, memory1: MemoryData, memory2: MemoryData) -> float:
        """计算程序性记忆相似度"""
        # 获取记忆元数据
        metadata1 = getattr(memory1, 'metadata', {})
        metadata2 = getattr(memory2, 'metadata', {})
        
        # 技能相似度
        skill1 = metadata1.get('skill', '')
        skill2 = metadata2.get('skill', '')
        skill_similarity = 1.0 if skill1 == skill2 and skill1 else 0.0
        
        # 目标相似度
        goal1 = metadata1.get('goal', '')
        goal2 = metadata2.get('goal', '')
        goal_similarity = 1.0 if goal1 == goal2 and goal1 else 0.0
        
        # 上下文相似度
        context1 = metadata1.get('context', '')
        context2 = metadata2.get('context', '')
        context_similarity = 1.0 if context1 == context2 and context1 else 0.0
        
        # 语义相似度
        if memory1.content_vector is not None and memory2.content_vector is not None:
            semantic_similarity = np.dot(memory1.content_vector, memory2.content_vector) / (
                np.linalg.norm(memory1.content_vector) * np.linalg.norm(memory2.content_vector)
            )
        else:
            semantic_similarity = 0.0
        
        # 综合相似度
        return (
            skill_similarity * 0.4 +
            goal_similarity * 0.3 +
            context_similarity * 0.1 +
            semantic_similarity * 0.2
        )

        # 生成概念名称和描述

        description = f"与{', '.join(common_keywords)}相关的抽象概念"
        
        return {

        }

    def _calculate_consolidated_importance(self, memories: List[MemoryData]) -> MemoryImportance:
        """计算巩固后记忆的重要性"""
        # 计算平均重要性
        importance_scores = []
        for memory in memories:
            importance_scores.append(self._importance_to_score(memory.importance))
        
        avg_importance = sum(importance_scores) / len(importance_scores)
        
        # 巩固后记忆的重要性略高于原始记忆的平均值
        consolidated_importance = min(avg_importance * 1.1, 1.0)
        
        if consolidated_importance >= 0.8:
            return MemoryImportance.HIGH
        elif consolidated_importance >= 0.5:
            return MemoryImportance.MEDIUM
        elif consolidated_importance >= 0.2:
            return MemoryImportance.LOW
        else:
            return MemoryImportance.MINIMAL

        elif importance == MemoryImportance.MEDIUM:
            return 0.7
        elif importance == MemoryImportance.LOW:
            return 0.4
        elif importance == MemoryImportance.MINIMAL:
            return 0.1
        else:
            return 0.5
```

```python
from typing import Dict, List, Optional, Tuple, Any
from datetime import datetime, timedelta

class MemoryForgetting:

                    result = self._forget_procedural_memories(type_memories)
                elif memory_type == MemoryType.EMOTIONAL:

                else:
                    result = {"status": "skipped", "message": f"Unsupported memory type: {memory_type}"}
                
                forgetting_results[memory_type.value] = result
            
            # 更新记忆状态
            self._update_memory_forgetting_status(memory_ids)
            
            return {
                "status": "success",
                "forgotten_count": len(memory_ids),

            logger.error(f"Error forgetting memories: {e}")
            return {"status": "error", "message": str(e)}
    
    def _get_memories_for_forgetting(self) -> List[str]:
        """获取需要遗忘的记忆ID"""
        # 获取低重要性且长时间未访问的记忆
        cutoff_time = datetime.now() - timedelta(days=30)  # 30天未访问
        memories = self.relational_storage.get_memories_for_forgetting(
            importance_threshold=self.config.forgetting_threshold,
            last_accessed_before=cutoff_time
        )
        
        # 按遗忘优先级排序
        memories.sort(key=lambda x: self._calculate_forgetting_priority(x), reverse=True)
        
        # 限制数量
        max_forgetting_count = 50  # 每次最多遗忘50个记忆
        return [memory.memory_id for memory in memories[:max_forgetting_count]]
    
    def _group_memories_by_type(self, memories: List[MemoryData]) -> Dict[MemoryType, List[MemoryData]]:
        """按类型分组记忆"""
        memories_by_type = {}
        for memory in memories:
            if memory.memory_type not in memories_by_type:
                memories_by_type[memory.memory_type] = []
            memories_by_type[memory.memory_type].append(memory)
        return memories_by_type
    
    def _forget_episodic_memories(self, memories: List[MemoryData]) -> Dict[str, Any]:
        """遗忘情景记忆"""
        try:
            # 检查是否有关联的重要记忆
            protected_memories = []
            forgotten_memories = []
            
            for memory in memories:
                if self._is_protected_episodic_memory(memory):
                    protected_memories.append(memory)
                else:
                    forgotten_memories.append(memory)
            
            # 删除遗忘的记忆
            for memory in forgotten_memories:
                self._delete_memory(memory.memory_id)
            
            # 对保护的记忆降低重要性
            for memory in protected_memories:
                self._decrease_memory_importance(memory.memory_id)
            
            return {
                "status": "success",
                "original_count": len(memories),
                "forgotten_count": len(forgotten_memories),
                "protected_count": len(protected_memories)
            }
            
        except Exception as e:
            logger.error(f"Error forgetting episodic memories: {e}")
            return {"status": "error", "message": str(e)}
    
    def _forget_semantic_memories(self, memories: List[MemoryData]) -> Dict[str, Any]:
        """遗忘语义记忆"""
        try:
            # 检查是否是基础概念
            protected_memories = []
            forgotten_memories = []
            
            for memory in memories:
                if self._is_fundamental_concept(memory):
                    protected_memories.append(memory)
                else:
                    forgotten_memories.append(memory)
            
            # 删除遗忘的记忆
            for memory in forgotten_memories:
                self._delete_memory(memory.memory_id)
            
            # 对保护的记忆降低重要性
            for memory in protected_memories:
                self._decrease_memory_importance(memory.memory_id)
            
            return {
                "status": "success",
                "original_count": len(memories),
                "forgotten_count": len(forgotten_memories),
                "protected_count": len(protected_memories)
            }
            
        except Exception as e:
            logger.error(f"Error forgetting semantic memories: {e}")
            return {"status": "error", "message": str(e)}
    
    def _forget_procedural_memories(self, memories: List[MemoryData]) -> Dict[str, Any]:
        """遗忘程序性记忆"""
        try:
            # 检查是否是常用技能
            protected_memories = []
            forgotten_memories = []
            
            for memory in memories:
                if self._is_frequently_used_skill(memory):
                    protected_memories.append(memory)
                else:
                    forgotten_memories.append(memory)
            
            # 删除遗忘的记忆
            for memory in forgotten_memories:
                self._delete_memory(memory.memory_id)
            
            # 对保护的记忆降低重要性
            for memory in protected_memories:

            return {
                "status": "success",
                "original_count": len(memories),
                "forgotten_count": len(forgotten_memories),
                "protected_count": len(protected_memories)
            }
            
        except Exception as e:

            return {"status": "error", "message": str(e)}
    
    def _forget_emotional_memories(self, memories: List[MemoryData]) -> Dict[str, Any]:
        """遗忘情感记忆"""
        try:
            # 检查是否是强烈情感记忆
            protected_memories = []
            forgotten_memories = []
            
            for memory in memories:
                if self._is_strong_emotional_memory(memory):
                    protected_memories.append(memory)
                else:
                    forgotten_memories.append(memory)
            
            # 删除遗忘的记忆
            for memory in forgotten_memories:
                self._delete_memory(memory.memory_id)
            
            # 对保护的记忆降低重要性
            for memory in protected_memories:
                self._decrease_memory_importance(memory.memory_id)
            
            return {
                "status": "success",
                "original_count": len(memories),
                "forgotten_count": len(forgotten_memories),
                "protected_count": len(protected_memories)
            }
            
        except Exception as e:
            logger.error(f"Error forgetting emotional memories: {e}")
            return {"status": "error", "message": str(e)}
    
    def _is_protected_episodic_memory(self, memory: MemoryData) -> bool:
        """检查是否是受保护的情景记忆"""
        # 检查是否与重要记忆相关联
        associated_memories = self.graph_storage.get_associated_memories(
            [memory.memory_id], 
            max_depth=2
        )
        
        for associated_id in associated_memories:
            associated_memory = self.relational_storage.get_memory(associated_id)
            if associated_memory and self._importance_to_score(associated_memory.importance) > 0.8:
                return True
        
        # 检查是否是重要事件
        metadata = getattr(memory, 'metadata', {})
        if metadata.get('is_important_event', False):
            return True
        
        return False
    
    def _is_fundamental_concept(self, memory: MemoryData) -> bool:
        """检查是否是基础概念"""
        # 检查是否被许多其他概念引用
        referencing_memories = self.graph_storage.get_referencing_memories(memory.memory_id)
        if len(referencing_memories) > 5:  # 被超过5个其他记忆引用
            return True
        
        # 检查是否是基础类别
        metadata = getattr(memory, 'metadata', {})
        if metadata.get('is_fundamental', False):
            return True
        
        return False
    
    def _is_frequently_used_skill(self, memory: MemoryData) -> bool:
        """检查是否是常用技能"""
        # 检查最近使用频率
        metadata = getattr(memory, 'metadata', {})
        usage_count = metadata.get('usage_count', 0)
        last_used = metadata.get('last_used', None)
        
        # 最近使用过或使用频率高
        if last_used and (datetime.now() - last_used).days < 7:
            return True
        
        if usage_count > 10:
            return True
        
        return False
    
    def _is_strong_emotional_memory(self, memory: MemoryData) -> bool:
        """检查是否是强烈情感记忆"""
        # 检查情感强度
        metadata = getattr(memory, 'metadata', {})
        emotional_intensity = metadata.get('emotional_intensity', 0.5)
        
        if emotional_intensity > 0.8:
            return True
        
        # 检查是否是重要情感事件
        if metadata.get('is_important_emotion', False):
            return True
        
        return False
    
    def _delete_memory(self, memory_id: str):
        """删除记忆"""
        # 从向量数据库删除
        self.vector_storage.delete(memory_id)
        
        # 从关系数据库删除
        self.relational_storage.delete_memory(memory_id)
        
        # 从图数据库删除
        self.graph_storage.delete_memory_node(memory_id)
    
    def _decrease_memory_importance(self, memory_id: str):
        """降低记忆重要性"""
        memory = self.relational_storage.get_memory(memory_id)
        if not memory:
            return
        
        # 降低重要性
        current_score = self._importance_to_score(memory.importance)
        new_score = max(current_score - 0.2, 0.1)
        
        if new_score >= 0.8:
            new_importance = MemoryImportance.HIGH
        elif new_score >= 0.5:
            new_importance = MemoryImportance.MEDIUM
        elif new_score >= 0.2:
            new_importance = MemoryImportance.LOW
        else:
            new_importance = MemoryImportance.MINIMAL
        
        # 更新记忆
        self.relational_storage.update_memory(memory_id, {"importance": new_importance})
    
    def _calculate_forgetting_priority(self, memory: MemoryData) -> float:
        """计算遗忘优先级"""
        # 重要性越低，优先级越高
        importance_score = 1.0 - self._importance_to_score(memory.importance)
        
        # 访问时间越早，优先级越高
        last_accessed = getattr(memory, 'last_accessed', memory.created_at)
        days_since_access = (datetime.now() - last_accessed).days
        time_score = min(days_since_access / 30, 1.0)  # 30天以上访问的得满分
        
        # 访问次数越少，优先级越高
        access_count = getattr(memory, 'access_count', 0)
        access_score = 1.0 - min(access_count / 10, 1.0)  # 10次以上访问的得0分
        
        # 综合优先级
        return (
            importance_score * 0.4 +
            time_score * 0.4 +
            access_score * 0.2
        )
    
    def _importance_to_score(self, importance) -> float:
        """将重要性转换为分数"""
        if importance == MemoryImportance.HIGH:
            return 1.0
        elif importance == MemoryImportance.MEDIUM:
            return 0.7
        elif importance == MemoryImportance.LOW:
            return 0.4
        elif importance == MemoryImportance.MINIMAL:
            return 0.1
        else:
            return 0.5
    
    def _update_memory_forgetting_status(self, memory_ids: List[str]):
        """更新记忆遗忘状态"""
        for memory_id in memory_ids:
            self.relational_storage.update_memory_forgetting_status(memory_id, True)
```
