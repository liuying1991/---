# 自我调整模块实现工作流

## 1. 模块概述

### 1.1 模块功能与职责

自我调整模块是婴儿AI管家系统自我意识子系统的核心组件之一，负责根据系统内部状态和外部反馈动态调整自身行为和参数。该模块具有以下主要功能：

1. **参数自适应调整**：根据系统运行状态自动调整内部参数
2. **行为策略优化**：基于反馈优化行为决策策略
3. **学习率调整**：动态调整学习组件的学习率
4. **资源分配优化**：根据任务优先级和系统负载优化资源分配
5. **错误恢复机制**：检测并纠正系统行为偏差

### 1.2 模块在系统中的位置

自我调整模块位于自我意识子系统中，与自我评价模块、自我识别模块等紧密协作：

```
自我意识子系统
├── 自我识别模块 (识别自身状态和能力) 
├── 自我评价模块 (评估自身表现)
├── 自我调整模块 (调整自身行为和参数)  ← 当前实现
└── 自我意识整合模块 (整合自我意识信息)
```

### 1.3 关键技术选型

1. **强化学习算法**：基于PPO、A3C等算法实现自适应调整
2. **贝叶斯优化**：用于超参数自动调优
3. **反馈控制理论**：基于PID控制器实现精确调整
4. **元学习**：实现快速适应新环境的能力
5. **多目标优化**：平衡多个调整目标的冲突

```python
# core/parameter_adjuster.py
import numpy as np
from typing import Dict, List, Any, Optional, Tuple
import logging
from abc import ABC, abstractmethod

from models.adjustment_request import AdjustmentRequest
from models.system_state import SystemState
from models.feedback_data import FeedbackData
from algorithms.bayesian_optimization import BayesianOptimizer
from controllers.pid_controller import PIDController
from config.adjustment_config import AdjustmentConfig

logger = logging.getLogger(__name__)

class ParameterAdjuster(ABC):
    """参数调整器抽象基类"""
    
    @abstractmethod
    def adjust(self, request: AdjustmentRequest) -> Dict[str, Any]:
        """调整参数"""
        pass

class RLParameterAdjuster(ParameterAdjuster):
    """基于强化学习的参数调整器"""
    
    def __init__(self, config: AdjustmentConfig):
        self.config = config
        self.optimizer = BayesianOptimizer(config.optimization)
        self.pid_controllers = {}
        
        # 初始化PID控制器
        for param_name, param_config in config.parameters.items():
            if param_config.use_pid:
                self.pid_controllers[param_name] = PIDController(
                    kp=param_config.kp,
                    ki=param_config.ki,
                    kd=param_config.kd,
                    setpoint=param_config.target_value
                )
        
        logger.info("RLParameterAdjuster initialized")
    
    def adjust(self, request: AdjustmentRequest) -> Dict[str, Any]:
        """使用强化学习调整参数"""
        try:
            current_state = request.current_state
            target_state = request.target_state
            feedback = request.feedback
            
            adjustments = {}
            
            # 对每个参数进行调整
            for param_name, param_config in self.config.parameters.items():
                if param_name in current_state.parameters:
                    current_value = current_state.parameters[param_name]
                    
                    # 使用PID控制器进行初步调整
                    if param_name in self.pid_controllers:
                        pid_adjustment = self.pid_controllers[param_name].update(
                            current_value, target_state.parameters.get(param_name, current_value)
                        )
                        
                        # 使用贝叶斯优化进一步调整
                        final_adjustment = self.optimizer.optimize(
                            param_name, current_value, pid_adjustment, feedback
                        )
                        
                        # 确保调整在允许范围内
                        final_adjustment = np.clip(
                            final_adjustment, 
                            param_config.min_value, 
                            param_config.max_value
                        )
                        
                        adjustments[param_name] = {
                            "old_value": current_value,
                            "new_value": final_adjustment,
                            "adjustment_magnitude": abs(final_adjustment - current_value),
                            "adjustment_direction": "increase" if final_adjustment > current_value else "decrease"
                        }
            
            return {
                "adjustments": adjustments,
                "adjustment_strategy": "reinforcement_learning",
                "confidence": self._calculate_confidence(adjustments, feedback),
                "expected_impact": self._estimate_impact(adjustments)
            }
            
        except Exception as e:
            logger.error(f"Error adjusting parameters: {e}")
            raise
    
    def _calculate_confidence(self, adjustments: Dict[str, Any], feedback: FeedbackData) -> float:
        """计算调整置信度"""
        # 基于历史调整成功率和当前反馈质量计算置信度
        base_confidence = 0.7
        
        # 根据反馈质量调整置信度
        feedback_quality = feedback.quality_score if hasattr(feedback, 'quality_score') else 0.5
        confidence = base_confidence * (0.5 + feedback_quality)
        
        # 根据调整幅度调整置信度
        avg_adjustment = np.mean([adj["adjustment_magnitude"] for adj in adjustments.values()])
        if avg_adjustment > 0.5:  # 大幅调整降低置信度
            confidence *= 0.8
            
        return min(confidence, 1.0)
    
    def _estimate_impact(self, adjustments: Dict[str, Any]) -> Dict[str, float]:
        """估算调整的影响"""
        impact = {
            "performance_impact": 0.0,
            "stability_impact": 0.0,
            "resource_impact": 0.0
        }
        
        for param_name, adj in adjustments.items():
            param_config = self.config.parameters.get(param_name)
            if param_config:
                # 根据参数配置估算影响
                impact["performance_impact"] += adj["adjustment_magnitude"] * param_config.performance_weight
                impact["stability_impact"] += adj["adjustment_magnitude"] * param_config.stability_weight
                impact["resource_impact"] += adj["adjustment_magnitude"] * param_config.resource_weight
        
        return impact

class MetaLearningParameterAdjuster(ParameterAdjuster):
```

### 3.2 行为优化器

```python
# core/behavior_optimizer.py
import numpy as np
from typing import Dict, List, Any, Optional, Tuple
import logging
from enum import Enum

from models.adjustment_request import AdjustmentRequest
from models.system_state import SystemState
from models.feedback_data import FeedbackData
from algorithms.reinforcement_learning import PPOAgent, A3CAgent
from algorithms.multi_objective_opt import MultiObjectiveOptimizer
from config.adjustment_config import AdjustmentConfig

logger = logging.getLogger(__name__)

class BehaviorStrategy(Enum):
    """行为策略枚举"""
    CONSERVATIVE = "conservative"  # 保守策略
    AGGRESSIVE = "aggressive"      # 激进策略
    BALANCED = "balanced"          # 平衡策略
    ADAPTIVE = "adaptive"          # 自适应策略

class BehaviorOptimizer:
    """行为优化器"""
    
    def __init__(self, config: AdjustmentConfig):
        self.config = config
        self.rl_agents = {}
        self.multi_objective_optimizer = MultiObjectiveOptimizer(config.multi_objective)
        
        # 初始化强化学习智能体
        for strategy in BehaviorStrategy:
            self.rl_agents[strategy] = PPOAgent(config.rl[strategy.value])
        
        self.current_strategy = BehaviorStrategy.BALANCED
        self.strategy_performance = {strategy: [] for strategy in BehaviorStrategy}
        
        logger.info("BehaviorOptimizer initialized")
    
    def optimize_behavior(self, request: AdjustmentRequest) -> Dict[str, Any]:
        """优化行为策略"""
        try:
            current_state = request.current_state
            feedback = request.feedback
            
            # 1. 评估当前策略性能
            current_performance = self._evaluate_strategy_performance(
                self.current_strategy, feedback
            )
            
            # 2. 选择最佳策略
            best_strategy = self._select_best_strategy(current_state, feedback)
            
            # 3. 如果策略改变，执行策略切换
            if best_strategy != self.current_strategy:
                self._switch_strategy(best_strategy)
            
            # 4. 使用当前策略的智能体生成行为
            agent = self.rl_agents[self.current_strategy]
            behavior = agent.generate_behavior(current_state, feedback)
            
            # 5. 多目标优化
            optimized_behavior = self.multi_objective_optimizer.optimize(
                behavior, current_state, feedback
            )
            
            # 6. 更新策略性能记录
            self.strategy_performance[self.current_strategy].append(current_performance)
            
            return {
                "optimized_behavior": optimized_behavior,
                "strategy": self.current_strategy.value,
                "strategy_performance": current_performance,
                "strategy_switch": best_strategy != self.current_strategy,
                "confidence": self._calculate_behavior_confidence(optimized_behavior, feedback)
            }
            
        except Exception as e:
            logger.error(f"Error optimizing behavior: {e}")
            raise
    
    def _evaluate_strategy_performance(self, strategy: BehaviorStrategy, feedback: FeedbackData) -> float:
        """评估策略性能"""
        if not feedback.metrics:
            return 0.5
        
        # 计算综合性能指标
        performance = 0.0
        weights = self.config.strategy_performance_weights
        
        for metric_name, metric_value in feedback.metrics.items():
            if metric_name in weights:
                performance += metric_value * weights[metric_name]
        
        # 考虑历史性能
        if self.strategy_performance[strategy]:
            recent_performance = np.mean(self.strategy_performance[strategy][-5:])  # 最近5次
            performance = 0.7 * performance + 0.3 * recent_performance
        
        return performance
    
    def _select_best_strategy(self, current_state: SystemState, feedback: FeedbackData) -> BehaviorStrategy:
        """选择最佳策略"""
        # 基于当前状态和反馈选择策略
        strategy_scores = {}
        
        for strategy in BehaviorStrategy:
            # 评估每个策略在当前状态下的预期性能
            agent = self.rl_agents[strategy]
            expected_performance = agent.predict_performance(current_state, feedback)
            
            # 考虑历史平均性能
            if self.strategy_performance[strategy]:
                avg_performance = np.mean(self.strategy_performance[strategy])
                expected_performance = 0.6 * expected_performance + 0.4 * avg_performance
            
            strategy_scores[strategy] = expected_performance
        
        # 选择得分最高的策略
        best_strategy = max(strategy_scores, key=strategy_scores.get)
        
        # 如果当前策略性能接近最佳策略，保持不变（避免频繁切换）
        current_score = strategy_scores[self.current_strategy]
        best_score = strategy_scores[best_strategy]
        
        if best_score - current_score < self.config.strategy_switch_threshold:
            return self.current_strategy
        
        return best_strategy
    
    def _switch_strategy(self, new_strategy: BehaviorStrategy) -> None:
        """切换策略"""
        logger.info(f"Switching behavior strategy from {self.current_strategy.value} to {new_strategy.value}")
        
        # 保存当前智能体状态
        current_agent = self.rl_agents[self.current_strategy]
        current_agent.save_state()
        
        # 切换到新策略
        self.current_strategy = new_strategy
        
        # 加载新智能体状态
        new_agent = self.rl_agents[self.current_strategy]
        new_agent.load_state()
    
    def _calculate_behavior_confidence(self, behavior: Dict[str, Any], feedback: FeedbackData) -> float:
        """计算行为置信度"""
        # 基于行为一致性和反馈质量计算置信度
        base_confidence = 0.7
        
        # 根据反馈质量调整
        feedback_quality = feedback.quality_score if hasattr(feedback, 'quality_score') else 0.5
        confidence = base_confidence * (0.5 + feedback_quality)
        
        # 根据行为一致性调整
        behavior_consistency = self._calculate_behavior_consistency(behavior)
        confidence *= (0.5 + 0.5 * behavior_consistency)
        
        return min(confidence, 1.0)
    
    def _calculate_behavior_consistency(self, behavior: Dict[str, Any]) -> float:
        """计算行为一致性"""
        # 实现行为一致性计算逻辑
        # 可以比较当前行为与历史行为的相似度
        return 0.8  # 占位符
```

### 3.3 学习率控制器

```python
# core/learning_rate_controller.py
import numpy as np
from typing import Dict, List, Any, Optional, Tuple
import logging
from enum import Enum

from models.adjustment_request import AdjustmentRequest
from models.system_state import SystemState
from models.feedback_data import FeedbackData
from algorithms.meta_learning import MetaLearningController
from controllers.adaptive_controller import AdaptiveController
from config.adjustment_config import AdjustmentConfig

logger = logging.getLogger(__name__)

class LearningRateSchedule(Enum):
    """学习率调度策略枚举"""
    FIXED = "fixed"                  # 固定学习率
    STEP_DECAY = "step_decay"        # 步长衰减
    EXPONENTIAL_DECAY = "exp_decay"  # 指数衰减
    COSINE_ANNEALING = "cosine"      # 余弦退火
    ADAPTIVE = "adaptive"            # 自适应
    META_LEARNING = "meta_learning"  # 元学习控制

class LearningRateController:
    """学习率控制器"""
    
    def __init__(self, config: AdjustmentConfig):
        self.config = config
        self.adaptive_controllers = {}
        self.meta_controller = MetaLearningController(config.meta_learning)
        
        # 初始化自适应控制器
        for component_name, component_config in config.learning_components.items():
            self.adaptive_controllers[component_name] = AdaptiveController(
                component_config.initial_lr,
                component_config.min_lr,
                component_config.max_lr,
                component_config.adjustment_factor
            )
        
        logger.info("LearningRateController initialized")
    
    def adjust_learning_rates(self, request: AdjustmentRequest) -> Dict[str, Any]:
        """调整学习率"""
        try:
            current_state = request.current_state
            feedback = request.feedback
            
            adjustments = {}
            
            # 对每个组件的学习率进行调整
            for component_name, controller in self.adaptive_controllers.items():
                if component_name in current_state.learning_rates:
                    current_lr = current_state.learning_rates[component_name]
                    
                    # 获取组件特定的反馈
                    component_feedback = self._get_component_feedback(
                        component_name, feedback
                    )
                    
                    # 根据配置的策略调整学习率
                    schedule = self.config.learning_components[component_name].schedule
                    
                    if schedule == LearningRateSchedule.FIXED:
                        new_lr = current_lr  # 固定不变
                    elif schedule == LearningRateSchedule.STEP_DECAY:
                        new_lr = self._step_decay_adjustment(
                            component_name, current_lr, current_state
                        )
                    elif schedule == LearningRateSchedule.EXPONENTIAL_DECAY:
                        new_lr = self._exponential_decay_adjustment(
                            component_name, current_lr, current_state
                        )
                    elif schedule == LearningRateSchedule.COSINE_ANNEALING:
                        new_lr = self._cosine_annealing_adjustment(
                            component_name, current_lr, current_state
                        )
                    elif schedule == LearningRateSchedule.ADAPTIVE:
                        new_lr = controller.adapt(current_lr, component_feedback)
                    elif schedule == LearningRateSchedule.META_LEARNING:
                        new_lr = self.meta_controller.adjust(
                            component_name, current_lr, component_feedback, current_state
                        )
                    else:
                        new_lr = current_lr  # 默认不变
                    
                    # 确保学习率在允许范围内
                    component_config = self.config.learning_components[component_name]
                    new_lr = np.clip(new_lr, component_config.min_lr, component_config.max_lr)
                    
                    adjustments[component_name] = {
                        "old_lr": current_lr,
                        "new_lr": new_lr,
                        "adjustment_ratio": new_lr / current_lr if current_lr > 0 else 1.0,
                        "schedule": schedule.value
                    }
            
            return {
                "learning_rate_adjustments": adjustments,
                "overall_strategy": self._determine_overall_strategy(),
                "expected_impact": self._estimate_lr_impact(adjustments)
            }
            
        except Exception as e:
            logger.error(f"Error adjusting learning rates: {e}")
            raise
    
    def _get_component_feedback(self, component_name: str, feedback: FeedbackData) -> Dict[str, Any]:
        """获取组件特定的反馈"""
        component_feedback = {}
        
        # 如果反馈中有组件特定的指标，提取出来
        if hasattr(feedback, 'component_metrics') and component_name in feedback.component_metrics:
            component_feedback = feedback.component_metrics[component_name]
        else:
            # 否则使用通用反馈指标
            component_feedback = {
                "loss": feedback.metrics.get("loss", 0.0),
                "accuracy": feedback.metrics.get("accuracy", 0.0),
                "gradient_norm": feedback.metrics.get("gradient_norm", 0.0),
                "validation_loss": feedback.metrics.get("validation_loss", 0.0)
            }
        
        return component_feedback
    
    def _step_decay_adjustment(self, component_name: str, current_lr: float, state: SystemState) -> float:
        """步长衰减调整"""
        component_config = self.config.learning_components[component_name]
        steps = state.training_steps.get(component_name, 0)
        
        # 计算衰减步数
        decay_steps = steps // component_config.decay_steps
        
        # 应用衰减
        new_lr = current_lr * (component_config.decay_rate ** decay_steps)
        
        return new_lr
    
    def _exponential_decay_adjustment(self, component_name: str, current_lr: float, state: SystemState) -> float:
        """指数衰减调整"""
        component_config = self.config.learning_components[component_name]
        steps = state.training_steps.get(component_name, 0)
        
        # 应用指数衰减
        new_lr = current_lr * np.exp(-component_config.decay_rate * steps)
        
        return new_lr
    
    def _cosine_annealing_adjustment(self, component_name: str, current_lr: float, state: SystemState) -> float:
        """余弦退火调整"""
        component_config = self.config.learning_components[component_name]
        steps = state.training_steps.get(component_name, 0)
        t_max = component_config.t_max
        
        # 计算余弦退火
        new_lr = component_config.min_lr + (current_lr - component_config.min_lr) * (
            1 + np.cos(np.pi * steps / t_max)
        ) / 2
        
        return new_lr
    
    def _determine_overall_strategy(self) -> str:
        """确定整体学习率调整策略"""
        # 统计各种调度策略的使用情况
        schedule_counts = {}
        for component_config in self.config.learning_components.values():
            schedule = component_config.schedule
            schedule_counts[schedule] = schedule_counts.get(schedule, 0) + 1
        
        # 找出最常用的策略
        most_common_schedule = max(schedule_counts, key=schedule_counts.get)
        
        return most_common_schedule.value
    
    def _estimate_lr_impact(self, adjustments: Dict[str, Dict[str, Any]]) -> Dict[str, float]:
        """估算学习率调整的影响"""
        impact = {
            "convergence_speed_impact": 0.0,
            "stability_impact": 0.0,
            "final_performance_impact": 0.0
        }
        
        for component_name, adj in adjustments.items():
            component_config = self.config.learning_components.get(component_name)
            if component_config:
                # 根据组件配置估算影响
                adjustment_ratio = adj["adjustment_ratio"]
                
                # 收敛速度影响
                if adjustment_ratio > 1.0:  # 学习率增加
                    impact["convergence_speed_impact"] += 0.2 * min(adjustment_ratio - 1.0, 1.0)
                else:  # 学习率减少
                    impact["convergence_speed_impact"] -= 0.1 * min(1.0 - adjustment_ratio, 1.0)
                
                # 稳定性影响
                if adjustment_ratio > 1.5:  # 学习率大幅增加可能影响稳定性
                    impact["stability_impact"] -= 0.2 * min(adjustment_ratio - 1.5, 1.0)
                elif adjustment_ratio < 0.5:  # 学习率大幅减少可能影响稳定性
                    impact["stability_impact"] -= 0.1 * min(0.5 - adjustment_ratio, 1.0)
                
                # 最终性能影响
                impact["final_performance_impact"] += component_config.performance_weight * abs(adjustment_ratio - 1.0)
        
        return impact
```

```python
# utils/model_optimization.py
import torch
import torch.nn as nn
from typing import Dict, Any, List, Tuple
import logging

logger = logging.getLogger(__name__)

class ModelOptimizer:
    """模型优化工具类"""
    
    @staticmethod
    def quantize_model(model: nn.Module, quantization_type: str = "dynamic") -> nn.Module:
        """量化模型以减少内存使用和提高推理速度"""
        try:
            if quantization_type == "dynamic":
                quantized_model = torch.quantization.quantize_dynamic(
                    model, {nn.Linear, nn.Conv2d}, dtype=torch.qint8
                )
            elif quantization_type == "static":
                # 静态量化需要校准数据
                quantized_model = torch.quantization.quantize_static(
                    model, None, dtype=torch.qint8
                )
            else:
                raise ValueError(f"Unsupported quantization type: {quantization_type}")
            
            logger.info(f"Model quantized using {quantization_type} quantization")
            return quantized_model
        except Exception as e:
            logger.error(f"Error quantizing model: {e}")
            raise
    
    @staticmethod
    def prune_model(model: nn.Module, pruning_ratio: float = 0.2) -> nn.Module:
        """剪枝模型以减少参数数量"""
        try:
            import torch.nn.utils.prune as prune
            
            # 全局剪枝
            parameters_to_prune = []
            for name, module in model.named_modules():
                if isinstance(module, (nn.Conv2d, nn.Linear)):
                    parameters_to_prune.append((module, 'weight'))
            
            prune.global_unstructured(
                parameters_to_prune,
                pruning_method=prune.L1Unstructured,
                amount=pruning_ratio,
            )
            
            # 移除剪枝重参数化以使剪枝永久化
            for module, param_name in parameters_to_prune:
                prune.remove(module, param_name)
            
            logger.info(f"Model pruned with ratio {pruning_ratio}")
            return model
        except Exception as e:
            logger.error(f"Error pruning model: {e}")
            raise
    
    @staticmethod
    def optimize_for_inference(model: nn.Module) -> nn.Module:
        """优化模型以进行推理"""
        try:
            # 设置模型为评估模式
            model.eval()
            
            # 融合操作
            if hasattr(model, 'fuse_model'):
                model.fuse_model()
            
            # 编译模型（PyTorch 2.0+）
            if hasattr(torch, 'compile'):
                model = torch.compile(model)
            
            logger.info("Model optimized for inference")
            return model
        except Exception as e:
            logger.error(f"Error optimizing model for inference: {e}")
            raise
```

```python
# utils/cache_optimization.py
import redis
import pickle
import hashlib
import json
from typing import Any, Dict, Optional, Union
import logging
from functools import wraps

logger = logging.getLogger(__name__)

class CacheManager:
    """缓存管理器"""
    
    def __init__(self, redis_url: str = "redis://localhost:6379", default_ttl: int = 3600):
        self.redis_client = redis.from_url(redis_url)
        self.default_ttl = default_ttl
    
    def _generate_key(self, prefix: str, **kwargs) -> str:
        """生成缓存键"""
        key_data = {k: v for k, v in kwargs.items() if v is not None}
        key_str = json.dumps(key_data, sort_keys=True)
        key_hash = hashlib.md5(key_str.encode()).hexdigest()
        return f"{prefix}:{key_hash}"
    
    def get(self, prefix: str, **kwargs) -> Optional[Any]:
        """获取缓存值"""
        try:
            key = self._generate_key(prefix, **kwargs)
            cached_data = self.redis_client.get(key)
            
            if cached_data:
                return pickle.loads(cached_data)
            return None
        except Exception as e:
            logger.error(f"Error getting from cache: {e}")
            return None
    
    def set(self, prefix: str, value: Any, ttl: Optional[int] = None, **kwargs) -> bool:
        """设置缓存值"""
        try:
            key = self._generate_key(prefix, **kwargs)
            ttl = ttl or self.default_ttl
            serialized_value = pickle.dumps(value)
            return self.redis_client.setex(key, ttl, serialized_value)
        except Exception as e:
            logger.error(f"Error setting cache: {e}")
            return False
    
    def delete(self, prefix: str, **kwargs) -> bool:
        """删除缓存值"""
        try:
            key = self._generate_key(prefix, **kwargs)
            return bool(self.redis_client.delete(key))
        except Exception as e:
            logger.error(f"Error deleting cache: {e}")
            return False
    
    def clear_prefix(self, prefix: str) -> int:
        """清除特定前缀的所有缓存"""
        try:
            pattern = f"{prefix}:*"
            keys = self.redis_client.keys(pattern)
            if keys:
                return self.redis_client.delete(*keys)
            return 0
        except Exception as e:
            logger.error(f"Error clearing cache prefix: {e}")
            return 0

def cached_result(prefix: str, ttl: Optional[int] = None):
    """缓存装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 尝试从缓存获取结果
            cache_manager = getattr(wrapper, '_cache_manager', None)
            if not cache_manager:
                cache_manager = CacheManager()
                wrapper._cache_manager = cache_manager
            
            # 生成缓存键
            cache_key_kwargs = {
                'args': str(args),
                'kwargs': str(kwargs)
            }
            
            # 尝试从缓存获取
            cached_result = cache_manager.get(prefix, **cache_key_kwargs)
            if cached_result is not None:
                return cached_result
            
            # 执行函数并缓存结果
            result = func(*args, **kwargs)
            cache_manager.set(prefix, result, ttl, **cache_key_kwargs)
            
            return result
        return wrapper
    return decorator
```

```python
# utils/monitoring.py
import time
import psutil
import torch
from typing import Dict, Any, List, Optional
import logging
from functools import wraps

logger = logging.getLogger(__name__)

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self):
        self.metrics_history = []
        self.start_time = time.time()
    
    def record_metrics(self, metrics: Dict[str, Any]) -> None:
        """记录性能指标"""
        timestamp = time.time() - self.start_time
        system_metrics = self._get_system_metrics()
        
        record = {
            "timestamp": timestamp,
            "custom_metrics": metrics,
            "system_metrics": system_metrics
        }
        
        self.metrics_history.append(record)
        
        # 保持历史记录在合理范围内
        if len(self.metrics_history) > 1000:
            self.metrics_history = self.metrics_history[-500:]
    
    def _get_system_metrics(self) -> Dict[str, Any]:
        """获取系统性能指标"""
        try:
            # CPU使用率
            cpu_percent = psutil.cpu_percent(interval=0.1)
            
            # 内存使用情况
            memory = psutil.virtual_memory()
            memory_percent = memory.percent
            
            # GPU使用情况（如果有）
            gpu_metrics = {}
            if torch.cuda.is_available():
                gpu_metrics = {
                    "gpu_memory_used": torch.cuda.memory_allocated() / 1024**3,  # GB
                    "gpu_memory_total": torch.cuda.get_device_properties(0).total_memory / 1024**3,  # GB
                    "gpu_utilization": 0  # 需要nvidia-ml-py库获取
                }
            
            return {
                "cpu_percent": cpu_percent,
                "memory_percent": memory_percent,
                "memory_used_gb": memory.used / 1024**3,
                "memory_total_gb": memory.total / 1024**3,
                **gpu_metrics
            }
        except Exception as e:
            logger.error(f"Error getting system metrics: {e}")
            return {}
    
    def get_average_metrics(self, window_size: int = 100) -> Dict[str, float]:
        """获取平均性能指标"""
        if not self.metrics_history:
            return {}
        
        recent_metrics = self.metrics_history[-window_size:]
        
        avg_metrics = {}
        
        # 计算系统指标平均值
        system_keys = recent_metrics[0]["system_metrics"].keys()
        for key in system_keys:
            values = [m["system_metrics"].get(key, 0) for m in recent_metrics if key in m["system_metrics"]]
            if values:
                avg_metrics[f"avg_{key}"] = sum(values) / len(values)
        
        # 计算自定义指标平均值
        if recent_metrics and "custom_metrics" in recent_metrics[0]:
            custom_keys = recent_metrics[0]["custom_metrics"].keys()
            for key in custom_keys:
                values = [m["custom_metrics"].get(key, 0) for m in recent_metrics if key in m["custom_metrics"]]
                if values:
                    avg_metrics[f"avg_{key}"] = sum(values) / len(values)
        
        return avg_metrics
    
    def get_metrics_summary(self) -> Dict[str, Any]:
        """获取性能指标摘要"""
        if not self.metrics_history:
            return {"status": "No metrics recorded"}
        
        avg_metrics = self.get_average_metrics()
        latest_metrics = self.metrics_history[-1]
        
        return {
            "status": "Active",
            "recording_duration": time.time() - self.start_time,
            "total_records": len(self.metrics_history),
            "latest_system_metrics": latest_metrics["system_metrics"],
            "latest_custom_metrics": latest_metrics["custom_metrics"],
            "average_metrics": avg_metrics
        }

def monitor_performance(prefix: str = "performance"):
    """性能监控装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 获取或创建性能监控器
            monitor = getattr(wrapper, '_monitor', None)
            if not monitor:
                monitor = PerformanceMonitor()
                wrapper._monitor = monitor
            
            # 记录开始时间
            start_time = time.time()
            
            # 执行函数
            result = func(*args, **kwargs)
            
            # 计算执行时间
            execution_time = time.time() - start_time
            
            # 记录性能指标
            metrics = {
                "function": func.__name__,
                "execution_time": execution_time,
                "args_count": len(args),
                "kwargs_count": len(kwargs)
            }
            
            monitor.record_metrics(metrics)
            
            return result
        return wrapper
    return decorator
```
