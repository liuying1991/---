# 表情生成模块6A工作流

## 模块概述

### 模块名称
表情生成模块

### 所属子系统
交互表达子系统

### 模块功能描述
表情生成模块负责根据文本内容、情感色彩和交互场景，生成适合婴儿认知和情感需求的虚拟表情。该模块能够创建生动、友好、富有情感的表情形象，增强与婴儿的交互体验，促进婴儿的情感发展和认知能力提升。

### 模块职责
1. 表情生成：根据内容和情感生成相应的表情
2. 情感表达：在表情中融入适当的情感色彩
3. 年龄适配：根据婴儿年龄调整表情特征
4. 动态表情：生成动态变化的表情序列
5. 个性化表情：根据婴儿偏好生成个性化表情
6. 表情优化：优化表情质量和自然度

### 模块接口
- **输入接口**：文本内容、情感参数、婴儿状态
- **输出接口**：表情图像/动画、表情元数据
- **依赖模块**：文本分析模块、情感识别模块
- **被依赖模块**：显示输出模块、交互管理模块

# 表情生成器实现
class ExpressionGenerator:
    def __init__(self, base_generator, emotion_generator, dynamic_generator, 
                 age_adapter, personalization_adapter, expression_optimizer, quality_assessor):
        self.base_generator = base_generator
        self.emotion_generator = emotion_generator
        self.dynamic_generator = dynamic_generator
        self.age_adapter = age_adapter
        self.personalization_adapter = personalization_adapter
        self.expression_optimizer = expression_optimizer
        self.quality_assessor = quality_assessor
        self.content_analyzer = ContentAnalyzer()
        
    def generate(self, text, generation_params, baby_state):
        # 输入处理
        processed_input = self._process_input(text, generation_params, baby_state)
        
        # 内容分析
        content_analysis = self.content_analyzer.analyze(processed_input["text"])
        
        # 基础表情生成
        base_expression = self.base_generator.generate(
            processed_input["text"],
            content_analysis,
            processed_input["base_params"]
        )
        
        # 情感表情生成
        if generation_params.get("apply_emotion", True):
            emotion_expression = self.emotion_generator.add_emotion(
                base_expression,
                content_analysis["emotion"],
                generation_params.get("emotion_intensity", 0.8)
            )
        else:
            emotion_expression = base_expression
        
        # 年龄适配
        if "age" in baby_state:
            age_adapted_expression = self.age_adapter.adapt_for_age(
                emotion_expression,
                baby_state["age"]
            )
        else:
            age_adapted_expression = emotion_expression
        
        # 个性化适配
        if generation_params.get("apply_personalization", True) and "user_id" in baby_state:
            personalized_expression = self.personalization_adapter.adapt_for_user(
                age_adapted_expression,
                baby_state["user_id"]
            )
        else:
            personalized_expression = age_adapted_expression
        
        # 动态表情生成
        if generation_params.get("generate_dynamic", False):
            dynamic_expression = self.dynamic_generator.generate_dynamic(
                personalized_expression,
                content_analysis,
                generation_params.get("animation_params", {})
            )
        else:
            dynamic_expression = personalized_expression
        
        # 表情优化
        if generation_params.get("optimize_expression", True):
            optimized_expression = self.expression_optimizer.optimize(
                dynamic_expression,
                generation_params
            )
        else:
            optimized_expression = dynamic_expression
        
        # 质量评估
        quality_score = self.quality_assessor.assess(optimized_expression, content_analysis)
        
        # 如果质量评分不达标，尝试重新生成
        if quality_score < 3.5:
            # 尝试调整生成参数并重新生成
            adjusted_params = self._adjust_generation_params(generation_params, quality_score)
            return self.generate(text, adjusted_params, baby_state)
        
        return {
            "text": text,
            "expression": optimized_expression,
            "metadata": {
                "emotion": content_analysis["emotion"],
                "age": baby_state.get("age", None),
                "user_id": baby_state.get("user_id", None),
                "is_dynamic": generation_params.get("generate_dynamic", False),
                "quality_score": quality_score
            },
            "timestamp": time.time()
        }

class BaseExpressionGenerator:
    def __init__(self):
        self.expression_model = self._load_expression_model()
        self.expression_templates = self._load_expression_templates()
        
    def generate(self, text, content_analysis, generation_params):
        # 确定基础表情类型
        expression_type = self._determine_expression_type(content_analysis)
        
        # 获取表情模板
        template = self.expression_templates[expression_type]
        
        # 生成基础表情
        base_expression = self._generate_base_expression(template, text, generation_params)
        
        return base_expression
    
    def _determine_expression_type(self, content_analysis):
        # 根据内容分析确定基础表情类型
        if "question" in content_analysis["content_type"]:
            return "curious"
        elif "exclamation" in content_analysis["content_type"]:
            return "surprised"
        elif "story" in content_analysis["content_type"]:
            return "engaged"
        elif "song" in content_analysis["content_type"]:
            return "happy"
        else:
            return "neutral"
    
    def _generate_base_expression(self, template, text, generation_params):
        # 使用模板生成基础表情
        base_expression = self.expression_model.generate(
            template=template,
            text=text,
            params=generation_params
        )
        
        return base_expression

# 情感表情生成器实现
class EmotionExpressionGenerator:
    def __init__(self):
        self.emotion_converter = self._load_emotion_converter()
        self.emotion_features = self._load_emotion_features()
        
    def add_emotion(self, base_expression, emotion, intensity=0.8):
        if emotion not in self.emotion_features:
            return base_expression
        
        emotion_features = self.emotion_features[emotion]
        
        # 应用情感特征
        emotion_expression = self._apply_emotion_features(
            base_expression,
            emotion_features,
            intensity
        )
        
        return emotion_expression
    
    def _apply_emotion_features(self, base_expression, emotion_features, intensity):
        # 应用情感相关的面部特征
        emotion_expression = base_expression.copy()
        
        # 眼部特征调整
        if "eyes" in emotion_features:
            emotion_expression = self._adjust_eye_features(
                emotion_expression,
                emotion_features["eyes"],
                intensity
            )
        
        # 嘴部特征调整
        if "mouth" in emotion_features:
            emotion_expression = self._adjust_mouth_features(
                emotion_expression,
                emotion_features["mouth"],
                intensity
            )
        
        # 眉毛特征调整
        if "eyebrows" in emotion_features:
            emotion_expression = self._adjust_eyebrow_features(
                emotion_expression,
                emotion_features["eyebrows"],
                intensity
            )
        
        # 整体面部表情调整
        if "face" in emotion_features:
            emotion_expression = self._adjust_face_features(
                emotion_expression,
                emotion_features["face"],
                intensity
            )
        
        return emotion_expression
    
    def _adjust_eye_features(self, expression, eye_features, intensity):
        # 调整眼部特征
        # 根据情感类型调整眼睛形状、大小和位置
        
        # 获取当前眼部特征
        current_eye_features = self._extract_eye_features(expression)
        
        # 计算目标眼部特征
        target_eye_features = {}
        for feature, value in eye_features.items():
            current_value = current_eye_features.get(feature, 0)
            target_value = current_value + (value - current_value) * intensity
            target_eye_features[feature] = target_value
        
        # 应用目标眼部特征
        expression = self._apply_eye_features(expression, target_eye_features)
        
        return expression
    
    def _adjust_mouth_features(self, expression, mouth_features, intensity):
        # 调整嘴部特征
        # 根据情感类型调整嘴巴形状、大小和位置
        
        # 获取当前嘴部特征
        current_mouth_features = self._extract_mouth_features(expression)
        
        # 计算目标嘴部特征
        target_mouth_features = {}
        for feature, value in mouth_features.items():
            current_value = current_mouth_features.get(feature, 0)
            target_value = current_value + (value - current_value) * intensity
            target_mouth_features[feature] = target_value
        
        # 应用目标嘴部特征
        expression = self._apply_mouth_features(expression, target_mouth_features)
        
        return expression

class DynamicExpressionGenerator:
    def __init__(self):
        self.animation_templates = self._load_animation_templates()
        self.transition_generator = TransitionGenerator()
        
    def generate_dynamic(self, static_expression, content_analysis, animation_params):
        # 确定动画类型
        animation_type = self._determine_animation_type(content_analysis, animation_params)
        
        # 获取动画模板
        template = self.animation_templates[animation_type]
        
        # 生成动态表情
        dynamic_expression = self._generate_dynamic_expression(
            static_expression,
            template,
            animation_params
        )
        
        return dynamic_expression
    
    def _determine_animation_type(self, content_analysis, animation_params):
        # 根据内容分析和动画参数确定动画类型
        if "animation_type" in animation_params:
            return animation_params["animation_type"]
        
        # 根据内容类型自动选择动画类型
        if "question" in content_analysis["content_type"]:
            return "thinking"
        elif "story" in content_analysis["content_type"]:
            return "storytelling"
        elif "song" in content_analysis["content_type"]:
            return "singing"
        else:
            return "breathing"
    
    def _generate_dynamic_expression(self, static_expression, template, animation_params):
        # 生成动态表情序列
        frames = []
        num_frames = animation_params.get("num_frames", 30)
        
        # 生成过渡帧

            # 计算当前帧的参数
            frame_params = self._calculate_frame_params(
                template,
                i,
                num_frames,
                animation_params
            )
            
            # 生成当前帧
            frame = self._generate_frame(static_expression, frame_params)
            frames.append(frame)
        
        # 组合帧为动态表情
        dynamic_expression = self._combine_frames(frames)
        
        return dynamic_expression
    
    def _calculate_frame_params(self, template, frame_index, num_frames, animation_params):
        # 计算当前帧的参数
        progress = frame_index / num_frames
        
        # 基础参数
        frame_params = {
            "progress": progress,
            "intensity": animation_params.get("intensity", 0.8)
        }
        
        # 根据动画类型计算特定参数
        if template["type"] == "breathing":
            # 呼吸动画
            frame_params["scale"] = 1.0 + 0.05 * math.sin(2 * math.pi * progress)
            frame_params["brightness"] = 1.0 + 0.1 * math.sin(2 * math.pi * progress)
        elif template["type"] == "blinking":
            # 眨眼动画
            if 0.4 < progress < 0.6:
                frame_params["eye_openness"] = 1.0 - 10 * (progress - 0.4)
            else:
                frame_params["eye_openness"] = 1.0
        elif template["type"] == "talking":
            # 说话动画
            frame_params["mouth_openness"] = 0.3 + 0.5 * abs(math.sin(4 * math.pi * progress))
        
        return frame_params
    
    def _generate_frame(self, static_expression, frame_params):
        # 生成单帧表情
        frame = static_expression.copy()
        
        # 应用帧参数
        if "scale" in frame_params:
            frame = self._apply_scale(frame, frame_params["scale"])
        
        if "brightness" in frame_params:
            frame = self._apply_brightness(frame, frame_params["brightness"])
        
        if "eye_openness" in frame_params:
            frame = self._apply_eye_openness(frame, frame_params["eye_openness"])
        
        if "mouth_openness" in frame_params:
            frame = self._apply_mouth_openness(frame, frame_params["mouth_openness"])
        
        return frame
    
    def _combine_frames(self, frames):
        # 组合帧为动态表情
        # 使用GIF格式或视频格式
        dynamic_expression = self._create_gif(frames)
        
        return dynamic_expression
