# video_collector.py
import cv2
import numpy as np
from typing import Optional, Callable, Dict, Any
import threading
import queue
import logging

class VideoCollector:
    """视频采集器类，负责从各种视频源采集视频数据"""
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化视频采集器
        
        Args:
            config: 配置字典，包含视频源类型、路径等参数
        """
        self.config = config
        self.source_type = config.get('source_type', 'camera')
        self.source_path = config.get('source_path', 0)
        self.fps = config.get('fps', 30)
        self.resolution = config.get('resolution', (1920, 1080))
        
        self.cap = None
        self.frame_queue = queue.Queue(maxsize=config.get('buffer_size', 10))
        self.is_collecting = False
        self.collection_thread = None
        
        self.logger = logging.getLogger(__name__)
    
    def start_collection(self) -> bool:
        """
        开始视频采集
        
        Returns:
            bool: 采集是否成功启动
        """
        try:
            # 初始化视频捕获
            if self.source_type == 'camera':
                self.cap = cv2.VideoCapture(self.source_path)
            elif self.source_type == 'file':
                self.cap = cv2.VideoCapture(self.source_path)
            elif self.source_type == 'stream':
                self.cap = cv2.VideoCapture(self.source_path)
            else:
                raise ValueError(f"不支持的视频源类型: {self.source_type}")
            
            if not self.cap.isOpened():
                raise RuntimeError("无法打开视频源")
            
            # 设置视频参数
            self.cap.set(cv2.CAP_PROP_FPS, self.fps)
            self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, self.resolution[0])
            self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, self.resolution[1])
            
            # 启动采集线程
            self.is_collecting = True
            self.collection_thread = threading.Thread(target=self._collect_frames)
            self.collection_thread.daemon = True
            self.collection_thread.start()
            
            self.logger.info(f"视频采集已启动，源类型: {self.source_type}")
            return True
            
        except Exception as e:
            self.logger.error(f"启动视频采集失败: {str(e)}")
            return False
    
    def stop_collection(self):
        """停止视频采集"""
        self.is_collecting = False
        if self.collection_thread and self.collection_thread.is_alive():
            self.collection_thread.join(timeout=1.0)
        
        if self.cap:
            self.cap.release()
            self.cap = None
            
        self.logger.info("视频采集已停止")
    
    def get_frame(self, timeout: float = 1.0) -> Optional[np.ndarray]:
        """
        获取最新视频帧
        
        Args:
            timeout: 获取超时时间(秒)
            
        Returns:
            np.ndarray: 视频帧数据，如果超时或出错则返回None
        """
        try:
            return self.frame_queue.get(timeout=timeout)
        except queue.Empty:
            return None
    
    def _collect_frames(self):
        """采集视频帧的内部方法"""
        while self.is_collecting:
            ret, frame = self.cap.read()
            if not ret:
                self.logger.warning("无法读取视频帧")
                break
                
            # 如果队列已满，丢弃最旧的帧
            if self.frame_queue.full():
                try:
                    self.frame_queue.get_nowait()
                except queue.Empty:
                    pass
                    
            self.frame_queue.put(frame)
    
    def get_video_info(self) -> Dict[str, Any]:
        """
        获取视频信息
        
        Returns:
            Dict[str, Any]: 包含视频信息的字典
        """
        if not self.cap:
            return {}
            
        return {
            'fps': self.cap.get(cv2.CAP_PROP_FPS),
            'width': int(self.cap.get(cv2.CAP_PROP_FRAME_WIDTH)),
            'height': int(self.cap.get(cv2.CAP_PROP_FRAME_HEIGHT)),
            'frame_count': int(self.cap.get(cv2.CAP_PROP_FRAME_COUNT)),
            'fourcc': int(self.cap.get(cv2.CAP_PROP_FOURCC))
        }
```

# object_detector.py
import torch
import cv2
import numpy as np
from typing import List, Dict, Any, Tuple
import logging
import os

class YOLOv8Detector:
    """基于YOLOv8的目标检测器"""
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化YOLOv8检测器
        
        Args:
            config: 配置字典，包含模型路径、置信度阈值等参数
        """
        self.config = config
        self.model_path = config.get('model_path', 'yolov8n.pt')
        self.confidence_threshold = config.get('confidence_threshold', 0.5)
        self.iou_threshold = config.get('iou_threshold', 0.45)
        self.device = config.get('device', 'cuda' if torch.cuda.is_available() else 'cpu')
        self.target_classes = config.get('target_classes', ['person'])
        
        self.model = None
        self.logger = logging.getLogger(__name__)
        
        self._load_model()
    
    def _load_model(self):
        """加载YOLOv8模型"""
        try:
            # 检查模型文件是否存在
            if not os.path.exists(self.model_path):
                self.logger.warning(f"模型文件不存在: {self.model_path}，将使用预训练模型")
                self.model = torch.hub.load('ultralytics/yolov8', 'yolov8n', pretrained=True)
            else:
                self.model = torch.hub.load('ultralytics/yolov8', 'custom', path=self.model_path)
            
            # 设置模型参数
            self.model.conf = self.confidence_threshold
            self.model.iou = self.iou_threshold
            self.model.to(self.device)
            
            self.logger.info(f"YOLOv8模型已加载，设备: {self.device}")
            
        except Exception as e:
            self.logger.error(f"加载YOLOv8模型失败: {str(e)}")
            raise
    
    def detect(self, frame: np.ndarray) -> List[Dict[str, Any]]:
        """
        检测视频帧中的目标
        
        Args:
            frame: 输入视频帧
            
        Returns:
            List[Dict[str, Any]]: 检测结果列表，每个元素包含类别、边界框、置信度等信息
        """
        try:
            # 使用模型进行推理
            results = self.model(frame)
            
            # 解析检测结果
            detections = []
            for result in results.xyxy[0].cpu().numpy():
                x1, y1, x2, y2, conf, cls = result
                class_name = self.model.names[int(cls)]
                
                # 只保留目标类别的检测结果
                if self.target_classes and class_name not in self.target_classes:
                    continue
                
                detection = {
                    'class': class_name,
                    'bbox': [int(x1), int(y1), int(x2), int(y2)],
                    'confidence': float(conf),
                    'center': [int((x1 + x2) / 2), int((y1 + y2) / 2)],
                    'area': int((x2 - x1) * (y2 - y1))
                }
                detections.append(detection)
            
            return detections
            
        except Exception as e:
            self.logger.error(f"目标检测失败: {str(e)}")
            return []
    
    def detect_batch(self, frames: List[np.ndarray]) -> List[List[Dict[str, Any]]]:
        """
        批量检测多帧中的目标
        
        Args:
            frames: 输入视频帧列表
            
        Returns:
            List[List[Dict[str, Any]]]: 每帧的检测结果列表
        """
        try:
            # 使用模型进行批量推理
            results = self.model(frames)
            
            # 解析每帧的检测结果
            all_detections = []
            for frame_results in results.xyxy:
                frame_detections = []
                for result in frame_results.cpu().numpy():
                    x1, y1, x2, y2, conf, cls = result
                    class_name = self.model.names[int(cls)]
                    
                    # 只保留目标类别的检测结果
                    if self.target_classes and class_name not in self.target_classes:
                        continue
                    
                    detection = {
                        'class': class_name,
                        'bbox': [int(x1), int(y1), int(x2), int(y2)],
                        'confidence': float(conf),
                        'center': [int((x1 + x2) / 2), int((y1 + y2) / 2)],
                        'area': int((x2 - x1) * (y2 - y1))
                    }
                    frame_detections.append(detection)
                
                all_detections.append(frame_detections)
            
            return all_detections
            
        except Exception as e:
            self.logger.error(f"批量目标检测失败: {str(e)}")
            return [[] for _ in frames]

class MediaPipeFaceDetector:
    """基于MediaPipe的人脸检测器"""
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化MediaPipe人脸检测器
        
        Args:
            config: 配置字典，包含模型选择、检测阈值等参数
        """
        self.config = config
        self.model_selection = config.get('model_selection', 0)  # 0: 短距离, 1: 长距离
        self.min_detection_confidence = config.get('min_detection_confidence', 0.5)
        
        self.mp_face_detection = None
        self.face_detection = None
        self.logger = logging.getLogger(__name__)
        
        self._load_model()
    
    def _load_model(self):
        """加载MediaPipe人脸检测模型"""
        try:
            import mediapipe as mp
            self.mp_face_detection = mp.solutions.face_detection
            self.face_detection = self.mp_face_detection.FaceDetection(
                model_selection=self.model_selection,
                min_detection_confidence=self.min_detection_confidence
            )
            
            self.logger.info("MediaPipe人脸检测模型已加载")
            
        except ImportError:
            self.logger.error("MediaPipe未安装，无法加载人脸检测模型")
            raise
        except Exception as e:
            self.logger.error(f"加载MediaPipe人脸检测模型失败: {str(e)}")
            raise
    
    def detect(self, frame: np.ndarray) -> List[Dict[str, Any]]:
        """
        检测视频帧中的人脸
        
        Args:
            frame: 输入视频帧
            
        Returns:
            List[Dict[str, Any]]: 人脸检测结果列表
        """
        try:
            # 转换颜色空间
            rgb_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            
            # 进行人脸检测
            results = self.face_detection.process(rgb_frame)
            
            # 解析检测结果
            detections = []
            if results.detections:
                for detection in results.detections:
                    bbox = detection.location_data.relative_bounding_box
                    h, w, _ = frame.shape
                    
                    # 转换相对坐标为绝对坐标
                    x1 = int(bbox.xmin * w)
                    y1 = int(bbox.ymin * h)
                    x2 = int((bbox.xmin + bbox.width) * w)
                    y2 = int((bbox.ymin + bbox.height) * h)
                    
                    face_detection = {
                        'class': 'face',
                        'bbox': [x1, y1, x2, y2],
                        'confidence': detection.score[0],
                        'center': [int((x1 + x2) / 2), int((y1 + y2) / 2)],
                        'area': int((x2 - x1) * (y2 - y1)),
                        'keypoints': self._extract_keypoints(detection, w, h)
                    }
                    detections.append(face_detection)
            
            return detections
            
        except Exception as e:
            self.logger.error(f"人脸检测失败: {str(e)}")
            return []
    
    def _extract_keypoints(self, detection, w: int, h: int) -> List[Dict[str, Any]]:
        """
        提取人脸关键点
        
        Args:
            detection: MediaPipe检测结果
            w: 图像宽度
            h: 图像高度
            
        Returns:
            List[Dict[str, Any]]: 关键点列表
        """
        keypoints = []
        if detection.location_data.relative_keypoints:
            for i, keypoint in enumerate(detection.location_data.relative_keypoints):
                x = int(keypoint.x * w)
                y = int(keypoint.y * h)
                keypoints.append({
                    'id': i,
                    'x': x,
                    'y': y,
                    'name': self._get_keypoint_name(i)
                })
        
        return keypoints
    
    def _get_keypoint_name(self, keypoint_id: int) -> str:
        """获取关键点名称"""
        keypoint_names = [
            'right_eye', 'left_eye', 'nose_tip', 'mouth_center', 
            'right_ear_tragion', 'left_ear_tragion'
        ]
        return keypoint_names[keypoint_id] if keypoint_id < len(keypoint_names) else f'keypoint_{keypoint_id}'

```python
# behavior_recognizer.py
import torch
import torch.nn as nn
import cv2
import numpy as np
from typing import List, Dict, Any, Optional
import logging
import os
from collections import deque

class BehaviorRecognizer:
    """基于3D CNN和LSTM的行为识别器"""
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化行为识别器
        
        Args:
            config: 配置字典，包含模型路径、序列长度等参数
        """
        self.config = config
        self.model_path = config.get('model_path', 'behavior_model.pth')
        self.sequence_length = config.get('sequence_length', 16)
        self.frame_size = config.get('frame_size', (224, 224))
        self.behavior_classes = config.get('behavior_classes', 
                                         ['sleeping', 'crying', 'playing', 'eating'])
        self.device = config.get('device', 'cuda' if torch.cuda.is_available() else 'cpu')
        
        self.model = None
        self.frame_buffer = deque(maxlen=self.sequence_length)
        self.logger = logging.getLogger(__name__)
        
        self._load_model()
    
    def _load_model(self):
        """加载行为识别模型"""
        try:
            # 检查模型文件是否存在
            if not os.path.exists(self.model_path):
                self.logger.warning(f"模型文件不存在: {self.model_path}，将使用预定义模型结构")
                self.model = self._create_default_model()
            else:
                self.model = torch.load(self.model_path, map_location=self.device)
            
            self.model.to(self.device)
            self.model.eval()
            
            self.logger.info(f"行为识别模型已加载，设备: {self.device}")
            
        except Exception as e:
            self.logger.error(f"加载行为识别模型失败: {str(e)}")
            raise
    
    def _create_default_model(self) -> nn.Module:
        """创建默认的行为识别模型"""
        class BehaviorModel(nn.Module):
            def __init__(self, num_classes: int):
                super(BehaviorModel, self).__init__()
                
                # 3D CNN特征提取器
                self.feature_extractor = nn.Sequential(
                    nn.Conv3d(3, 16, kernel_size=(3, 3, 3), padding=(1, 1, 1)),
                    nn.BatchNorm3d(16),
                    nn.ReLU(inplace=True),
                    nn.MaxPool3d(kernel_size=(1, 2, 2)),
                    
                    nn.Conv3d(16, 32, kernel_size=(3, 3, 3), padding=(1, 1, 1)),
                    nn.BatchNorm3d(32),
                    nn.ReLU(inplace=True),
                    nn.MaxPool3d(kernel_size=(2, 2, 2)),
                    
                    nn.Conv3d(32, 64, kernel_size=(3, 3, 3), padding=(1, 1, 1)),
                    nn.BatchNorm3d(64),
                    nn.ReLU(inplace=True),
                    nn.MaxPool3d(kernel_size=(2, 2, 2)),
                )
                
                # LSTM时序建模
                self.lstm = nn.LSTM(64 * 4 * 7 * 7, 128, batch_first=True)
                
                # 分类器
                self.classifier = nn.Sequential(
                    nn.Linear(128, 64),
                    nn.ReLU(inplace=True),
                    nn.Dropout(0.5),
                    nn.Linear(64, num_classes)
                )
            
            def forward(self, x):
                # x shape: (batch_size, seq_length, channels, height, width)
                batch_size, seq_len, c, h, w = x.shape
                
                # 3D CNN特征提取
                c_out = self.feature_extractor(x.view(batch_size, c, seq_len, h, w))
                
                # 重塑为LSTM输入格式
                c_out = c_out.permute(0, 2, 1, 3, 4).contiguous()
                c_out = c_out.view(batch_size, c_out.size(1), -1)
                
                # LSTM时序建模
                lstm_out, _ = self.lstm(c_out)
                
                # 使用最后一个时间步的输出进行分类
                output = self.classifier(lstm_out[:, -1, :])
                
                return output
        
        return BehaviorModel(len(self.behavior_classes))
    
    def recognize(self, frame: np.ndarray, detections: List[Dict[str, Any]] = None) -> Dict[str, Any]:
        """
        识别当前帧中的行为
        
        Args:
            frame: 输入视频帧
            detections: 目标检测结果，用于聚焦感兴趣区域
            
        Returns:
            Dict[str, Any]: 行为识别结果
        """
        try:
            # 预处理帧
            processed_frame = self._preprocess_frame(frame, detections)
            
            # 添加到帧缓冲区
            self.frame_buffer.append(processed_frame)
            
            # 如果缓冲区未满，返回未知行为
            if len(self.frame_buffer) < self.sequence_length:
                return {
                    'behavior': 'unknown',
                    'confidence': 0.0,
                    'reason': 'insufficient_frames'
                }
            
            # 构建输入序列
            input_sequence = torch.stack(list(self.frame_buffer)).unsqueeze(0).to(self.device)
            
            # 模型推理
            with torch.no_grad():
                outputs = self.model(input_sequence)
                probabilities = torch.softmax(outputs, dim=1).cpu().numpy()[0]
                
                # 获取最高概率的行为类别
                max_prob_idx = np.argmax(probabilities)
                behavior = self.behavior_classes[max_prob_idx]
                confidence = float(probabilities[max_prob_idx])
                
                # 构建所有行为的概率分布
                behavior_probs = {
                    self.behavior_classes[i]: float(probabilities[i]) 
                    for i in range(len(self.behavior_classes))
                }
                
                return {
                    'behavior': behavior,
                    'confidence': confidence,
                    'probabilities': behavior_probs
                }
                
        except Exception as e:
            self.logger.error(f"行为识别失败: {str(e)}")
            return {
                'behavior': 'unknown',
                'confidence': 0.0,
                'reason': str(e)
            }
    
    def _preprocess_frame(self, frame: np.ndarray, detections: List[Dict[str, Any]] = None) -> torch.Tensor:
        """
        预处理视频帧
        
        Args:
            frame: 输入视频帧
            detections: 目标检测结果，用于聚焦感兴趣区域
            
        Returns:
            torch.Tensor: 预处理后的帧张量
        """
        # 如果有检测结果，聚焦于最大的目标（通常是婴儿）
        if detections:
            # 找到面积最大的目标
            max_area = 0
            target_bbox = None
            for detection in detections:
                if detection['class'] == 'person' and detection['area'] > max_area:
                    max_area = detection['area']
                    target_bbox = detection['bbox']
            
            # 如果找到目标，裁剪感兴趣区域
            if target_bbox:
                x1, y1, x2, y2 = target_bbox
                # 扩展边界框，确保包含完整目标
                margin = 20
                x1 = max(0, x1 - margin)
                y1 = max(0, y1 - margin)
                x2 = min(frame.shape[1], x2 + margin)
                y2 = min(frame.shape[0], y2 + margin)
                frame = frame[y1:y2, x1:x2]
        
        # 调整大小
        frame = cv2.resize(frame, self.frame_size)
        
        # 归一化
        frame = frame.astype(np.float32) / 255.0
        
        # 转换为张量
        frame = torch.from_numpy(frame).permute(2, 0, 1)  # HWC -> CHW
        
        return frame
    
    def reset(self):
        """重置帧缓冲区"""
        self.frame_buffer.clear()

# video_service.py
import asyncio
import threading
import time
from typing import Dict, Any, List, Optional, Callable
import logging
import numpy as np
from concurrent.futures import ThreadPoolExecutor

from .core.video_collector import VideoCollector
from .core.video_preprocessor import VideoPreprocessor
from .core.object_detector import YOLOv8Detector, MediaPipeFaceDetector
from .core.behavior_recognizer import BehaviorRecognizer
from .core.feature_extractor import FeatureExtractor
from .utils.config import load_config

class VideoProcessingService:
    """视频处理服务类，协调各个组件完成视频处理流程"""
    
    def __init__(self, config_path: str = None):
        """
        初始化视频处理服务
        
        Args:
            config_path: 配置文件路径
        """
        self.config = load_config(config_path) if config_path else {}
        self.logger = logging.getLogger(__name__)
        
        # 初始化组件
        self.video_collector = None
        self.video_preprocessor = None
        self.object_detector = None
        self.face_detector = None
        self.behavior_recognizer = None
        self.feature_extractor = None
        
        # 处理状态
        self.is_processing = False
        self.processing_thread = None
        self.executor = ThreadPoolExecutor(max_workers=4)
        
        # 回调函数
        self.frame_callbacks = []
        self.detection_callbacks = []
        self.behavior_callbacks = []
        
        # 性能统计
        self.stats = {
            'frames_processed': 0,
            'detections_count': 0,
            'behaviors_recognized': 0,
            'processing_time': 0.0,
            'fps': 0.0
        }
        
        self._initialize_components()
    
    def _initialize_components(self):
        """初始化各个组件"""
        try:
            # 初始化视频采集器
            self.video_collector = VideoCollector(self.config.get('video_collector', {}))
            
            # 初始化视频预处理器
            self.video_preprocessor = VideoPreprocessor(self.config.get('video_preprocessor', {}))
            
            # 初始化目标检测器
            self.object_detector = YOLOv8Detector(self.config.get('object_detector', {}))
            self.face_detector = MediaPipeFaceDetector(self.config.get('face_detector', {}))
            
            # 初始化行为识别器
            self.behavior_recognizer = BehaviorRecognizer(self.config.get('behavior_recognizer', {}))
            
            # 初始化特征提取器
            self.feature_extractor = FeatureExtractor(self.config.get('feature_extractor', {}))
            
            self.logger.info("视频处理服务组件初始化完成")
            
        except Exception as e:
            self.logger.error(f"初始化视频处理服务组件失败: {str(e)}")
            raise
    
    def start_processing(self) -> bool:
        """
        开始视频处理
        
        Returns:
            bool: 处理是否成功启动
        """
        try:
            if self.is_processing:
                self.logger.warning("视频处理已在运行中")
                return False
            
            # 启动视频采集
            if not self.video_collector.start_collection():
                self.logger.error("无法启动视频采集")
                return False
            
            # 启动处理线程
            self.is_processing = True
            self.processing_thread = threading.Thread(target=self._processing_loop)
            self.processing_thread.daemon = True
            self.processing_thread.start()
            
            self.logger.info("视频处理已启动")
            return True
            
        except Exception as e:
            self.logger.error(f"启动视频处理失败: {str(e)}")
            return False
    
    def stop_processing(self):
        """停止视频处理"""
        self.is_processing = False
        
        if self.processing_thread and self.processing_thread.is_alive():
            self.processing_thread.join(timeout=1.0)
        
        if self.video_collector:
            self.video_collector.stop_collection()
        
        self.logger.info("视频处理已停止")
    
    def _processing_loop(self):
        """视频处理主循环"""
        last_time = time.time()
        frame_times = []
        
        while self.is_processing:
            try:
                # 获取视频帧
                frame = self.video_collector.get_frame(timeout=0.1)
                if frame is None:
                    continue
                
                # 记录处理开始时间
                start_time = time.time()
                
                # 预处理
                processed_frame = self.video_preprocessor.process(frame)
                
                # 并行执行目标检测和人脸检测
                detection_future = self.executor.submit(self.object_detector.detect, processed_frame)
                face_future = self.executor.submit(self.face_detector.detect, processed_frame)
                
                # 等待检测结果
                detections = detection_future.result()
                face_detections = face_future.result()
                
                # 合并检测结果
                all_detections = detections + face_detections
                
                # 行为识别
                behavior_result = self.behavior_recognizer.recognize(processed_frame, all_detections)
                
                # 特征提取
                features = self.feature_extractor.extract(processed_frame, all_detections, behavior_result)
                
                # 记录处理结束时间
                end_time = time.time()
                processing_time = end_time - start_time
                
                # 更新统计信息
                self.stats['frames_processed'] += 1
                self.stats['detections_count'] += len(all_detections)
                if behavior_result['behavior'] != 'unknown':
                    self.stats['behaviors_recognized'] += 1
                self.stats['processing_time'] += processing_time
                
                # 计算FPS
                frame_times.append(end_time - last_time)
                if len(frame_times) > 10:
                    frame_times.pop(0)
                self.stats['fps'] = len(frame_times) / sum(frame_times)
                last_time = end_time
                
                # 调用回调函数
                self._call_frame_callbacks(frame, processed_frame)
                self._call_detection_callbacks(all_detections)
                self._call_behavior_callbacks(behavior_result)
                
            except Exception as e:
                self.logger.error(f"视频处理循环出错: {str(e)}")
    
    def add_frame_callback(self, callback: Callable):
        """添加帧处理回调函数"""
        self.frame_callbacks.append(callback)
    
    def add_detection_callback(self, callback: Callable):
        """添加检测结果回调函数"""
        self.detection_callbacks.append(callback)
    
    def add_behavior_callback(self, callback: Callable):
        """添加行为识别回调函数"""
        self.behavior_callbacks.append(callback)
    
    def _call_frame_callbacks(self, original_frame: np.ndarray, processed_frame: np.ndarray):
        """调用帧处理回调函数"""
        for callback in self.frame_callbacks:
            try:
                callback(original_frame, processed_frame)
            except Exception as e:
                self.logger.error(f"帧回调函数执行失败: {str(e)}")
    
    def _call_detection_callbacks(self, detections: List[Dict[str, Any]]):
        """调用检测结果回调函数"""
        for callback in self.detection_callbacks:
            try:
                callback(detections)
            except Exception as e:
                self.logger.error(f"检测回调函数执行失败: {str(e)}")
    
    def _call_behavior_callbacks(self, behavior_result: Dict[str, Any]):
        """调用行为识别回调函数"""
        for callback in self.behavior_callbacks:
            try:
                callback(behavior_result)
            except Exception as e:
                self.logger.error(f"行为回调函数执行失败: {str(e)}")
    
    def get_stats(self) -> Dict[str, Any]:
        """获取处理统计信息"""
        return self.stats.copy()
    
    def get_video_info(self) -> Dict[str, Any]:
        """获取视频信息"""
        if self.video_collector:
            return self.video_collector.get_video_info()
        return {}

            # 编码处理后的帧为JPEG
            _, buffer = cv2.imencode('.jpg', processed_frame)
            frame_data = base64.b64encode(buffer).decode('utf-8')
            
            # 发送帧数据
            asyncio.create_task(websocket.send_text(json.dumps({
                "type": "frame",
                "data": frame_data
            })))

            # 发送检测结果
            asyncio.create_task(websocket.send_text(json.dumps({
                "type": "detection",
                "data": detections
            })))

            # 发送行为识别结果
            asyncio.create_task(websocket.send_text(json.dumps({
                "type": "behavior",
                "data": behavior_result
            })))
        
        # 添加回调函数
        service.add_frame_callback(frame_callback)
        service.add_detection_callback(detection_callback)
        service.add_behavior_callback(behavior_callback)
        
        # 保持连接
        while True:
            try:
                # 等待客户端消息
                message = await websocket.receive_text()
                data = json.loads(message)
                
                # 处理客户端请求
                if data.get("action") == "start":
                    if not service.is_processing:
                        service.start_processing()
                        await websocket.send_text(json.dumps({
                            "type": "status",
                            "data": {"processing": True}
                        }))

                        service.stop_processing()

                            "type": "status",
                            "data": {"processing": False}
                        }))

        if service and service.is_processing:
            service.stop_processing()

# 视频处理模块开发文档

## 概述

视频处理模块是真实婴儿AI管家系统的核心组件之一，负责从各种视频源采集视频数据，进行目标检测、行为识别和特征提取。

## 架构设计

### 组件架构

```
视频处理模块
├── 视频采集器 (VideoCollector)
├── 视频预处理器 (VideoPreprocessor)
├── 目标检测器 (ObjectDetector)
│   ├── YOLOv8检测器 (YOLOv8Detector)
│   └── MediaPipe人脸检测器 (MediaPipeFaceDetector)
├── 行为识别器 (BehaviorRecognizer)
├── 特征提取器 (FeatureExtractor)
└── 视频处理服务 (VideoProcessingService)
```

### 数据流

```
视频源 → 视频采集器 → 视频预处理器 → 目标检测器 → 行为识别器 → 特征提取器 → 输出结果
```

## 核心组件

### 1. 视频采集器 (VideoCollector)

视频采集器负责从各种视频源（摄像头、文件、流）采集视频数据。

#### 主要方法

- `start_collection()`: 开始视频采集
- `stop_collection()`: 停止视频采集
- `get_frame()`: 获取最新视频帧
- `get_video_info()`: 获取视频信息

#### 配置参数

```yaml
video_collector:
  source_type: "camera|file|stream"
  source_path: "string"
  fps: 30
  resolution: [1920, 1080]
  buffer_size: 10
```

### 2. 视频预处理器 (VideoPreprocessor)

视频预处理器负责对采集的视频帧进行预处理，如尺寸调整、颜色空间转换、降噪等。

#### 主要方法

- `process()`: 处理视频帧
- `resize()`: 调整帧尺寸
- `convert_color_space()`: 转换颜色空间
- `denoise()`: 降噪处理

#### 配置参数

```yaml
video_preprocessor:
  target_size: [640, 480]
  color_space: "RGB|HSV|GRAY"
  denoise_method: "gaussian|bilateral|median"
  denoise_params:
    kernel_size: 5
    sigma: 1.0
```

### 3. 目标检测器 (ObjectDetector)

目标检测器负责检测视频帧中的目标，如人物、物体等。

#### YOLOv8检测器

YOLOv8检测器基于YOLOv8模型，提供高效的目标检测功能。

#### 主要方法

- `detect()`: 检测单帧中的目标
- `detect_batch()`: 批量检测多帧中的目标

#### 配置参数

```yaml
object_detector:
  model_path: "yolov8n.pt"
  confidence_threshold: 0.5
  iou_threshold: 0.45
  device: "cpu|cuda"
  target_classes: ["person", "car", ...]
```

#### MediaPipe人脸检测器

MediaPipe人脸检测器基于MediaPipe库，专门用于人脸检测。

#### 主要方法

- `detect()`: 检测单帧中的人脸
- `_extract_keypoints()`: 提取人脸关键点

#### 配置参数

```yaml
face_detector:
  model_selection: 0  # 0: 短距离, 1: 长距离
  min_detection_confidence: 0.5
```

### 4. 行为识别器 (BehaviorRecognizer)

行为识别器基于3D CNN和LSTM模型，识别视频中的行为。

#### 主要方法

- `recognize()`: 识别当前帧中的行为
- `_preprocess_frame()`: 预处理视频帧
- `reset()`: 重置帧缓冲区

#### 配置参数

```yaml
behavior_recognizer:
  model_path: "behavior_model.pth"
  sequence_length: 16
  frame_size: [224, 224]
  behavior_classes: ["sleeping", "crying", "playing", "eating"]
  device: "cpu|cuda"
```

### 5. 特征提取器 (FeatureExtractor)

特征提取器从视频帧和检测结果中提取各种特征。

#### 主要方法

- `extract()`: 提取特征
- `_extract_color_features()`: 提取颜色特征
- `_extract_texture_features()`: 提取纹理特征
- `_extract_motion_features()`: 提取运动特征

#### 配置参数

```yaml
feature_extractor:
  color_features: true
  texture_features: true
  motion_features: true
  color_bins: 16
  texture_method: "lbp|glcm"
  motion_method: "optical_flow|frame_diff"
```

### 6. 视频处理服务 (VideoProcessingService)

视频处理服务协调各个组件，完成整个视频处理流程。

#### 主要方法

- `start_processing()`: 开始视频处理
- `stop_processing()`: 停止视频处理
- `add_frame_callback()`: 添加帧处理回调函数
- `add_detection_callback()`: 添加检测结果回调函数
- `add_behavior_callback()`: 添加行为识别回调函数
- `get_stats()`: 获取处理统计信息
- `get_video_info()`: 获取视频信息
