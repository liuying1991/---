# 中期记忆模块6A工作流

## 模块概述

中期记忆模块是AI系统中负责中期信息存储和管理的关键组件，介于短期记忆和长期记忆之间，主要存储需要保留数小时到数周的信息。该模块采用分布式数据库和缓存系统相结合的架构，实现了高效的数据存储、检索和管理功能。

### 模块职责

1. 中期数据存储：存储需要保留数小时到数周的信息
2. 数据检索：提供高效的数据检索和查询功能
3. 数据管理：实现数据的组织、分类和索引
4. 数据迁移：实现数据从短期记忆到长期记忆的迁移
5. 数据一致性：确保数据的一致性和完整性

### 技术特点

1. 分布式存储：采用分布式数据库，支持大规模数据存储
2. 高性能访问：通过缓存和索引优化，实现高性能数据访问
3. 数据生命周期管理：自动管理数据的生命周期，实现数据的自动迁移
4. 数据压缩：采用高效的数据压缩算法，减少存储空间占用
5. 数据安全：实现数据加密和访问控制，确保数据安全

   @Service
   @Transactional
   public class MemoryStorageService {
       
       @Autowired
       private MemoryDataRepository memoryDataRepository;
       
       @Autowired
       private MetadataRepository metadataRepository;
       
       @Autowired
       private IndexService indexService;
       
       @Autowired
       private CompressionService compressionService;
       
       /**
        * 存储数据
        */
       public MemoryData storeData(MemoryDataRequest request) {
           // 1. 数据验证
           validateData(request);
           
           // 2. 数据压缩
           byte[] compressedContent = compressionService.compress(request.getContent());
           
           // 3. 创建数据对象
           MemoryData data = MemoryData.builder()
               .id(UUID.randomUUID().toString())
               .type(request.getType())
               .content(compressedContent)
               .metadata(request.getMetadata())
               .tags(request.getTags())
               .createdAt(LocalDateTime.now())
               .updatedAt(LocalDateTime.now())
               .expiresAt(calculateExpiryTime(request))
               .accessCount(0)
               .accessFrequency(0.0)
               .importance(calculateImportance(request))
               .build();
           
           // 4. 存储数据
           MemoryData savedData = memoryDataRepository.save(data);
           
           // 5. 创建元数据
           Metadata metadata = createMetadata(savedData, request);
           metadataRepository.save(metadata);
           
           // 6. 更新索引
           indexService.updateIndex(savedData);
           
           return savedData;
       }
       
       /**
        * 获取数据
        */
       public MemoryData getData(String id) {
           MemoryData data = memoryDataRepository.findById(id)
               .orElseThrow(() -> new DataNotFoundException("Data not found: " + id));
           
           // 更新访问统计
           updateAccessStatistics(data);
           
           // 解压缩数据
           byte[] decompressedContent = compressionService.decompress(data.getContent());
           data.setContent(decompressedContent);
           
           return data;
       }
       
       /**
        * 批量存储数据
        */
       public List<MemoryData> batchStoreData(List<MemoryDataRequest> requests) {
           List<MemoryData> results = new ArrayList<>();
           
           // 分批处理，避免内存溢出
           int batchSize = 100;
           for (int i = 0; i < requests.size(); i += batchSize) {
               int end = Math.min(i + batchSize, requests.size());
               List<MemoryDataRequest> batch = requests.subList(i, end);
               
               List<MemoryData> batchResults = batch.stream()
                   .map(this::storeData)
                   .collect(Collectors.toList());
               
               results.addAll(batchResults);
           }
           
           return results;
       }
       
       private void validateData(MemoryDataRequest request) {
           if (request == null || request.getContent() == null) {
               throw new InvalidDataException("Data content cannot be null");
           }
           
           if (request.getContent().length > MAX_DATA_SIZE) {
               throw new DataSizeExceededException("Data size exceeds maximum limit");
           }
       }
       
       private LocalDateTime calculateExpiryTime(MemoryDataRequest request) {
           Duration ttl = request.getTtl() != null ? request.getTtl() : DEFAULT_TTL;
           return LocalDateTime.now().plus(ttl);
       }
       
       private double calculateImportance(MemoryDataRequest request) {
           // 基于数据类型、大小、标签等因素计算重要性
           double importance = 0.5; // 默认重要性
           
           // 根据数据类型调整重要性
           if (request.getTags() != null && request.getTags().contains("important")) {
               importance += 0.3;
           }
           
           // 根据数据大小调整重要性
           if (request.getContent().length > LARGE_DATA_THRESHOLD) {
               importance += 0.1;
           }
           
           return Math.min(1.0, importance);
       }
       
       private Metadata createMetadata(MemoryData data, MemoryDataRequest request) {
           return Metadata.builder()
               .dataId(data.getId())
               .source(request.getSource())
               .format(request.getFormat())
               .size(request.getContent().length)
               .checksum(calculateChecksum(request.getContent()))
               .version(1)
               .owner(request.getOwner())
               .permissions(request.getPermissions())
               .customFields(request.getCustomFields())
               .build();
       }
       
       private String calculateChecksum(byte[] content) {
           try {
               MessageDigest digest = MessageDigest.getInstance("SHA-256");
               byte[] hash = digest.digest(content);
               return DatatypeConverter.printHexBinary(hash);
           } catch (NoSuchAlgorithmException e) {
               throw new RuntimeException("Failed to calculate checksum", e);
           }
       }
       
       private void updateAccessStatistics(MemoryData data) {
           data.setAccessCount(data.getAccessCount() + 1);
           
           // 计算访问频率（简化版）
           long daysSinceCreation = ChronoUnit.DAYS.between(data.getCreatedAt(), LocalDateTime.now());
           if (daysSinceCreation > 0) {
               data.setAccessFrequency((double) data.getAccessCount() / daysSinceCreation);
           }
           
           data.setUpdatedAt(LocalDateTime.now());
           memoryDataRepository.save(data);
       }
   }

   @Service
   public class MemoryRetrievalService {
       
       @Autowired
       private MemoryDataRepository memoryDataRepository;
       
       @Autowired
       private ElasticsearchTemplate elasticsearchTemplate;
       
       @Autowired
       private IndexService indexService;
       
       /**
        * 根据ID获取数据
        */
       public MemoryData getById(String id) {

       /**
        * 根据条件查询数据
        */
       public Page<MemoryData> query(MemoryQueryRequest request) {
           // 构建查询条件
           QueryBuilder queryBuilder = buildQueryBuilder(request);
           
           // 构建排序条件
           SortBuilder sortBuilder = buildSortBuilder(request);
           
           // 构建分页参数
           Pageable pageable = PageRequest.of(
               request.getPage(),
               request.getSize(),
               Sort.by(sortBuilder.getField()).descending()
           );
           
           // 执行查询
           NativeSearchQuery searchQuery = new NativeSearchQueryBuilder()
               .withQuery(queryBuilder)
               .withPageable(pageable)
               .build();
           
           SearchHits<MemoryData> searchHits = elasticsearchTemplate
               .search(searchQuery, MemoryData.class);
           
           List<MemoryData> content = searchHits.stream()
               .map(SearchHit::getContent)
               .collect(Collectors.toList());
           
           return new PageImpl<>(content, pageable, searchHits.getTotalHits());
       }
       
       /**
        * 全文搜索
        */
       public Page<MemoryData> fullTextSearch(String keyword, int page, int size) {
           QueryBuilder queryBuilder = QueryBuilders.multiMatchQuery(keyword)
               .field("content")
               .field("metadata.title")
               .field("metadata.description")
               .type(MultiMatchQueryBuilder.Type.BEST_FIELDS);
           
           Pageable pageable = PageRequest.of(page, size);
           
           NativeSearchQuery searchQuery = new NativeSearchQueryBuilder()
               .withQuery(queryBuilder)
               .withPageable(pageable)
               .build();
           
           SearchHits<MemoryData> searchHits = elasticsearchTemplate
               .search(searchQuery, MemoryData.class);
           
           List<MemoryData> content = searchHits.stream()
               .map(SearchHit::getContent)
               .collect(Collectors.toList());
           
           return new PageImpl<>(content, pageable, searchHits.getTotalHits());
       }
       
       /**
        * 根据标签查询
        */
       public List<MemoryData> getByTags(List<String> tags) {
           return memoryDataRepository.findByTagsIn(tags);
       }
       
       /**
        * 根据时间范围查询
        */
       public List<MemoryData> getByTimeRange(LocalDateTime startTime, LocalDateTime endTime) {
           return memoryDataRepository.findByCreatedAtBetween(startTime, endTime);
       }
       
       private QueryBuilder buildQueryBuilder(MemoryQueryRequest request) {
           BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
           
           // 数据类型
           if (request.getType() != null) {
               boolQuery.must(QueryBuilders.termQuery("type", request.getType()));
           }
           
           // 标签
           if (request.getTags() != null && !request.getTags().isEmpty()) {
               boolQuery.must(QueryBuilders.termsQuery("tags", request.getTags()));
           }
           
           // 时间范围
           if (request.getStartTime() != null && request.getEndTime() != null) {
               boolQuery.must(QueryBuilders.rangeQuery("createdAt")
                   .gte(request.getStartTime())
                   .lte(request.getEndTime()));
           }
           
           // 重要性范围
           if (request.getMinImportance() != null) {
               boolQuery.must(QueryBuilders.rangeQuery("importance")
                   .gte(request.getMinImportance()));
           }
           
           if (request.getMaxImportance() != null) {
               boolQuery.must(QueryBuilders.rangeQuery("importance")
                   .lte(request.getMaxImportance()));
           }
           
           // 自定义字段
           if (request.getCustomFields() != null && !request.getCustomFields().isEmpty()) {
               for (Map.Entry<String, Object> entry : request.getCustomFields().entrySet()) {
                   boolQuery.must(QueryBuilders.termQuery(
                       "metadata.customFields." + entry.getKey(),
                       entry.getValue()
                   ));
               }
           }
           
           return boolQuery;
       }
       
       private SortBuilder buildSortBuilder(MemoryQueryRequest request) {
           String sortBy = request.getSortBy() != null ? request.getSortBy() : "createdAt";
           return new FieldSortBuilder(sortBy);
       }
   }

   @Service
   public class MemoryMigrationService {
       
       @Autowired
       private MemoryDataRepository memoryDataRepository;
       
       @Autowired
       private LongTermMemoryService longTermMemoryService;
       
       @Autowired
       private MigrationRuleEngine migrationRuleEngine;
       
       @Autowired
       private MigrationJobRepository migrationJobRepository;
       
       /**
        * 执行数据迁移
        */
       @Async
       public CompletableFuture<MigrationResult> executeMigration(MigrationRequest request) {
           // 创建迁移任务
           MigrationJob job = createMigrationJob(request);
           migrationJobRepository.save(job);
           
           try {
               // 1. 扫描需要迁移的数据
               List<MemoryData> dataToMigrate = scanDataToMigrate(request);
               
               // 2. 分批迁移数据
               int batchSize = request.getBatchSize() != null ? request.getBatchSize() : DEFAULT_BATCH_SIZE;
               int totalBatches = (int) Math.ceil((double) dataToMigrate.size() / batchSize);
               
               int successCount = 0;
               int failureCount = 0;
               List<String> failedIds = new ArrayList<>();
               
               for (int i = 0; i < totalBatches; i++) {
                   int start = i * batchSize;
                   int end = Math.min(start + batchSize, dataToMigrate.size());
                   List<MemoryData> batch = dataToMigrate.subList(start, end);
                   
                   // 更新任务进度
                   updateJobProgress(job, i + 1, totalBatches);
                   
                   // 迁移批次数据
                   for (MemoryData data : batch) {
                       try {
                           // 迁移到长期记忆
                           boolean migrated = longTermMemoryService.migrateFromMediumTerm(data);
                           if (migrated) {
                               // 从中期记忆中删除
                               memoryDataRepository.delete(data);
                               successCount++;
                           } else {
                               failureCount++;
                               failedIds.add(data.getId());
                           }
                       } catch (Exception e) {
                           failureCount++;
                           failedIds.add(data.getId());
                           log.error("Failed to migrate data: " + data.getId(), e);
                       }
                   }
                   
                   // 批次间延迟，避免系统负载过高
                   if (i < totalBatches - 1) {
                       Thread.sleep(BATCH_DELAY_MS);
                   }
               }
               
               // 3. 更新任务状态
               MigrationResult result = MigrationResult.builder()
                   .jobId(job.getId())
                   .totalCount(dataToMigrate.size())
                   .successCount(successCount)
                   .failureCount(failureCount)
                   .failedIds(failedIds)
                   .build();
               
               job.setStatus(MigrationJobStatus.COMPLETED);
               job.setCompletedAt(LocalDateTime.now());
               job.setResult(result);
               migrationJobRepository.save(job);
               
               return CompletableFuture.completedFuture(result);
               
           } catch (Exception e) {
               // 更新任务状态为失败
               job.setStatus(MigrationJobStatus.FAILED);
               job.setCompletedAt(LocalDateTime.now());
               job.setErrorMessage(e.getMessage());
               migrationJobRepository.save(job);

           for (MemoryData data : dataToClassify) {
               // 基于分类规则对数据进行分类

               List<MemoryData> expiredData = memoryDataRepository.findByExpiresAtBefore(LocalDateTime.now());
               dataToCleanup.addAll(expiredData);
           }
           
           // 查找低访问频率数据
           if (request.isCleanupLowFrequency()) {
               List<MemoryData> lowFrequencyData = memoryDataRepository.findByAccessFrequencyLessThan(
                   request.getFrequencyThreshold()
               );
               dataToCleanup.addAll(lowFrequencyData);
           }
           
           // 查找低重要性数据
           if (request.isCleanupLowImportance()) {
               List<MemoryData> lowImportanceData = memoryDataRepository.findByImportanceLessThan(
                   request.getImportanceThreshold()
               );
               dataToCleanup.addAll(lowImportanceData);
           }
           
           // 去重
           List<String> idsToDelete = dataToCleanup.stream()
               .map(MemoryData::getId)
               .distinct()
               .collect(Collectors.toList());
           
           // 删除数据
           List<MemoryData> deletedData = memoryDataRepository.findAllById(idsToDelete);
           memoryDataRepository.deleteAll(deletedData);
           
           // 删除元数据
           List<Metadata> metadataToDelete = metadataRepository.findAllByDataIdIn(idsToDelete);
           metadataRepository.deleteAll(metadataToDelete);
           
           return CleanupResult.builder()
               .deletedCount(deletedData.size())
               .deletedIds(idsToDelete)
               .build();
       }

           long endTime = System.currentTimeMillis();
           
           // 验证结果
           assertEquals(threadCount * operationsPerThread, allResults.size());
           
           long duration = endTime - startTime;
           double throughput = allResults.size() / (duration / 1000.0); // 数据/秒
           assertTrue("Concurrent throughput should be at least 200 data/sec, but was " + throughput, 
                      throughput >= 200);
           
           executorService.shutdown();
       }
       
       @Test
       void testConcurrentDataRetrievePerformance() throws InterruptedException, ExecutionException {
           // 准备测试数据
           List<MemoryDataRequest> requests = new ArrayList<>();
           for (int i = 0; i < 100; i++) {
               byte[] content = new byte[1024];
               new Random().nextBytes(content);
               
               requests.add(MemoryDataRequest.builder()
                   .type("binary")
                   .content(content)
                   .metadata(Map.of("index", i))
                   .build());
           }
           
           List<MemoryData> storedData = memoryStorageService.batchStoreData(requests);
           
           int threadCount = 10;
           ExecutorService executorService = Executors.newFixedThreadPool(threadCount);
           List<Future<Void>> futures = new ArrayList<>();
           
           // 创建并发检索任务
           for (int i = 0; i < threadCount; i++) {
               futures.add(executorService.submit(() -> {
                   for (MemoryData data : storedData) {
                       memoryStorageService.getData(data.getId());
                   }
                   return null;
               }));
           }
           
           // 等待所有任务完成
           long startTime = System.currentTimeMillis();
           
           for (Future<Void> future : futures) {
               future.get();
           }
           
           long endTime = System.currentTimeMillis();
           
           // 验证结果
           long duration = endTime - startTime;
           double throughput = (threadCount * storedData.size()) / (duration / 1000.0); // 数据/秒
           assertTrue("Concurrent retrieval throughput should be at least 500 data/sec, but was " + throughput, 
                      throughput >= 500);
           
           executorService.shutdown();
       }
