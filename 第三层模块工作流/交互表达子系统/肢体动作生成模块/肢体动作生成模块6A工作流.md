# 肢体动作生成模块6A工作流

## 模块概述

### 模块名称
肢体动作生成模块

### 所属子系统
交互表达子系统

### 模块功能描述
肢体动作生成模块负责根据文本内容、情感色彩和交互场景，生成适合婴儿认知和情感需求的虚拟角色肢体动作。该模块能够创建生动、友好、富有表现力的肢体动作，增强与婴儿的交互体验，促进婴儿的情感发展和认知能力提升。

### 模块职责
1. 动作生成：根据内容和情感生成相应的肢体动作
2. 情感表达：在肢体动作中融入适当的情感色彩
3. 年龄适配：根据婴儿年龄调整动作特征
4. 动态动作：生成动态变化的动作序列
5. 个性化动作：根据婴儿偏好生成个性化动作
6. 动作优化：优化动作质量和自然度

### 模块接口
- **输入接口**：文本内容、情感参数、婴儿状态
- **输出接口**：动作数据、动作元数据
- **依赖模块**：文本分析模块、情感识别模块
- **被依赖模块**：显示输出模块、交互管理模块

# 动作生成器实现
class ActionGenerator:
    def __init__(self, base_generator, emotion_generator, dynamic_generator, 
                 age_adapter, personalization_adapter, action_optimizer, quality_assessor):
        self.base_generator = base_generator
        self.emotion_generator = emotion_generator
        self.dynamic_generator = dynamic_generator
        self.age_adapter = age_adapter
        self.personalization_adapter = personalization_adapter
        self.action_optimizer = action_optimizer
        self.quality_assessor = quality_assessor
        self.content_analyzer = ContentAnalyzer()
        
    def generate(self, text, generation_params, baby_state):
        # 输入处理
        processed_input = self._process_input(text, generation_params, baby_state)
        
        # 内容分析
        content_analysis = self.content_analyzer.analyze(processed_input["text"])
        
        # 基础动作生成
        base_action = self.base_generator.generate(
            processed_input["text"],
            content_analysis,
            processed_input["base_params"]
        )
        
        # 情感动作生成
        if generation_params.get("apply_emotion", True):
            emotion_action = self.emotion_generator.add_emotion(
                base_action,
                content_analysis["emotion"],
                generation_params.get("emotion_intensity", 0.8)
            )
        else:
            emotion_action = base_action
        
        # 年龄适配
        if "age" in baby_state:
            age_adapted_action = self.age_adapter.adapt_for_age(
                emotion_action,
                baby_state["age"]
            )
        else:
            age_adapted_action = emotion_action
        
        # 个性化适配
        if generation_params.get("apply_personalization", True) and "user_id" in baby_state:
            personalized_action = self.personalization_adapter.adapt_for_user(
                age_adapted_action,
                baby_state["user_id"]
            )
        else:
            personalized_action = age_adapted_action
        
        # 动态动作生成
        if generation_params.get("generate_dynamic", False):
            dynamic_action = self.dynamic_generator.generate_dynamic(
                personalized_action,
                content_analysis,
                generation_params.get("animation_params", {})
            )
        else:
            dynamic_action = personalized_action
        
        # 动作优化
        if generation_params.get("optimize_action", True):
            optimized_action = self.action_optimizer.optimize(
                dynamic_action,
                generation_params
            )
        else:
            optimized_action = dynamic_action
        
        # 质量评估
        quality_score = self.quality_assessor.assess(optimized_action, content_analysis)
        
        # 如果质量评分不达标，尝试重新生成
        if quality_score < 3.5:
            # 尝试调整生成参数并重新生成
            adjusted_params = self._adjust_generation_params(generation_params, quality_score)
            return self.generate(text, adjusted_params, baby_state)
        
        return {
            "text": text,
            "action": optimized_action,
            "metadata": {
                "emotion": content_analysis["emotion"],
                "age": baby_state.get("age", None),
                "user_id": baby_state.get("user_id", None),
                "is_dynamic": generation_params.get("generate_dynamic", False),
                "quality_score": quality_score
            },
            "timestamp": time.time()
        }

class BaseActionGenerator:

            return "storytelling"

            return "neutral"

        base_action = self.action_model.generate(
            template=template,
            text=text,
            params=generation_params
        )
        
        return base_action

class EmotionActionGenerator:
    def __init__(self):
        self.emotion_converter = self._load_emotion_converter()
        self.emotion_features = self._load_emotion_features()
        
    def add_emotion(self, base_action, emotion, intensity=0.8):
        if emotion not in self.emotion_features:
            return base_action
        
        emotion_features = self.emotion_features[emotion]
        
        # 应用情感特征
        emotion_action = self._apply_emotion_features(
            base_action,
            emotion_features,
            intensity
        )
        
        return emotion_action
    
    def _apply_emotion_features(self, base_action, emotion_features, intensity):
        # 应用情感相关的动作特征
        emotion_action = base_action.copy()
        
        # 手部动作调整
        if "hands" in emotion_features:
            emotion_action = self._adjust_hand_features(
                emotion_action,
                emotion_features["hands"],
                intensity
            )
        
        # 身体动作调整
        if "body" in emotion_features:
            emotion_action = self._adjust_body_features(
                emotion_action,
                emotion_features["body"],
                intensity
            )
        
        # 头部动作调整
        if "head" in emotion_features:
            emotion_action = self._adjust_head_features(
                emotion_action,
                emotion_features["head"],
                intensity
            )
        
        # 整体动作调整
        if "overall" in emotion_features:
            emotion_action = self._adjust_overall_features(
                emotion_action,
                emotion_features["overall"],
                intensity
            )
        
        return emotion_action
    
    def _adjust_hand_features(self, action, hand_features, intensity):
        # 调整手部动作特征
        # 根据情感类型调整手部动作的幅度、速度和形状
        
        # 获取当前手部特征
        current_hand_features = self._extract_hand_features(action)
        
        # 计算目标手部特征
        target_hand_features = {}
        for feature, value in hand_features.items():
            current_value = current_hand_features.get(feature, 0)
            target_value = current_value + (value - current_value) * intensity
            target_hand_features[feature] = target_value
        
        # 应用目标手部特征
        action = self._apply_hand_features(action, target_hand_features)
        
        return action
    
    def _adjust_body_features(self, action, body_features, intensity):
        # 调整身体动作特征
        # 根据情感类型调整身体动作的幅度、速度和姿态
        
        # 获取当前身体特征
        current_body_features = self._extract_body_features(action)
        
        # 计算目标身体特征
        target_body_features = {}
        for feature, value in body_features.items():
            current_value = current_body_features.get(feature, 0)
            target_value = current_value + (value - current_value) * intensity
            target_body_features[feature] = target_value
        
        # 应用目标身体特征
        action = self._apply_body_features(action, target_body_features)
        
        return action

class DynamicActionGenerator:
    def __init__(self):
        self.animation_templates = self._load_animation_templates()
        self.transition_generator = TransitionGenerator()
        
    def generate_dynamic(self, static_action, content_analysis, animation_params):
        # 确定动画类型
        animation_type = self._determine_animation_type(content_analysis, animation_params)
        
        # 获取动画模板
        template = self.animation_templates[animation_type]
        
        # 生成动态动作
        dynamic_action = self._generate_dynamic_action(
            static_action,
            template,
            animation_params
        )
        
        return dynamic_action
    
    def _determine_animation_type(self, content_analysis, animation_params):
        # 根据内容分析和动画参数确定动画类型
        if "animation_type" in animation_params:
            return animation_params["animation_type"]
        
        # 根据内容类型自动选择动画类型
        if "question" in content_analysis["content_type"]:
            return "thinking"
        elif "story" in content_analysis["content_type"]:
            return "storytelling"
        elif "song" in content_analysis["content_type"]:
            return "singing"
        else:
            return "breathing"
    
    def _generate_dynamic_action(self, static_action, template, animation_params):
        # 生成动态动作序列
        frames = []
        num_frames = animation_params.get("num_frames", 90)  # 3秒，30fps
        
        # 生成过渡帧
        for i in range(num_frames):
            # 计算当前帧的参数
            frame_params = self._calculate_frame_params(
                template,
                i,
                num_frames,
                animation_params
            )
            
            # 生成当前帧
            frame = self._generate_frame(static_action, frame_params)
            frames.append(frame)
        
        # 组合帧为动态动作
        dynamic_action = self._combine_frames(frames)
        
        return dynamic_action
    
    def _calculate_frame_params(self, template, frame_index, num_frames, animation_params):
        # 计算当前帧的参数
        progress = frame_index / num_frames
        
        # 基础参数
        frame_params = {
            "progress": progress,
            "intensity": animation_params.get("intensity", 0.8)
        }
        
        # 根据动画类型计算特定参数
        if template["type"] == "breathing":
            # 呼吸动画
            frame_params["chest_expansion"] = 1.0 + 0.1 * math.sin(2 * math.pi * progress)
            frame_params["shoulder_movement"] = 0.05 * math.sin(2 * math.pi * progress)
        elif template["type"] == "nodding":
            # 点头动画
            if 0.3 < progress < 0.7:
                frame_params["head_rotation_x"] = 15 * math.sin(math.pi * (progress - 0.3) / 0.4)
            else:
                frame_params["head_rotation_x"] = 0
        elif template["type"] == "waving":
            # 挥手动画
            frame_params["arm_rotation_z"] = 30 * math.sin(4 * math.pi * progress)
            frame_params["elbow_bend"] = 20 + 10 * math.sin(4 * math.pi * progress)
        
        return frame_params
    
    def _generate_frame(self, static_action, frame_params):
        # 生成单帧动作
        frame = static_action.copy()
        
        # 应用帧参数
        if "chest_expansion" in frame_params:
            frame = self._apply_chest_expansion(frame, frame_params["chest_expansion"])
        
        if "shoulder_movement" in frame_params:
            frame = self._apply_shoulder_movement(frame, frame_params["shoulder_movement"])
        
        if "head_rotation_x" in frame_params:
            frame = self._apply_head_rotation_x(frame, frame_params["head_rotation_x"])
        
        if "arm_rotation_z" in frame_params:
            frame = self._apply_arm_rotation_z(frame, frame_params["arm_rotation_z"])
        
        if "elbow_bend" in frame_params:
            frame = self._apply_elbow_bend(frame, frame_params["elbow_bend"])
        
        return frame
    
    def _combine_frames(self, frames):
        # 组合帧为动态动作
        # 使用BVH或FBX格式
        dynamic_action = self._create_bvh(frames)
        
        return dynamic_action
