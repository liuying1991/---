# 认知处理模块实现工作流

## 模块信息

### 模块名称
认知处理模块

### 所属子系统
认知决策子系统

### 模块功能描述
认知处理模块是真实婴儿AI管家系统的认知核心，负责模拟人类认知过程，对多源信息进行高级处理和理解。该模块整合感知信息、记忆内容和推理结果，通过认知算法和模型，实现对复杂情境的理解、分析和决策支持。

### 模块职责
1. 整合多源信息，构建统一的认知模型
2. 执行高级认知功能，如理解、分析、综合和评估
3. 管理认知过程，协调不同认知组件的工作
4. 维护认知状态，跟踪认知过程和结果
5. 提供认知解释，使认知过程透明化和可解释
6. 支持认知学习，从经验中优化认知模型和算法

### 模块接口
- **输入接口**：感知信息、记忆内容、推理结果、情境上下文
- **输出接口**：认知结果、决策支持、认知解释、学习更新
- **依赖模块**：感知处理模块、记忆存储模块、推理规划模块
- **被依赖模块**：决策制定模块、决策执行模块、交互表达模块

```python
# src/components/information_integration.py
from typing import Dict, List, Any, Optional
import numpy as np
from datetime import datetime

class InformationIntegration:
    """信息整合组件，负责整合多源信息，构建统一的认知模型"""
    
    def __init__(self):
        self.integration_strategies = {
            "fusion": self._fusion_integration,
            "aggregation": self._aggregation_integration,
            "consensus": self._consensus_integration
        }
        self.default_strategy = "fusion"
    
    def integrate(self, input_data: Dict[str, Any]) -> Dict[str, Any]:
        """
        整合多源信息
        
        Args:
            input_data: 包含多源信息的字典
            
        Returns:
            整合后的信息
        """
        # 验证输入数据
        if not input_data:
            raise ValueError("输入数据不能为空")
        
        # 获取整合策略
        strategy = input_data.get("integration_strategy", self.default_strategy)
        integration_func = self.integration_strategies.get(strategy, self._fusion_integration)
        
        # 执行信息整合
        integrated_data = integration_func(input_data)
        
        # 添加元数据
        integrated_data["metadata"] = {
            "integration_time": datetime.now().isoformat(),
            "integration_strategy": strategy,
            "source_count": len([k for k in input_data.keys() if k.endswith("_data")])
        }
        
        return integrated_data
    
    def _fusion_integration(self, input_data: Dict[str, Any]) -> Dict[str, Any]:
        """融合整合策略"""
        # 提取各源数据
        perception_data = input_data.get("perception_data", {})
        memory_data = input_data.get("memory_data", {})
        reasoning_data = input_data.get("reasoning_data", {})
        context_data = input_data.get("context_data", {})
        
        # 创建统一的数据结构
        fused_data = {
            "situation": self._extract_situation(perception_data, context_data),
            "entities": self._extract_entities(perception_data, memory_data),
            "relationships": self._extract_relationships(memory_data, reasoning_data),
            "temporal_context": self._extract_temporal_context(context_data, memory_data),
            "spatial_context": self._extract_spatial_context(perception_data, context_data)
        }
        
        return fused_data
    
    def _aggregation_integration(self, input_data: Dict[str, Any]) -> Dict[str, Any]:
        """聚合整合策略"""
        # 简单聚合各源数据
        aggregated_data = {}
        for key, value in input_data.items():
            if key.endswith("_data"):
                source_name = key[:-5]  # 移除"_data"后缀
                aggregated_data[source_name] = value
        
        return aggregated_data
    
    def _consensus_integration(self, input_data: Dict[str, Any]) -> Dict[str, Any]:
        """共识整合策略"""
        # 提取各源数据
        perception_data = input_data.get("perception_data", {})
        memory_data = input_data.get("memory_data", {})
        reasoning_data = input_data.get("reasoning_data", {})
        
        # 寻找共识信息
        consensus_data = {}
        
        # 实体共识
        perception_entities = set(perception_data.get("entities", []))
        memory_entities = set(memory_data.get("entities", []))
        reasoning_entities = set(reasoning_data.get("entities", []))
        
        consensus_entities = perception_entities & memory_entities & reasoning_entities
        consensus_data["entities"] = list(consensus_entities)
        
        # 关系共识
        perception_relations = set(perception_data.get("relations", []))
        memory_relations = set(memory_data.get("relations", []))
        reasoning_relations = set(reasoning_data.get("relations", []))
        
        consensus_relations = perception_relations & memory_relations & reasoning_relations
        consensus_data["relations"] = list(consensus_relations)
        
        return consensus_data
    
    def _extract_situation(self, perception_data: Dict, context_data: Dict) -> Dict:
        """提取情境信息"""
        situation = {
            "description": "",
            "type": "",
            "urgency": "normal",
            "participants": [],
            "environment": ""
        }
        
        # 从感知数据中提取情境描述
        if "scene_description" in perception_data:
            situation["description"] = perception_data["scene_description"]
        
        # 从上下文数据中提取情境类型
        if "situation_type" in context_data:
            situation["type"] = context_data["situation_type"]
        
        # 从感知数据中提取参与者
        if "detected_objects" in perception_data:
            situation["participants"] = perception_data["detected_objects"]
        
        # 从上下文数据中提取环境信息
        if "environment" in context_data:
            situation["environment"] = context_data["environment"]
        
        return situation
    
    def _extract_entities(self, perception_data: Dict, memory_data: Dict) -> List[Dict]:
        """提取实体信息"""
        entities = []
        
        # 从感知数据中提取实体
        if "detected_objects" in perception_data:
            for obj in perception_data["detected_objects"]:
                entity = {
                    "id": obj.get("id", ""),
                    "name": obj.get("name", ""),
                    "type": obj.get("type", ""),
                    "attributes": obj.get("attributes", {}),
                    "confidence": obj.get("confidence", 0.0),
                    "source": "perception"
                }
                entities.append(entity)
        
        # 从记忆数据中提取实体
        if "entities" in memory_data:
            for entity in memory_data["entities"]:
                # 检查是否已存在相同实体
                existing_ids = [e["id"] for e in entities]
                if entity.get("id", "") not in existing_ids:
                    entity["source"] = "memory"
                    entities.append(entity)
        
        return entities
    
    def _extract_relationships(self, memory_data: Dict, reasoning_data: Dict) -> List[Dict]:
        """提取关系信息"""
        relationships = []
        
        # 从记忆数据中提取关系
        if "relationships" in memory_data:
            for rel in memory_data["relationships"]:
                relationship = {
                    "id": rel.get("id", ""),
                    "source_entity": rel.get("source_entity", ""),
                    "target_entity": rel.get("target_entity", ""),
                    "type": rel.get("type", ""),
                    "attributes": rel.get("attributes", {}),
                    "confidence": rel.get("confidence", 0.0),
                    "source": "memory"
                }
                relationships.append(relationship)
        
        # 从推理数据中提取关系
        if "inferred_relationships" in reasoning_data:
            for rel in reasoning_data["inferred_relationships"]:
                # 检查是否已存在相同关系
                existing_ids = [r["id"] for r in relationships]
                if rel.get("id", "") not in existing_ids:
                    rel["source"] = "reasoning"
                    relationships.append(rel)
        
        return relationships
    
    def _extract_temporal_context(self, context_data: Dict, memory_data: Dict) -> Dict:
        """提取时间上下文"""
        temporal_context = {
            "current_time": "",
            "time_of_day": "",
            "day_of_week": "",
            "season": "",
            "recent_events": []
        }
        
        # 从上下文数据中提取时间信息
        if "timestamp" in context_data:
            timestamp = context_data["timestamp"]
            temporal_context["current_time"] = timestamp
            
            # 解析时间信息
            try:
                dt = datetime.fromisoformat(timestamp)
                temporal_context["time_of_day"] = self._get_time_of_day(dt.hour)
                temporal_context["day_of_week"] = dt.strftime("%A")
                temporal_context["season"] = self._get_season(dt.month)
            except:
                pass
        
        # 从记忆数据中提取最近事件
        if "recent_events" in memory_data:
            temporal_context["recent_events"] = memory_data["recent_events"]
        
        return temporal_context
    
    def _extract_spatial_context(self, perception_data: Dict, context_data: Dict) -> Dict:
        """提取空间上下文"""
        spatial_context = {
            "location": "",
            "room": "",
            "objects_positions": [],
            "layout": ""
        }
        
        # 从上下文数据中提取位置信息
        if "location" in context_data:
            spatial_context["location"] = context_data["location"]
        
        if "room" in context_data:
            spatial_context["room"] = context_data["room"]
        
        # 从感知数据中提取物体位置
        if "detected_objects" in perception_data:
            for obj in perception_data["detected_objects"]:
                if "position" in obj:
                    spatial_context["objects_positions"].append({
                        "object_id": obj.get("id", ""),
                        "position": obj["position"]
                    })
        
        return spatial_context
    
    def _get_time_of_day(self, hour: int) -> str:
        """获取一天中的时段"""
        if 5 <= hour < 12:
            return "morning"
        elif 12 <= hour < 17:
            return "afternoon"
        elif 17 <= hour < 21:
            return "evening"
        else:
            return "night"
    
    def _get_season(self, month: int) -> str:
        """获取季节"""
        if month in [12, 1, 2]:
            return "winter"
        elif month in [3, 4, 5]:
            return "spring"
        elif month in [6, 7, 8]:
            return "summer"
        else:
            return "autumn"
```

```python
# src/components/situation_understanding.py
from typing import Dict, List, Any, Optional, Tuple
import numpy as np
from datetime import datetime
import re

class SituationUnderstanding:
    """情境理解组件，负责理解复杂情境，识别关键因素和关系"""
    
    def __init__(self):
        self.understanding_models = {
            "semantic": self._semantic_understanding,
            "pattern": self._pattern_understanding,
            "causal": self._causal_understanding
        }
        self.default_model = "semantic"
    
    def understand(self, integrated_data: Dict[str, Any]) -> Dict[str, Any]:
        """
        理解情境
        
        Args:
            integrated_data: 整合后的信息
            
        Returns:
            情境理解结果
        """
        # 验证输入数据
        if not integrated_data:
            raise ValueError("输入数据不能为空")
        
        # 获取理解模型
        model = integrated_data.get("understanding_model", self.default_model)
        understanding_func = self.understanding_models.get(model, self._semantic_understanding)
        
        # 执行情境理解
        understanding_result = understanding_func(integrated_data)
        
        # 添加元数据
        understanding_result["metadata"] = {
            "understanding_time": datetime.now().isoformat(),
            "understanding_model": model
        }
        
        return understanding_result
    
    def _semantic_understanding(self, integrated_data: Dict[str, Any]) -> Dict[str, Any]:
        """语义理解模型"""
        # 提取情境信息
        situation = integrated_data.get("situation", {})
        entities = integrated_data.get("entities", [])
        relationships = integrated_data.get("relationships", [])
        temporal_context = integrated_data.get("temporal_context", {})
        spatial_context = integrated_data.get("spatial_context", {})
        
        # 语义分析
        semantic_analysis = {
            "situation_type": self._classify_situation_type(situation),
            "key_entities": self._identify_key_entities(entities),
            "key_relationships": self._identify_key_relationships(relationships),
            "temporal_patterns": self._analyze_temporal_patterns(temporal_context),
            "spatial_patterns": self._analyze_spatial_patterns(spatial_context),
            "semantic_summary": self._generate_semantic_summary(situation, entities, relationships)
        }
        
        # 情境评估
        situation_assessment = {
            "complexity": self._assess_complexity(semantic_analysis),
            "urgency": self._assess_urgency(situation, temporal_context),
            "familiarity": self._assess_familiarity(semantic_analysis),
            "predictability": self._assess_predictability(semantic_analysis)
        }
        
        return {
            "semantic_analysis": semantic_analysis,
            "situation_assessment": situation_assessment
        }
    
    def _pattern_understanding(self, integrated_data: Dict[str, Any]) -> Dict[str, Any]:
        """模式理解模型"""
        # 提取情境信息
        situation = integrated_data.get("situation", {})
        entities = integrated_data.get("entities", [])
        relationships = integrated_data.get("relationships", [])
        temporal_context = integrated_data.get("temporal_context", {})
        
        # 模式识别
        pattern_analysis = {
            "entity_patterns": self._identify_entity_patterns(entities),
            "relationship_patterns": self._identify_relationship_patterns(relationships),
            "temporal_patterns": self._identify_temporal_patterns(temporal_context),
            "behavioral_patterns": self._identify_behavioral_patterns(situation, entities),
            "contextual_patterns": self._identify_contextual_patterns(situation, temporal_context)
        }
        
        # 模式评估
        pattern_assessment = {
            "pattern_strength": self._assess_pattern_strength(pattern_analysis),
            "pattern_stability": self._assess_pattern_stability(pattern_analysis),
            "pattern_significance": self._assess_pattern_significance(pattern_analysis)
        }
        
        return {
            "pattern_analysis": pattern_analysis,
            "pattern_assessment": pattern_assessment
        }
    
    def _causal_understanding(self, integrated_data: Dict[str, Any]) -> Dict[str, Any]:
        """因果理解模型"""
        # 提取情境信息
        situation = integrated_data.get("situation", {})
        entities = integrated_data.get("entities", [])
        relationships = integrated_data.get("relationships", [])
        temporal_context = integrated_data.get("temporal_context", {})
        
        # 因果分析
        causal_analysis = {
            "causal_relationships": self._identify_causal_relationships(relationships),
            "causal_chains": self._identify_causal_chains(relationships, temporal_context),
            "causal_factors": self._identify_causal_factors(entities, relationships),
            "causal_effects": self._identify_causal_effects(entities, relationships),
            "causal_explanations": self._generate_causal_explanations(situation, entities, relationships)
        }
        
        # 因果评估
        causal_assessment = {

        for rel in relationships:

    def _identify_causal_chains(self, relationships: List[Dict], temporal_context: Dict) -> List[List[Dict]]:
        """识别因果链"""
        causal_relationships = self._identify_causal_relationships(relationships)
        if not causal_relationships:
            return []
        
        # 构建因果图
        causal_graph = {}
        for rel in causal_relationships:
            source = rel.get("source_entity", "")
            target = rel.get("target_entity", "")
            
            if source not in causal_graph:
                causal_graph[source] = []
            causal_graph[source].append((target, rel))
        
        # 识别因果链
        causal_chains = []
        visited = set()
        
        for source in causal_graph:
            if source in visited:
                continue
            
            chain = []
            self._find_causal_chain(source, causal_graph, chain, visited)
            
            if len(chain) > 1:
                causal_chains.append(chain)
        
        return causal_chains
    
    def _find_causal_chain(self, node: str, graph: Dict, chain: List[Dict], visited: set):
        """递归查找因果链"""
        if node in visited:
            return
        
        visited.add(node)
        
        if node in graph:
            for target, rel in graph[node]:
                chain.append(rel)
                self._find_causal_chain(target, graph, chain, visited)

    def _generate_causal_explanations(self, situation: Dict, entities: List[Dict], relationships: List[Dict]) -> List[str]:

        # 识别因果关系
        causal_relationships = self._identify_causal_relationships(relationships)
        if not causal_relationships:
            return ["No clear causal relationships identified."]
        
        # 为每个因果关系生成解释
        for rel in causal_relationships:
            source = rel.get("source_entity", "")
            target = rel.get("target_entity", "")
            rel_type = rel.get("type", "")
            
            # 查找对应的实体名称
            source_name = ""
            target_name = ""
            
            for entity in entities:
                entity_id = entity.get("id", "")
                if entity_id == source:
                    source_name = entity.get("name", "")
                elif entity_id == target:
                    target_name = entity.get("name", "")
            
            # 生成解释
            if source_name and target_name:
                explanation = f"{source_name} {rel_type.replace('_', ' ')} {target_name}."
                explanations.append(explanation)
        
        return explanations

            return {
                "pattern": "repetitive",
```

```python
        for attr_name, attr_value in base_attributes.items():
            is_common = True
            for item in items[1:]:
                item_attributes = item.get("attributes", {})
                if attr_name not in item_attributes or item_attributes[attr_name] != attr_value:

        attribute_frequencies = {}
        for attr_name, value_counts in all_attributes.items():
            total_count = sum(value_counts.values())
            attribute_frequencies[attr_name] = {
                value: count / total_count for value, count in value_counts.items()
            }
```

```python
# learning/learning_component.py

from typing import Dict, List, Any, Optional, Tuple

class LearningComponent:

    def learn(self, experience: Dict[str, Any], feedback: Dict[str, Any]) -> Dict[str, Any]:
        """从经验中学习

        """

        learning_func = self.learning_strategies.get(strategy, self._reinforcement_learning)
        
        # 执行学习
        learning_result = learning_func(experience, feedback)
        
        # 更新学习历史
        self._update_learning_history(experience, feedback, learning_result)
        
        # 更新性能指标
        self._update_performance_metrics(learning_result)
        
        # 添加元数据
        learning_result["metadata"] = {
            "learning_time": datetime.now().isoformat(),
            "learning_strategy": strategy,
            "experience_id": experience.get("id", ""),
            "feedback_id": feedback.get("id", "")
        }
        
        return learning_result
    
    def _reinforcement_learning(self, experience: Dict[str, Any], feedback: Dict[str, Any]) -> Dict[str, Any]:
        """强化学习"""
        # 提取状态、动作、奖励
        state = experience.get("state", {})

        reward = feedback.get("reward", 0)
        next_state = feedback.get("next_state", {})
        
        # 更新Q值（简化实现）
        q_value_update = self._update_q_value(state, action, reward, next_state)

        policy_update = self._update_policy(state, action, q_value_update)
        
        return {
            "learning_type": "reinforcement",

            "policy_update": policy_update,

    def _supervised_learning(self, experience: Dict[str, Any], feedback: Dict[str, Any]) -> Dict[str, Any]:

        # 模型预测

    def _update_q_value(self, state: Dict[str, Any], action: Dict[str, Any], reward: float, next_state: Dict[str, Any]) -> Dict[str, Any]:
        """更新Q值"""
        # 简化的Q值更新
        state_key = self._state_to_key(state)
        action_key = self._action_to_key(action)
        
        # 获取当前Q值
        current_q_value = self._get_q_value(state_key, action_key)
        
        # 获取下一状态的最大Q值
        next_max_q_value = self._get_max_q_value(next_state)
        
        # 计算新的Q值
        new_q_value = current_q_value + self.learning_rate * (reward + self.discount_factor * next_max_q_value - current_q_value)
        
        # 更新Q值表
        self._set_q_value(state_key, action_key, new_q_value)
        
        return {
            "state_key": state_key,
            "action_key": action_key,
            "old_q_value": current_q_value,
            "new_q_value": new_q_value,
            "reward": reward,
            "next_max_q_value": next_max_q_value
        }
    
    def _update_policy(self, state: Dict[str, Any], action: Dict[str, Any], q_value_update: Dict[str, Any]) -> Dict[str, Any]:
        """更新策略"""
        # 简化的策略更新
        state_key = q_value_update.get("state_key", "")
        action_key = q_value_update.get("action_key", "")
        new_q_value = q_value_update.get("new_q_value", 0)
        
        # 获取状态的所有可能动作
        possible_actions = self._get_possible_actions(state)
        
        # 计算每个动作的概率
        action_probabilities = {}
        total_q_value = 0
        
        for possible_action in possible_actions:
            possible_action_key = self._action_to_key(possible_action)
            q_value = self._get_q_value(state_key, possible_action_key)
            action_probabilities[possible_action_key] = q_value
            total_q_value += q_value
        
        # 归一化概率
        if total_q_value > 0:
            for action_key in action_probabilities:
                action_probabilities[action_key] /= total_q_value
        else:
            # 如果所有Q值为0，使用均匀分布
            for action_key in action_probabilities:
                action_probabilities[action_key] = 1.0 / len(action_probabilities)
        
        # 更新策略
        self._set_policy(state_key, action_probabilities)
        
        return {
            "state_key": state_key,
            "action_probabilities": action_probabilities,
            "updated_action": action_key,
            "updated_probability": action_probabilities.get(action_key, 0)
        }
    
    def _predict(self, input_data: Dict[str, Any]) -> Dict[str, Any]:

        # 实际实现中应该使用训练好的模型
        
        # 提取输入特征
        features = self._extract_features(input_data)
        
        # 模拟预测结果
        prediction = {}
        
        # 基于特征生成预测
        for feature_name, feature_value in features.items():
            if isinstance(feature_value, (int, float)):

                prediction[feature_name] = feature_value * 0.8 + np.random.normal(0, 0.1)
            else:
                # 分类预测
                prediction[feature_name] = f"predicted_{feature_value}"
        
        return prediction
    
    def _calculate_loss(self, predicted_output: Dict[str, Any], target_output: Dict[str, Any]) -> Dict[str, Any]:
        """计算损失"""
        loss = {}
        
        # 计算每个特征的损失
        for key, target_value in target_output.items():
            if key in predicted_output:
                predicted_value = predicted_output[key]
                
                if isinstance(target_value, (int, float)) and isinstance(predicted_value, (int, float)):
                    # 回归损失（均方误差）
                    loss[key] = (predicted_value - target_value) ** 2
                else:
                    # 分类损失（0-1损失）
                    loss[key] = 0 if predicted_value == target_value else 1
            else:
                # 缺失预测
                loss[key] = 1.0
        
        # 计算总损失
        total_loss = sum(loss.values()) / len(loss) if loss else 0
        
        return {
            "feature_losses": loss,
            "total_loss": total_loss
        }
    
    def _update_parameters(self, input_data: Dict[str, Any], predicted_output: Dict[str, Any], 
                          target_output: Dict[str, Any], loss: Dict[str, Any]) -> Dict[str, Any]:
        """更新模型参数"""
        # 简化的参数更新
        # 实际实现中应该使用梯度下降等优化算法
        
        parameter_updates = {}
        
        # 基于损失计算参数更新
        total_loss = loss.get("total_loss", 0)
        
        # 模拟参数更新
        for key in predicted_output:
            if key in target_output:
                # 计算梯度（简化）
                gradient = predicted_output[key] - target_output[key]
                
                # 参数更新
                parameter_update = -self.learning_rate * gradient
                parameter_updates[key] = parameter_update
        
        return {
            "parameter_updates": parameter_updates,
            "total_loss": total_loss,
            "learning_rate": self.learning_rate
        }
    
    def _extract_features(self, input_data: Dict[str, Any]) -> Dict[str, Any]:
        """提取特征"""
        features = {}
        
        # 提取数值特征
        for key, value in input_data.items():
            if isinstance(value, (int, float)):
                features[key] = value
            elif isinstance(value, str):
                # 文本特征（简化）
                features[f"{key}_length"] = len(value)
                features[f"{key}_words"] = len(value.split())
            elif isinstance(value, list):
                # 列表特征
                features[f"{key}_count"] = len(value)
            elif isinstance(value, dict):
                # 字典特征
                features[f"{key}_keys"] = len(value.keys())
        
        return features
    
    def _identify_patterns(self, features: Dict[str, Any]) -> List[Dict[str, Any]]:
```

            union = current_entity_types.union(case_entity_types)
            jaccard_similarity = len(intersection) / len(union) if union else 0
            similarity += jaccard_similarity
            feature_count += 1

            similarity += jaccard_similarity

```python
from typing import Dict, List, Any, Optional, Tuple
from datetime import datetime
import logging

logger = logging.getLogger(__name__)

class DecisionComponent:
    """决策组件
    
    负责基于推理结果和当前情境做出决策，生成行动计划。

    def __init__(self, config: Dict[str, Any]):

    def decide(self, reasoning_result: Dict[str, Any], context: Dict[str, Any]) -> Dict[str, Any]:

        if not context:
            raise ValueError("情境不能为空")
        
        # 获取决策策略

        decision_func = self.decision_strategies.get(strategy, self._utility_based_decision)
        
        # 执行决策
        decision_result = decision_func(reasoning_result, context)
        
        # 更新决策历史
        self._update_decision_history(reasoning_result, context, decision_result)
        
        # 添加元数据
        decision_result["metadata"] = {
            "decision_time": datetime.now().isoformat(),
            "decision_strategy": strategy,
            "reasoning_id": reasoning_result.get("metadata", {}).get("reasoning_time", ""),
            "context_id": context.get("id", "")
        }
        
        return decision_result
    
    def _utility_based_decision(self, reasoning_result: Dict[str, Any], context: Dict[str, Any]) -> Dict[str, Any]:
        """基于效用的决策

        """
        # 生成决策选项
        options = self._generate_options(reasoning_result, context)
        
        # 计算每个选项的效用

        for option in options:

                "option": option,
                "utility": utility
            })
        
        # 选择效用最大的选项
        best_option = max(option_utilities, key=lambda x: x.get("utility", 0))
        
        # 生成行动计划
        action_plan = self._generate_action_plan(best_option.get("option"), context)
        
        return {
            "decision_type": "utility_based",
            "options": options,
            "option_utilities": option_utilities,

            "utility": best_option.get("utility", 0),

        # 匹配决策规则
        matched_rules = []
        for rule in self.decision_rules.get("rules", []):
            if self._match_rule(rule, conditions):
                matched_rules.append(rule)
        
        # 选择优先级最高的规则
        if matched_rules:
            selected_rule = max(matched_rules, key=lambda x: x.get("priority", 0))
        else:
            # 如果没有匹配的规则，使用默认决策
            selected_rule = {
                "id": "default_rule",
                "name": "默认决策",
                "action": "default_action",
                "priority": 0
            }
        
        # 生成行动计划
        action_plan = self._generate_action_plan_from_rule(selected_rule, context)
        
        return {
            "decision_type": "rule_based",
            "conditions": conditions,
            "matched_rules": matched_rules,
            "selected_rule": selected_rule,
            "action_plan": action_plan,
            "confidence": self._calculate_rule_confidence(selected_rule, matched_rules)
        }
    
    def _case_based_decision(self, reasoning_result: Dict[str, Any], context: Dict[str, Any]) -> Dict[str, Any]:
        """基于案例的决策
        
        基于相似的历史案例进行决策
        """
        # 提取当前情境特征
        current_features = self._extract_decision_features(reasoning_result, context)
        
        # 检索相似案例
        similar_cases = self._retrieve_similar_decision_cases(current_features)
        
        # 基于案例生成决策选项
        options = self._generate_options_from_cases(similar_cases, context)
        
        # 评估选项
        evaluated_options = []
        for option in options:
            evaluation = self._evaluate_option(option, reasoning_result, context)
            evaluated_options.append(evaluation)
        
        # 选择最佳选项
        if evaluated_options:
            best_option = max(evaluated_options, key=lambda x: x.get("score", 0))
        else:
            # 如果没有选项，使用默认决策
            best_option = {
                "option": {"action": "default_action"},
                "score": 0.5
            }
        
        # 生成行动计划
        action_plan = self._generate_action_plan(best_option.get("option"), context)
        
        return {
            "decision_type": "case_based",
            "current_features": current_features,
            "similar_cases": similar_cases,
            "options": options,
            "evaluated_options": evaluated_options,
            "selected_option": best_option.get("option"),
            "score": best_option.get("score", 0),
            "action_plan": action_plan,
            "confidence": self._calculate_case_confidence(best_option.get("score", 0), evaluated_options)
        }
    
    def _risk_aware_decision(self, reasoning_result: Dict[str, Any], context: Dict[str, Any]) -> Dict[str, Any]:
        """风险感知决策
        
        考虑决策风险和不确定性进行决策
        """
        # 生成决策选项
        options = self._generate_options(reasoning_result, context)
        
        # 评估每个选项的风险
        option_risks = []
        for option in options:
            risk = self._assess_risk(option, reasoning_result, context)
            option_risks.append({
                "option": option,
                "risk": risk
            })
        
        # 计算风险调整后的效用
        risk_adjusted_utilities = []
        risk_tolerance = context.get("risk_tolerance", 0.5)  # 风险容忍度，0-1之间
        
        for option_risk in option_risks:
            option = option_risk.get("option")
            risk = option_risk.get("risk", 0)
            
            # 计算效用
            utility = self._calculate_utility(option, reasoning_result, context)
            
            # 风险调整
            risk_adjusted_utility = utility * (1 - risk * (1 - risk_tolerance))
            
            risk_adjusted_utilities.append({
                "option": option,
                "utility": utility,
                "risk": risk,
                "risk_adjusted_utility": risk_adjusted_utility
            })
        
        # 选择风险调整后效用最大的选项
        best_option = max(risk_adjusted_utilities, key=lambda x: x.get("risk_adjusted_utility", 0))
        
        # 生成行动计划
        action_plan = self._generate_action_plan(best_option.get("option"), context)
        
        return {
            "decision_type": "risk_aware",
            "options": options,
            "option_risks": option_risks,
            "risk_adjusted_utilities": risk_adjusted_utilities,
            "selected_option": best_option.get("option"),
            "utility": best_option.get("utility", 0),
            "risk": best_option.get("risk", 0),
            "risk_adjusted_utility": best_option.get("risk_adjusted_utility", 0),
            "action_plan": action_plan,
            "confidence": self._calculate_risk_confidence(best_option.get("risk_adjusted_utility", 0), risk_adjusted_utilities)
        }
    
    def _multi_criteria_decision(self, reasoning_result: Dict[str, Any], context: Dict[str, Any]) -> Dict[str, Any]:
        """多标准决策
        
        基于多个评估标准进行决策
        """
        # 生成决策选项
        options = self._generate_options(reasoning_result, context)
        
        # 获取决策标准
        criteria = self._get_decision_criteria(context)
        
        # 评估每个选项在各标准下的表现
        option_evaluations = []
        for option in options:
            evaluation = {}
            total_score = 0
            total_weight = 0
            
            for criterion in criteria:
                criterion_name = criterion.get("name", "")
                criterion_weight = criterion.get("weight", 1.0)
                
                # 评估选项在当前标准下的表现
                score = self._evaluate_option_criterion(option, criterion, reasoning_result, context)
                
                evaluation[criterion_name] = {
                    "score": score,
                    "weight": criterion_weight
                }
                
                total_score += score * criterion_weight
                total_weight += criterion_weight
            
            # 计算加权总分
            if total_weight > 0:
                weighted_score = total_score / total_weight
            else:
                weighted_score = 0
            
            option_evaluations.append({
                "option": option,
                "evaluation": evaluation,
                "weighted_score": weighted_score
            })
        
        # 选择加权总分最高的选项
        best_option = max(option_evaluations, key=lambda x: x.get("weighted_score", 0))
        
        # 生成行动计划
        action_plan = self._generate_action_plan(best_option.get("option"), context)
        
        return {
            "decision_type": "multi_criteria",
            "options": options,
            "criteria": criteria,
            "option_evaluations": option_evaluations,
            "selected_option": best_option.get("option"),
            "evaluation": best_option.get("evaluation", {}),
            "weighted_score": best_option.get("weighted_score", 0),
            "action_plan": action_plan,
            "confidence": self._calculate_multi_criteria_confidence(best_option.get("weighted_score", 0), option_evaluations)
        }

    def _calculate_utility(self, option: Dict[str, Any], reasoning_result: Dict[str, Any], 
                          context: Dict[str, Any]) -> float:
        """计算选项效用"""
        # 简化的效用计算
        # 实际实现中应该使用更复杂的效用函数
        
        option_type = option.get("type", "")
        
        # 基础效用
        base_utility = 0.5
        
        # 基于选项类型调整效用
        if option_type == "deductive_option":
            base_utility = 0.8
        elif option_type == "inductive_option":
            base_utility = 0.7
        elif option_type == "abductive_option":
            base_utility = 0.6
        elif option_type == "analogical_option":
            base_utility = 0.6
        elif option_type == "causal_option":
            base_utility = 0.7
        elif option_type == "default_option":
            base_utility = 0.4
        
        # 基于推理置信度调整效用
        reasoning_confidence = reasoning_result.get("confidence", 0)
        adjusted_utility = base_utility * (0.5 + 0.5 * reasoning_confidence)
        
        # 基于情境调整效用
        situation_type = context.get("type", "")
        if situation_type == "emergency":
            adjusted_utility *= 1.2
        elif situation_type == "routine":
            adjusted_utility *= 0.9
        
        # 确保效用在0-1之间
        return max(0, min(1, adjusted_utility))

    def _calculate_decision_confidence(self, best_utility: float, option_utilities: List[Dict[str, Any]]) -> float:
        """计算决策置信度"""
        if not option_utilities:
            return 0.0
        
        # 计算效用分布
        utilities = [u.get("utility", 0) for u in option_utilities]
        
        # 计算平均效用
        avg_utility = sum(utilities) / len(utilities)
        
        # 计算效用标准差
        variance = sum((u - avg_utility) ** 2 for u in utilities) / len(utilities)
        std_dev = variance ** 0.5
        
        # 计算置信度：基于最佳效用与平均效用的差异，以及标准差
        if std_dev > 0:
            confidence = min(1.0, (best_utility - avg_utility) / std_dev)
        else:
            confidence = best_utility
        
        # 确保置信度在0-1之间
        return max(0, min(1, confidence))

    def _calculate_decision_similarity(self, current_features: Dict[str, Any], case_features: Dict[str, Any]) -> float:
        """计算决策特征相似度"""
        similarity = 0.0
        feature_count = 0
        
        # 比较推理类型
        current_reasoning_type = current_features.get("reasoning_type", "")
        case_reasoning_type = case_features.get("reasoning_type", "")
        if current_reasoning_type and case_reasoning_type:
            if current_reasoning_type == case_reasoning_type:
                similarity += 1.0
            feature_count += 1
        
        # 比较推理置信度
        current_reasoning_confidence = current_features.get("reasoning_confidence", 0)
        case_reasoning_confidence = case_features.get("reasoning_confidence", 0)
        if current_reasoning_confidence > 0 and case_reasoning_confidence > 0:
            confidence_diff = abs(current_reasoning_confidence - case_reasoning_confidence)
            confidence_similarity = 1 - confidence_diff
            similarity += confidence_similarity
            feature_count += 1
        
        # 比较情境类型
        current_situation_type = current_features.get("situation_type", "")
        case_situation_type = case_features.get("situation_type", "")
        if current_situation_type and case_situation_type:
            if current_situation_type == case_situation_type:
                similarity += 1.0
            feature_count += 1
        
        # 比较紧急程度
        current_urgency = current_features.get("urgency", "")
        case_urgency = case_features.get("urgency", "")
        if current_urgency and case_urgency:
            if current_urgency == case_urgency:
                similarity += 1.0
            feature_count += 1
        
        # 比较风险等级
        current_risk_level = current_features.get("risk_level", "")
        case_risk_level = case_features.get("risk_level", "")
        if current_risk_level and case_risk_level:
            if current_risk_level == case_risk_level:
                similarity += 1.0
            feature_count += 1
        
        # 计算平均相似度
        if feature_count > 0:
            similarity /= feature_count
        
        return similarity
    
    def _generate_options_from_cases(self, similar_cases: List[Dict[str, Any]], context: Dict[str, Any]) -> List[Dict[str, Any]]:
        """基于案例生成决策选项"""
        options = []
        
        for case in similar_cases:
            case_decision = case.get("decision", {})
            case_action = case_decision.get("action", "")
            case_outcome = case_decision.get("outcome", "")
            case_similarity = case.get("similarity", 0)
            
            if case_action:
                option = {
                    "type": "case_based_option",
                    "source_case": case.get("id", ""),
                    "action": case_action,
                    "description": f"基于案例{case.get('id', '')}的行动",
                    "expected_outcome": case_outcome,
                    "similarity": case_similarity
                }
                options.append(option)
        
        # 如果没有生成选项，添加默认选项
        if not options:
            option = {
                "type": "default_option",
                "source_case": "default",
                "action": "default_action",
                "description": "基于默认规则的行动",
                "expected_outcome": "unknown",
                "similarity": 0.0
            }
            options.append(option)
        
        return options
    
    def _evaluate_option(self, option: Dict[str, Any], reasoning_result: Dict[str, Any], 
                        context: Dict[str, Any]) -> Dict[str, Any]:
        """评估选项"""
        option_type = option.get("type", "")
        
        # 基础评分
        base_score = 0.5
        
        # 基于选项类型调整评分
        if option_type == "case_based_option":
            similarity = option.get("similarity", 0)
            base_score = 0.3 + 0.7 * similarity  # 基于相似度调整评分
        
        # 基于推理置信度调整评分
        reasoning_confidence = reasoning_result.get("confidence", 0)
        adjusted_score = base_score * (0.5 + 0.5 * reasoning_confidence)
        
        # 基于情境调整评分
        situation_type = context.get("type", "")
        if situation_type == "emergency":
            adjusted_score *= 1.1
        elif situation_type == "routine":
            adjusted_score *= 0.95
        
        # 确保评分在0-1之间
        final_score = max(0, min(1, adjusted_score))
        
        return {
            "option": option,
            "score": final_score,
            "evaluation_factors": {
                "base_score": base_score,
                "reasoning_confidence": reasoning_confidence,
                "situation_adjustment": 1.1 if situation_type == "emergency" else 0.95 if situation_type == "routine" else 1.0
            }
        }
    
    def _calculate_case_confidence(self, best_score: float, evaluated_options: List[Dict[str, Any]]) -> float:
        """计算案例决策置信度"""
        if not evaluated_options:
            return 0.0
        
        # 计算评分分布
        scores = [opt.get("score", 0) for opt in evaluated_options]
        
        # 计算平均评分
        avg_score = sum(scores) / len(scores)
        
        # 计算评分标准差
        variance = sum((s - avg_score) ** 2 for s in scores) / len(scores)
        std_dev = variance ** 0.5
        
        # 计算置信度：基于最佳评分与平均评分的差异，以及标准差
        if std_dev > 0:
            confidence = min(1.0, (best_score - avg_score) / std_dev)
        else:
            confidence = best_score
        
        # 确保置信度在0-1之间
        return max(0, min(1, confidence))
    
    def _assess_risk(self, option: Dict[str, Any], reasoning_result: Dict[str, Any], 
                    context: Dict[str, Any]) -> float:
        """评估选项风险"""
        option_type = option.get("type", "")
        
        # 基础风险
        base_risk = 0.5
        
        # 基于选项类型调整风险
        if option_type == "deductive_option":
            base_risk = 0.2  # 演绎推理风险较低
        elif option_type == "inductive_option":
            base_risk = 0.4  # 归纳推理风险中等
        elif option_type == "abductive_option":
            base_risk = 0.6  # 溯因推理风险较高
        elif option_type == "analogical_option":
            base_risk = 0.5  # 类比推理风险中等
        elif option_type == "causal_option":
            base_risk = 0.3  # 因果推理风险较低
        elif option_type == "case_based_option":
            similarity = option.get("similarity", 0)
            base_risk = 0.7 * (1 - similarity)  # 基于相似度调整风险
        elif option_type == "default_option":
            base_risk = 0.6  # 默认选项风险较高
        
        # 基于推理置信度调整风险
        reasoning_confidence = reasoning_result.get("confidence", 0)
        adjusted_risk = base_risk * (1.5 - 0.5 * reasoning_confidence)  # 置信度越高，风险越低
        
        # 基于情境调整风险
        situation_type = context.get("type", "")
        if situation_type == "emergency":
            adjusted_risk *= 1.2  # 紧急情况风险增加
        elif situation_type == "routine":
            adjusted_risk *= 0.8  # 常规情况风险降低
        
        # 确保风险在0-1之间
        return max(0, min(1, adjusted_risk))
    
    def _calculate_risk_confidence(self, best_risk_adjusted_utility: float, 
                                  risk_adjusted_utilities: List[Dict[str, Any]]) -> float:
        """计算风险感知决策置信度"""
        if not risk_adjusted_utilities:
            return 0.0
        
        # 计算风险调整效用分布
        utilities = [u.get("risk_adjusted_utility", 0) for u in risk_adjusted_utilities]
        
        # 计算平均风险调整效用
        avg_utility = sum(utilities) / len(utilities)
        
        # 计算风险调整效用标准差
        variance = sum((u - avg_utility) ** 2 for u in utilities) / len(utilities)
        std_dev = variance ** 0.5
        
        # 计算置信度：基于最佳风险调整效用与平均效用的差异，以及标准差
        if std_dev > 0:
            confidence = min(1.0, (best_risk_adjusted_utility - avg_utility) / std_dev)
        else:
            confidence = best_risk_adjusted_utility
        
        # 确保置信度在0-1之间
        return max(0, min(1, confidence))
```

### 7. 性能优化

#### 7.1 模型优化

```python
# src/utils/model_optimizer.py

import torch
import torch.nn.utils.prune as prune
from torch.quantization import quantize_dynamic
import logging

logger = logging.getLogger(__name__)

class ModelOptimizer:
    """模型优化器"""
    
    @staticmethod
    def prune_model(model, pruning_rate=0.2):
        """剪枝模型
        
        Args:
            model: 要剪枝的模型
            pruning_rate: 剪枝率
            
        Returns:
            剪枝后的模型
        """
        logger.info(f"开始剪枝模型，剪枝率: {pruning_rate}")
        
        # 获取所有参数
        parameters_to_prune = []
        for name, module in model.named_modules():
            if isinstance(module, torch.nn.Linear):
                parameters_to_prune.append((module, 'weight'))
        
        # 应用剪枝
        prune.global_unstructured(
            parameters_to_prune,
            pruning_method=prune.L1Unstructured,
            amount=pruning_rate,
        )
        
        # 移除剪枝重参数化
        for module, _ in parameters_to_prune:
            prune.remove(module, 'weight')
        
        logger.info("模型剪枝完成")
        return model
    
    @staticmethod
    def quantize_model(model, quantization_type='dynamic'):
        """量化模型
        
        Args:
            model: 要量化的模型
            quantization_type: 量化类型，'dynamic'或'static'
            
        Returns:
            量化后的模型
        """
        logger.info(f"开始量化模型，量化类型: {quantization_type}")
        
        if quantization_type == 'dynamic':
```

```python
# src/utils/cache_manager.py

import time
import hashlib
import pickle
import logging
from typing import Any, Dict, Optional
from functools import wraps

logger = logging.getLogger(__name__)

class CacheManager:
    """缓存管理器"""
    
    def __init__(self, max_size: int = 1000, ttl: int = 3600):
        """初始化缓存管理器
        
        Args:
            max_size: 最大缓存条目数
            ttl: 缓存生存时间（秒）
        """
        self.cache: Dict[str, Dict[str, Any]] = {}
        self.max_size = max_size
        self.ttl = ttl
    
    def _generate_key(self, func_name: str, args: tuple, kwargs: dict) -> str:
        """生成缓存键
        
        Args:
            func_name: 函数名
            args: 位置参数
            kwargs: 关键字参数
            
        Returns:
            缓存键
        """
        # 创建键的字符串表示
        key_str = f"{func_name}:{str(args)}:{str(sorted(kwargs.items()))}"
        
        # 生成哈希
        key_hash = hashlib.md5(key_str.encode()).hexdigest()
        
        return key_hash
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存值
        
        Args:
            key: 缓存键
            
        Returns:
            缓存值或None
        """
        if key in self.cache:
            entry = self.cache[key]
            
            # 检查是否过期
            if time.time() - entry['timestamp'] < self.ttl:
                return entry['value']
            else:
                # 过期，删除
                del self.cache[key]
        
        return None
    
    def set(self, key: str, value: Any) -> None:
        """设置缓存值
        
        Args:
            key: 缓存键
            value: 缓存值
        """
        # 如果缓存已满，删除最旧的条目
        if len(self.cache) >= self.max_size:
            oldest_key = min(self.cache.keys(), key=lambda k: self.cache[k]['timestamp'])
            del self.cache[oldest_key]
        
        # 添加新条目
        self.cache[key] = {
            'value': value,
            'timestamp': time.time()
        }
    
    def clear(self) -> None:
        """清空缓存"""
        self.cache.clear()
        logger.info("缓存已清空")
    
    def cache_result(self, func):
        """缓存结果的装饰器
        
        Args:
            func: 要缓存的函数
            
        Returns:
            装饰后的函数
        """
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 生成缓存键
            key = self._generate_key(func.__name__, args, kwargs)
            
            # 尝试从缓存获取
            result = self.get(key)
            if result is not None:
                logger.debug(f"从缓存获取结果: {func.__name__}")
                return result
            
            # 执行函数
            result = func(*args, **kwargs)
            
            # 存储到缓存
            self.set(key, result)
            logger.debug(f"结果已缓存: {func.__name__}")
            
            return result
        
        return wrapper
```

```python
import asyncio
import logging
from typing import List, Callable, Any, Dict
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor
import multiprocessing as mp

logger = logging.getLogger(__name__)

class ConcurrentProcessor:
    """并发处理器"""
    
    def __init__(self, max_workers: int = None):
        """初始化并发处理器
        
        Args:
            max_workers: 最大工作线程/进程数
        """
        if max_workers is None:
            max_workers = mp.cpu_count()
        
        self.max_workers = max_workers
        self.thread_executor = ThreadPoolExecutor(max_workers=max_workers)
        self.process_executor = ProcessPoolExecutor(max_workers=max_workers)
    
    async def process_async(self, func: Callable, args_list: List[tuple]) -> List[Any]:
        """异步处理多个任务
        
        Args:
            func: 要执行的函数
            args_list: 参数列表
            
        Returns:
            结果列表
        """
        logger.info(f"开始异步处理 {len(args_list)} 个任务")
        
        # 创建任务
        loop = asyncio.get_event_loop()
        tasks = [
            loop.run_in_executor(self.thread_executor, func, *args)
            for args in args_list
        ]
        
        # 等待所有任务完成
        results = await asyncio.gather(*tasks)
        
        logger.info(f"异步处理完成，共处理 {len(results)} 个任务")
        return results
    
    def process_thread(self, func: Callable, args_list: List[tuple]) -> List[Any]:
        """使用线程池处理多个任务
        
        Args:
            func: 要执行的函数
            args_list: 参数列表
            
        Returns:
            结果列表
        """
        logger.info(f"开始线程池处理 {len(args_list)} 个任务")
        
        # 提交任务
        futures = [
            self.thread_executor.submit(func, *args)
            for args in args_list
        ]
        
        # 等待所有任务完成
        results = [future.result() for future in futures]
        
        logger.info(f"线程池处理完成，共处理 {len(results)} 个任务")
        return results
    
    def process_process(self, func: Callable, args_list: List[tuple]) -> List[Any]:
        """使用进程池处理多个任务
        
        Args:
            func: 要执行的函数
            args_list: 参数列表
            
        Returns:
            结果列表
        """
        logger.info(f"开始进程池处理 {len(args_list)} 个任务")
        
        # 提交任务
        futures = [
            self.process_executor.submit(func, *args)
            for args in args_list
        ]
        
        # 等待所有任务完成
        results = [future.result() for future in futures]
        
        logger.info(f"进程池处理完成，共处理 {len(results)} 个任务")
        return results
    
    def shutdown(self):
        """关闭执行器"""
        self.thread_executor.shutdown(wait=True)
        self.process_executor.shutdown(wait=True)
        logger.info("并发处理器已关闭")
```

```python
import time

import logging
from typing import Dict, Any, Optional
from functools import wraps
from collections import defaultdict, deque

logger = logging.getLogger(__name__)

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self, history_size: int = 1000):
        """初始化性能监控器
        
        Args:
            history_size: 历史记录大小
        """
        self.history_size = history_size
        self.metrics: Dict[str, deque] = defaultdict(lambda: deque(maxlen=history_size))
        self.start_times: Dict[str, float] = {}
    
    def start_timer(self, name: str) -> None:
        """开始计时
        
        Args:
            name: 计时器名称
        """
        self.start_times[name] = time.time()
    
    def end_timer(self, name: str) -> float:
        """结束计时
        
        Args:
            name: 计时器名称
            
        Returns:
            经过的时间（秒）
        """
        if name not in self.start_times:
            logger.warning(f"计时器 {name} 未启动")
            return 0.0
        
        elapsed = time.time() - self.start_times[name]
        self.metrics[f"{name}_time"].append(elapsed)
        del self.start_times[name]
        
        return elapsed
    
    def record_metric(self, name: str, value: Any) -> None:
        """记录指标
        
        Args:
            name: 指标名称
            value: 指标值
        """
        self.metrics[name].append(value)
    
    def get_metric_stats(self, name: str) -> Optional[Dict[str, float]]:
        """获取指标统计信息
        
        Args:
            name: 指标名称
            
        Returns:
            统计信息字典
        """
        if name not in self.metrics or len(self.metrics[name]) == 0:
            return None
        
        values = list(self.metrics[name])
        return {
            'count': len(values),
            'min': min(values),
            'max': max(values),
            'mean': sum(values) / len(values),
            'latest': values[-1]
        }
    
    def get_system_metrics(self) -> Dict[str, float]:
        """获取系统指标
        
        Returns:
            系统指标字典
        """
        return {
            'cpu_percent': psutil.cpu_percent(),
            'memory_percent': psutil.virtual_memory().percent,
            'disk_percent': psutil.disk_usage('/').percent
        }
    
    def monitor_function(self, func):
        """函数性能监控装饰器
        
        Args:
            func: 要监控的函数
            
        Returns:
            装饰后的函数
        """
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 开始计时
            start_time = time.time()
            
            # 记录系统指标
            start_metrics = self.get_system_metrics()
            
            # 执行函数
            try:
                result = func(*args, **kwargs)
                success = True
            except Exception as e:
                result = e
                success = False
            
            # 结束计时
            elapsed_time = time.time() - start_time
            
            # 记录指标
            self.record_metric(f"{func.__name__}_time", elapsed_time)
            self.record_metric(f"{func.__name__}_success", 1 if success else 0)
            
            # 记录系统指标变化
            end_metrics = self.get_system_metrics()
            for metric_name in start_metrics:
                delta = end_metrics[metric_name] - start_metrics[metric_name]
                self.record_metric(f"{func.__name__}_{metric_name}_delta", delta)
            
            # 记录日志
            logger.info(
                f"函数 {func.__name__} 执行完成，"
                f"耗时: {elapsed_time:.4f}s，"
                f"成功: {success}"
            )
            
            # 如果函数执行失败，重新抛出异常
            if not success:
                raise result
            
            return result
        
        return wrapper
    
    def monitor_async_function(self, func):
        """异步函数性能监控装饰器
        
        Args:
            func: 要监控的异步函数
            
        Returns:
            装饰后的异步函数
        """

        async def wrapper(*args, **kwargs):
            # 开始计时

            start_metrics = self.get_system_metrics()
            
            # 执行函数
            try:
                result = await func(*args, **kwargs)

                result = e
                success = False

            self.record_metric(f"{func.__name__}_time", elapsed_time)
            self.record_metric(f"{func.__name__}_success", 1 if success else 0)
            
            # 记录系统指标变化
            end_metrics = self.get_system_metrics()
            for metric_name in start_metrics:
                delta = end_metrics[metric_name] - start_metrics[metric_name]
```
