# 决策执行模块实现工作流

## 模块信息

- **模块名称**：决策执行模块
- **所属子系统**：认知决策子系统
- **模块功能**：负责将决策制定模块生成的决策指令转换为具体的执行动作，并协调各子系统完成执行任务
- **模块职责**：
  - 解析和验证决策指令
  - 将决策指令分解为可执行任务
  - 调度和协调各子系统执行任务
  - 监控任务执行状态和进度
  - 处理执行过程中的异常情况
  - 收集和整合执行结果
  - 将执行结果反馈给决策系统
- **模块接口**：
  - 输入接口：接收决策制定模块的决策指令
  - 输出接口：向各子系统发送执行指令，向决策系统反馈执行结果
  - 监控接口：提供执行状态和进度的查询接口

#### 指令解析组件 (InstructionParser)

```python
# src/core/instruction_parser.py

from typing import Dict, List, Optional, Any
from pydantic import BaseModel, Field, validator
from langchain.schema import BasePromptTemplate
from langchain.chains import LLMChain
from langchain.llms import OpenAI

from ..models.instruction import Instruction, InstructionType, InstructionStatus
from ..config.settings import get_settings

settings = get_settings()


class InstructionParser:
    """指令解析组件，负责解析和验证决策指令"""
    
    def __init__(self):
        self.llm = OpenAI(
            temperature=0.1,
            openai_api_key=settings.OPENAI_API_KEY
        )
        
        # 指令解析提示模板
        self.instruction_template = BasePromptTemplate(
            input_variables=["instruction"],
            template="""
            你是一个决策指令解析专家，负责解析和验证来自决策系统的指令。
            
            请分析以下指令，并提取关键信息：
            
            指令内容: {instruction}
            
            请以JSON格式返回以下信息：
            1. instruction_type: 指令类型 (action/decision/query/feedback)
            2. priority: 优先级 (high/medium/low)
            3. target_subsystems: 目标子系统列表
            4. required_resources: 所需资源列表
            5. estimated_duration: 预估执行时间(秒)
            6. dependencies: 依赖的其他指令ID列表
            7. validation_score: 指令有效性评分(0-1)
            8. confidence: 解析置信度(0-1)
            """
        )
        
        self.chain = LLMChain(llm=self.llm, prompt=self.instruction_template)
    
    async def parse_instruction(self, instruction_text: str) -> Instruction:
        """
        解析指令文本，生成结构化指令对象
        
        Args:
            instruction_text: 原始指令文本
            
        Returns:
            Instruction: 结构化指令对象
        """
        try:
            # 使用LLM解析指令
            result = await self.chain.arun(instruction=instruction_text)
            
            # 解析JSON结果
            import json
            parsed_data = json.loads(result)
            
            # 创建指令对象
            instruction = Instruction(
                id=self._generate_instruction_id(),
                original_text=instruction_text,
                type=InstructionType(parsed_data.get("instruction_type", "action")),
                priority=parsed_data.get("priority", "medium"),
                target_subsystems=parsed_data.get("target_subsystems", []),
                required_resources=parsed_data.get("required_resources", []),
                estimated_duration=parsed_data.get("estimated_duration", 60),
                dependencies=parsed_data.get("dependencies", []),
                validation_score=parsed_data.get("validation_score", 0.8),
                confidence=parsed_data.get("confidence", 0.8),
                status=InstructionStatus.PARSED
            )
            
            return instruction
            
        except Exception as e:
            # 解析失败，创建错误指令
            return Instruction(
                id=self._generate_instruction_id(),
                original_text=instruction_text,
                type=InstructionType.ERROR,
                priority="low",
                target_subsystems=[],
                required_resources=[],
                estimated_duration=0,
                dependencies=[],
                validation_score=0.0,
                confidence=0.0,
                status=InstructionStatus.ERROR,
                error_message=str(e)
            )
    
    async def validate_instruction(self, instruction: Instruction) -> bool:
        """
        验证指令的有效性
        
        Args:
            instruction: 待验证的指令对象
            
        Returns:
            bool: 指令是否有效
        """
        # 基本验证
        if not instruction.original_text or not instruction.type:
            return False
            
        # 检查目标子系统是否有效
        valid_subsystems = ["perception", "memory", "interaction", "self_awareness"]
        for subsystem in instruction.target_subsystems:
            if subsystem not in valid_subsystems:
                return False
                
        # 检查依赖指令是否存在
        # 这里应该查询数据库或缓存来验证依赖指令
        # 简化实现，假设所有依赖都有效
        
        return True
    
    def _generate_instruction_id(self) -> str:
        """生成唯一的指令ID"""
        import uuid
        return str(uuid.uuid4())
```

#### 任务分解组件 (TaskDecomposer)

```python
# src/core/task_decomposer.py

from typing import List, Dict, Optional, Any
from pydantic import BaseModel, Field
from langchain.schema import BasePromptTemplate
from langchain.chains import LLMChain
from langchain.llms import OpenAI

from ..models.instruction import Instruction
from ..models.task import Task, TaskStatus, TaskType
from ..config.settings import get_settings

settings = get_settings()


class TaskDecomposer:
    """任务分解组件，负责将指令分解为可执行的任务"""
    
    def __init__(self):
        self.llm = OpenAI(
            temperature=0.1,
            openai_api_key=settings.OPENAI_API_KEY
        )
        
        # 任务分解提示模板
        self.decomposition_template = BasePromptTemplate(
            input_variables=["instruction", "context"],
            template="""
            你是一个任务分解专家，负责将决策指令分解为可执行的任务序列。
            
            请分析以下指令和上下文，并生成任务分解方案：
            
            指令内容: {instruction}
            上下文信息: {context}
            
            请以JSON格式返回以下信息：
            1. tasks: 任务列表，每个任务包含:
               - name: 任务名称
               - type: 任务类型 (data_processing/analysis/action/communication)
               - description: 任务描述
               - target_subsystem: 目标子系统
               - required_resources: 所需资源
               - estimated_duration: 预估执行时间(秒)
               - dependencies: 依赖的其他任务ID
               - parameters: 任务参数
            2. execution_order: 任务执行顺序
            3. parallel_groups: 可并行执行的任务组
            4. critical_path: 关键路径上的任务ID列表
            """
        )
        
        self.chain = LLMChain(llm=self.llm, prompt=self.decomposition_template)
    
    async def decompose_instruction(self, instruction: Instruction, context: Dict[str, Any] = None) -> List[Task]:
        """
        将指令分解为可执行的任务列表
        
        Args:
            instruction: 待分解的指令
            context: 执行上下文
            
        Returns:
            List[Task]: 分解后的任务列表
        """
        try:
            # 准备上下文信息
            context_str = self._format_context(context) if context else "无额外上下文信息"
            
            # 使用LLM分解指令
            result = await self.chain.arun(
                instruction=instruction.original_text,
                context=context_str
            )
            
            # 解析JSON结果
            import json
            parsed_data = json.loads(result)
            
            # 创建任务对象
            tasks = []
            task_id_map = {}  # 用于建立任务ID映射
            
            # 第一遍：创建所有任务
            for i, task_data in enumerate(parsed_data.get("tasks", [])):
                task_id = self._generate_task_id()
                task_id_map[i] = task_id
                
                task = Task(
                    id=task_id,
                    instruction_id=instruction.id,
                    name=task_data.get("name", f"任务{i+1}"),
                    type=TaskType(task_data.get("type", "action")),
                    description=task_data.get("description", ""),
                    target_subsystem=task_data.get("target_subsystem", ""),
                    required_resources=task_data.get("required_resources", []),
                    estimated_duration=task_data.get("estimated_duration", 60),
                    dependencies=[],  # 先设为空，第二遍再填充
                    parameters=task_data.get("parameters", {}),
                    status=TaskStatus.CREATED
                )
                tasks.append(task)
            
            # 第二遍：填充任务依赖关系
            for i, task_data in enumerate(parsed_data.get("tasks", [])):
                task_id = task_id_map[i]
                dependency_indices = task_data.get("dependencies", [])
                
                # 将索引转换为实际的任务ID
                dependencies = [task_id_map[dep_idx] for dep_idx in dependency_indices if dep_idx in task_id_map]
                
                # 更新任务的依赖关系
                for task in tasks:
                    if task.id == task_id:
                        task.dependencies = dependencies
                        break
            
            return tasks
            
        except Exception as e:
            # 分解失败，创建一个错误任务
            error_task = Task(
                id=self._generate_task_id(),
                instruction_id=instruction.id,
                name="指令分解失败",
                type=TaskType.ERROR,
                description=f"指令分解过程中发生错误: {str(e)}",
                target_subsystem="",
                required_resources=[],
                estimated_duration=0,
                dependencies=[],
                parameters={},
                status=TaskStatus.ERROR,
                error_message=str(e)
            )
            return [error_task]
    
    def _format_context(self, context: Dict[str, Any]) -> str:
        """格式化上下文信息为字符串"""
        if not context:
            return "无上下文信息"
            
        context_parts = []
        for key, value in context.items():
            context_parts.append(f"{key}: {value}")
            
        return "\n".join(context_parts)
    
    def _generate_task_id(self) -> str:
        """生成唯一的任务ID"""
        import uuid
        return str(uuid.uuid4())
```

#### 任务调度组件 (TaskScheduler)

```python
# src/core/task_scheduler.py

from typing import List, Dict, Optional, Any, Set
from datetime import datetime, timedelta
import asyncio
from enum import Enum

from ..models.task import Task, TaskStatus
from ..models.execution import Execution
from ..config.settings import get_settings

settings = get_settings()


class SchedulingStrategy(Enum):
    """任务调度策略"""
    FIFO = "fifo"  # 先进先出
    PRIORITY = "priority"  # 优先级调度
    SHORTEST_JOB_FIRST = "sjf"  # 最短作业优先
    DEPENDENCY_AWARE = "dependency_aware"  # 依赖感知调度


class TaskScheduler:
    """任务调度组件，负责任务的调度和执行协调"""
    
    def __init__(self, max_concurrent_tasks: int = None):
        self.max_concurrent_tasks = max_concurrent_tasks or settings.MAX_CONCURRENT_TASKS
        self.scheduling_strategy = SchedulingStrategy.DEPENDENCY_AWARE
        self.task_queue = asyncio.Queue()
        self.running_tasks = {}  # task_id -> Execution
        self.completed_tasks = {}  # task_id -> Execution
        self.task_dependencies = {}  # task_id -> Set[task_id]
        self.dependents = {}  # task_id -> Set[task_id]
        self.execution_semaphore = asyncio.Semaphore(self.max_concurrent_tasks)
        
    async def schedule_tasks(self, tasks: List[Task]) -> List[Execution]:
        """
        调度任务列表，创建执行对象
        
        Args:
            tasks: 待调度的任务列表
            
        Returns:
            List[Execution]: 创建的执行对象列表
        """
        # 构建任务依赖图
        self._build_dependency_graph(tasks)
        
        # 根据调度策略排序任务
        sorted_tasks = self._sort_tasks(tasks)
        
        # 创建执行对象
        executions = []
        for task in sorted_tasks:
            execution = Execution(
                id=self._generate_execution_id(),
                task_id=task.id,
                status=TaskStatus.PENDING,
                start_time=None,
                end_time=None,
                duration=None,
                result=None,
                error=None
            )
            executions.append(execution)
            
            # 将任务加入队列
            await self.task_queue.put((task, execution))
        
        return executions
    
    async def start_execution(self) -> None:
        """启动任务执行循环"""
        while True:
            try:
                # 获取待执行任务
                task, execution = await self.task_queue.get()
                
                # 检查任务依赖是否满足
                if not self._are_dependencies_satisfied(task):
                    # 依赖未满足，重新放回队列
                    await self.task_queue.put((task, execution))
                    await asyncio.sleep(1)  # 等待一段时间再检查
                    continue
                
                # 使用信号量控制并发执行
                async with self.execution_semaphore:
                    # 更新执行状态
                    execution.status = TaskStatus.RUNNING
                    execution.start_time = datetime.now()
                    self.running_tasks[task.id] = execution
                    
                    # 启动任务执行
                    asyncio.create_task(self._execute_task(task, execution))
                    
            except asyncio.CancelledError:
                break
            except Exception as e:
                print(f"任务调度错误: {str(e)}")
    
    async def _execute_task(self, task: Task, execution: Execution) -> None:
        """执行单个任务"""
        try:
            # 这里应该调用具体的执行器来执行任务
            # 简化实现，模拟任务执行
            await asyncio.sleep(task.estimated_duration / 1000)  # 转换为秒
            
            # 更新执行状态
            execution.status = TaskStatus.COMPLETED
            execution.end_time = datetime.now()
            execution.duration = (execution.end_time - execution.start_time).total_seconds()
            execution.result = {"status": "success", "message": "任务执行成功"}
            
            # 从运行中任务移到已完成任务
            self.running_tasks.pop(task.id, None)
            self.completed_tasks[task.id] = execution
            
        except Exception as e:
            # 任务执行失败
            execution.status = TaskStatus.FAILED
            execution.end_time = datetime.now()
            execution.duration = (execution.end_time - execution.start_time).total_seconds()
            execution.error = str(e)
            
            # 从运行中任务移到已完成任务
            self.running_tasks.pop(task.id, None)
            self.completed_tasks[task.id] = execution
    
    def _build_dependency_graph(self, tasks: List[Task]) -> None:
        """构建任务依赖图"""
        # 清空现有依赖图
        self.task_dependencies.clear()
        self.dependents.clear()
        
        # 构建依赖关系
        for task in tasks:
            self.task_dependencies[task.id] = set(task.dependencies)
            
            # 构建反向依赖关系
            for dep_id in task.dependencies:
                if dep_id not in self.dependents:
                    self.dependents[dep_id] = set()
                self.dependents[dep_id].add(task.id)
    
    def _sort_tasks(self, tasks: List[Task]) -> List[Task]:
        """根据调度策略对任务排序"""
        if self.scheduling_strategy == SchedulingStrategy.FIFO:
            return tasks  # 保持原顺序
        elif self.scheduling_strategy == SchedulingStrategy.PRIORITY:
            # 按优先级排序
            priority_order = {"high": 0, "medium": 1, "low": 2}
            return sorted(tasks, key=lambda t: priority_order.get(t.priority, 2))
        elif self.scheduling_strategy == SchedulingStrategy.SHORTEST_JOB_FIRST:
            # 按预估执行时间排序
            return sorted(tasks, key=lambda t: t.estimated_duration)
        elif self.scheduling_strategy == SchedulingStrategy.DEPENDENCY_AWARE:
            # 拓扑排序，考虑依赖关系
            return self._topological_sort(tasks)
        else:
            return tasks
    
    def _topological_sort(self, tasks: List[Task]) -> List[Task]:
        """拓扑排序，确保依赖关系正确的任务顺序"""
        # 创建任务ID到任务的映射
        task_map = {task.id: task for task in tasks}
        
        # 计算每个任务的入度
        in_degree = {task.id: len(task.dependencies) for task in tasks}
        
        # 初始化队列，包含所有入度为0的任务
        queue = [task_id for task_id, degree in in_degree.items() if degree == 0]
        result = []
        
        while queue:
            # 取出一个入度为0的任务
            current_id = queue.pop(0)
            result.append(task_map[current_id])
            
            # 更新依赖当前任务的其他任务的入度
            if current_id in self.dependents:
                for dependent_id in self.dependents[current_id]:
                    in_degree[dependent_id] -= 1
                    if in_degree[dependent_id] == 0:
                        queue.append(dependent_id)
        
        return result
    
    def _are_dependencies_satisfied(self, task: Task) -> bool:
        """检查任务依赖是否满足"""
        for dep_id in task.dependencies:
            if dep_id not in self.completed_tasks:
                return False
        return True
    
    def _generate_execution_id(self) -> str:
        """生成唯一的执行ID"""
        import uuid
        return str(uuid.uuid4())
    
    async def get_task_status(self, task_id: str) -> Optional[Execution]:
        """获取任务执行状态"""
        if task_id in self.running_tasks:
            return self.running_tasks[task_id]
        elif task_id in self.completed_tasks:
            return self.completed_tasks[task_id]
        else:
            return None
    
    async def cancel_task(self, task_id: str) -> bool:
        """取消任务执行"""
        if task_id in self.running_tasks:
            # 这里应该实现任务取消逻辑
            # 简化实现，标记为已取消
            execution = self.running_tasks[task_id]
            execution.status = TaskStatus.CANCELLED
            execution.end_time = datetime.now()
            
            # 从运行中任务移到已完成任务
            self.running_tasks.pop(task_id, None)
            self.completed_tasks[task_id] = execution
            return True
        return False
```

```python
from typing import Dict, List, Optional, Any
from datetime import datetime, timedelta
import asyncio
from enum import Enum

from ..models.task import Task, TaskStatus
from ..models.execution import Execution

settings = get_settings()


class AlertLevel(Enum):
    """告警级别"""
    INFO = "info"
    WARNING = "warning"
    ERROR = "error"
    CRITICAL = "critical"


class StatusMonitor:
    """状态监控组件，负责监控任务执行状态和系统健康状态"""
    
    def __init__(self):
        self.running_tasks = {}  # task_id -> Execution
        self.completed_tasks = {}  # task_id -> Execution
        self.alerts = []  # 告警列表
        self.metrics = {}  # 监控指标
        self.monitoring_active = False
        self.monitoring_task = None
    
    async def start_monitoring(self):
        """启动状态监控"""
        if self.monitoring_active:
            return
            
        self.monitoring_active = True
        self.monitoring_task = asyncio.create_task(self._monitoring_loop())
    
    async def stop_monitoring(self):
        """停止状态监控"""
        self.monitoring_active = False
        if self.monitoring_task:
            self.monitoring_task.cancel()
            try:
                await self.monitoring_task
            except asyncio.CancelledError:
                pass
    
    async def _monitoring_loop(self):
        """监控循环"""
        while self.monitoring_active:
            try:
                # 检查任务执行状态
                await self._check_task_status()
                
                # 检查系统健康状态
                await self._check_system_health()
                
                # 更新监控指标
                await self._update_metrics()
                
                # 等待下一次检查
                await asyncio.sleep(settings.MONITORING_INTERVAL)
                
            except asyncio.CancelledError:
                break
            except Exception as e:
                await self._add_alert(
                    level=AlertLevel.ERROR,
                    message=f"监控循环错误: {str(e)}",
                    source="status_monitor"
                )
    
    async def _check_task_status(self):
        """检查任务执行状态"""
        current_time = datetime.now()
        
        # 检查运行中的任务是否超时
        for task_id, execution in self.running_tasks.items():
            if execution.start_time:
                elapsed = (current_time - execution.start_time).total_seconds()
                task_timeout = settings.TASK_TIMEOUT
                
                if elapsed > task_timeout:
                    # 任务超时
                    await self._add_alert(
                        level=AlertLevel.WARNING,
                        message=f"任务 {task_id} 执行超时 ({elapsed:.1f}s > {task_timeout}s)",
                        source="task_monitor"
                    )
    
    async def _check_system_health(self):
        """检查系统健康状态"""
        # 检查内存使用率
        import psutil
        memory = psutil.virtual_memory()
        if memory.percent > 90:
            await self._add_alert(
                level=AlertLevel.WARNING,
                message=f"系统内存使用率过高: {memory.percent:.1f}%",
                source="system_health"
            )
        
        # 检查CPU使用率
        cpu_percent = psutil.cpu_percent(interval=1)
        if cpu_percent > 90:
            await self._add_alert(
                level=AlertLevel.WARNING,
                message=f"系统CPU使用率过高: {cpu_percent:.1f}%",
                source="system_health"
            )
        
        # 检查磁盘使用率
        disk = psutil.disk_usage('/')
        disk_percent = disk.used / disk.total * 100
        if disk_percent > 90:
            await self._add_alert(
                level=AlertLevel.ERROR,
                message=f"磁盘使用率过高: {disk_percent:.1f}%",
                source="system_health"
            )
    
    async def _update_metrics(self):
        """更新监控指标"""
        current_time = datetime.now()
        
        # 计算任务执行统计
        total_tasks = len(self.running_tasks) + len(self.completed_tasks)
        completed_tasks = len(self.completed_tasks)
        running_tasks = len(self.running_tasks)
        
        # 计算平均执行时间
        if completed_tasks > 0:
            avg_duration = sum(
                execution.duration for execution in self.completed_tasks.values()
                if execution.duration
            ) / completed_tasks
        else:
            avg_duration = 0
        
        # 计算成功率
        success_tasks = sum(
            1 for execution in self.completed_tasks.values()
            if execution.status == TaskStatus.COMPLETED
        )
        success_rate = success_tasks / completed_tasks if completed_tasks > 0 else 0
        
        # 更新指标
        self.metrics = {
            "timestamp": current_time.isoformat(),
            "total_tasks": total_tasks,
            "completed_tasks": completed_tasks,
            "running_tasks": running_tasks,
            "avg_duration": avg_duration,
            "success_rate": success_rate,
            "alerts_count": len(self.alerts)
        }
    
    async def _add_alert(self, level: AlertLevel, message: str, source: str):
        """添加告警"""
        alert = {
            "id": self._generate_alert_id(),
            "level": level.value,
            "message": message,
            "source": source,
            "timestamp": datetime.now().isoformat()
        }
        self.alerts.append(alert)
        
        # 限制告警数量，保留最近的1000条
        if len(self.alerts) > 1000:
            self.alerts = self.alerts[-1000:]
    
    def _generate_alert_id(self) -> str:
        """生成唯一的告警ID"""
        import uuid
        return str(uuid.uuid4())
    
    def register_task(self, task_id: str, execution: Execution):
        """注册任务到监控"""
        self.running_tasks[task_id] = execution
    
    def unregister_task(self, task_id: str):
        """从监控中注销任务"""
        if task_id in self.running_tasks:
            execution = self.running_tasks.pop(task_id)
            self.completed_tasks[task_id] = execution
    
    async def get_task_status(self, task_id: str) -> Optional[Execution]:
        """获取任务状态"""
        if task_id in self.running_tasks:
            return self.running_tasks[task_id]
        elif task_id in self.completed_tasks:
            return self.completed_tasks[task_id]
        else:
            return None
    
    async def get_metrics(self) -> Dict[str, Any]:
        """获取监控指标"""
        return self.metrics.copy()
    
    async def get_alerts(self, level: AlertLevel = None, limit: int = 100) -> List[Dict[str, Any]]:
        """获取告警列表"""
        alerts = self.alerts
        
        # 按级别过滤
        if level:
            alerts = [alert for alert in alerts if alert["level"] == level.value]
        
        # 按时间倒序排列，并限制数量
        alerts = sorted(alerts, key=lambda a: a["timestamp"], reverse=True)[:limit]
        
        return alerts
    
    async def clear_alerts(self, before: datetime = None):
        """清除告警"""
        if before:
            self.alerts = [
                alert for alert in self.alerts
                if datetime.fromisoformat(alert["timestamp"]) >= before
            ]
        else:
            self.alerts.clear()
```

```python
        }

        self.retry_policies[ExceptionType.UNKNOWN_ERROR] = {

            "retry_delay": 10,  # seconds
            "backoff_factor": 1.0
        }
    
    async def handle_exception(self, task: Task, execution: Execution, exception: Exception) -> Dict[str, Any]:
        """
        处理异常

            execution: 任务执行对象
            exception: 异常对象
            
        Returns:
            Dict[str, Any]: 处理结果
        """
        # 确定异常类型和严重程度
        exception_type = self._classify_exception(exception)
        severity = self._determine_severity(exception_type, task)
        
        # 记录异常
        await self._log_exception(task, execution, exception, exception_type, severity)
        
        # 获取异常处理器
        handler = self.exception_handlers.get(exception_type, self._handle_unknown_error)
        
        # 处理异常
        result = await handler(task, execution, exception)
        
        return result
    
    def _classify_exception(self, exception: Exception) -> ExceptionType:
        """分类异常类型"""
        exception_name = type(exception).__name__.lower()
        exception_message = str(exception).lower()
        
        if "connection" in exception_name or "network" in exception_message:
            return ExceptionType.NETWORK_ERROR
        elif "timeout" in exception_name or "timeout" in exception_message:
            return ExceptionType.TIMEOUT_ERROR
        elif "memory" in exception_message or "resource" in exception_message:
            return ExceptionType.RESOURCE_ERROR
        elif "validation" in exception_name or "value" in exception_name:
            return ExceptionType.VALIDATION_ERROR
        elif "system" in exception_message or "os" in exception_name:
            return ExceptionType.SYSTEM_ERROR
        else:
            return ExceptionType.UNKNOWN_ERROR
    
    def _determine_severity(self, exception_type: ExceptionType, task: Task) -> ExceptionSeverity:
        """确定异常严重程度"""
        # 基于异常类型和任务优先级确定严重程度
        if exception_type == ExceptionType.SYSTEM_ERROR:
            return ExceptionSeverity.CRITICAL
        elif exception_type in [ExceptionType.NETWORK_ERROR, ExceptionType.RESOURCE_ERROR]:
            return ExceptionSeverity.HIGH
        elif exception_type == ExceptionType.TIMEOUT_ERROR:
            return ExceptionSeverity.MEDIUM
        elif exception_type == ExceptionType.VALIDATION_ERROR:

        else:
            return ExceptionSeverity.MEDIUM

    async def _handle_validation_error(self, task: Task, execution: Execution, exception: Exception) -> Dict[str, Any]:
        """处理验证错误"""
        # 验证错误不重试，直接标记为失败
        execution.status = TaskStatus.FAILED
        execution.error = str(exception)
        
        return {
            "action": "fail",
            "message": f"验证错误: {str(exception)}"
        }
    
    async def _handle_system_error(self, task: Task, execution: Execution, exception: Exception) -> Dict[str, Any]:
        """处理系统错误"""
        retry_policy = self.retry_policies.get(ExceptionType.SYSTEM_ERROR)
        if execution.retry_count < retry_policy["max_retries"]:
            # 计算重试延迟
            delay = retry_policy["retry_delay"] * (retry_policy["backoff_factor"] ** execution.retry_count)
            
            # 更新执行状态
            execution.retry_count += 1
            execution.status = TaskStatus.PENDING
            
            return {
                "action": "retry",
                "delay": delay,
                "message": f"系统错误，将在 {delay} 秒后重试 (第 {execution.retry_count} 次)"
            }
        else:
            # 超过最大重试次数，标记为失败
            execution.status = TaskStatus.FAILED
            execution.error = str(exception)
            
            return {
                "action": "fail",
                "message": f"系统错误，已达到最大重试次数 {retry_policy['max_retries']}"
            }
    
    async def _handle_unknown_error(self, task: Task, execution: Execution, exception: Exception) -> Dict[str, Any]:
        """处理未知错误"""
        retry_policy = self.retry_policies.get(ExceptionType.UNKNOWN_ERROR)
        if execution.retry_count < retry_policy["max_retries"]:
            # 计算重试延迟
            delay = retry_policy["retry_delay"] * (retry_policy["backoff_factor"] ** execution.retry_count)
            
            # 更新执行状态
            execution.retry_count += 1
            execution.status = TaskStatus.PENDING
            
            return {
                "action": "retry",
                "delay": delay,
                "message": f"未知错误，将在 {delay} 秒后重试 (第 {execution.retry_count} 次)"
            }
        else:
            # 超过最大重试次数，标记为失败

            return {
                "action": "fail",
                "message": f"未知错误，已达到最大重试次数 {retry_policy['max_retries']}"
            }
```

```python
class ResultCollector:

    async def collect_result(self, task: Task, execution: Execution) -> Dict[str, Any]:
        """
        收集任务执行结果
        
        Args:
            task: 已完成的任务
            execution: 任务执行对象
            
        Returns:
            Dict[str, Any]: 收集的结果
        """
        # 创建结果对象
        result = {
            "task_id": task.id,
            "task_type": task.type.value,
            "task_name": task.name,
            "task_description": task.description,
            "execution_id": execution.id,
            "status": execution.status.value,
            "start_time": execution.start_time.isoformat() if execution.start_time else None,
            "end_time": execution.end_time.isoformat() if execution.end_time else None,
            "duration": execution.duration,
            "retry_count": execution.retry_count,
            "result": execution.result,
            "error": execution.error,
            "parent_task_id": task.parent_task_id,
            "subtask_ids": task.subtask_ids,
            "dependencies": task.dependencies,
            "priority": task.priority.value,
            "target_subsystem": task.target_subsystem,
            "parameters": task.parameters,
            "metadata": task.metadata,
            "timestamp": datetime.now().isoformat()
        }
        
        # 存储结果
        self.results[task.id] = result
        
        # 如果有父任务，更新聚合结果
        if task.parent_task_id:
            await self._update_aggregated_result(task.parent_task_id)
        
        return result
    
    async def _update_aggregated_result(self, parent_task_id: str):
        """更新聚合结果"""
        # 获取父任务的所有子任务结果
        subtask_results = [
            result for result in self.results.values()
            if result["parent_task_id"] == parent_task_id
        ]
        
        if not subtask_results:
            return
        
        # 计算聚合指标
        total_tasks = len(subtask_results)
        completed_tasks = sum(
            1 for result in subtask_results
            if result["status"] == TaskStatus.COMPLETED.value
        )
        failed_tasks = sum(
            1 for result in subtask_results
            if result["status"] == TaskStatus.FAILED.value
        )
        cancelled_tasks = sum(
            1 for result in subtask_results
            if result["status"] == TaskStatus.CANCELLED.value
        )
        running_tasks = sum(
            1 for result in subtask_results
            if result["status"] == TaskStatus.RUNNING.value
        )
        pending_tasks = sum(
            1 for result in subtask_results
            if result["status"] == TaskStatus.PENDING.value
        )
        
        # 计算总执行时间
        start_times = [
            datetime.fromisoformat(result["start_time"])
            for result in subtask_results
            if result["start_time"]
        ]
        end_times = [
            datetime.fromisoformat(result["end_time"])
            for result in subtask_results
            if result["end_time"]
        ]
        
        total_duration = 0
        if start_times and end_times:
            min_start_time = min(start_times)
            max_end_time = max(end_times)
            total_duration = (max_end_time - min_start_time).total_seconds()
        
        # 计算平均执行时间
        avg_duration = sum(
            result["duration"] for result in subtask_results
            if result["duration"]
        ) / total_tasks if total_tasks > 0 else 0
        
        # 计算成功率
        success_rate = completed_tasks / total_tasks if total_tasks > 0 else 0
        
        # 聚合结果
        aggregated_result = {
            "parent_task_id": parent_task_id,
            "total_tasks": total_tasks,
            "completed_tasks": completed_tasks,
            "failed_tasks": failed_tasks,
            "cancelled_tasks": cancelled_tasks,
            "running_tasks": running_tasks,
            "pending_tasks": pending_tasks,
            "success_rate": success_rate,
            "total_duration": total_duration,
            "avg_duration": avg_duration,
            "subtask_results": subtask_results,
            "timestamp": datetime.now().isoformat()
        }
        
        # 存储聚合结果
        self.aggregated_results[parent_task_id] = aggregated_result
    
    async def get_result(self, task_id: str) -> Optional[Dict[str, Any]]:
        """获取单个任务结果"""
        return self.results.get(task_id)
    
    async def get_aggregated_result(self, parent_task_id: str) -> Optional[Dict[str, Any]]:
        """获取聚合结果"""
        return self.aggregated_results.get(parent_task_id)
    
    async def get_results_by_status(self, status: TaskStatus) -> List[Dict[str, Any]]:
        """根据状态获取结果列表"""
        return [
            result for result in self.results.values()
            if result["status"] == status.value
        ]
    
    async def get_results_by_type(self, task_type: str) -> List[Dict[str, Any]]:
        """根据任务类型获取结果列表"""
        return [
            result for result in self.results.values()
            if result["task_type"] == task_type
        ]
    
    async def get_results_by_subsystem(self, subsystem: str) -> List[Dict[str, Any]]:
        """根据目标子系统获取结果列表"""
        return [
            result for result in self.results.values()
            if result["target_subsystem"] == subsystem
        ]
    
    async def get_results_by_time_range(self, start_time: datetime, end_time: datetime) -> List[Dict[str, Any]]:
        """根据时间范围获取结果列表"""
        return [
            result for result in self.results.values()
            if result["end_time"] and
               start_time <= datetime.fromisoformat(result["end_time"]) <= end_time
        ]
    
    async def get_all_results(self, limit: int = 100) -> List[Dict[str, Any]]:
        """获取所有结果"""
        results = list(self.results.values())
        # 按时间倒序排列，并限制数量
        results = sorted(results, key=lambda r: r["timestamp"], reverse=True)[:limit]
        return results
    
    async def export_results(self, format: str = "json", file_path: str = None) -> str:
        """
        导出结果
        
        Args:
            format: 导出格式，支持json、csv、xlsx
            file_path: 导出文件路径，如果不指定则返回内容
            
        Returns:
            str: 导出内容或文件路径
        """
        if format.lower() == "json":
            content = json.dumps(self.results, indent=2, ensure_ascii=False)
            if file_path:
                with open(file_path, "w", encoding="utf-8") as f:
                    f.write(content)
                return file_path
            return content
        
        elif format.lower() == "csv":
            import csv
            import io
            
            output = io.StringIO()
            if self.results:
                fieldnames = list(next(iter(self.results.values())).keys())
                writer = csv.DictWriter(output, fieldnames=fieldnames)
                writer.writeheader()
                writer.writerows(self.results.values())
            
            content = output.getvalue()
            if file_path:
                with open(file_path, "w", encoding="utf-8", newline="") as f:
                    f.write(content)
                return file_path
            return content
        
        elif format.lower() == "xlsx":
            import pandas as pd
            
            df = pd.DataFrame(list(self.results.values()))
            if file_path:
                df.to_excel(file_path, index=False)
                return file_path
            return df.to_string()
        
        else:
            raise ValueError(f"不支持的导出格式: {format}")
    
    async def clear_results(self, before: datetime = None):
        """清除结果"""
        if before:
            self.results = {
                task_id: result for task_id, result in self.results.items()
                if datetime.fromisoformat(result["timestamp"]) >= before
            }
        else:
            self.results.clear()
            self.aggregated_results.clear()
```

```python
        results = {}
        for channel in channels:

                    results[channel.value] = {"status": "success", "result": result}
                else:
```

```python
@router.post("/tasks", response_model=TaskResponse)
async def create_task(
    request: TaskRequest,
    background_tasks: BackgroundTasks,
    execution_engine: ExecutionEngine = Depends(get_execution_engine)
):
    """创建任务"""
    try:
        # 转换任务类型和优先级
        task_type = TaskType(request.type)
        task_priority = TaskPriority(request.priority)
        
        # 创建任务
        task = await execution_engine.create_task(
            name=request.name,
            description=request.description,
            type=task_type,
            priority=task_priority,
            target_subsystem=request.target_subsystem,
            parameters=request.parameters,
            dependencies=request.dependencies,
            parent_task_id=request.parent_task_id,
            metadata=request.metadata
        )
        
        # 在后台执行任务
        background_tasks.add_task(execution_engine.execute_task, task.id)
        
        # 返回任务响应
        return TaskResponse(
            task_id=task.id,
            name=task.name,
            description=task.description,
            type=task.type.value,
            priority=task.priority.value,
            status=task.status.value,
            target_subsystem=task.target_subsystem,
            parameters=task.parameters,
            dependencies=task.dependencies,
            parent_task_id=task.parent_task_id,
            subtask_ids=task.subtask_ids,
            metadata=task.metadata,
            created_at=task.created_at.isoformat(),
            updated_at=task.updated_at.isoformat()
        )
    
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"创建任务失败: {str(e)}")
```

```python
@pytest.mark.asyncio
async def test_task_throughput():
    """测试任务吞吐量"""
    # 创建执行引擎
    execution_engine = ExecutionEngine()
    await execution_engine.initialize()
    
    try:
        # 模拟子系统客户端
        mock_client = AsyncMock()
        mock_client.execute_task.return_value = {
            "status": "success",
            "result": {"data": "test_result"}
        }
        execution_engine.subsystem_clients["感知处理子系统"] = mock_client
        
        # 测试参数
        duration_seconds = 10  # 测试持续时间（秒）
        task_batch_size = 5    # 每批任务数量
        task_interval = 0.5    # 任务间隔（秒）
        
        # 记录开始时间
        start_time = time.time()
        end_time = start_time + duration_seconds
        
        # 任务计数器
        task_count = 0
        
        # 循环创建和执行任务
        current_time = time.time()
        while current_time < end_time:
            # 创建一批任务
            batch_tasks = []
            for i in range(task_batch_size):
                task = await execution_engine.create_task(
                    name=f"吞吐量测试任务{task_count + i}",
                    description=f"这是吞吐量测试任务{task_count + i}",
                    type=TaskType.DATA_PROCESSING,
                    target_subsystem="感知处理子系统"
                )
                batch_tasks.append(task)
            
            # 并发执行这批任务
            await asyncio.gather(*[
                execution_engine.execute_task(task.id) for task in batch_tasks
            ])
            
            # 更新任务计数
            task_count += task_batch_size
            
            # 等待一段时间
            await asyncio.sleep(task_interval)
            
            # 更新当前时间
            current_time = time.time()
        
        # 计算实际测试时间
        actual_duration = time.time() - start_time
        
        # 计算吞吐量（任务/秒）
        throughput = task_count / actual_duration
        
        print(f"在{actual_duration:.2f}秒内执行了{task_count}个任务")
        print(f"吞吐量: {throughput:.2f}任务/秒")
        
        # 验证吞吐量（应该大于一定的阈值）
        assert throughput > 5.0  # 每秒至少5个任务
        
    finally:
        # 清理资源
        await execution_engine.shutdown()
```
