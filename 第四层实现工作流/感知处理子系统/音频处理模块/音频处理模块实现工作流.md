```python
# src/collector/audio_collector.py
import pyaudio
import numpy as np
import threading
import queue
from typing import Optional, Callable
from src.utils.exceptions import AudioCollectionError
from src.utils.config import AudioConfig

class AudioCollector:
    """音频采集器类，负责从音频输入设备采集实时音频数据"""
    
    def __init__(self, config: AudioConfig):
        self.config = config
        self.audio = None
        self.stream = None
        self.is_recording = False
        self.audio_queue = queue.Queue(maxsize=config.buffer_size)
        self.recording_thread = None
        self.callback = None
        
    def initialize(self):
        """初始化音频采集器"""
        try:
            self.audio = pyaudio.PyAudio()
        except Exception as e:
            raise AudioCollectionError(f"Failed to initialize audio: {str(e)}")
    
    def start_recording(self, callback: Optional[Callable] = None):
        """开始录音"""
        if self.is_recording:
            return
            
        self.callback = callback
        self.is_recording = True
        
        try:
            self.stream = self.audio.open(
                format=getattr(pyaudio, f"pa{self.config.format.upper()}"),
                channels=self.config.channels,
                rate=self.config.sample_rate,
                input=True,
                input_device_index=self.config.device_index,
                frames_per_buffer=self.config.chunk_size,
                stream_callback=self._audio_callback
            )
            
            self.recording_thread = threading.Thread(target=self._process_audio)
            self.recording_thread.daemon = True
            self.recording_thread.start()
            
            self.stream.start_stream()
        except Exception as e:
            self.is_recording = False
            raise AudioCollectionError(f"Failed to start recording: {str(e)}")
    
    def stop_recording(self):
        """停止录音"""
        if not self.is_recording:
            return
            
        self.is_recording = False
        
        if self.stream:
            self.stream.stop_stream()
            self.stream.close()
            self.stream = None
            
        if self.recording_thread:
            self.recording_thread.join(timeout=1.0)
            self.recording_thread = None
    
    def get_audio_chunk(self, timeout: float = 1.0) -> Optional[np.ndarray]:
        """获取音频数据块"""
        try:
            return self.audio_queue.get(timeout=timeout)
        except queue.Empty:
            return None
    
    def _audio_callback(self, in_data, frame_count, time_info, status):
        """音频回调函数"""
        if status:
            print(f"Stream callback error: {status}")
            
        audio_data = np.frombuffer(in_data, dtype=np.int16)
        
        try:
            self.audio_queue.put_nowait(audio_data)
        except queue.Full:
            # 如果队列满了，丢弃最旧的数据
            try:
                self.audio_queue.get_nowait()
                self.audio_queue.put_nowait(audio_data)
            except queue.Empty:
                pass
                
        return (in_data, pyaudio.paContinue)
    
    def _process_audio(self):
        """处理音频数据的线程函数"""
        while self.is_recording:
            audio_chunk = self.get_audio_chunk(timeout=0.1)
            if audio_chunk is not None and self.callback:
                self.callback(audio_chunk)
    
    def get_device_list(self):
        """获取可用音频设备列表"""
        if not self.audio:
            self.initialize()
            
        device_count = self.audio.get_device_count()
        devices = []
        
        for i in range(device_count):
            device_info = self.audio.get_device_info_by_index(i)
            if device_info['maxInputChannels'] > 0:
                devices.append({
                    'index': i,
                    'name': device_info['name'],
                    'channels': device_info['maxInputChannels'],
                    'sample_rate': int(device_info['defaultSampleRate'])
                })
                
        return devices
    
    def cleanup(self):
        """清理资源"""
        self.stop_recording()
        
        if self.audio:
            self.audio.terminate()
            self.audio = None
```

```python
# src/preprocessor/deep_learning_enhancer.py
import torch
import torch.nn as nn
import torchaudio
import numpy as np
from typing import Optional, Tuple
from src.utils.config import AudioConfig

class AudioEnhancementModel(nn.Module):
    """深度学习音频增强模型"""
    
    def __init__(self, input_size=257, hidden_size=256, num_layers=2):
        super(AudioEnhancementModel, self).__init__()
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers, batch_first=True, bidirectional=True)
        self.fc = nn.Linear(hidden_size * 2, input_size)
        self.sigmoid = nn.Sigmoid()
        
    def forward(self, x):
        # x shape: (batch_size, time_steps, freq_bins)
        output, _ = self.lstm(x)
        output = self.fc(output)
        mask = self.sigmoid(output)
        return mask

class DeepLearningAudioEnhancer:
    """基于深度学习的音频增强器"""
    
    def __init__(self, config: AudioConfig):
        self.config = config
        self.device = torch.device("cuda" if torch.cuda.is_available() and config.use_gpu else "cpu")
        self.model = None
        self.window_length = config.n_fft
        self.hop_length = config.hop_length
        self._load_model()
    
    def _load_model(self):
        """加载预训练模型"""
        try:
            self.model = AudioEnhancementModel(
                input_size=self.config.n_fft // 2 + 1,
                hidden_size=256,
                num_layers=2
            )
            
            if torch.cuda.is_available() and self.config.use_gpu:
                self.model.load_state_dict(torch.load(self.config.model_path))
            else:
                self.model.load_state_dict(
                    torch.load(self.config.model_path, map_location=torch.device('cpu'))
                )
                
            self.model.to(self.device)
            self.model.eval()
        except Exception as e:
            print(f"Warning: Failed to load enhancement model: {str(e)}")
            self.model = None
    
    def enhance(self, audio: np.ndarray) -> np.ndarray:
        """增强音频"""
        if self.model is None:
            return audio
            
        # 转换为tensor
        audio_tensor = torch.from_numpy(audio).float().unsqueeze(0).unsqueeze(0)  # (1, 1, samples)
        audio_tensor = audio_tensor.to(self.device)
        
        # 计算STFT
        stft = torch.stft(
            audio_tensor.squeeze(0),
            n_fft=self.window_length,
            hop_length=self.hop_length,
            window=torch.hann_window(self.window_length).to(self.device),
            return_complex=True
        )
        
        magnitude = torch.abs(stft)  # (freq_bins, time_steps)
        phase = torch.angle(stft)
        
        # 准备模型输入
        log_magnitude = torch.log1p(magnitude).T.unsqueeze(0)  # (1, time_steps, freq_bins)
        
        # 模型推理
        with torch.no_grad():
            mask = self.model(log_magnitude)
            
        # 应用掩码
        enhanced_magnitude = magnitude * mask.squeeze(0).T
        
        # 重建复数谱
        enhanced_stft = enhanced_magnitude * torch.exp(1j * phase)
        
        # 逆STFT
        enhanced_audio = torch.istft(
            enhanced_stft,
            n_fft=self.window_length,
            hop_length=self.hop_length,
            window=torch.hann_window(self.window_length).to(self.device)
        )
        
        # 转换为numpy数组
        enhanced_audio_np = enhanced_audio.cpu().numpy()
        
        # 确保输出长度与输入相同
        if len(enhanced_audio_np) > len(audio):
            enhanced_audio_np = enhanced_audio_np[:len(audio)]
        elif len(enhanced_audio_np) < len(audio):
            enhanced_audio_np = np.pad(enhanced_audio_np, (0, len(audio) - len(enhanced_audio_np)), 'constant')
            
        return enhanced_audio_np
```

##### 高效MFCC特征提取器
```python
# src/extractor/mfcc_extractor.py
import librosa
import numpy as np
import torch
import nnAudio
from nnAudio import features
from typing import Optional, Tuple
from src.utils.config import AudioConfig

class EfficientMFCCExtractor:
    """高效MFCC特征提取器，结合传统方法和深度学习加速"""
    
    def __init__(self, config: AudioConfig):
        self.config = config
        self.device = torch.device("cuda" if torch.cuda.is_available() and config.use_gpu else "cpu")
        
        # 初始化nnAudio的梅尔频谱变换器（GPU加速）
        self.mel_transform = None
        if torch.cuda.is_available() and config.use_gpu:
            self.mel_transform = features.MelSpectrogram(
                sr=config.sample_rate,
                n_fft=config.n_fft,
                hop_length=config.hop_length,
                n_mels=config.n_mels,
                fmin=config.fmin,
                fmax=config.fmax
            ).to(self.device)
    
    def extract(self, audio: np.ndarray) -> np.ndarray:
        """提取MFCC特征"""
        # 确保音频是浮点类型
        if audio.dtype != np.float32 and audio.dtype != np.float64:
            audio = audio.astype(np.float32) / 32768.0  # 假设是16位整数
            
        # 使用GPU加速的梅尔频谱变换（如果可用）
        if self.mel_transform is not None:
            return self._extract_with_gpu(audio)
        else:
            return self._extract_with_cpu(audio)
    
    def _extract_with_gpu(self, audio: np.ndarray) -> np.ndarray:
        """使用GPU加速提取MFCC特征"""
        # 转换为tensor
        audio_tensor = torch.from_numpy(audio).float().unsqueeze(0).to(self.device)
        
        # 计算梅尔频谱
        mel_spec = self.mel_transform(audio_tensor)  # (1, n_mels, time_steps)
        
        # 转换为对数梅尔频谱
        log_mel_spec = torch.log1p(mel_spec)
        
        # 离散余弦变换（DCT）得到MFCC
        mfcc = torch.dct(log_mel_spec, norm='ortho')
        
        # 只保留前n_mfcc个系数
        mfcc = mfcc[:, :self.config.n_mfcc, :]
        
        # 转换为numpy并转置为(time_steps, n_mfcc)
        mfcc_np = mfcc.squeeze(0).cpu().numpy().T
        
        return mfcc_np
    
    def _extract_with_cpu(self, audio: np.ndarray) -> np.ndarray:
        """使用CPU提取MFCC特征（传统方法）"""
        # 使用librosa提取MFCC
        mfcc = librosa.feature.mfcc(
            y=audio,
            sr=self.config.sample_rate,
            n_mfcc=self.config.n_mfcc,
            n_fft=self.config.n_fft,
            hop_length=self.config.hop_length,
            n_mels=self.config.n_mels,
            fmin=self.config.fmin,
            fmax=self.config.fmax
        )
        
        # 转置为(time_steps, n_mfcc)
        return mfcc.T
    
    def extract_delta(self, mfcc: np.ndarray) -> np.ndarray:
        """提取一阶差分特征"""
        delta = librosa.feature.delta(mfcc.T)
        return delta.T
    
    def extract_delta_delta(self, mfcc: np.ndarray) -> np.ndarray:
        """提取二阶差分特征"""
        delta_delta = librosa.feature.delta(mfcc.T, order=2)
        return delta_delta.T
    
    def extract_with_deltas(self, audio: np.ndarray) -> np.ndarray:
        """提取包含一阶和二阶差分的完整MFCC特征"""
        mfcc = self.extract(audio)
        delta = self.extract_delta(mfcc)
        delta_delta = self.extract_delta_delta(mfcc)
        
        # 拼接特征
        combined_features = np.concatenate([mfcc, delta, delta_delta], axis=1)
        
        return combined_features
```

```python
# src/controller/quality_controller.py
import numpy as np
import librosa
from typing import Dict, Any, Tuple
from src.utils.config import AudioConfig

class QualityController:
    """音频质量控制器，负责评估音频质量并提供反馈"""
    
    def __init__(self, config: AudioConfig):
        self.config = config
        self.snr_threshold = 10.0  # 信噪比阈值(dB)
        self.spectral_flatness_threshold = 0.5  # 频谱平坦度阈值
        
    def evaluate(self, audio: np.ndarray, enhanced_audio: np.ndarray = None) -> Dict[str, Any]:
        """评估音频质量"""
        metrics = {}
        
        # 计算基本音频指标
        metrics['rms'] = self._calculate_rms(audio)
        metrics['zero_crossing_rate'] = self._calculate_zero_crossing_rate(audio)
        metrics['spectral_centroid'] = self._calculate_spectral_centroid(audio)
        metrics['spectral_bandwidth'] = self._calculate_spectral_bandwidth(audio)
        metrics['spectral_rolloff'] = self._calculate_spectral_rolloff(audio)
        metrics['spectral_flatness'] = self._calculate_spectral_flatness(audio)
        
        # 如果提供了增强后的音频，计算增强效果指标
        if enhanced_audio is not None:
            metrics['snr_improvement'] = self._calculate_snr_improvement(audio, enhanced_audio)
            metrics['enhancement_score'] = self._calculate_enhancement_score(audio, enhanced_audio)
        
        # 计算整体质量分数
        metrics['overall_quality'] = self._calculate_overall_quality(metrics)
        
        return metrics
    
    def _calculate_rms(self, audio: np.ndarray) -> float:
        """计算均方根能量"""
        return float(np.sqrt(np.mean(audio ** 2)))
    
    def _calculate_zero_crossing_rate(self, audio: np.ndarray) -> float:
        """计算过零率"""
        zero_crossings = np.where(np.diff(np.sign(audio)))[0]
        return float(len(zero_crossings) / len(audio))
    
    def _calculate_spectral_centroid(self, audio: np.ndarray) -> float:
        """计算频谱质心"""
        stft = np.abs(librosa.stft(audio))
        freqs = librosa.fft_frequencies(sr=self.config.sample_rate)
        return float(np.sum(freqs[:, None] * stft, axis=0) / np.sum(stft, axis=0).mean())
    
    def _calculate_spectral_bandwidth(self, audio: np.ndarray) -> float:
        """计算频谱带宽"""
        stft = np.abs(librosa.stft(audio))
        freqs = librosa.fft_frequencies(sr=self.config.sample_rate)
        centroid = self._calculate_spectral_centroid(audio)
        return float(np.sqrt(np.sum(((freqs[:, None] - centroid) ** 2) * stft, axis=0) / np.sum(stft, axis=0)).mean())
    
    def _calculate_spectral_rolloff(self, audio: np.ndarray, roll_percent=0.85) -> float:
        """计算频谱滚降点"""
        stft = np.abs(librosa.stft(audio))
        total_energy = np.sum(stft, axis=0)
        cumsum_energy = np.cumsum(stft, axis=0)
        rolloff_freqs = np.zeros(total_energy.shape)
        
        for i in range(total_energy.shape[0]):
            if total_energy[i] > 0:
                rolloff_idx = np.where(cumsum_energy[:, i] >= roll_percent * total_energy[i])[0]
                if len(rolloff_idx) > 0:
                    rolloff_freqs[i] = librosa.fft_frequencies(sr=self.config.sample_rate)[rolloff_idx[0]]
                    
        return float(np.mean(rolloff_freqs))
    
    def _calculate_spectral_flatness(self, audio: np.ndarray) -> float:
        """计算频谱平坦度"""
        stft = np.abs(librosa.stft(audio)) ** 2
        geometric_mean = np.exp(np.mean(np.log(stft + 1e-10), axis=0))
        arithmetic_mean = np.mean(stft, axis=0)
        return float(np.mean(geometric_mean / (arithmetic_mean + 1e-10)))
    
    def _calculate_snr_improvement(self, original_audio: np.ndarray, enhanced_audio: np.ndarray) -> float:
        """计算信噪比改善"""
        original_snr = self._estimate_snr(original_audio)
        enhanced_snr = self._estimate_snr(enhanced_audio)
        return float(enhanced_snr - original_snr)
    
    def _estimate_snr(self, audio: np.ndarray) -> float:
        """估计音频的信噪比"""
        # 简单的SNR估计方法，假设静音部分为噪声
        # 在实际应用中，可能需要更复杂的噪声估计方法
        frame_length = 2048
        hop_length = 512
        
        # 计算短时能量
        stft = librosa.stft(audio, n_fft=frame_length, hop_length=hop_length)
        magnitude = np.abs(stft) ** 2
        
        # 假设能量最低的10%帧为噪声
        frame_energy = np.sum(magnitude, axis=0)
        noise_frames = np.argsort(frame_energy)[:int(0.1 * len(frame_energy))]
        noise_power = np.mean(frame_energy[noise_frames])
        
        # 信号功率为总功率减去噪声功率
        signal_power = np.mean(frame_energy) - noise_power
        
        if noise_power > 0:
            snr_db = 10 * np.log10(signal_power / noise_power)
        else:
            snr_db = 40.0  # 如果噪声功率为0，设置一个较高的SNR值
            
        return float(snr_db)
    
    def _calculate_enhancement_score(self, original_audio: np.ndarray, enhanced_audio: np.ndarray) -> float:
        """计算音频增强分数"""
        # 结合多个指标计算增强分数
        snr_improvement = self._calculate_snr_improvement(original_audio, enhanced_audio)
        
        # 计算频谱平坦度改善
        original_flatness = self._calculate_spectral_flatness(original_audio)
        enhanced_flatness = self._calculate_spectral_flatness(enhanced_audio)
        flatness_improvement = original_flatness - enhanced_flatness
        
        # 综合评分
        enhancement_score = 0.7 * min(snr_improvement / 10.0, 1.0) + 0.3 * min(flatness_improvement * 2, 1.0)
        
        return float(enhancement_score)
    
    def _calculate_overall_quality(self, metrics: Dict[str, Any]) -> float:
        """计算整体质量分数"""
        # 基于多个指标计算加权平均质量分数
        weights = {
            'snr_improvement': 0.3,
            'enhancement_score': 0.4,
            'spectral_flatness': 0.2,
            'spectral_centroid': 0.1
        }
        
        score = 0.0
        total_weight = 0.0
        
        for metric, weight in weights.items():
            if metric in metrics:
                # 归一化指标到0-1范围
                if metric == 'snr_improvement':
                    normalized_value = min(metrics[metric] / 20.0, 1.0)
                elif metric == 'spectral_flatness':
                    normalized_value = max(0, 1.0 - metrics[metric])
                elif metric == 'spectral_centroid':
                    # 假设理想的频谱质心在1000-3000Hz之间
                    centroid = metrics[metric]
                    if 1000 <= centroid <= 3000:
                        normalized_value = 1.0
                    else:
                        normalized_value = max(0, 1.0 - abs(centroid - 2000) / 2000)
                else:
                    normalized_value = metrics[metric]
                    
                score += normalized_value * weight
                total_weight += weight
        
        if total_weight > 0:
            score /= total_weight
            
        return float(score)
    
    def is_acceptable_quality(self, metrics: Dict[str, Any]) -> bool:
        """判断音频质量是否可接受"""
        # 检查关键指标是否满足阈值
        if 'overall_quality' in metrics and metrics['overall_quality'] < 0.6:
            return False
            
        if 'snr_improvement' in metrics and metrics['snr_improvement'] < 5.0:
            return False
            
        if 'spectral_flatness' in metrics and metrics['spectral_flatness'] > self.spectral_flatness_threshold:
            return False
            
        return True
```

```python
logger = logging.getLogger(__name__)

def handle_audio_processing_errors(func: Callable) -> Callable:
    """音频处理错误装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs) -> Any:
        try:
            return func(*args, **kwargs)
        except AudioProcessingError as e:
            logger.error(f"Audio processing error in {func.__name__}: {str(e)}")
            raise HTTPException(status_code=400, detail=str(e))
        except Exception as e:
            logger.error(f"Unexpected error in {func.__name__}: {str(e)}")
            raise HTTPException(status_code=500, detail="Internal server error")
    return wrapper

        import time
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        execution_time = end_time - start_time
        logger.info(f"{func.__name__} executed in {execution_time:.4f} seconds")
        return result
    return wrapper
```

## 使用指南

### 基本使用

```python
from src.collector.audio_collector import AudioCollector
from src.preprocessor.deep_learning_enhancer import DeepLearningAudioEnhancer
from src.extractor.mfcc_extractor import EfficientMFCCExtractor
from src.controller.quality_controller import QualityController
from src.utils.config import AudioConfig

# 配置音频处理参数

    sample_rate=16000,
    chunk_size=1024,
    n_mfcc=13,
    use_gpu=True
)

# 初始化组件

enhancer = DeepLearningAudioEnhancer(config)
extractor = EfficientMFCCExtractor(config)
controller = QualityController(config)

# 音频处理回调函数
def process_audio(audio_chunk):
    # 增强音频
    enhanced_audio = enhancer.enhance(audio_chunk)
    
    # 提取特征
    mfcc = extractor.extract(enhanced_audio)
    
    # 评估质量
    quality_metrics = controller.evaluate(audio_chunk, enhanced_audio)
    
    return {
        "mfcc": mfcc.tolist(),
        "quality": quality_metrics
    }

# 开始音频采集
collector.initialize()
collector.start_recording(callback=process_audio)

# 获取处理结果
try:
    while True:
        result = collector.get_audio_chunk()
        if result is not None:
            print(f"Processed audio chunk: {len(result)} samples")
except KeyboardInterrupt:
    print("Stopping audio collection...")
    collector.stop_recording()
    collector.cleanup()
```

```python
import threading
import queue
import time

class RealTimeAudioProcessor:
    def __init__(self, config):
        self.config = config
        self.collector = AudioCollector(config)
        self.enhancer = DeepLearningAudioEnhancer(config)
        self.extractor = EfficientMFCCExtractor(config)
        self.controller = QualityController(config)
        self.result_queue = queue.Queue()
        self.is_running = False
        
    def start(self):
        """启动实时音频处理"""
        self.collector.initialize()
        self.collector.start_recording(callback=self._process_audio)
        self.is_running = True
        
        # 启动结果处理线程
        self.result_thread = threading.Thread(target=self._process_results)
        self.result_thread.daemon = True
        self.result_thread.start()
        
    def stop(self):
        """停止实时音频处理"""
        self.is_running = False
        self.collector.stop_recording()
        self.collector.cleanup()
        
    def _process_audio(self, audio_chunk):
        """处理音频数据"""
        if not self.is_running:
            return
            
        # 增强音频
        enhanced_audio = self.enhancer.enhance(audio_chunk)
        
        # 提取特征
        mfcc = self.extractor.extract(enhanced_audio)
        
        # 评估质量
        quality_metrics = controller.evaluate(audio_chunk, enhanced_audio)
        
        # 将结果放入队列
        result = {
            "timestamp": time.time(),
            "mfcc": mfcc.tolist(),
            "quality": quality_metrics
        }
        
        try:
            self.result_queue.put_nowait(result)
        except queue.Full:
            # 如果队列满了，丢弃最旧的结果
            try:
                self.result_queue.get_nowait()
                self.result_queue.put_nowait(result)
            except queue.Empty:
                pass
    
    def _process_results(self):
        """处理结果"""
        while self.is_running:
            try:
                result = self.result_queue.get(timeout=0.1)
                # 在这里处理结果
                print(f"处理结果: 时间戳={result['timestamp']}, 质量评分={result['quality']['overall_quality']:.2f}")
            except queue.Empty:
                pass
    
    def get_latest_result(self, timeout=1.0):
        """获取最新的处理结果"""
        try:
            return self.result_queue.get(timeout=timeout)
        except queue.Empty:
            return None

# 使用示例
config = AudioConfig()
processor = RealTimeAudioProcessor(config)

try:
    processor.start()
    while True:
        result = processor.get_latest_result()
        if result:
            print(f"最新结果: {result}")
        time.sleep(0.1)
except KeyboardInterrupt:
    print("停止处理...")
    processor.stop()
```

### 4.2 批量音频处理

对于需要处理大量音频文件的应用场景，可以使用批量处理模式：

```python
import os
import glob
from concurrent.futures import ThreadPoolExecutor
import json

class BatchAudioProcessor:
    def __init__(self, config, max_workers=4):
        self.config = config
        self.enhancer = DeepLearningAudioEnhancer(config)
        self.extractor = EfficientMFCCExtractor(config)
        self.controller = QualityController(config)
        self.max_workers = max_workers
        
    def process_file(self, file_path):
        """处理单个音频文件"""
        try:
            # 加载音频文件
            audio, sr = librosa.load(file_path, sr=self.config.sample_rate)
            audio = (audio * 32767).astype(np.int16)
            
            # 处理音频
            enhanced_audio = self.enhancer.enhance(audio)
            mfcc = self.extractor.extract(enhanced_audio)
            quality_metrics = self.controller.evaluate(audio, enhanced_audio)
            
            # 返回结果
            result = {
                "file_path": file_path,
                "success": True,
                "mfcc_shape": mfcc.shape,
                "quality_score": quality_metrics['overall_quality'],
                "is_acceptable": self.controller.is_acceptable_quality(quality_metrics)
            }
            
            return result
        except Exception as e:
            return {
                "file_path": file_path,
                "success": False,
                "error": str(e)
            }
    
    def process_directory(self, input_dir, output_file):
        """处理目录中的所有音频文件"""
        # 查找所有音频文件
        audio_files = glob.glob(os.path.join(input_dir, "*.wav"))
        
        # 使用线程池并行处理
        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            results = list(executor.map(self.process_file, audio_files))
        
        # 保存结果
        with open(output_file, 'w') as f:
            json.dump(results, f, indent=2)
        
        # 打印统计信息
        success_count = sum(1 for r in results if r['success'])
        acceptable_count = sum(1 for r in results if r['success'] and r['is_acceptable'])
        
        print(f"处理完成: {success_count}/{len(audio_files)} 成功")
        print(f"质量可接受: {acceptable_count}/{success_count}")
        
        return results

# 使用示例
config = AudioConfig()
processor = BatchAudioProcessor(config, max_workers=8)
results = processor.process_directory("input_audio_dir", "output_results.json")
```

```python
        return response.json()
    
    def process_file(self, file_path, **kwargs):
        """处理音频文件"""
        with open(file_path, 'rb') as f:
            files = {"file": f}
            data = kwargs
            response = requests.post(
                f"{self.base_url}/api/v1/audio/process",
                headers=self.headers,
                files=files,
                data=data
            )
        return response.json()
    
    def process_url(self, url, **kwargs):
        """处理远程音频文件"""
        data = {"url": url, **kwargs}
        response = requests.post(
            f"{self.base_url}/api/v1/audio/process_url",
            headers=self.headers,
            json=data
        )
        return response.json()
    
    def get_features(self, audio_id, feature_types=None):
        """获取音频特征"""

            headers=self.headers,
            json={"audio_id": audio_id, **data}
        )
        return response.json()

        """获取音频质量指标"""
        response = requests.post(
            f"{self.base_url}/api/v1/audio/quality",
            headers=self.headers,
            json={"audio_id": audio_id}
        )
        return response.json()
```

## 5. 常见问题

### 5.1 音频采集问题

**Q: 为什么无法采集到音频？**

A: 可能的原因和解决方案：
1. 检查音频设备是否正确连接
2. 确认设备权限设置，确保应用有访问麦克风的权限
3. 尝试使用不同的音频设备
4. 检查设备是否被其他应用占用

**Q: 采集到的音频质量很差，有很多噪声？**

A: 可能的原因和解决方案：
1. 检查麦克风硬件质量
2. 确认麦克风位置，避免靠近噪声源
3. 调整音频采集参数，如采样率和增益
4. 使用降噪算法处理音频

### 5.2 处理性能问题

**Q: 音频处理速度很慢，无法满足实时要求？**

A: 可能的原因和解决方案：
1. 启用GPU加速：设置`use_gpu=True`
2. 减小音频块大小：降低`chunk_size`值
3. 使用更简单的处理算法：如将`enhancement_method`从"deep_learning"改为"wiener"
4. 优化代码性能：使用更高效的算法和数据结构

**Q: 内存使用过高，导致系统不稳定？**

A: 可能的原因和解决方案：
1. 减小缓冲区大小：降低`buffer_size`值
2. 启用内存优化：调用`memory_optimizer.enable_memory_optimization()`
3. 定期清理缓存：定期调用`gc.collect()`
4. 使用批处理模式：避免频繁的小块处理

### 5.3 特征提取问题

**Q: 提取的MFCC特征不准确？**

A: 可能的原因和解决方案：
1. 调整MFCC参数：如`n_mfcc`、`n_fft`、`hop_length`等
2. 检查音频质量：确保音频没有严重的噪声或失真
3. 尝试不同的特征提取方法：如使用深度学习特征提取器
4. 标准化音频：确保音频采样率和格式一致

**Q: 特征提取速度很慢？**

A: 可能的原因和解决方案：
1. 启用GPU加速：设置`use_gpu=True`
2. 减小特征维度：降低`n_mfcc`值
3. 使用更高效的算法：如使用nnAudio库代替librosa
4. 批量处理：一次性处理多个音频帧
