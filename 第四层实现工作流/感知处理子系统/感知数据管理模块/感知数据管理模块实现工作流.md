#### 配置加载器实现
```python
# src/utils/config.py
import os
import yaml
from typing import Dict, Any, Optional

class ConfigLoader:
    """配置加载器，支持从YAML文件和环境变量加载配置"""
    
    def __init__(self, config_dir: str = "config"):
        self.config_dir = config_dir
        self._config_cache = {}
    
    def load_config(self, config_name: str, env_var_prefix: Optional[str] = None) -> Dict[str, Any]:
        """
        加载配置文件，并可选择性地覆盖环境变量
        
        Args:
            config_name: 配置文件名（不含扩展名）
            env_var_prefix: 环境变量前缀，用于覆盖配置
            
        Returns:
            配置字典
        """
        cache_key = f"{config_name}_{env_var_prefix}"
        if cache_key in self._config_cache:
            return self._config_cache[cache_key]
            
        config_path = os.path.join(self.config_dir, f"{config_name}.yaml")
        
        # 加载YAML配置
        with open(config_path, 'r', encoding='utf-8') as f:
            config = yaml.safe_load(f)
        
        # 应用环境变量覆盖
        if env_var_prefix:
            config = self._apply_env_overrides(config, env_var_prefix)
        
        self._config_cache[cache_key] = config
        return config
    
    def _apply_env_overrides(self, config: Dict[str, Any], prefix: str) -> Dict[str, Any]:
        """
        应用环境变量覆盖配置
        
        Args:
            config: 原始配置
            prefix: 环境变量前缀
            
        Returns:
            覆盖后的配置
        """
        for key, value in os.environ.items():
            if key.startswith(f"{prefix}_"):
                config_key = key[len(prefix)+1:].lower()
                # 支持嵌套配置，使用双下划线分隔
                keys = config_key.split('__')
                self._set_nested_value(config, keys, value)
        
        return config
    
    def _set_nested_value(self, config: Dict[str, Any], keys: list, value: str):
        """设置嵌套配置值"""
        current = config
        for key in keys[:-1]:
            if key not in current:
                current[key] = {}
            current = current[key]
        
        # 尝试转换值类型
        try:
            if value.lower() in ('true', 'false'):
                current[keys[-1]] = value.lower() == 'true'
            elif value.isdigit():
                current[keys[-1]] = int(value)
            elif '.' in value and all(part.isdigit() for part in value.split('.')):
                current[keys[-1]] = float(value)
            else:
                current[keys[-1]] = value
        except ValueError:
            current[keys[-1]] = value
```

```python
# src/ingestion/data_ingestor.py
import asyncio
import logging
from typing import Dict, Any, List, Optional, Union
from datetime import datetime
from concurrent.futures import ThreadPoolExecutor
from bson import ObjectId
from pymongo import MongoClient
from redis import Redis
from minio import Minio

from ..storage.mongodb_storage import MongoDBStorage
from ..storage.redis_storage import RedisStorage
from ..storage.minio_storage import MinIOStorage
from ..storage.elasticsearch_storage import ElasticsearchStorage
from ..utils.exceptions import DataIngestionError, ValidationError
from .data_validator import DataValidator
from .data_preprocessor import DataPreprocessor

logger = logging.getLogger(__name__)

class DataIngestor:
    """数据接入器，负责接收和预处理感知数据"""
    
    def __init__(
        self,
        mongodb_storage: MongoDBStorage,
        redis_storage: RedisStorage,
        minio_storage: MinIOStorage,
        elasticsearch_storage: ElasticsearchStorage,
        validator: DataValidator,
        preprocessor: DataPreprocessor,
        max_workers: int = 10
    ):
        self.mongodb_storage = mongodb_storage
        self.redis_storage = redis_storage
        self.minio_storage = minio_storage
        self.elasticsearch_storage = elasticsearch_storage
        self.validator = validator
        self.preprocessor = preprocessor
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self._ingestion_stats = {
            "total_ingested": 0,
            "successful": 0,
            "failed": 0,
            "by_type": {}
        }
    
    async def ingest_data(
        self,
        data: Dict[str, Any],
        data_type: str,
        metadata: Optional[Dict[str, Any]] = None
    ) -> Dict[str, Any]:
        """
        接入感知数据
        
        Args:
            data: 感知数据
            data_type: 数据类型 (audio, video, multimodal)
            metadata: 元数据
            
        Returns:
            包含数据ID和状态的字典
            
        Raises:
            DataIngestionError: 数据接入失败
            ValidationError: 数据验证失败
        """
        try:
            # 更新统计信息
            self._ingestion_stats["total_ingested"] += 1
            if data_type not in self._ingestion_stats["by_type"]:
                self._ingestion_stats["by_type"][data_type] = 0
            self._ingestion_stats["by_type"][data_type] += 1
            
            # 验证数据
            validation_result = await self.validator.validate(data, data_type)
            if not validation_result.is_valid:
                self._ingestion_stats["failed"] += 1
                raise ValidationError(f"数据验证失败: {validation_result.errors}")
            
            # 预处理数据
            processed_data = await self.preprocessor.process(data, data_type)
            
            # 创建数据记录
            data_id = str(ObjectId())
            timestamp = datetime.utcnow()
            
            # 准备元数据
            if metadata is None:
                metadata = {}
            metadata.update({
                "data_id": data_id,
                "data_type": data_type,
                "timestamp": timestamp,
                "ingestion_time": timestamp,
                "storage_tier": "hot"  # 新数据默认为热数据
            })
            
            # 根据数据类型选择存储策略
            if data_type == "audio":
                storage_result = await self._ingest_audio_data(processed_data, metadata)
            elif data_type == "video":
                storage_result = await self._ingest_video_data(processed_data, metadata)
            elif data_type == "multimodal":
                storage_result = await self._ingest_multimodal_data(processed_data, metadata)
            else:
                raise DataIngestionError(f"不支持的数据类型: {data_type}")
            
            # 更新统计信息
            self._ingestion_stats["successful"] += 1
            
            logger.info(f"成功接入数据: {data_id}, 类型: {data_type}")
            return {
                "data_id": data_id,
                "status": "success",
                "storage_info": storage_result,
                "timestamp": timestamp.isoformat()
            }
            
        except Exception as e:
            self._ingestion_stats["failed"] += 1
            logger.error(f"数据接入失败: {str(e)}")
            raise DataIngestionError(f"数据接入失败: {str(e)}")
    
    async def _ingest_audio_data(self, data: Dict[str, Any], metadata: Dict[str, Any]) -> Dict[str, Any]:
        """接入音频数据"""
        # 将音频文件存储到MinIO
        if "audio_file" in data:
            file_path = data["audio_file"]
            object_name = f"audio/{metadata['data_id']}/{file_path.split('/')[-1]}"
            await self.minio_storage.store_file(file_path, object_name)
            metadata["minio_object"] = object_name
        
        # 将元数据和特征存储到MongoDB
        await self.mongodb_storage.store_metadata(metadata, data.get("features", {}))
        
        # 在Elasticsearch中创建索引
        await self.elasticsearch_storage.index_document(metadata)
        
        # 在Redis中缓存热点数据
        await self.redis_storage.cache_data(metadata["data_id"], {
            "metadata": metadata,
            "features": data.get("features", {})
        })
        
        return {
            "mongodb_id": metadata["data_id"],
            "minio_object": metadata.get("minio_object"),
            "elasticsearch_id": metadata["data_id"]
        }
    
    async def _ingest_video_data(self, data: Dict[str, Any], metadata: Dict[str, Any]) -> Dict[str, Any]:
        """接入视频数据"""
        # 将视频文件存储到MinIO
        if "video_file" in data:
            file_path = data["video_file"]
            object_name = f"video/{metadata['data_id']}/{file_path.split('/')[-1]}"
            await self.minio_storage.store_file(file_path, object_name)
            metadata["minio_object"] = object_name
        
        # 将元数据和特征存储到MongoDB
        await self.mongodb_storage.store_metadata(metadata, data.get("features", {}))
        
        # 在Elasticsearch中创建索引
        await self.elasticsearch_storage.index_document(metadata)
        
        # 在Redis中缓存热点数据
        await self.redis_storage.cache_data(metadata["data_id"], {
            "metadata": metadata,
            "features": data.get("features", {})
        })
        
        return {
            "mongodb_id": metadata["data_id"],
            "minio_object": metadata.get("minio_object"),
            "elasticsearch_id": metadata["data_id"]
        }
    
    async def _ingest_multimodal_data(self, data: Dict[str, Any], metadata: Dict[str, Any]) -> Dict[str, Any]:
        """接入多模态数据"""
        # 存储音频文件
        if "audio_file" in data:
            file_path = data["audio_file"]
            object_name = f"multimodal/{metadata['data_id']}/audio/{file_path.split('/')[-1]}"
            await self.minio_storage.store_file(file_path, object_name)
            metadata["audio_minio_object"] = object_name
        
        # 存储视频文件
        if "video_file" in data:
            file_path = data["video_file"]
            object_name = f"multimodal/{metadata['data_id']}/video/{file_path.split('/')[-1]}"
            await self.minio_storage.store_file(file_path, object_name)
            metadata["video_minio_object"] = object_name
        
        # 将元数据和特征存储到MongoDB
        await self.mongodb_storage.store_metadata(metadata, data.get("features", {}))
        
        # 在Elasticsearch中创建索引
        await self.elasticsearch_storage.index_document(metadata)
        
        # 在Redis中缓存热点数据
        await self.redis_storage.cache_data(metadata["data_id"], {
            "metadata": metadata,
            "features": data.get("features", {}),
            "alignment_info": data.get("alignment_info", {})
        })
        
        return {
            "mongodb_id": metadata["data_id"],
            "audio_minio_object": metadata.get("audio_minio_object"),
            "video_minio_object": metadata.get("video_minio_object"),
            "elasticsearch_id": metadata["data_id"]
        }
    
    def get_ingestion_stats(self) -> Dict[str, Any]:
        """获取数据接入统计信息"""
        return self._ingestion_stats.copy()
    
    async def close(self):
        """关闭数据接入器，释放资源"""
        self.executor.shutdown(wait=True)
```

#### 数据检索器实现
```python
# src/retrieval/data_retriever.py
import asyncio
import logging
import json
from typing import Dict, Any, List, Optional, Union
from datetime import datetime
from bson import ObjectId

from ..storage.mongodb_storage import MongoDBStorage
from ..storage.redis_storage import RedisStorage
from ..storage.minio_storage import MinIOStorage
from ..storage.elasticsearch_storage import ElasticsearchStorage
from ..models.data_models import DataRecord, RetrievalQuery, RetrievalResult
from ..utils.exceptions import DataRetrievalError, NotFoundError

logger = logging.getLogger(__name__)

class DataRetriever:
    """数据检索器 - 负责从不同存储层检索数据"""
    
    def __init__(
        self,
        mongodb_storage: MongoDBStorage,
        redis_storage: RedisStorage,
        minio_storage: MinIOStorage,
        elasticsearch_storage: ElasticsearchStorage,
        max_concurrent_retrievals: int = 10
    ):
        """
        初始化数据检索器
        
        Args:
            mongodb_storage: MongoDB存储实例
            redis_storage: Redis存储实例
            minio_storage: MinIO存储实例
            elasticsearch_storage: Elasticsearch存储实例
            max_concurrent_retrievals: 最大并发检索数
        """
        self.mongodb_storage = mongodb_storage
        self.redis_storage = redis_storage
        self.minio_storage = minio_storage
        self.elasticsearch_storage = elasticsearch_storage
        self.semaphore = asyncio.Semaphore(max_concurrent_retrievals)
        self._retrieval_stats = {
            "total_retrievals": 0,
            "cache_hits": 0,
            "db_hits": 0,
            "minio_hits": 0,
            "es_hits": 0,
            "by_type": {}
        }
    
    async def retrieve_data(
        self,
        query: Union[str, RetrievalQuery],
        retrieve_content: bool = True,
        max_results: int = 100
    ) -> RetrievalResult:
        """
        检索数据
        
        Args:
            query: 检索查询，可以是字符串或RetrievalQuery对象
            retrieve_content: 是否检索实际内容（文件）
            max_results: 最大结果数
            
        Returns:
            检索结果对象
            
        Raises:
            DataRetrievalError: 数据检索失败
        """
        try:
            # 更新统计信息
            self._retrieval_stats["total_retrievals"] += 1
            
            # 解析查询
            if isinstance(query, str):
                query = RetrievalQuery(text=query)
            
            # 首先尝试从缓存中检索
            cache_results = await self._retrieve_from_cache(query, max_results)
            if cache_results and len(cache_results) >= max_results:
                self._retrieval_stats["cache_hits"] += 1
                logger.info(f"从缓存检索到 {len(cache_results)} 条结果")
                return RetrievalResult(
                    records=cache_results[:max_results],
                    total_count=len(cache_results),
                    query=query,
                    source="cache"
                )
            
            # 从数据库检索
            db_results = await self._retrieve_from_database(query, max_results)
            if not db_results:
                raise NotFoundError("未找到匹配的数据")
            
            self._retrieval_stats["db_hits"] += 1
            
            # 如果需要，检索实际内容
            if retrieve_content:
                db_results = await self._retrieve_content(db_results)
            
            # 更新缓存
            await self._update_cache(db_results)
            
            # 更新统计信息
            data_type = query.filters.get("data_type", "unknown")
            if data_type not in self._retrieval_stats["by_type"]:
                self._retrieval_stats["by_type"][data_type] = 0
            self._retrieval_stats["by_type"][data_type] += 1
            
            logger.info(f"从数据库检索到 {len(db_results)} 条结果")
            return RetrievalResult(
                records=db_results[:max_results],
                total_count=len(db_results),
                query=query,
                source="database"
            )
            
        except Exception as e:
            logger.error(f"数据检索失败: {str(e)}")
            raise DataRetrievalError(f"数据检索失败: {str(e)}")
    
    async def retrieve_by_id(
        self,
        data_id: str,
        retrieve_content: bool = True
    ) -> Optional[DataRecord]:
        """
        根据ID检索数据
        
        Args:
            data_id: 数据ID
            retrieve_content: 是否检索实际内容（文件）
            
        Returns:
            数据记录对象，如果未找到则返回None
        """
        try:
            # 首先尝试从缓存中检索
            cached_data = await self.redis_storage.get_cached_data(data_id)
            if cached_data:
                self._retrieval_stats["cache_hits"] += 1
                record = DataRecord.from_dict(cached_data)
                
                # 如果需要且缓存中没有内容，从MinIO检索
                if retrieve_content and not record.content:
                    record = await self._retrieve_content_for_record(record)
                
                return record
            
            # 从MongoDB检索
            metadata = await self.mongodb_storage.get_metadata(data_id)
            if not metadata:
                return None
            
            self._retrieval_stats["db_hits"] += 1
            
            # 创建数据记录
            record = DataRecord.from_metadata(metadata)
            
            # 如果需要，检索实际内容
            if retrieve_content:
                record = await self._retrieve_content_for_record(record)
            
            # 更新缓存
            await self.redis_storage.cache_data(data_id, record.to_dict())
            
            return record
            
        except Exception as e:
            logger.error(f"根据ID检索数据失败: {str(e)}")
            return None
    
    async def _retrieve_from_cache(
        self,
        query: RetrievalQuery,
        max_results: int
    ) -> List[DataRecord]:
        """从缓存中检索数据"""
        # 尝试从Redis缓存中检索
        cache_results = await self.redis_storage.search_cached_data(query)
        return cache_results[:max_results]
    
    async def _retrieve_from_database(
        self,
        query: RetrievalQuery,
        max_results: int
    ) -> List[DataRecord]:
        """从数据库中检索数据"""
        # 使用Elasticsearch进行全文搜索
        es_results = await self.elasticsearch_storage.search_documents(query, max_results)
        
        if not es_results:
            # 如果Elasticsearch没有结果，尝试MongoDB
            db_results = await self.mongodb_storage.search_metadata(query, max_results)
            self._retrieval_stats["db_hits"] += 1
        else:
            self._retrieval_stats["es_hits"] += 1
            # 从MongoDB获取完整元数据
            data_ids = [result["_source"]["data_id"] for result in es_results]
            db_results = await self.mongodb_storage.get_metadata_batch(data_ids)
        
        # 转换为DataRecord对象
        records = [DataRecord.from_metadata(metadata) for metadata in db_results]
        return records
    
    async def _retrieve_content(self, records: List[DataRecord]) -> List[DataRecord]:
        """为记录检索实际内容"""
        async with self.semaphore:  # 限制并发数
            tasks = [self._retrieve_content_for_record(record) for record in records]
            return await asyncio.gather(*tasks)
    
    async def _retrieve_content_for_record(self, record: DataRecord) -> DataRecord:
        """为单个记录检索实际内容"""
        if record.data_type == "audio" and record.metadata.get("minio_object"):
            # 从MinIO检索音频文件
```

```python
# src/lifecycle/lifecycle_manager.py
import asyncio
import logging
from typing import Dict, Any, List, Optional, Callable
from datetime import datetime, timedelta
from enum import Enum

from ..storage.mongodb_storage import MongoDBStorage
from ..storage.redis_storage import RedisStorage
from ..storage.minio_storage import MinIOStorage
from ..models.data_models import DataRecord, StorageTier
from ..utils.exceptions import LifecycleManagementError

logger = logging.getLogger(__name__)

class LifecyclePolicy(Enum):
    """生命周期策略枚举"""
    TIME_BASED = "time_based"  # 基于时间的策略
    ACCESS_BASED = "access_based"  # 基于访问频率的策略
    SIZE_BASED = "size_based"  # 基于大小的策略
    CUSTOM = "custom"  # 自定义策略

class LifecycleManager:
    """生命周期管理器 - 负责管理数据的生命周期"""
    
    def __init__(
        self,
        mongodb_storage: MongoDBStorage,
        redis_storage: RedisStorage,
        minio_storage: MinIOStorage,
        hot_data_ttl: int = 7,  # 热数据保存7天
        warm_data_ttl: int = 30,  # 温数据保存30天
        cold_data_ttl: int = 365,  # 冷数据保存1年
        check_interval: int = 24 * 60 * 60  # 检查间隔，默认24小时
    ):
        """
        初始化生命周期管理器
        
        Args:
            mongodb_storage: MongoDB存储实例
            redis_storage: Redis存储实例
            minio_storage: MinIO存储实例
            hot_data_ttl: 热数据保存时间（天）
            warm_data_ttl: 温数据保存时间（天）
            cold_data_ttl: 冷数据保存时间（天）
            check_interval: 检查间隔（秒）
        """
        self.mongodb_storage = mongodb_storage
        self.redis_storage = redis_storage
        self.minio_storage = minio_storage
        self.hot_data_ttl = timedelta(days=hot_data_ttl)
        self.warm_data_ttl = timedelta(days=warm_data_ttl)
        self.cold_data_ttl = timedelta(days=cold_data_ttl)
        self.check_interval = check_interval
        self._running = False
        self._task = None
        self._policies = {}
        self._stats = {
            "total_processed": 0,
            "hot_to_warm": 0,
            "warm_to_cold": 0,
            "cold_to_deleted": 0,
            "errors": 0
        }
    
    async def start(self):
        """启动生命周期管理器"""
        if self._running:
            logger.warning("生命周期管理器已在运行")
            return
        
        self._running = True
        self._task = asyncio.create_task(self._lifecycle_loop())
        logger.info("生命周期管理器已启动")
    
    async def stop(self):
        """停止生命周期管理器"""
        if not self._running:
            logger.warning("生命周期管理器未在运行")
            return
        
        self._running = False
        if self._task:
            self._task.cancel()
            try:
                await self._task
            except asyncio.CancelledError:
                pass
        
        logger.info("生命周期管理器已停止")
    
    async def _lifecycle_loop(self):
        """生命周期管理循环"""
        while self._running:
            try:
                await self._process_lifecycle()
                await asyncio.sleep(self.check_interval)
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"生命周期管理出错: {str(e)}")
                self._stats["errors"] += 1
                await asyncio.sleep(60)  # 出错后等待1分钟再重试
    
    async def _process_lifecycle(self):
        """处理数据生命周期"""
        logger.info("开始处理数据生命周期")
        
        # 获取所有需要处理的数据
        all_data = await self.mongodb_storage.get_all_metadata()
        
        for data in all_data:
            try:
                await self._process_single_data(data)
                self._stats["total_processed"] += 1
            except Exception as e:
                logger.error(f"处理数据 {data.get('data_id')} 生命周期出错: {str(e)}")
                self._stats["errors"] += 1
        
        logger.info(f"生命周期处理完成，处理了 {self._stats['total_processed']} 条数据")
    
    async def _process_single_data(self, data: Dict[str, Any]):
        """处理单条数据的生命周期"""
        data_id = data.get("data_id")
        current_tier = data.get("storage_tier", "hot")
        ingestion_time = data.get("ingestion_time")
        last_accessed = data.get("last_accessed", ingestion_time)
        
        if not ingestion_time:
            logger.warning(f"数据 {data_id} 缺少接入时间")
            return
        
        # 计算数据年龄
        now = datetime.utcnow()
        if isinstance(ingestion_time, str):
            ingestion_time = datetime.fromisoformat(ingestion_time)
        if isinstance(last_accessed, str):
            last_accessed = datetime.fromisoformat(last_accessed)
        
        data_age = now - ingestion_time
        time_since_access = now - last_accessed
        
        # 应用生命周期策略
        new_tier = await self._apply_lifecycle_policies(
            data, current_tier, data_age, time_since_access
        )
        
        # 如果存储层级发生变化，执行迁移
        if new_tier != current_tier:
            await self._migrate_data_tier(data, current_tier, new_tier)
    
    async def _apply_lifecycle_policies(
        self,
        data: Dict[str, Any],
        current_tier: str,
        data_age: timedelta,
        time_since_access: timedelta
    ) -> str:
        """应用生命周期策略"""
        # 默认时间策略
        if current_tier == "hot" and data_age > self.hot_data_ttl:
            return "warm"
        elif current_tier == "warm" and data_age > self.warm_data_ttl:
            return "cold"
        elif current_tier == "cold" and data_age > self.cold_data_ttl:
            return "delete"
        
        # 应用自定义策略
        data_type = data.get("data_type")
        if data_type in self._policies:
            policy_func = self._policies[data_type]
            return await policy_func(data, current_tier, data_age, time_since_access)
        
        return current_tier
    
    async def _migrate_data_tier(
        self,
        data: Dict[str, Any],
        from_tier: str,
        to_tier: str
    ):
        """迁移数据存储层级"""
        data_id = data.get("data_id")
        logger.info(f"迁移数据 {data_id} 从 {from_tier} 到 {to_tier}")
        
        if from_tier == "hot" and to_tier == "warm":
            # 从热数据迁移到温数据
            await self._migrate_hot_to_warm(data)
            self._stats["hot_to_warm"] += 1
        elif from_tier == "warm" and to_tier == "cold":
            # 从温数据迁移到冷数据
            await self._migrate_warm_to_cold(data)
            self._stats["warm_to_cold"] += 1
        elif from_tier == "cold" and to_tier == "delete":
            # 删除冷数据
            await self._delete_cold_data(data)
            self._stats["cold_to_deleted"] += 1
    
    async def _migrate_hot_to_warm(self, data: Dict[str, Any]):
        """将热数据迁移到温数据"""
        data_id = data.get("data_id")
        
        # 从Redis缓存中移除
        await self.redis_storage.delete_cached_data(data_id)
        
        # 更新MongoDB中的存储层级
        await self.mongodb_storage.update_metadata(data_id, {
            "storage_tier": "warm",
            "tier_change_time": datetime.utcnow()
        })
    
    async def _migrate_warm_to_cold(self, data: Dict[str, Any]):
        """将温数据迁移到冷数据"""
        data_id = data.get("data_id")
        
        # 更新MongoDB中的存储层级
        await self.mongodb_storage.update_metadata(data_id, {
            "storage_tier": "cold",
            "tier_change_time": datetime.utcnow()
        })
        
        # 如果需要，可以将MinIO中的数据迁移到冷存储
        # 这里简化处理，实际可能需要迁移到不同的存储类别
    
    async def _delete_cold_data(self, data: Dict[str, Any]):
        """删除冷数据"""
        data_id = data.get("data_id")
        
        # 从MinIO删除文件
        if data.get("minio_object"):
            await self.minio_storage.delete_file(data["minio_object"])
        if data.get("audio_minio_object"):
            await self.minio_storage.delete_file(data["audio_minio_object"])
        if data.get("video_minio_object"):
            await self.minio_storage.delete_file(data["video_minio_object"])
        
        # 从MongoDB删除元数据
        await self.mongodb_storage.delete_metadata(data_id)
        
        # 从Elasticsearch删除索引
        await self.elasticsearch_storage.delete_document(data_id)
    
    def register_policy(self, data_type: str, policy_func: Callable):
        """注册自定义生命周期策略"""
        self._policies[data_type] = policy_func
        logger.info(f"已为数据类型 {data_type} 注册自定义生命周期策略")
    
    def get_stats(self) -> Dict[str, Any]:
        """获取生命周期管理统计信息"""
        return self._stats.copy()
    
    async def manual_process_data(self, data_id: str):
        """手动处理单条数据的生命周期"""
        data = await self.mongodb_storage.get_metadata(data_id)
        if not data:
            raise NotFoundError(f"未找到数据: {data_id}")
        
        await self._process_single_data(data)
        logger.info(f"手动处理数据 {data_id} 生命周期完成")
```

```python
import asyncio
import logging

import base64
from typing import Dict, Any, List, Optional, Tuple

logger = logging.getLogger(__name__)

class SecurityManager:
    """安全管理器 - 负责数据安全访问和管理"""
    
    def __init__(
        self,
        mongodb_storage: MongoDBStorage,
        redis_storage: RedisStorage,
        minio_storage: MinIOStorage,
        encryption_key: Optional[bytes] = None,
        session_timeout: int = 3600  # 会话超时时间（秒）
    ):
        """
        初始化安全管理器
        
        Args:
            mongodb_storage: MongoDB存储实例
            redis_storage: Redis存储实例
            minio_storage: MinIO存储实例
            encryption_key: 加密密钥，如果为None则自动生成
            session_timeout: 会话超时时间（秒）
        """
        self.mongodb_storage = mongodb_storage
        self.redis_storage = redis_storage
        self.minio_storage = minio_storage
        self.session_timeout = session_timeout
        
        # 初始化加密
        if encryption_key is None:
            self.encryption_key = Fernet.generate_key()
        else:
            self.encryption_key = encryption_key
        
        self.cipher = Fernet(self.encryption_key)
        
        # 会话管理
        self._sessions = {}
        
        # 访问控制列表（ACL）
        self._acl = {
            "admin": ["read", "write", "delete", "manage"],
            "analyst": ["read", "write"],
            "viewer": ["read"]
        }
        
        # 统计信息
        self._stats = {
            "total_access": 0,
            "authenticated": 0,
            "unauthorized": 0,
            "encrypted_files": 0,
            "decrypted_files": 0
        }
    
    async def authenticate(
        self,
        username: str,
        password: str
    ) -> str:
        """
        用户认证
        
        Args:
            username: 用户名
            password: 密码
            
        Returns:
            会话令牌
            
        Raises:
            AuthenticationError: 认证失败
        """
        # 从数据库获取用户信息
        user = await self.mongodb_storage.get_user(username)
        if not user:
            raise AuthenticationError("用户不存在")
        
        # 验证密码
        if not self._verify_password(password, user.get("password_hash", "")):
            raise AuthenticationError("密码错误")
        
        # 创建会话
        session_id = self._generate_session_id()
        session_data = {
            "username": username,
            "role": user.get("role", "viewer"),
            "created_at": datetime.utcnow(),
            "last_accessed": datetime.utcnow()
        }
        
        # 存储会话
        await self.redis_storage.set_session(
            session_id, 
            session_data, 
            expire=self.session_timeout
        )
        
        self._stats["authenticated"] += 1
        logger.info(f"用户 {username} 认证成功，会话ID: {session_id}")
        
        return session_id
    
    async def authorize(
        self,
        session_id: str,
        required_permission: str,
        data_id: Optional[str] = None
    ) -> bool:
        """
        用户授权
        
        Args:
            session_id: 会话ID
            required_permission: 所需权限
            data_id: 数据ID（用于数据级权限控制）
            
        Returns:
            是否授权成功
            
        Raises:
            AuthorizationError: 授权失败
        """
        # 获取会话信息
        session_data = await self.redis_storage.get_session(session_id)
        if not session_data:
            raise AuthorizationError("会话不存在或已过期")
        
        # 更新最后访问时间
        session_data["last_accessed"] = datetime.utcnow()
        await self.redis_storage.set_session(
            session_id, 
            session_data, 
            expire=self.session_timeout
        )
        
        # 检查权限
        user_role = session_data.get("role", "viewer")
        user_permissions = self._acl.get(user_role, [])
        
        if required_permission not in user_permissions:
            self._stats["unauthorized"] += 1
            raise AuthorizationError(f"权限不足，需要权限: {required_permission}")
        
        # 如果指定了数据ID，检查数据级权限
        if data_id:
            data_owner = await self._get_data_owner(data_id)
            if data_owner and data_owner != session_data.get("username"):
                # 检查是否有访问他人数据的权限
                if "access_all" not in user_permissions:
                    self._stats["unauthorized"] += 1
                    raise AuthorizationError("无权访问此数据")
        
        self._stats["total_access"] += 1
        return True
    
    async def encrypt_file(self, file_path: str) -> str:
        """
        加密文件
        
        Args:
            file_path: 文件路径
            
        Returns:
            加密后的文件路径
        """
        try:
            # 读取文件
            with open(file_path, "rb") as f:
                file_data = f.read()
            
            # 加密数据
            encrypted_data = self.cipher.encrypt(file_data)
            
            # 保存加密文件
            encrypted_path = f"{file_path}.encrypted"
            with open(encrypted_path, "wb") as f:
                f.write(encrypted_data)
            
            self._stats["encrypted_files"] += 1
            logger.info(f"文件加密完成: {file_path} -> {encrypted_path}")
            
            return encrypted_path
            
        except Exception as e:
            logger.error(f"文件加密失败: {str(e)}")
            raise SecurityError(f"文件加密失败: {str(e)}")
    
    async def decrypt_file(self, encrypted_file_path: str) -> str:
        """
        解密文件
        
        Args:
            encrypted_file_path: 加密文件路径
            
        Returns:
            解密后的文件路径
        """
        try:
            # 读取加密文件
            with open(encrypted_file_path, "rb") as f:
                encrypted_data = f.read()
            
            # 解密数据
            decrypted_data = self.cipher.decrypt(encrypted_data)
            
            # 保存解密文件
            if encrypted_file_path.endswith(".encrypted"):
                decrypted_path = encrypted_file_path[:-10]  # 移除".encrypted"
            else:
                decrypted_path = f"{encrypted_file_path}.decrypted"
            
            with open(decrypted_path, "wb") as f:
                f.write(decrypted_data)
            
            self._stats["decrypted_files"] += 1
            logger.info(f"文件解密完成: {encrypted_file_path} -> {decrypted_path}")
            
            return decrypted_path
            
        except Exception as e:
            logger.error(f"文件解密失败: {str(e)}")
            raise SecurityError(f"文件解密失败: {str(e)}")
    
    async def log_access(
        self,
        session_id: str,
        data_id: str,
        action: str,
        result: str
    ):
        """
        记录访问日志
        
        Args:
            session_id: 会话ID
            data_id: 数据ID
            action: 操作类型
            result: 操作结果
        """
        try:
            # 获取会话信息
            session_data = await self.redis_storage.get_session(session_id)
            if not session_data:
                return
            
            # 创建访问日志
            access_log = AccessLog(
                username=session_data.get("username"),
                data_id=data_id,
                action=action,
                result=result,
                timestamp=datetime.utcnow(),
                ip_address=session_data.get("ip_address", "unknown")
            )
            
            # 存储访问日志
            await self.mongodb_storage.store_access_log(access_log.to_dict())
            
        except Exception as e:
            logger.error(f"记录访问日志失败: {str(e)}")
    
    def _generate_session_id(self) -> str:
        """生成会话ID"""
        import uuid
        return str(uuid.uuid4())
    
    def _verify_password(self, password: str, password_hash: str) -> bool:
        """验证密码"""
        # 这里简化处理，实际应该使用更安全的密码验证方法
        return hmac.compare_digest(
            hashlib.sha256(password.encode()).hexdigest(),
            password_hash
        )
    
    async def _get_data_owner(self, data_id: str) -> Optional[str]:
        """获取数据所有者"""
        metadata = await self.mongodb_storage.get_metadata(data_id)
        if metadata:
            return metadata.get("owner")
        return None
    
    def get_stats(self) -> Dict[str, Any]:
        """获取安全管理统计信息"""
        return self._stats.copy()
```

logger = logging.getLogger(__name__)

class DataRetriever:
    """数据检索器，提供高效的数据检索和查询功能"""
    
    def __init__(
        self,
        mongodb_storage: MongoDBStorage,
        redis_storage: RedisStorage,
        minio_storage: MinIOStorage,
        elasticsearch_storage: ElasticsearchStorage
    ):
        self.mongodb_storage = mongodb_storage
        self.redis_storage = redis_storage
        self.minio_storage = minio_storage
        self.elasticsearch_storage = elasticsearch_storage
        self._retrieval_stats = {
            "total_queries": 0,
            "cache_hits": 0,
            "cache_misses": 0,
            "avg_response_time": 0
        }
    
    async def retrieve_by_id(self, data_id: str, include_content: bool = False) -> Dict[str, Any]:
        """
        根据ID检索数据
        
        Args:
            data_id: 数据ID
            include_content: 是否包含文件内容
            
        Returns:
            检索到的数据
            
        Raises:
            DataRetrievalError: 数据检索失败
        """
        start_time = datetime.now()
        self._retrieval_stats["total_queries"] += 1
        
        try:
            # 首先从Redis缓存中查找
            cache_key = f"data:{data_id}"
            cached_data = await self.redis_storage.get_cached_data(cache_key)
            
            if cached_data:
                self._retrieval_stats["cache_hits"] += 1
                logger.debug(f"从缓存中检索到数据: {data_id}")
                
                # 如果需要文件内容，从MinIO获取
                if include_content:
                    await self._add_file_content(cached_data)
                
                return cached_data
            else:
                self._retrieval_stats["cache_misses"] += 1
                
                # 从MongoDB中查询
                data = await self.mongodb_storage.get_metadata(data_id)
                if not data:
                    raise DataRetrievalError(f"未找到数据: {data_id}")
                
                # 如果需要文件内容，从MinIO获取
                if include_content:
                    await self._add_file_content(data)
                
                # 缓存查询结果
                await self.redis_storage.cache_data(cache_key, data)
                
                return data
                
        except Exception as e:
            logger.error(f"数据检索失败: {str(e)}")
            raise DataRetrievalError(f"数据检索失败: {str(e)}")
        finally:
            # 更新平均响应时间
            response_time = (datetime.now() - start_time).total_seconds()
            self._update_avg_response_time(response_time)
    
    async def retrieve_by_time_range(
        self,
        start_time: datetime,
        end_time: datetime,
        data_type: Optional[str] = None,
        limit: int = 100,
        offset: int = 0,
        include_content: bool = False
    ) -> List[Dict[str, Any]]:
        """
        根据时间范围检索数据
        
        Args:
            start_time: 开始时间
            end_time: 结束时间
            data_type: 数据类型 (audio, video, multimodal)
            limit: 返回结果数量限制
            offset: 偏移量
            include_content: 是否包含文件内容
            
        Returns:
            检索到的数据列表
        """
        start_time_query = start_time
        cache_key = f"range:{start_time_query.isoformat()}:{end_time.isoformat()}:{data_type}:{limit}:{offset}"
        
        # 首先从Redis缓存中查找
        cached_result = await self.redis_storage.get_cached_data(cache_key)
        if cached_result:
            self._retrieval_stats["cache_hits"] += 1
            logger.debug(f"从缓存中检索到时间范围数据")
            
            # 如果需要文件内容，从MinIO获取
            if include_content:
                for data in cached_result:
                    await self._add_file_content(data)
            
            return cached_result
        
        # 从MongoDB中查询
        results = await self.mongodb_storage.query_by_time_range(
            start_time_query, end_time, data_type, limit, offset
        )
        
        # 缓存查询结果
        await self.redis_storage.cache_data(cache_key, results, ttl=3600)  # 缓存1小时
        
        self._retrieval_stats["cache_misses"] += 1
        
        # 如果需要文件内容，从MinIO获取
        if include_content:
            for data in results:
                await self._add_file_content(data)
        
        return results
    
    async def retrieve_by_content(
        self,
        query: str,
        data_type: Optional[str] = None,
        limit: int = 100,
        offset: int = 0,
        include_content: bool = False
    ) -> List[Dict[str, Any]]:
        """
        根据内容检索数据
        
        Args:
            query: 查询内容

            limit: 返回结果数量限制
            offset: 偏移量
            include_content: 是否包含文件内容
            
        Returns:

        # 使用Elasticsearch进行全文搜索
        search_results = await self.elasticsearch_storage.search_content(

        )
        
        # 从搜索结果中提取数据ID
        data_ids = [result["_id"] for result in search_results]

        results = await self.mongodb_storage.get_by_ids(data_ids)
        
        # 如果需要文件内容，从MinIO获取
        if include_content:
            for data in results:
                await self._add_file_content(data)

        total_queries = self._retrieval_stats["total_queries"]
        current_avg = self._retrieval_stats["avg_response_time"]
        
        # 计算新的平均响应时间
        new_avg = (current_avg * (total_queries - 1) + response_time) / total_queries
        self._retrieval_stats["avg_response_time"] = new_avg
    
    def get_retrieval_stats(self) -> Dict[str, Any]:
        """获取检索统计信息"""
        stats = self._retrieval_stats.copy()
        
        # 计算缓存命中率
        total_requests = stats["cache_hits"] + stats["cache_misses"]
        if total_requests > 0:
            stats["cache_hit_rate"] = stats["cache_hits"] / total_requests
        else:
            stats["cache_hit_rate"] = 0
            
        return stats

```python
import asyncio

from enum import Enum

from ..storage.mongodb_storage import MongoDBStorage

logger = logging.getLogger(__name__)

class StorageTier(Enum):
    """存储层级枚举"""
    HOT = "hot"      # 热数据：频繁访问，存储在高速存储
    WARM = "warm"    # 温数据：较少访问，存储在标准存储
    COLD = "cold"    # 冷数据：很少访问，存储在低成本存储
    ARCHIVED = "archived"  # 归档数据：长期保存，压缩存储

class LifecycleManager:
    """生命周期管理器，负责数据的生命周期管理"""
    
    def __init__(
        self,
        mongodb_storage: MongoDBStorage,
        minio_storage: MinIOStorage,
        hot_threshold_days: int = 7,
        warm_threshold_days: int = 30,
        cold_threshold_days: int = 90,
        archived_threshold_days: int = 365
    ):
        self.mongodb_storage = mongodb_storage
        self.minio_storage = minio_storage
        
        # 存储层级阈值（天数）
        self.hot_threshold = timedelta(days=hot_threshold_days)
        self.warm_threshold = timedelta(days=warm_threshold_days)
        self.cold_threshold = timedelta(days=cold_threshold_days)
        self.archived_threshold = timedelta(days=archived_threshold_days)
        
        self._lifecycle_stats = {
            "total_processed": 0,
            "hot_to_warm": 0,
            "warm_to_cold": 0,
            "cold_to_archived": 0,
            "deleted": 0
        }
    
    async def manage_data_lifecycle(self, batch_size: int = 100) -> Dict[str, Any]:
        """
        管理数据生命周期
        
        Args:
            batch_size: 每批处理的数据量
            
        Returns:
            生命周期管理统计信息
        """
        try:
            current_time = datetime.utcnow()
            
            # 1. 查找需要从热数据迁移到温数据的数据
            hot_to_warm_data = await self.mongodb_storage.find_data_for_tier_migration(
                current_time - self.hot_threshold,
                StorageTier.HOT.value,
                batch_size
            )
            
            # 2. 查找需要从温数据迁移到冷数据的数据
            warm_to_cold_data = await self.mongodb_storage.find_data_for_tier_migration(
                current_time - self.warm_threshold,
                StorageTier.WARM.value,
                batch_size
            )
            
            # 3. 查找需要从冷数据归档的数据
            cold_to_archived_data = await self.mongodb_storage.find_data_for_tier_migration(
                current_time - self.cold_threshold,
                StorageTier.COLD.value,
                batch_size
            )
            
            # 4. 查找需要删除的过期归档数据
            expired_archived_data = await self.mongodb_storage.find_data_for_tier_migration(
                current_time - self.archived_threshold,
                StorageTier.ARCHIVED.value,
                batch_size
            )
            
            # 执行数据迁移和删除
            await self._migrate_hot_to_warm(hot_to_warm_data)
            await self._migrate_warm_to_cold(warm_to_cold_data)
            await self._archive_cold_data(cold_to_archived_data)
            await self._delete_expired_archived_data(expired_archived_data)
            
            return self.get_lifecycle_stats()
            
        except Exception as e:
            logger.error(f"生命周期管理失败: {str(e)}")
            raise LifecycleManagementError(f"生命周期管理失败: {str(e)}")

                if "minio_object" in data:

                    new_object_name = old_object_name.replace("hot/", "warm/")
                    await self.minio_storage.move_object(old_object_name, new_object_name)
                
                # 从Redis缓存中移除
                await self.mongodb_storage.remove_from_cache(data_id)
                
                self._lifecycle_stats["hot_to_warm"] += 1
                self._lifecycle_stats["total_processed"] += 1
                
                logger.info(f"数据 {data_id} 已从热数据迁移到温数据")
                
            except Exception as e:
                logger.error(f"热数据迁移到温数据失败: {str(e)}")

            current_tier = data.get("storage_tier", StorageTier.HOT.value)
            
            # 检查是否可以提升
            if not self._can_promote(current_tier, target_tier):
                logger.warning(f"无法从 {current_tier} 提升到 {target_tier.value}")
                return False
            
            # 更新存储层级
            await self.mongodb_storage.update_storage_tier(data_id, target_tier.value)
            
            # 在MinIO中移动文件
            if "minio_object" in data:
                old_object_name = data["minio_object"]
                new_object_name = self._get_tier_object_name(old_object_name, target_tier)
                await self.minio_storage.move_object(old_object_name, new_object_name)
                
                # 如果提升到热数据，添加到Redis缓存
                if target_tier == StorageTier.HOT:
                    await self.mongodb_storage.add_to_cache(data_id)
            
            logger.info(f"数据 {data_id} 已提升到 {target_tier.value}")
            return True
            
        except Exception as e:
            logger.error(f"提升数据层级失败: {str(e)}")
            return False
    
    def _can_promote(self, current_tier: str, target_tier: StorageTier) -> bool:
        """检查是否可以提升存储层级"""
        tier_hierarchy = {
            StorageTier.HOT.value: 0,
            StorageTier.WARM.value: 1,
            StorageTier.COLD.value: 2,
            StorageTier.ARCHIVED.value: 3
        }
        
        current_level = tier_hierarchy.get(current_tier, 0)
        target_level = tier_hierarchy.get(target_tier.value, 0)
        
        return target_level < current_level
    
    def _get_tier_object_name(self, object_name: str, tier: StorageTier) -> str:
        """获取存储层级对应的对象名称"""
        # 替换存储层级前缀
        for t in StorageTier:
            if object_name.startswith(f"{t.value}/"):
                return object_name.replace(f"{t.value}/", f"{tier.value}/", 1)
        
        # 如果没有前缀，添加目标层级前缀
        return f"{tier.value}/{object_name}"
    
    def get_lifecycle_stats(self) -> Dict[str, Any]:
        """获取生命周期管理统计信息"""
        return self._lifecycle_stats.copy()
    
    def reset_stats(self):
        """重置统计信息"""
        self._lifecycle_stats = {
            "total_processed": 0,
            "hot_to_warm": 0,
            "warm_to_cold": 0,
            "cold_to_archived": 0,
            "deleted": 0
        }
```

```python
import asyncio
import logging

from typing import Dict, Any, Callable
from datetime import datetime

from ..ingestion.data_ingestor import DataIngestor
from ..retrieval.data_retriever import DataRetriever
from ..management.lifecycle_manager import LifecycleManager
from ..utils.exceptions import PerceptionDataException

logger = logging.getLogger(__name__)

class EventHandler:
    """事件处理器，处理系统内部事件"""
    
    def __init__(
        self,
        data_ingestor: DataIngestor,
        data_retriever: DataRetriever,
        lifecycle_manager: LifecycleManager
    ):
        self.data_ingestor = data_ingestor
        self.data_retriever = data_retriever
        self.lifecycle_manager = lifecycle_manager
        
        # 事件处理器映射
        self.event_handlers = {
            "data_ingestion_request": self._handle_data_ingestion_request,
            "data_retrieval_request": self._handle_data_retrieval_request,
            "lifecycle_management_request": self._handle_lifecycle_management_request,
            "data_accessed": self._handle_data_accessed,
            "system_maintenance": self._handle_system_maintenance
        }
    
    async def handle_event(self, event_type: str, event_data: Dict[str, Any]) -> Dict[str, Any]:
        """
        处理事件
        
        Args:
            event_type: 事件类型
            event_data: 事件数据
            
        Returns:
            处理结果
        """
        try:
            if event_type not in self.event_handlers:
                raise PerceptionDataException(f"不支持的事件类型: {event_type}")
            
            handler = self.event_handlers[event_type]
            result = await handler(event_data)
            
            logger.info(f"事件处理成功: {event_type}")
            return {
                "success": True,
                "event_type": event_type,
                "timestamp": datetime.utcnow().isoformat(),
                "result": result
            }
            
        except Exception as e:
            logger.error(f"事件处理失败: {str(e)}")
            return {
                "success": False,
                "event_type": event_type,
                "timestamp": datetime.utcnow().isoformat(),
                "error": str(e)
            }
    
    async def _handle_data_ingestion_request(self, event_data: Dict[str, Any]) -> Dict[str, Any]:
        """处理数据接入请求事件"""
        data_type = event_data.get("data_type")
        data = event_data.get("data")
        metadata = event_data.get("metadata")
        
        result = await self.data_ingestor.ingest_data(data, data_type, metadata)
        return result
    
    async def _handle_data_retrieval_request(self, event_data: Dict[str, Any]) -> Dict[str, Any]:
        """处理数据检索请求事件"""
        retrieval_type = event_data.get("type", "by_id")
        
        if retrieval_type == "by_id":
            data_id = event_data.get("data_id")
            include_content = event_data.get("include_content", False)
```
