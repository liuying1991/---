### 功能概述
记忆管理模块是真实婴儿AI管家系统的记忆核心，负责管理和协调系统的各种记忆功能，包括短期记忆、长期记忆和工作记忆。该模块提供统一的记忆接口，支持记忆的存储、检索、更新和遗忘，确保记忆系统的高效运行和数据的完整性。

```python
# src/models/memory.py
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Dict, List, Optional, Any
import uuid

class MemoryType(Enum):
    """记忆类型枚举"""
    SHORT_TERM = "short_term"
    LONG_TERM = "long_term"
    WORKING = "working"

class MemoryStatus(Enum):
    """记忆状态枚举"""
    ACTIVE = "active"
    ARCHIVED = "archived"
    DELETED = "deleted"

class MemoryPriority(Enum):
    """记忆优先级枚举"""
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    CRITICAL = 4

@dataclass
class MemoryMetadata:
    """记忆元数据"""
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    last_accessed_at: Optional[datetime] = None
    access_count: int = 0
    importance_score: float = 0.5  # 0.0-1.0
    relevance_score: float = 0.5  # 0.0-1.0
    tags: List[str] = field(default_factory=list)
    source: Optional[str] = None
    context: Dict[str, Any] = field(default_factory=dict)
    retention_policy: Optional[str] = None

@dataclass
class Memory:
    """记忆数据模型"""
    id: str = field(default_factory=lambda: str(uuid.uuid4()))
    type: MemoryType = MemoryType.SHORT_TERM
    status: MemoryStatus = MemoryStatus.ACTIVE
    priority: MemoryPriority = MemoryPriority.MEDIUM
    content: Dict[str, Any] = field(default_factory=dict)
    metadata: MemoryMetadata = field(default_factory=MemoryMetadata)
    embedding: Optional[List[float]] = None
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "id": self.id,
            "type": self.type.value,
            "status": self.status.value,
            "priority": self.priority.value,
            "content": self.content,
            "metadata": {
                "created_at": self.metadata.created_at.isoformat(),
                "updated_at": self.metadata.updated_at.isoformat(),
                "last_accessed_at": self.metadata.last_accessed_at.isoformat() if self.metadata.last_accessed_at else None,
                "access_count": self.metadata.access_count,
                "importance_score": self.metadata.importance_score,
                "relevance_score": self.metadata.relevance_score,
                "tags": self.metadata.tags,
                "source": self.metadata.source,
                "context": self.metadata.context,
                "retention_policy": self.metadata.retention_policy
            },
            "embedding": self.embedding
        }
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "Memory":
        """从字典创建记忆对象"""
        metadata = MemoryMetadata(
            created_at=datetime.fromisoformat(data["metadata"]["created_at"]),
            updated_at=datetime.fromisoformat(data["metadata"]["updated_at"]),
            last_accessed_at=datetime.fromisoformat(data["metadata"]["last_accessed_at"]) if data["metadata"]["last_accessed_at"] else None,
            access_count=data["metadata"]["access_count"],
            importance_score=data["metadata"]["importance_score"],
            relevance_score=data["metadata"]["relevance_score"],
            tags=data["metadata"]["tags"],
            source=data["metadata"]["source"],
            context=data["metadata"]["context"],
            retention_policy=data["metadata"]["retention_policy"]
        )
        
        return cls(
            id=data["id"],
            type=MemoryType(data["type"]),
            status=MemoryStatus(data["status"]),
            priority=MemoryPriority(data["priority"]),
            content=data["content"],
            metadata=metadata,
            embedding=data.get("embedding")
        )
```

#### 向量存储实现
```python
# src/storage/vector_storage.py
from abc import ABC, abstractmethod
from typing import List, Dict, Any, Optional, Tuple
import numpy as np
from pymilvus import connections, Collection, FieldSchema, CollectionSchema, DataType, utility
import chromadb
from chromadb.config import Settings
import faiss
import logging

logger = logging.getLogger(__name__)

class VectorStorage(ABC):
    """向量存储抽象基类"""
    
    @abstractmethod
    def connect(self) -> bool:
        """连接到向量数据库"""
        pass
    
    @abstractmethod
    def disconnect(self) -> bool:
        """断开向量数据库连接"""
        pass
    
    @abstractmethod
    def create_collection(self, collection_name: str, dimension: int, index_type: str = "IVF_FLAT") -> bool:
        """创建集合"""
        pass
    
    @abstractmethod
    def insert(self, collection_name: str, ids: List[str], embeddings: List[List[float]], metadatas: List[Dict[str, Any]]) -> bool:
        """插入向量数据"""
        pass
    
    @abstractmethod
    def search(self, collection_name: str, query_embedding: List[float], limit: int = 10, filter_expr: Optional[str] = None) -> List[Dict[str, Any]]:
        """搜索向量数据"""
        pass
    
    @abstractmethod
    def delete(self, collection_name: str, ids: List[str]) -> bool:
        """删除向量数据"""
        pass
    
    @abstractmethod
    def update(self, collection_name: str, ids: List[str], embeddings: List[List[float]], metadatas: List[Dict[str, Any]]) -> bool:
        """更新向量数据"""
        pass

class MilvusVectorStorage(VectorStorage):
    """Milvus向量存储实现"""
    
    def __init__(self, host: str = "localhost", port: int = 19530):
        self.host = host
        self.port = port
        self.connected = False
    
    def connect(self) -> bool:
        """连接到Milvus服务器"""
        try:
            connections.connect(host=self.host, port=self.port)
            self.connected = True
            logger.info(f"Connected to Milvus at {self.host}:{self.port}")
            return True
        except Exception as e:
            logger.error(f"Failed to connect to Milvus: {str(e)}")
            return False
    
    def disconnect(self) -> bool:
        """断开Milvus连接"""
        try:
            connections.disconnect("default")
            self.connected = False
            logger.info("Disconnected from Milvus")
            return True
        except Exception as e:
            logger.error(f"Failed to disconnect from Milvus: {str(e)}")
            return False
    
    def create_collection(self, collection_name: str, dimension: int, index_type: str = "IVF_FLAT") -> bool:
        """创建Milvus集合"""
        try:
            # 检查集合是否已存在
            if utility.has_collection(collection_name):
                logger.info(f"Collection {collection_name} already exists")
                return True
            
            # 定义字段
            id_field = FieldSchema(name="id", dtype=DataType.VARCHAR, max_length=100, is_primary=True)
            embedding_field = FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=dimension)
            metadata_field = FieldSchema(name="metadata", dtype=DataType.JSON)
            
            # 创建集合模式
            schema = CollectionSchema(
                fields=[id_field, embedding_field, metadata_field],
                description=f"Collection for {collection_name}"
            )
            
            # 创建集合
            collection = Collection(name=collection_name, schema=schema)
            
            # 创建索引
            index_params = {
                "metric_type": "COSINE",
                "index_type": index_type,
                "params": {"nlist": 128}
            }
            collection.create_index(field_name="embedding", index_params=index_params)
            
            logger.info(f"Created collection {collection_name} with dimension {dimension} and index type {index_type}")
            return True
        except Exception as e:
            logger.error(f"Failed to create collection {collection_name}: {str(e)}")
            return False
    
    def insert(self, collection_name: str, ids: List[str], embeddings: List[List[float]], metadatas: List[Dict[str, Any]]) -> bool:
        """插入向量数据到Milvus"""
        try:
            collection = Collection(collection_name)
            
            # 准备数据
            data = [
                ids,
                embeddings,
                metadatas
            ]
            
            # 插入数据
            insert_result = collection.insert(data)
            collection.flush()
            
            logger.info(f"Inserted {len(ids)} vectors into collection {collection_name}")
            return True
        except Exception as e:
            logger.error(f"Failed to insert vectors into collection {collection_name}: {str(e)}")
            return False
    
    def search(self, collection_name: str, query_embedding: List[float], limit: int = 10, filter_expr: Optional[str] = None) -> List[Dict[str, Any]]:
        """在Milvus中搜索向量"""
        try:
            collection = Collection(collection_name)
            collection.load()
            
            # 设置搜索参数
            search_params = {
                "metric_type": "COSINE",
                "params": {"nprobe": 10}
            }
            
            # 执行搜索
            results = collection.search(
                data=[query_embedding],
                anns_field="embedding",
                param=search_params,
                limit=limit,
                expr=filter_expr,
                output_fields=["id", "metadata"]
            )
            
            # 处理结果
            search_results = []
            for hit in results[0]:
                search_results.append({
                    "id": hit.entity.get("id"),
                    "score": hit.score,
                    "metadata": hit.entity.get("metadata")
                })
            
            logger.info(f"Found {len(search_results)} results in collection {collection_name}")
            return search_results
        except Exception as e:
            logger.error(f"Failed to search in collection {collection_name}: {str(e)}")
            return []
    
    def delete(self, collection_name: str, ids: List[str]) -> bool:
        """从Milvus删除向量数据"""
        try:
            collection = Collection(collection_name)
            
            # 构建删除表达式
            id_list = "', '".join(ids)
            expr = f"id in ['{id_list}']"
            
            # 执行删除
            collection.delete(expr)
            collection.flush()
            
            logger.info(f"Deleted {len(ids)} vectors from collection {collection_name}")
            return True
        except Exception as e:
            logger.error(f"Failed to delete vectors from collection {collection_name}: {str(e)}")
            return False
    
    def update(self, collection_name: str, ids: List[str], embeddings: List[List[float]], metadatas: List[Dict[str, Any]]) -> bool:
        """更新Milvus中的向量数据"""
        try:
            # Milvus不支持直接更新，需要先删除再插入
            self.delete(collection_name, ids)
            self.insert(collection_name, ids, embeddings, metadatas)
            
            logger.info(f"Updated {len(ids)} vectors in collection {collection_name}")
            return True
        except Exception as e:
            logger.error(f"Failed to update vectors in collection {collection_name}: {str(e)}")
            return False

class FaissVectorStorage(VectorStorage):
    """Faiss向量存储实现"""
    
    def __init__(self, index_type: str = "IVF_FLAT", dimension: int = 768):
        self.index_type = index_type
        self.dimension = dimension
        self.index = None
        self.id_map = {}  # ID到索引的映射
        self.metadata_map = {}  # ID到元数据的映射
        self.connected = False
    
    def connect(self) -> bool:
        """初始化Faiss索引"""
        try:
            if self.index_type == "IVF_FLAT":
                quantizer = faiss.IndexFlatIP(self.dimension)
                self.index = faiss.IndexIVFFlat(quantizer, self.dimension, 100)
            elif self.index_type == "HNSW":
                self.index = faiss.IndexHNSWFlat(self.dimension, 32)
            else:
                self.index = faiss.IndexFlatIP(self.dimension)
            
            self.connected = True
            logger.info(f"Initialized Faiss index with type {self.index_type} and dimension {self.dimension}")
            return True
        except Exception as e:
            logger.error(f"Failed to initialize Faiss index: {str(e)}")
            return False
    
    def disconnect(self) -> bool:
        """释放Faiss索引"""
        try:
            self.index = None
            self.id_map = {}
            self.metadata_map = {}
            self.connected = False
            logger.info("Released Faiss index")
            return True
        except Exception as e:
            logger.error(f"Failed to release Faiss index: {str(e)}")
            return False
    
    def create_collection(self, collection_name: str, dimension: int, index_type: str = "IVF_FLAT") -> bool:
        """Faiss不需要显式创建集合"""
        self.dimension = dimension
        self.index_type = index_type
        return self.connect()
    
    def insert(self, collection_name: str, ids: List[str], embeddings: List[List[float]], metadatas: List[Dict[str, Any]]) -> bool:
        """插入向量数据到Faiss"""
        try:
            # 转换为numpy数组
            embeddings_np = np.array(embeddings).astype('float32')
            
            # 归一化向量（使用内积需要归一化）
            faiss.normalize_L2(embeddings_np)
            
            # 添加到索引
            if self.index_type == "IVF_FLAT" and not self.index.is_trained:
                self.index.train(embeddings_np)
            
            start_idx = len(self.id_map)
            self.index.add(embeddings_np)
            
            # 更新映射
            for i, id in enumerate(ids):
                self.id_map[id] = start_idx + i
                self.metadata_map[id] = metadatas[i]
            
            logger.info(f"Inserted {len(ids)} vectors into Faiss index")
            return True
        except Exception as e:
            logger.error(f"Failed to insert vectors into Faiss index: {str(e)}")
            return False
    
    def search(self, collection_name: str, query_embedding: List[float], limit: int = 10, filter_expr: Optional[str] = None) -> List[Dict[str, Any]]:
        """在Faiss中搜索向量"""
        try:
            # 转换为numpy数组并归一化
            query_np = np.array([query_embedding]).astype('float32')
            faiss.normalize_L2(query_np)
            
            # 执行搜索
            scores, indices = self.index.search(query_np, limit)
            
            # 处理结果
            search_results = []
            for i, idx in enumerate(indices[0]):
                if idx == -1:  # 无效索引
                    continue
                
                # 查找对应的ID
                id = None
                for mem_id, mem_idx in self.id_map.items():
                    if mem_idx == idx:
                        id = mem_id
                        break
                
                if id and (not filter_expr or eval(filter_expr, {"metadata": self.metadata_map.get(id, {})})):
                    search_results.append({
                        "id": id,
                        "score": float(scores[0][i]),
                        "metadata": self.metadata_map.get(id, {})
                    })
            
            logger.info(f"Found {len(search_results)} results in Faiss index")
            return search_results
        except Exception as e:
            logger.error(f"Failed to search in Faiss index: {str(e)}")
            return []
    
    def delete(self, collection_name: str, ids: List[str]) -> bool:
        """从Faiss删除向量数据"""
        try:
            # Faiss不支持直接删除，需要重建索引
            # 这里简化处理，仅从映射中删除
            for id in ids:
                if id in self.id_map:
                    del self.id_map[id]
                if id in self.metadata_map:
                    del self.metadata_map[id]
            
            logger.info(f"Deleted {len(ids)} vectors from Faiss index mappings")
            return True
        except Exception as e:
            logger.error(f"Failed to delete vectors from Faiss index: {str(e)}")
            return False
    
    def update(self, collection_name: str, ids: List[str], embeddings: List[List[float]], metadatas: List[Dict[str, Any]]) -> bool:
        """更新Faiss中的向量数据"""
        try:
            # Faiss不支持直接更新，需要先删除再插入
            self.delete(collection_name, ids)
            self.insert(collection_name, ids, embeddings, metadatas)
            
            logger.info(f"Updated {len(ids)} vectors in Faiss index")
            return True
        except Exception as e:
            logger.error(f"Failed to update vectors in Faiss index: {str(e)}")
            return False
```

```python
class TextEmbedding(ABC):
    """文本嵌入抽象基类"""
    
    @abstractmethod
    def embed(self, texts: List[str]) -> List[List[float]]:
        """生成文本嵌入向量"""
        pass
    
    @abstractmethod
    def get_dimension(self) -> int:
        """获取嵌入向量维度"""

class SentenceTransformerEmbedding(TextEmbedding):
    """SentenceTransformer文本嵌入实现"""
    
    def __init__(self, model_name: str = "all-MiniLM-L6-v2"):
        self.model_name = model_name
        self.model = None
        self._load_model()
    
    def _load_model(self):
        """加载模型"""
        try:
            self.model = SentenceTransformer(self.model_name)
            logger.info(f"Loaded SentenceTransformer model: {self.model_name}")
        except Exception as e:
            logger.error(f"Failed to load SentenceTransformer model {self.model_name}: {str(e)}")
            raise
    
    def embed(self, texts: List[str]) -> List[List[float]]:
        """生成文本嵌入向量"""
        try:
            embeddings = self.model.encode(texts, convert_to_tensor=False)
            return embeddings.tolist()
        except Exception as e:
            logger.error(f"Failed to generate embeddings: {str(e)}")
            return []
    
    def get_dimension(self) -> int:
        """获取嵌入向量维度"""
        return self.model.get_sentence_embedding_dimension()

class OpenAIEmbedding(TextEmbedding):
    """OpenAI文本嵌入实现"""
    
    def __init__(self, model_name: str = "text-embedding-ada-002", api_key: Optional[str] = None):
        self.model_name = model_name

class HuggingFaceEmbedding(TextEmbedding):
    """HuggingFace文本嵌入实现"""
    
    def __init__(self, model_name: str = "sentence-transformers/all-MiniLM-L6-v2"):
        self.model_name = model_name
        self.tokenizer = None
        self.model = None
        self.device = "cuda" if torch.cuda.is_available() else "cpu"
        self._load_model()
    
    def _load_model(self):
        """加载模型"""
        try:
            from transformers import AutoTokenizer, AutoModel
            self.tokenizer = AutoTokenizer.from_pretrained(self.model_name)
            self.model = AutoModel.from_pretrained(self.model_name)
            self.model.to(self.device)
            logger.info(f"Loaded HuggingFace model: {self.model_name}")
        except Exception as e:
            logger.error(f"Failed to load HuggingFace model {self.model_name}: {str(e)}")
            raise
    
    def embed(self, texts: List[str]) -> List[List[float]]:
        """生成文本嵌入向量"""
        try:
            # 分词
            inputs = self.tokenizer(texts, padding=True, truncation=True, return_tensors="pt")
            inputs = {k: v.to(self.device) for k, v in inputs.items()}
            
            # 获取模型输出
            with torch.no_grad():
                outputs = self.model(**inputs)
                
                # 使用平均池化获取句子嵌入
                embeddings = self.mean_pooling(outputs.last_hidden_state, inputs["attention_mask"])
                
                # 归一化
                embeddings = torch.nn.functional.normalize(embeddings, p=2, dim=1)
            
            return embeddings.cpu().numpy().tolist()
        except Exception as e:
            logger.error(f"Failed to generate embeddings with HuggingFace: {str(e)}")
            return []
    
    def mean_pooling(self, last_hidden_state: torch.Tensor, attention_mask: torch.Tensor) -> torch.Tensor:
        """平均池化"""
        input_mask_expanded = attention_mask.unsqueeze(-1).expand(last_hidden_state.size()).float()
        sum_embeddings = torch.sum(last_hidden_state * input_mask_expanded, 1)
        sum_mask = torch.clamp(input_mask_expanded.sum(1), min=1e-9)
        return sum_embeddings / sum_mask
    
    def get_dimension(self) -> int:
        """获取嵌入向量维度"""
        return self.model.config.hidden_size
```

```python
class ForgettingStrategy(ABC):
    """遗忘策略抽象基类"""
    
    @abstractmethod

class TimeBasedForgetting(ForgettingStrategy):
    """基于时间的遗忘策略"""
    
    def __init__(self, short_term_retention_days: int = 7, long_term_retention_days: int = 365):
        self.short_term_retention_days = short_term_retention_days
        self.long_term_retention_days = long_term_retention_days
    
    def should_forget(self, memory: Memory) -> bool:
        """判断是否应该遗忘记忆"""
        now = datetime.now()
        created_at = memory.metadata.created_at
        
        if memory.type == MemoryType.SHORT_TERM:
            retention_period = timedelta(days=self.short_term_retention_days)
        elif memory.type == MemoryType.LONG_TERM:
            retention_period = timedelta(days=self.long_term_retention_days)
        else:
            # 工作记忆不基于时间遗忘
            return False
        
        return (now - created_at) > retention_period
    
    def get_forgetting_score(self, memory: Memory) -> float:
        """获取遗忘分数"""
        now = datetime.now()
        created_at = memory.metadata.created_at
        
        if memory.type == MemoryType.SHORT_TERM:
            retention_period = timedelta(days=self.short_term_retention_days)
        elif memory.type == MemoryType.LONG_TERM:
            retention_period = timedelta(days=self.long_term_retention_days)
        else:
            # 工作记忆不基于时间遗忘
            return 0.0
        
        elapsed = (now - created_at).total_seconds()
        retention_seconds = retention_period.total_seconds()
        
        # 使用线性函数计算遗忘分数，超过保留期后分数增加
        if elapsed < retention_seconds:
            return 0.0
        else:
            # 超过保留期后，分数线性增加，最大为1.0
            return min(1.0, (elapsed - retention_seconds) / retention_seconds)

class AccessBasedForgetting(ForgettingStrategy):
    """基于访问频率的遗忘策略"""
    
    def __init__(self, min_access_count: int = 5, days_since_last_access: int = 30):
        self.min_access_count = min_access_count
        self.days_since_last_access = days_since_last_access
    
    def should_forget(self, memory: Memory) -> bool:
        """判断是否应该遗忘记忆"""
        # 工作记忆不基于访问频率遗忘
        if memory.type == MemoryType.WORKING:
            return False
        
        # 如果访问次数足够多，不遗忘
        if memory.metadata.access_count >= self.min_access_count:
            return False
        
        # 如果最近有访问，不遗忘
        if memory.metadata.last_accessed_at:
            days_since_access = (datetime.now() - memory.metadata.last_accessed_at).days
            if days_since_access < self.days_since_last_access:
                return False
        
        return True
    
    def get_forgetting_score(self, memory: Memory) -> float:
        """获取遗忘分数"""
        # 工作记忆不基于访问频率遗忘
        if memory.type == MemoryType.WORKING:
            return 0.0
        
        # 访问次数分数
        access_score = max(0.0, 1.0 - memory.metadata.access_count / self.min_access_count)
        
        # 时间分数
        time_score = 0.0
        if memory.metadata.last_accessed_at:
            days_since_access = (datetime.now() - memory.metadata.last_accessed_at).days
            time_score = min(1.0, days_since_access / self.days_since_last_access)
        else:
            # 如果从未访问，使用创建时间
            days_since_creation = (datetime.now() - memory.metadata.created_at).days
            time_score = min(1.0, days_since_creation / self.days_since_last_access)
        
        # 综合分数
        return (access_score + time_score) / 2

class ImportanceBasedForgetting(ForgettingStrategy):
    """基于重要性的遗忘策略"""
    
    def __init__(self, importance_threshold: float = 0.2):
        self.importance_threshold = importance_threshold
```

```python
                if "cluster" not in memory.metadata.context:
```

```python
        if request.memory_type:
            memory_type = MemoryType(request.memory_type)
        
        # 检索记忆
        memories = memory_service.retrieve_memory(
            query=request.query,
            retrieval_type=request.retrieval_type,
            memory_type=memory_type,
            limit=request.limit
        )
        
        # 转换为响应模型
        memory_responses = [
            MemoryResponse(
                id=memory.id,
                type=memory.type.value,
                status=memory.status.value,
                priority=memory.priority.value,
                content=memory.content,
                metadata=memory.metadata.__dict__
            )
            for memory in memories
        ]
        
        return MemoryListResponse(
            memories=memory_responses,
            total=len(memory_responses)
        )
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except Exception as e:
        logger.error(f"Error retrieving memories: {str(e)}")
        raise HTTPException(status_code=500, detail="Internal server error")

@router.post("/forget", response_model=Dict[str, int])
async def forget_memories(
    strategy: str = Query(..., description="遗忘策略", example="time_based"),
    limit: Optional[int] = Query(None, description="遗忘数量限制", example=100),
    memory_service: MemoryService = Depends(get_memory_service)
):
    """遗忘记忆"""
    try:
        forgotten_count = memory_service.forget_memories(strategy, limit)
        
        return {"forgotten_count": forgotten_count}
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except Exception as e:
        logger.error(f"Error forgetting memories: {str(e)}")
        raise HTTPException(status_code=500, detail="Internal server error")

@router.post("/optimize", response_model=Dict[str, bool])
async def optimize_memory(
    optimization_type: str = Query(..., description="优化类型", example="storage"),
    memory_service: MemoryService = Depends(get_memory_service)
):
    """优化记忆"""
    try:
        success = memory_service.optimize_memory(optimization_type)
        
        return {"success": success}
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except Exception as e:
        logger.error(f"Error optimizing memory: {str(e)}")
        raise HTTPException(status_code=500, detail="Internal server error")

@router.get("/stats", response_model=StatsResponse)
```

```python
# examples/api_client_example.py
import requests
import json

class MemoryAPIClient:
    """记忆API客户端"""
    
    def __init__(self, base_url="http://localhost:8000"):
        self.base_url = base_url
        self.api_url = f"{base_url}/api/v1/memory"
    
    def create_memory(self, memory_data):
        """创建记忆"""
        response = requests.post(f"{self.api_url}/", json=memory_data)
        response.raise_for_status()
        return response.json()
    
    def retrieve_memories(self, query, retrieval_type="semantic", limit=10):
        """检索记忆"""
        request_data = {
            "query": query,
            "retrieval_type": retrieval_type,
            "limit": limit
        }
        response = requests.post(f"{self.api_url}/retrieve", json=request_data)
        response.raise_for_status()
        return response.json()
    
    def delete_memory(self, memory_id):
        """删除记忆"""
        response = requests.delete(f"{self.api_url}/{memory_id}")
        response.raise_for_status()
        return response.status_code == 204
    
    def forget_memories(self, strategy="time_based", limit=None):
        """遗忘记忆"""
        params = {"strategy": strategy}
        if limit:
            params["limit"] = limit
        response = requests.post(f"{self.api_url}/forget", params=params)
        response.raise_for_status()
        return response.json()
    
    def optimize_memory(self, optimization_type="storage"):
        """优化记忆"""
        params = {"optimization_type": optimization_type}
        response = requests.post(f"{self.api_url}/optimize", params=params)
        response.raise_for_status()
        return response.json()
    
    def get_stats(self):
        """获取统计信息"""
        response = requests.get(f"{self.api_url}/stats")
        response.raise_for_status()
        return response.json()

def main():
    # 创建API客户端
    client = MemoryAPIClient()
    
    # 创建记忆
    memory_data = {
        "type": "short_term",
        "content": {"text": "宝宝今天第一次笑了", "emotion": "happy"},
        "priority": 4,
        "tags": ["宝宝", "第一次", "笑容"],
        "source": "父母观察"
    }
    
    memory = client.create_memory(memory_data)
    memory_id = memory["id"]
    print(f"创建记忆: {memory_id}")
    
    # 检索记忆
    query = "宝宝笑容"
    result = client.retrieve_memories(query)
    print(f"查询'{query}'返回{result['total']}个结果:")
    for memory in result["memories"]:
        print(f"- {memory['id']}: {memory['content']}")
    
    # 遗忘记忆
    forget_result = client.forget_memories("time_based", 10)
    print(f"遗忘了{forget_result['forgotten_count']}个记忆")
    
    # 优化记忆
    optimize_result = client.optimize_memory("storage")
    print(f"优化记忆{'成功' if optimize_result['success'] else '失败'}")
    
    # 获取统计信息
```

## 常见问题

### Q1: 如何选择合适的向量数据库？
A1: 选择向量数据库时需要考虑以下因素：
- **数据规模**：小规模数据可选择Faiss或Chroma，大规模数据建议选择Milvus
- **性能要求**：对检索速度要求高的场景可选择HNSW索引
- **功能需求**：需要丰富查询功能的场景可选择Milvus或Chroma
- **部署复杂度**：简单部署可选择Faiss或Chroma，复杂部署可选择Milvus

### Q2: 如何优化记忆检索性能？
A2: 优化记忆检索性能的方法：
- **选择合适的索引类型**：根据数据规模和查询模式选择IVF_FLAT、IVF_SQ8或HNSW
- **调整索引参数**：如nlist、nprobe等参数
- **使用缓存**：对频繁查询的结果进行缓存
- **批量处理**：对多个查询进行批量处理
- **异步处理**：对耗时操作使用异步处理

### Q3: 如何处理记忆遗忘？
A3: 处理记忆遗忘的方法：
- **时间策略**：根据记忆创建时间进行遗忘
- **访问策略**：根据记忆访问频率进行遗忘
- **重要性策略**：根据记忆重要性分数进行遗忘
- **组合策略**：结合多种策略进行遗忘
- **归档而非删除**：对重要但少用的记忆进行归档而非删除

### Q4: 如何确保记忆安全？
A4: 确保记忆安全的方法：
- **数据加密**：对敏感记忆进行加密存储
- **访问控制**：实施严格的访问控制策略
- **审计日志**：记录所有记忆操作的审计日志
- **备份恢复**：定期备份记忆数据并测试恢复流程
- **隐私保护**：对个人隐私信息进行脱敏处理
