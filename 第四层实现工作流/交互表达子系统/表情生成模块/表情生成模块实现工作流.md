# 表情生成模块实现工作流

## 模块信息

### 模块名称
表情生成模块

### 所属子系统
交互表达子系统

### 模块功能描述
表情生成模块负责根据文本内容、情感色彩和交互场景，生成适合婴儿认知和情感需求的虚拟表情。该模块能够创建生动、友好、富有情感的表情形象，增强与婴儿的交互体验，促进婴儿的情感发展和认知能力提升。

### 模块职责
1. 表情生成：根据内容和情感生成相应的表情
2. 情感表达：在表情中融入适当的情感色彩
3. 年龄适配：根据婴儿年龄调整表情特征
4. 动态表情：生成动态变化的表情序列
5. 个性化表情：根据婴儿偏好生成个性化表情
6. 表情优化：优化表情质量和自然度

### 模块接口
- **输入接口**：文本内容、情感参数、婴儿状态
- **输出接口**：表情图像/动画、表情元数据
- **依赖模块**：文本分析模块、情感识别模块
- **被依赖模块**：显示输出模块、交互管理模块

### 优先级
高

### 1. 表情生成管理器

```python
# src/core/expression_manager.py
import time
import logging
import os
import io
import base64
from typing import Dict, Any, List, Optional, Tuple, Union
from dataclasses import dataclass
from datetime import datetime
import numpy as np
from PIL import Image

from ..models.expression_models import (
    ExpressionRequest, 
    ExpressionResult, 
    ExpressionMetadata,
    ExpressionType,
    AnimationType
)
from ..core.input_processing.input_processor import InputProcessor
from ..core.input_processing.parameter_extractor import ParameterExtractor
from ..core.content_analysis.content_analyzer import ContentAnalyzer
from ..core.content_analysis.emotion_detector import EmotionDetector
from ..core.content_analysis.context_analyzer import ContextAnalyzer
from ..core.expression_generation.base_expression_generator import BaseExpressionGenerator
from ..core.expression_generation.emotion_expression_generator import EmotionExpressionGenerator
from ..core.expression_generation.dynamic_expression_generator import DynamicExpressionGenerator
from ..core.age_adaptation.age_adapter import AgeAdapter
from ..core.age_adaptation.expression_modifier import ExpressionModifier
from ..core.personalization.personalization_adapter import PersonalizationAdapter
from ..core.personalization.preference_learner import PreferenceLearner
from ..core.expression_optimization.expression_optimizer import ExpressionOptimizer
from ..core.expression_optimization.quality_enhancer import QualityEnhancer
from ..core.quality_assessment.quality_assessor import QualityAssessor
from ..core.quality_assessment.naturalness_evaluator import NaturalnessEvaluator
from ..ml.expression_generation_model import ExpressionGenerationModel
from ..ml.emotion_detection_model import EmotionDetectionModel
from ..ml.quality_assessment_model import QualityAssessmentModel
from ..utils.config import ExpressionConfig
from ..utils.image_utils import ImageUtils
from ..utils.animation_utils import AnimationUtils

logger = logging.getLogger(__name__)

@dataclass
class GenerationRequest:
    """表情生成请求"""
    text: str
    generation_params: Dict[str, Any]
    baby_state: Dict[str, Any]
    request_id: str
    timestamp: float

@dataclass
class GenerationResult:
    """表情生成结果"""
    image_data: bytes  # 图像数据
    metadata: Dict[str, Any]
    quality_score: float
    generation_time: float
    request_id: str
    timestamp: float

class ExpressionGenerationManager:
    """表情生成管理器"""
    
    def __init__(self, config: ExpressionConfig):
        """初始化表情生成管理器"""
        self.config = config
        
        # 初始化输入处理组件
        self.input_processor = InputProcessor(config)
        self.parameter_extractor = ParameterExtractor(config)
        
        # 初始化内容分析组件
        self.content_analyzer = ContentAnalyzer(config)
        self.emotion_detector = EmotionDetector(config)
        self.context_analyzer = ContextAnalyzer(config)
        
        # 初始化表情生成组件
        self.base_expression_generator = BaseExpressionGenerator(config)
        self.emotion_expression_generator = EmotionExpressionGenerator(config)
        self.dynamic_expression_generator = DynamicExpressionGenerator(config)
        
        # 初始化年龄适配组件
        self.age_adapter = AgeAdapter(config)
        self.expression_modifier = ExpressionModifier(config)
        
        # 初始化个性化组件
        self.personalization_adapter = PersonalizationAdapter(config)
        self.preference_learner = PreferenceLearner(config)
        
        # 初始化表情优化组件
        self.expression_optimizer = ExpressionOptimizer(config)
        self.quality_enhancer = QualityEnhancer(config)
        
        # 初始化质量评估组件
        self.quality_assessor = QualityAssessor(config)
        self.naturalness_evaluator = NaturalnessEvaluator(config)
        
        # 初始化工具类
        self.image_utils = ImageUtils(config)
        self.animation_utils = AnimationUtils(config)
        
        # 初始化机器学习模型
        self.expression_model = ExpressionGenerationModel(config)
        self.emotion_model = EmotionDetectionModel(config)
        self.quality_model = QualityAssessmentModel(config)
        
        # 初始化模型
        self._initialize_models()
        
        logger.info("ExpressionGenerationManager initialized")
    
    def _initialize_models(self):
        """初始化模型"""
        try:
            self.expression_model.load_model(self.config.base_model_path)
            self.emotion_model.load_model(self.config.emotion_model_path)
            self.quality_model.load_model(self.config.quality_model_path)
                
            logger.info("Models loaded successfully")
        except Exception as e:
            logger.error(f"Error loading models: {str(e)}")
            raise
    
    def generate_expression(self, request: GenerationRequest) -> GenerationResult:
        """生成表情"""
        start_time = time.time()
        
        try:
            # 1. 输入处理
            processed_input = self.input_processor.process_input(
                request.text, 
                request.generation_params, 
                request.baby_state
            )
            
            # 2. 参数提取
            extracted_params = self.parameter_extractor.extract_parameters(
                processed_input
            )
            
            # 3. 内容分析
            content_analysis = self.content_analyzer.analyze(processed_input["text"])
            
            # 4. 情感检测
            emotion = self.emotion_detector.detect_emotion(
                processed_input["text"],
                extracted_params
            )
            
            # 5. 上下文分析
            context = self.context_analyzer.analyze_context(
                processed_input["text"],
                request.baby_state
            )
            
            # 6. 确定年龄组
            age_group = self._determine_age_group(request.baby_state.get("age", None))
            
            # 7. 个性化适配
            personalization = None
            if self.config.personalization_enabled and "user_id" in request.baby_state:
                personalization = self.personalization_adapter.get_personalization(
                    request.baby_state["user_id"]
                )
            
            # 8. 基础表情生成
            base_expression = self.base_expression_generator.generate(
                content_analysis,
                extracted_params,
                age_group
            )
            
            # 9. 情感表情生成
            if emotion and request.generation_params.get("apply_emotion", True):
                emotion_expression = self.emotion_expression_generator.add_emotion(
                    base_expression,
                    emotion,
                    request.generation_params.get("emotion_intensity", 0.8)
                )
            else:
                emotion_expression = base_expression
            
            # 10. 年龄适配
            age_adapted_expression = self.age_adapter.adapt_expression(
                emotion_expression,
                age_group
            )
            
            # 11. 个性化适配
            if personalization:
                personalized_expression = self.personalization_adapter.apply_personalization(
                    age_adapted_expression,
                    personalization
                )
            else:
                personalized_expression = age_adapted_expression
            
            # 12. 动态表情生成（如果需要）
            if request.generation_params.get("generate_animation", False):
                animation_type = request.generation_params.get("animation_type", "emotion_transition")
                animated_expression = self.dynamic_expression_generator.generate_animation(
                    personalized_expression,
                    animation_type,
                    request.generation_params.get("animation_duration", 2.0)
                )
            else:
                animated_expression = personalized_expression
            
            # 13. 表情优化
            optimized_expression = self.expression_optimizer.optimize(
                animated_expression,
                content_analysis,
                age_group
            )
            
            # 14. 质量增强
            enhanced_expression = self.quality_enhancer.enhance(
                optimized_expression,
                age_group
            )
            
            # 15. 质量评估
            quality_metrics = self.quality_assessor.assess_quality(
                enhanced_expression,
                content_analysis,
                emotion
            )
            
            # 16. 自然度评估
            naturalness_score = self.naturalness_evaluator.evaluate_naturalness(
                enhanced_expression,
                age_group
            )
            
            # 17. 计算综合质量分数
            quality_score = self._calculate_quality_score(
                quality_metrics,
                naturalness_score
            )
            
            # 18. 质量检查和重试
            if quality_score < self.config.min_quality_score:
                # 尝试调整生成参数并重新生成
                adjusted_request = self._adjust_generation_params(
                    request,
                    quality_score
                )
                
                # 递归调用，但限制重试次数
                if adjusted_request.metadata.get("retry_count", 0) < self.config.max_generation_attempts:
                    adjusted_request.metadata["retry_count"] = adjusted_request.metadata.get("retry_count", 0) + 1
                    return self.generate_expression(adjusted_request)
            
            # 19. 转换为输出格式
            output_format = request.generation_params.get("output_format", "png")
            image_data = self._convert_to_output_format(
                enhanced_expression,
                output_format
            )
            
            generation_time = time.time() - start_time
            
            # 20. 创建表情元数据
            metadata = self._create_metadata(
                enhanced_expression,
                content_analysis,
                emotion,
                age_group,
                personalization,
                output_format,
                quality_metrics,
                naturalness_score
            )
            
            # 21. 学习用户偏好（如果有用户ID）
            if self.config.personalization_enabled and "user_id" in request.baby_state:
                self.preference_learner.update_preferences(
                    request.baby_state["user_id"],
                    {
                        "content_analysis": content_analysis,
                        "emotion": emotion,
                        "age_group": age_group,
                        "quality_score": quality_score,
                        "generation_params": request.generation_params
                    }
                )
            
            # 返回结果
            return GenerationResult(
                image_data=image_data,
                metadata=metadata,
                quality_score=quality_score,
                generation_time=generation_time,
                request_id=request.request_id,
                timestamp=time.time()
            )
            
        except Exception as e:
            logger.error(f"Error generating expression: {str(e)}")
            raise
    
    def _determine_age_group(self, age_months: Optional[int]) -> str:
        """确定年龄组"""
        if age_months is None:
            return "12-24_months"  # 默认年龄组
        
        if age_months <= 6:
            return "0-6_months"
        elif age_months <= 12:
            return "6-12_months"
        else:
            return "12-24_months"
    
    def _calculate_quality_score(self, quality_metrics: Dict[str, float], 
                                 naturalness_score: float) -> float:
        """计算综合质量分数"""
        # 使用质量模型计算质量分数
        model_score = self.quality_model.calculate_quality_score(quality_metrics)
        
        # 综合各项指标计算总分
        clarity = quality_metrics.get("clarity", 0.0)
        emotion_accuracy = quality_metrics.get("emotion_accuracy", 0.0)
        age_appropriateness = quality_metrics.get("age_appropriateness", 0.0)
        
        # 加权平均
        total_score = (
            clarity * 0.2 +
            emotion_accuracy * 0.2 +
            age_appropriateness * 0.2 +
            naturalness_score * 0.2 +
            model_score * 0.2
        )
        
        return total_score
    
    def _adjust_generation_params(self, request: GenerationRequest, 
                                   quality_score: float) -> GenerationRequest:
        """调整生成参数"""
        # 创建新的请求对象
        adjusted_request = GenerationRequest(
            text=request.text,
            generation_params=request.generation_params.copy(),
            baby_state=request.baby_state.copy(),
            request_id=f"{request.request_id}_retry",
            timestamp=time.time()
        )
        
        # 根据质量分数调整参数
        if quality_score < 3.0:
            # 质量很低，大幅调整参数
            adjusted_request.generation_params["simplify_expression"] = True
            adjusted_request.generation_params["increase_contrast"] = True
            adjusted_request.generation_params["enhance_features"] = True
        elif quality_score < 3.5:
            # 质量较低，小幅调整参数
            adjusted_request.generation_params["enhance_features"] = True
        
        # 添加重试次数
        adjusted_request.metadata = {"retry_count": request.metadata.get("retry_count", 0) + 1}
        
        return adjusted_request
    
    def _convert_to_output_format(self, expression: Any, output_format: str) -> bytes:
        """转换为输出格式"""
        if output_format.lower() == "png":
            # 转换为PNG格式
            if isinstance(expression, np.ndarray):
                image = Image.fromarray(expression)
            else:
                image = expression
            
            # 转换为RGB模式（如果需要）
            if image.mode != 'RGB':
                image = image.convert('RGB')
            
            # 调整大小
            if image.size != self.config.default_expression_size:
                image = image.resize(self.config.default_expression_size)
            
            # 转换为字节流
            with io.BytesIO() as output:
                image.save(output, format="PNG")
                return output.getvalue()
        
        elif output_format.lower() == "jpg" or output_format.lower() == "jpeg":
            # 转换为JPEG格式
            if isinstance(expression, np.ndarray):
                image = Image.fromarray(expression)
            else:
                image = expression
            
            # 转换为RGB模式（如果需要）
            if image.mode != 'RGB':
                image = image.convert('RGB')
            
            # 调整大小
            if image.size != self.config.default_expression_size:
                image = image.resize(self.config.default_expression_size)
            
            # 转换为字节流
            with io.BytesIO() as output:
                image.save(output, format="JPEG", quality=95)
                return output.getvalue()
        
        elif output_format.lower() == "gif":
            # 转换为GIF格式（动画）
            if isinstance(expression, list):
                # 假设是图像列表
                images = []
                for frame in expression:
                    if isinstance(frame, np.ndarray):
                        img = Image.fromarray(frame)
                    else:
                        img = frame
                    
                    # 转换为RGB模式（如果需要）
                    if img.mode != 'RGB':
                        img = img.convert('RGB')
                    
                    # 调整大小
                    if img.size != self.config.default_expression_size:
                        img = img.resize(self.config.default_expression_size)
                    
                    images.append(img)
                
                # 转换为字节流
                with io.BytesIO() as output:
                    images[0].save(
                        output,
                        format="GIF",
                        save_all=True,
                        append_images=images[1:],
                        duration=1000 // self.config.animation_fps,
                        loop=0
                    )
                    return output.getvalue()
            else:
                # 单张图像转换为GIF
                if isinstance(expression, np.ndarray):
                    image = Image.fromarray(expression)
                else:
                    image = expression
                
                # 转换为RGB模式（如果需要）
                if image.mode != 'RGB':
                    image = image.convert('RGB')
                
                # 调整大小
                if image.size != self.config.default_expression_size:
                    image = image.resize(self.config.default_expression_size)
                
                # 转换为字节流
                with io.BytesIO() as output:
                    image.save(output, format="GIF")
                    return output.getvalue()
        
        else:
            raise ValueError(f"Unsupported output format: {output_format}")
    
    def _create_metadata(self, expression: Any, content_analysis: Dict[str, Any], 
                         emotion: Optional[str], age_group: str, 
                         personalization: Optional[Dict[str, Any]], 
                         output_format: str, quality_metrics: Dict[str, float], 
                         naturalness_score: float) -> Dict[str, Any]:
        """创建表情元数据"""
        return {
            "expression_id": f"expr_{datetime.now().strftime('%Y%m%d_%H%M%S')}",
            "content_analysis": content_analysis,
            "emotion": emotion,
            "age_group": age_group,
            "personalization": personalization,
            "output_format": output_format,
            "quality_metrics": quality_metrics,
            "naturalness_score": naturalness_score,
            "created_at": datetime.now().isoformat()
        }
```

```python
# src/core/expression_generation/base_expression_generator.py
import os
import json
import random
import numpy as np
from typing import Dict, Any, List, Optional, Tuple
from PIL import Image, ImageDraw, ImageFont

from ...models.expression_models import ContentAnalysis, ExpressionParameters
from ...utils.config import ExpressionConfig
from ...utils.image_utils import ImageUtils

class BaseExpressionGenerator:
    """基础表情生成器"""
    
    def __init__(self, config: ExpressionConfig):
        self.config = config
        self.image_utils = ImageUtils(config)
        self.expression_templates = {}
        
        # 加载表情模板
        self._load_expression_templates()
    
    def _load_expression_templates(self):
        """加载表情模板"""
        template_path = self.config.expression_library_path
        
        # 加载基础表情模板
        basic_template_path = os.path.join(template_path, "basic_expressions")
        for file_name in os.listdir(basic_template_path):
            if file_name.endswith('.json'):
                template_name = file_name[:-5]  # 去掉.json后缀
                with open(os.path.join(basic_template_path, file_name), 'r', encoding='utf-8') as f:
                    self.expression_templates[template_name] = json.load(f)
        
        # 加载年龄特定表情模板
        age_template_path = os.path.join(template_path, "age_specific_expressions")
        for age_group in os.listdir(age_template_path):
            if os.path.isdir(os.path.join(age_template_path, age_group)):
                age_group_path = os.path.join(age_template_path, age_group)
                for file_name in os.listdir(age_group_path):
                    if file_name.endswith('.json'):
                        template_name = file_name[:-5]  # 去掉.json后缀
                        with open(os.path.join(age_group_path, file_name), 'r', encoding='utf-8') as f:
                            # 添加年龄组前缀
                            self.expression_templates[f"{age_group}_{template_name}"] = json.load(f)
    
    def generate(self, content_analysis: Dict[str, Any], 
                 generation_params: Dict[str, Any], 
                 age_group: str) -> Image.Image:
        """生成基础表情"""
        # 确定表情类型
        expression_type = self._determine_expression_type(content_analysis)
        
        # 获取模板
        template_id = f"{age_group}_{expression_type}"
        template = self.expression_templates.get(template_id)
        
        # 如果没有年龄特定模板，使用基础模板
        if not template:
            template = self.expression_templates.get(expression_type, 
                                                    self.expression_templates.get("neutral", {}))
        
        # 生成基础表情
        base_expression = self._generate_base_expression(template, content_analysis, generation_params)
        
        # 后处理
        processed_expression = self._post_process_expression(base_expression, age_group)
        
        return processed_expression
    
    def _determine_expression_type(self, content_analysis: Dict[str, Any]) -> str:
        """确定表情类型"""
        content_type = content_analysis.get("content_type", [])
        
        if "question" in content_type:
            return "curious"
        elif "exclamation" in content_type:
            return "surprised"
        elif "story" in content_type:
            return "engaged"
        elif "song" in content_type:
            return "happy"
        elif "praise" in content_type:
            return "proud"
        elif "comfort" in content_type:
            return "gentle"
        else:
            return "neutral"
    
    def _generate_base_expression(self, template: Dict[str, Any], 
                                  content_analysis: Dict[str, Any], 
                                  generation_params: Dict[str, Any]) -> Image.Image:
        """生成基础表情"""
        # 创建基础图像
        width, height = self.config.default_expression_size
        image = Image.new('RGB', (width, height), color='white')
        draw = ImageDraw.Draw(image)
        
        # 绘制脸部轮廓
        face_params = template.get("face", {})
        self._draw_face(draw, face_params, width, height)
        
        # 绘制眼睛
        eye_params = template.get("eyes", {})
        self._draw_eyes(draw, eye_params, width, height)
        
        # 绘制嘴巴
        mouth_params = template.get("mouth", {})
        self._draw_mouth(draw, mouth_params, width, height)
        
        # 绘制其他特征
        other_features = template.get("other_features", {})
        self._draw_other_features(draw, other_features, width, height)
        
        return image
    
    def _draw_face(self, draw: ImageDraw, face_params: Dict[str, Any], 
                   width: int, height: int):
        """绘制脸部轮廓"""
        # 获取脸部参数
        face_color = face_params.get("color", "#FFE0BD")
        face_shape = face_params.get("shape", "oval")
        face_size = face_params.get("size", 0.8)  # 相对于图像大小的比例
        
        # 计算脸部位置和大小
        face_width = int(width * face_size)
        face_height = int(height * face_size)
        face_x = (width - face_width) // 2
        face_y = (height - face_height) // 2
        
        # 绘制脸部
        if face_shape == "oval":
            draw.ellipse(
                [face_x, face_y, face_x + face_width, face_y + face_height],
                fill=face_color
            )
        elif face_shape == "round":
            radius = min(face_width, face_height) // 2
            center_x = width // 2
            center_y = height // 2
            draw.ellipse(
                [center_x - radius, center_y - radius, center_x + radius, center_y + radius],
                fill=face_color
            )
    
    def _draw_eyes(self, draw: ImageDraw, eye_params: Dict[str, Any], 
                   width: int, height: int):
        """绘制眼睛"""
        # 获取眼睛参数
        eye_color = eye_params.get("color", "#000000")
        eye_shape = eye_params.get("shape", "circle")
        eye_size = eye_params.get("size", 0.05)  # 相对于图像大小的比例
        eye_distance = eye_params.get("distance", 0.3)  # 两眼之间的距离比例
        eye_height = eye_params.get("height", 0.4)  # 眼睛在脸部的高度比例
        
        # 计算眼睛位置和大小
        eye_radius = int(min(width, height) * eye_size)
        eye_y = int(height * eye_height)
        
        # 左眼位置
        left_eye_x = int(width * (0.5 - eye_distance / 2))
        
        # 右眼位置
        right_eye_x = int(width * (0.5 + eye_distance / 2))
        
        # 绘制眼睛
        if eye_shape == "circle":
            draw.ellipse(
                [left_eye_x - eye_radius, eye_y - eye_radius, 
                 left_eye_x + eye_radius, eye_y + eye_radius],
                fill=eye_color
            )
            draw.ellipse(
                [right_eye_x - eye_radius, eye_y - eye_radius, 
                 right_eye_x + eye_radius, eye_y + eye_radius],
                fill=eye_color
            )
        elif eye_shape == "arc":
            # 绘制弧形眼睛（微笑）
            eye_width = eye_radius * 2
            eye_arc_height = eye_radius // 2
            
            # 左眼
            draw.arc(
                [left_eye_x - eye_radius, eye_y - eye_arc_height, 
                 left_eye_x + eye_radius, eye_y + eye_arc_height],
                start=0, end=180, fill=eye_color
            )
            
            # 右眼
            draw.arc(
                [right_eye_x - eye_radius, eye_y - eye_arc_height, 
                 right_eye_x + eye_radius, eye_y + eye_arc_height],
                start=0, end=180, fill=eye_color
            )
    
    def _draw_mouth(self, draw: ImageDraw, mouth_params: Dict[str, Any], 
                    width: int, height: int):
        """绘制嘴巴"""
        # 获取嘴巴参数
        mouth_color = mouth_params.get("color", "#FF0000")
        mouth_shape = mouth_params.get("shape", "smile")
        mouth_size = mouth_params.get("size", 0.2)  # 相对于图像大小的比例
        mouth_width = mouth_params.get("width", 0.3)  # 嘴巴宽度比例
        mouth_height = mouth_params.get("height", 0.7)  # 嘴巴在脸部的高度比例
        
        # 计算嘴巴位置和大小
        mouth_y = int(height * mouth_height)
        mouth_half_width = int(width * mouth_width / 2)
        mouth_arc_height = int(min(width, height) * mouth_size)
        
        # 绘制嘴巴
        if mouth_shape == "smile":
            # 绘制微笑
            draw.arc(
                [width // 2 - mouth_half_width, mouth_y - mouth_arc_height, 
                 width // 2 + mouth_half_width, mouth_y + mouth_arc_height],
                start=0, end=180, fill=mouth_color, width=3
            )
        elif mouth_shape == "open":
            # 绘制张开的嘴巴
            draw.ellipse(
                [width // 2 - mouth_half_width // 2, mouth_y - mouth_arc_height // 2, 
                 width // 2 + mouth_half_width // 2, mouth_y + mouth_arc_height // 2],
                fill=mouth_color
            )
        elif mouth_shape == "neutral":
            # 绘制中性嘴巴
            draw.line(
                [width // 2 - mouth_half_width, mouth_y, 
                 width // 2 + mouth_half_width, mouth_y],
                fill=mouth_color, width=3
            )
    
    def _draw_other_features(self, draw: ImageDraw, other_features: Dict[str, Any], 
                             width: int, height: int):
        """绘制其他特征"""
        # 绘制眉毛
        if "eyebrows" in other_features:
            eyebrow_params = other_features["eyebrows"]
            self._draw_eyebrows(draw, eyebrow_params, width, height)
        
        # 绘制鼻子
        if "nose" in other_features:
            nose_params = other_features["nose"]
            self._draw_nose(draw, nose_params, width, height)
        
        # 绘制腮红
        if "blush" in other_features:
            blush_params = other_features["blush"]
            self._draw_blush(draw, blush_params, width, height)
    
    def _draw_eyebrows(self, draw: ImageDraw, eyebrow_params: Dict[str, Any], 
                       width: int, height: int):
        """绘制眉毛"""
        # 获取眉毛参数
        eyebrow_color = eyebrow_params.get("color", "#000000")
        eyebrow_shape = eyebrow_params.get("shape", "straight")
        eyebrow_size = eyebrow_params.get("size", 0.02)  # 相对于图像大小的比例
        eyebrow_distance = eyebrow_params.get("distance", 0.3)  # 两眉之间的距离比例
        eyebrow_height = eyebrow_params.get("height", 0.3)  # 眉毛在脸部的高度比例
        
        # 计算眉毛位置和大小
        eyebrow_width = int(min(width, height) * eyebrow_size)
        eyebrow_y = int(height * eyebrow_height)
        
        # 左眉位置
        left_eyebrow_x = int(width * (0.5 - eyebrow_distance / 2))
        
        # 右眉位置
        right_eyebrow_x = int(width * (0.5 + eyebrow_distance / 2))
        
        # 绘制眉毛
        if eyebrow_shape == "straight":
            # 直眉毛
            draw.line(
                [left_eyebrow_x - eyebrow_width // 2, eyebrow_y, 
                 left_eyebrow_x + eyebrow_width // 2, eyebrow_y],
                fill=eyebrow_color, width=2
            )
            draw.line(
                [right_eyebrow_x - eyebrow_width // 2, eyebrow_y, 
                 right_eyebrow_x + eyebrow_width // 2, eyebrow_y],
                fill=eyebrow_color, width=2
            )
        elif eyebrow_shape == "raised":
            # 上扬眉毛
            draw.line(
                [left_eyebrow_x - eyebrow_width // 2, eyebrow_y + 2, 
                 left_eyebrow_x + eyebrow_width // 2, eyebrow_y - 2],
                fill=eyebrow_color, width=2
            )
            draw.line(
                [right_eyebrow_x - eyebrow_width // 2, eyebrow_y - 2, 
                 right_eyebrow_x + eyebrow_width // 2, eyebrow_y + 2],
                fill=eyebrow_color, width=2
            )
    
    def _draw_nose(self, draw: ImageDraw, nose_params: Dict[str, Any], 
                   width: int, height: int):
        """绘制鼻子"""
        # 获取鼻子参数
        nose_color = nose_params.get("color", "#000000")
        nose_shape = nose_params.get("shape", "dot")
        nose_size = nose_params.get("size", 0.01)  # 相对于图像大小的比例
        nose_height = nose_params.get("height", 0.5)  # 鼻子在脸部的高度比例
        
        # 计算鼻子位置和大小
        nose_radius = int(min(width, height) * nose_size)
        nose_x = width // 2
        nose_y = int(height * nose_height)
        
        # 绘制鼻子
        if nose_shape == "dot":
            draw.ellipse(
                [nose_x - nose_radius, nose_y - nose_radius, 
                 nose_x + nose_radius, nose_y + nose_radius],
                fill=nose_color
            )
        elif nose_shape == "line":
            draw.line(
                [nose_x, nose_y - nose_radius, 
                 nose_x, nose_y + nose_radius],
                fill=nose_color, width=2
            )
    
    def _draw_blush(self, draw: ImageDraw, blush_params: Dict[str, Any], 
                    width: int, height: int):
        """绘制腮红"""
        # 获取腮红参数
        blush_color = blush_params.get("color", "#FFB6C1")
        blush_size = blush_params.get("size", 0.05)  # 相对于图像大小的比例
        blush_distance = blush_params.get("distance", 0.4)  # 腮红与中心的距离比例
        blush_height = blush_params.get("height", 0.5)  # 腮红在脸部的高度比例
        
        # 计算腮红位置和大小
        blush_radius = int(min(width, height) * blush_size)
        blush_y = int(height * blush_height)
        
        # 左腮红位置
        left_blush_x = int(width * (0.5 - blush_distance / 2))
        
        # 右腮红位置
        right_blush_x = int(width * (0.5 + blush_distance / 2))
        
        # 绘制腮红（半透明）
        for i in range(3):
            radius = blush_radius - i * 2
            if radius > 0:
                # 左腮红
                draw.ellipse(
                    [left_blush_x - radius, blush_y - radius, 
                     left_blush_x + radius, blush_y + radius],
                    fill=blush_color
                )
                
                # 右腮红
                draw.ellipse(
                    [right_blush_x - radius, blush_y - radius, 
                     right_blush_x + radius, blush_y + radius],
                    fill=blush_color
                )
    
    def _post_process_expression(self, expression: Image.Image, age_group: str) -> Image.Image:
        """后处理表情"""
        # 获取年龄组配置
        age_config = self.config.age_groups.get(age_group, {})
        
        # 应用年龄特定处理
        if age_config.get("color_contrast") == "high":
            # 增强颜色对比度
            expression = self.image_utils.enhance_contrast(expression, factor=1.5)
        elif age_config.get("color_contrast") == "medium":
            # 中等颜色对比度
            expression = self.image_utils.enhance_contrast(expression, factor=1.2)
        
        # 应用特征夸张
        feature_exaggeration = age_config.get("feature_exaggeration", 0.0)
        if feature_exaggeration > 0:
            expression = self.image_utils.exaggerate_features(expression, factor=feature_exaggeration)
        
        return expression
```

### Python示例

```python
import requests
import base64
from PIL import Image
import io

# API基础URL
BASE_URL = "http://localhost:8000/api/expression"
# 认证令牌
TOKEN = "your_token_here"

# 生成表情
def generate_expression(text, baby_age, baby_id, output_format="png"):
    url = f"{BASE_URL}/generate"
    headers = {"Authorization": f"Bearer {TOKEN}"}
    
    data = {
        "text": text,
        "generation_params": {
            "output_format": output_format,
            "apply_emotion": True,
            "emotion_intensity": 0.8
        },
        "baby_state": {
            "age": baby_age,
            "user_id": baby_id
        }
    }
    
    response = requests.post(url, json=data, headers=headers)
    result = response.json()
    
    if result["success"]:
        # 解码图像数据
        image_data = base64.b64decode(result["data"]["image_data"])
        image = Image.open(io.BytesIO(image_data))
        
        # 显示图像
        image.show()
        
        return image, result["data"]["metadata"]
    else:
        print(f"Error: {result['error']['message']}")
        return None, None

# 使用示例
image, metadata = generate_expression("你好，宝宝！今天天气真好！", 12, "baby_001")
```
