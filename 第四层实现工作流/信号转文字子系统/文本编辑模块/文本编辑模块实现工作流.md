        # 保存编辑结果
        self.save_text(text)
        
        # 记录编辑历史
        self.history.append({
            'timestamp': datetime.now().isoformat(),
            'operation': 'style_edit',
            'original_text': original_text,
            'edited_text': text,
            'suggestions_count': len(suggestions)
        })
        
        return text, suggestions
    
    def get_style_suggestions(self, text: str) -> List[Dict[str, Any]]:
        """获取风格改进建议"""
        suggestions = []
        
        # 分析文本风格
        style_score = self.analyze_style(text)
        
        # 根据分析结果生成建议
        if style_score['formality'] < 0.3:
            suggestions.append({
                'type': 'formality',
                'message': '文本过于口语化，建议增加正式表达',
                'examples': ['将"很棒"改为"优秀"', '将"搞定"改为"完成"']
            })
        
        if style_score['readability'] < 0.5:
            suggestions.append({
                'type': 'readability',
                'message': '文本可读性较低，建议简化长句',
                'examples': ['将长句拆分为短句', '减少嵌套从句']
            })
        
        if style_score['engagement'] < 0.4:
            suggestions.append({
                'type': 'engagement',
                'message': '文本缺乏吸引力，建议增加生动的表达',
                'examples': ['使用比喻和类比', '增加具体例子']
            })
        
        return suggestions
    
    def analyze_style(self, text: str) -> Dict[str, float]:
        """分析文本风格"""
        # 这里简化处理，实际应用中可以使用更复杂的NLP模型
        sentences = nltk.sent_tokenize(text)
        words = nltk.word_tokenize(text)
        
        # 计算基本指标
        avg_sentence_length = len(words) / len(sentences) if sentences else 0
        complex_words = sum(1 for word in words if len(word) > 6)
        complex_word_ratio = complex_words / len(words) if words else 0
        
        # 评估正式性（简化）
        formal_indicators = ['因此', '此外', '然而', '综上所述']
        formality = sum(1 for indicator in formal_indicators if indicator in text) / len(formal_indicators)
        
        # 评估可读性（简化）
        readability = max(0, 1 - (avg_sentence_length - 15) / 30)  # 假设15词为理想句长
        
        # 评估吸引力（简化）
        engagement_indicators = ['!', '?', '例如', '比如']
        engagement = sum(1 for indicator in engagement_indicators if indicator in text) / len(engagement_indicators)
        
        return {
            'formality': min(1.0, formality),
            'readability': min(1.0, readability),
            'engagement': min(1.0, engagement),
            'avg_sentence_length': avg_sentence_length,
            'complex_word_ratio': complex_word_ratio
        }

                return {
                    "success": True,
                    "corrected_text": corrected_text,

                    "original_text": request.text
                }

                raise HTTPException(status_code=500, detail=str(e))
        
        @self.app.post("/edit/style")
        async def style_edit(request: StyleEditRequest):
            """风格编辑"""
            try:
                edited_text, suggestions = self.editor.edit_style(
                    request.text, 
                    request.style_type
                )
                return {
                    "success": True,
                    "edited_text": edited_text,
                    "suggestions": suggestions,
                    "original_text": request.text
                }
            except Exception as e:
                raise HTTPException(status_code=500, detail=str(e))
        
        @self.app.post("/edit/suggestions")
        async def get_suggestions(request: SuggestionsRequest):
            """获取编辑建议"""
            try:
                suggestions = self.editor.get_style_suggestions(request.text)
                return {
                    "success": True,
                    "suggestions": suggestions
                }
            except Exception as e:
                raise HTTPException(status_code=500, detail=str(e))
        
        @self.app.get("/edit/history")

                return {
                    "success": True,
                    "history": history
                }

        async def undo_edit():

                success = self.editor.undo_last_edit()
                return {
                    "success": success,
                    "message": "撤销成功" if success else "没有可撤销的操作"
                }

文本编辑模块负责对生成的文本内容进行编辑和优化，确保文本内容符合婴儿互动的需求。该模块能够对文本进行语法检查、风格调整、长度控制、内容简化等操作，使文本更适合婴儿理解和互动。

4. 内容简化：简化复杂词汇和句子结构
5. 个性化调整：根据婴儿特点调整文本内容
6. 质量评估：评估编辑后文本的质量和适宜性

### 模块接口
- **输入接口**：原始文本、编辑参数、婴儿状态
- **输出接口**：编辑后文本、编辑标签、质量评分
- **依赖模块**：文本生成模块
- **被依赖模块**：语音合成模块、交互表达子系统

#### 4. 核心文本编辑器 (src/core/text_editor.py)
```python
from typing import Dict, List, Any, Optional
from loguru import logger

from .text_analyzer import TextAnalyzer
from .grammar_checker import GrammarChecker
from .style_editor import StyleEditor
from .length_controller import LengthController
from .content_simplifier import ContentSimplifier
from .personalization_editor import PersonalizationEditor
from .quality_assessor import QualityAssessor
from ..models.internal_models import EditResult, EditOperation, TextAnalysis


class TextEditor:
    """核心文本编辑器"""
    
    def __init__(
        self,
        grammar_checker: GrammarChecker,
        style_editor: StyleEditor,
        length_controller: LengthController,
        content_simplifier: ContentSimplifier,
        personalization_editor: PersonalizationEditor,
        quality_assessor: QualityAssessor
    ):
        self.grammar_checker = grammar_checker
        self.style_editor = style_editor
        self.length_controller = length_controller
        self.content_simplifier = content_simplifier
        self.personalization_editor = personalization_editor
        self.quality_assessor = quality_assessor
        self.text_analyzer = TextAnalyzer()
        
    def edit(self, text: str, edit_params: Dict[str, Any], baby_state: Dict[str, Any]) -> EditResult:
        """
        编辑文本
        
        Args:
            text: 原始文本
            edit_params: 编辑参数
            baby_state: 婴儿状态
            
        Returns:
            编辑结果
        """
        logger.info(f"开始编辑文本，长度: {len(text)}")
        
        # 输入处理
        processed_input = self._process_input(text, edit_params, baby_state)
        
        # 文本分析
        text_analysis = self.text_analyzer.analyze(processed_input["text"])
        
        # 编辑操作
        edited_text = processed_input["text"]
        edit_history: List[EditOperation] = []
        
        # 语法检查
        if edit_params.get("check_grammar", True):
            logger.debug("执行语法检查")
            grammar_result = self.grammar_checker.check(edited_text)
            if grammar_result.has_errors:
                edited_text = grammar_result.corrected_text
                edit_history.append(EditOperation(
                    operation="grammar_check",
                    original=processed_input["text"],
                    edited=edited_text,
                    changes=grammar_result.corrections
                ))
        
        # 风格调整
        if edit_params.get("adjust_style", True):
            logger.debug("执行风格调整")
            target_style = edit_params.get("target_style", "friendly")
            style_result = self.style_editor.adjust(
                text=edited_text,
                target_style=target_style,
                style_params=edit_params.get("style_params", {})
            )
            if style_result.adjusted_text != edited_text:
                edited_text = style_result.adjusted_text
                edit_history.append(EditOperation(
                    operation="style_adjust",
                    original=edited_text,
                    edited=style_result.adjusted_text,
                    changes=style_result.changes
                ))
        
        # 长度控制
        if edit_params.get("control_length", True):
            logger.debug("执行长度控制")
            age_group = baby_state.get("age_group", "3-4")
            length_result = self.length_controller.control(
                text=edited_text,
                age_group=age_group,
                length_params=edit_params.get("length_params", {})
            )
            if length_result.controlled_text != edited_text:
                edited_text = length_result.controlled_text
                edit_history.append(EditOperation(
                    operation="length_control",
                    original=edited_text,
                    edited=length_result.controlled_text,
                    changes=length_result.changes
                ))
        
        # 内容简化
        if edit_params.get("simplify_content", True):
            logger.debug("执行内容简化")
            age_group = baby_state.get("age_group", "3-4")
            simplify_result = self.content_simplifier.simplify(
                text=edited_text,
                target_age=age_group,
                simplify_params=edit_params.get("simplify_params", {})
            )
            if simplify_result.simplified_text != edited_text:
                edited_text = simplify_result.simplified_text
                edit_history.append(EditOperation(
                    operation="content_simplification",
                    original=edited_text,
                    edited=simplify_result.simplified_text,
                    changes=simplify_result.changes
                ))
        
        # 个性化编辑
        if edit_params.get("personalize", True) and baby_state:
            logger.debug("执行个性化编辑")
            personalize_result = self.personalization_editor.personalize(
                text=edited_text,
                baby_profile=baby_state,
                personalization_params=edit_params.get("personalization_params", {})
            )
            if personalize_result.personalized_text != edited_text:
                edited_text = personalize_result.personalized_text
                edit_history.append(EditOperation(
                    operation="personalization",
                    original=edited_text,
                    edited=personalize_result.personalized_text,
                    changes=personalize_result.changes
                ))
        
        # 质量评估
        logger.debug("执行质量评估")
        age_group = baby_state.get("age_group", "3-4")
        quality_result = self.quality_assessor.assess(
            text=edited_text,
            age_group=age_group,
            assessment_params=edit_params.get("assessment_params", {})
        )
        
        # 生成编辑标签
        edit_tags = self._generate_edit_tags(edit_history)
        
        logger.info(f"文本编辑完成，最终长度: {len(edited_text)}")
        
        return EditResult(
            original_text=text,
            edited_text=edited_text,
            edit_history=edit_history,
            quality_score=quality_result.overall_score,
            edit_tags=edit_tags,
            text_analysis=text_analysis,
            quality_metrics=quality_result.detailed_metrics
        )
    
    def batch_edit(
        self, 
        texts: List[str], 
        edit_params: Dict[str, Any], 
        baby_state: Dict[str, Any]
    ) -> List[EditResult]:
        """
        批量编辑文本
        
        Args:
            texts: 文本列表
            edit_params: 编辑参数
            baby_state: 婴儿状态
            
        Returns:
            编辑结果列表
        """
        logger.info(f"开始批量编辑文本，数量: {len(texts)}")
        
        results = []
        for i, text in enumerate(texts):
            try:
                result = self.edit(text, edit_params, baby_state)
                results.append(result)
                logger.debug(f"文本 {i+1}/{len(texts)} 编辑完成")
            except Exception as e:
                logger.error(f"编辑文本 {i+1} 时出错: {str(e)}")
                # 创建错误结果
                error_result = EditResult(
                    original_text=text,
                    edited_text=text,  # 返回原始文本
                    edit_history=[],
                    quality_score=0.0,
                    edit_tags=["error"],
                    text_analysis=TextAnalysis(),
                    quality_metrics={}
                )
                results.append(error_result)
        
        logger.info(f"批量编辑完成，成功: {len(results)}")
        return results
    
    def _process_input(
        self, 
        text: str, 
        edit_params: Dict[str, Any], 
        baby_state: Dict[str, Any]
    ) -> Dict[str, Any]:
        """处理输入"""
        # 文本预处理
        processed_text = text.strip()
        
        # 参数验证和默认值设置
        processed_params = {
            "check_grammar": edit_params.get("check_grammar", True),
            "adjust_style": edit_params.get("adjust_style", True),
            "control_length": edit_params.get("control_length", True),
            "simplify_content": edit_params.get("simplify_content", True),
            "personalize": edit_params.get("personalize", True),
            "target_style": edit_params.get("target_style", "friendly"),
            "style_params": edit_params.get("style_params", {}),
            "length_params": edit_params.get("length_params", {}),
            "simplify_params": edit_params.get("simplify_params", {}),
            "personalization_params": edit_params.get("personalization_params", {}),
            "assessment_params": edit_params.get("assessment_params", {})
        }
        
        # 婴儿状态处理
        processed_baby_state = baby_state.copy()
        if "age_group" not in processed_baby_state:
            # 如果没有提供年龄组，根据年龄推断
            age_months = processed_baby_state.get("age_months", 36)  # 默认3岁
            if age_months <= 12:
                processed_baby_state["age_group"] = "0-1"
            elif age_months <= 24:
                processed_baby_state["age_group"] = "1-2"
            elif age_months <= 36:
                processed_baby_state["age_group"] = "2-3"
            elif age_months <= 48:
                processed_baby_state["age_group"] = "3-4"
            else:
                processed_baby_state["age_group"] = "4-5"
        
        return {
            "text": processed_text,
            "edit_params": processed_params,
            "baby_state": processed_baby_state
        }
    
    def _generate_edit_tags(self, edit_history: List[EditOperation]) -> List[str]:
        """生成编辑标签"""
        tags = []
        for operation in edit_history:
            if operation.operation == "grammar_check":
                tags.append("语法修正")
            elif operation.operation == "style_adjust":
                tags.append("风格调整")
            elif operation.operation == "length_control":
                tags.append("长度控制")
            elif operation.operation == "content_simplification":
                tags.append("内容简化")
            elif operation.operation == "personalization":
                tags.append("个性化编辑")
        
        return tags
```

```python
import language_tool_python
from typing import Dict, List, Any, Optional
from loguru import logger

from ..models.internal_models import GrammarCheckResult


class GrammarChecker:
    """语法检查器"""
    
    def __init__(self, language: str = "zh-CN"):
        """
        初始化语法检查器
        
        Args:
            language: 语言代码
        """
        self.language = language
        self.tool = None
        self._initialize_tool()
    
    def _initialize_tool(self):
        """初始化语法检查工具"""
        try:
            # 尝试初始化LanguageTool
            self.tool = language_tool_python.LanguageTool(self.language)
            logger.info(f"语法检查工具初始化成功，语言: {self.language}")
        except Exception as e:
            logger.error(f"初始化语法检查工具失败: {str(e)}")
            # 创建一个空的工具对象，避免后续调用出错
            self.tool = None
    
    def check(self, text: str, auto_correct: bool = True) -> GrammarCheckResult:
        """
        检查文本语法
        
        Args:
            text: 要检查的文本
            auto_correct: 是否自动修正
            
        Returns:
            语法检查结果
        """
        logger.debug(f"开始语法检查，文本长度: {len(text)}")
        
        if not self.tool:
            logger.warning("语法检查工具未初始化，返回原始文本")
            return GrammarCheckResult(
                has_errors=False,
                errors=[],
                corrected_text=text,
                corrections=[]
            )
        
        try:
            # 检查语法错误
            matches = self.tool.check(text)
            has_errors = len(matches) > 0
            
            # 格式化错误信息
            errors = []
            for match in matches:
                errors.append({
                    "message": match.message,
                    "offset": match.offset,
                    "length": match.errorLength,
                    "suggestions": match.replacements[:3] if match.replacements else [],
                    "category": match.category,
                    "rule_id": match.ruleId
                })
            
            # 自动修正
            corrected_text = text
            corrections = []
            
            if auto_correct and has_errors:
                corrected_text = self.tool.correct(text)
                
                # 记录修正内容
                for match in matches:
                    if match.replacements:
                        original = text[match.offset:match.offset + match.errorLength]
                        replacement = match.replacements[0]
                        corrections.append({
                            "type": "replacement",
                            "original": original,
                            "replacement": replacement,
                            "position": match.offset,
                            "rule": match.ruleId
                        })
            
            logger.debug(f"语法检查完成，发现 {len(errors)} 个错误")
            
            return GrammarCheckResult(
                has_errors=has_errors,
                errors=errors,
                corrected_text=corrected_text,
                corrections=corrections
            )
            
        except Exception as e:
            logger.error(f"语法检查过程中出错: {str(e)}")
            return GrammarCheckResult(
                has_errors=False,
                errors=[],
                corrected_text=text,
                corrections=[]
            )
    
    def check_batch(self, texts: List[str], auto_correct: bool = True) -> List[GrammarCheckResult]:
        """
        批量检查文本语法
        
        Args:
            texts: 要检查的文本列表
            auto_correct: 是否自动修正
            
        Returns:
            语法检查结果列表
        """
        logger.debug(f"开始批量语法检查，文本数量: {len(texts)}")
        
        results = []
        for text in texts:
            result = self.check(text, auto_correct)
            results.append(result)
        
        logger.debug(f"批量语法检查完成")
        return results
    
    def close(self):
        """关闭语法检查工具"""
        if self.tool:
            self.tool.close()
            logger.info("语法检查工具已关闭")
```

```python
            original_length = len(text)
            target_length = min(
                max_length,
                int(original_length * max_length_ratio)
            )
            target_length = max(
                target_length,
                int(original_length * min_length_ratio)
            )

            if original_length <= target_length:
                logger.debug("文本长度已在目标范围内，无需调整")
                return LengthControlResult(

            controlled_text = text

    def _remove_redundant_content(self, text: str, target_length: int) -> tuple[str, List[str]]:
        """删除冗余内容"""
        changes = []
        controlled_text = text
        
        # 删除重复的短语
        doc = self.nlp(controlled_text)
        phrases = [chunk.text for chunk in doc.noun_chunks]
        
        # 检查重复短语
        phrase_counts = {}
        for phrase in phrases:
            if phrase in phrase_counts:
                phrase_counts[phrase] += 1
            else:
                phrase_counts[phrase] = 1
        
        # 删除重复出现的短语（保留第一次出现）
        for phrase, count in phrase_counts.items():
            if count > 1 and len(phrase) > 3:  # 只处理长度大于3的短语
                # 删除除第一次外的所有出现
                first_occurrence = controlled_text.find(phrase)
                if first_occurrence != -1:
                    # 找到第二次出现的位置
                    second_occurrence = controlled_text.find(phrase, first_occurrence + 1)
                    while second_occurrence != -1 and len(controlled_text) > target_length:
                        controlled_text = controlled_text[:second_occurrence] + controlled_text[second_occurrence + len(phrase):]
                        changes.append(f"删除重复短语: {phrase}")
                        second_occurrence = controlled_text.find(phrase, second_occurrence)
        
        return controlled_text, changes
    
    def _simplify_expressions(self, text: str, target_length: int) -> tuple[str, List[str]]:
        """简化表达"""
        changes = []
        controlled_text = text
        
        # 替换冗长表达为简洁表达

            "以便于": "以便",
            "与此同时": "同时",
            "除此之外": "此外",

            "综上所述": "综上",
            "一方面...另一方面": "一方面和另一方面"
        }
        
        for long_expr, short_expr in replacements.items():
            if long_expr in controlled_text:
                controlled_text = controlled_text.replace(long_expr, short_expr)
                changes.append(f"简化表达: {long_expr} -> {short_expr}")
        
        return controlled_text, changes
    
    def _remove_secondary_content(self, text: str, target_length: int) -> tuple[str, List[str]]:
        """删除次要内容"""
        changes = []

        doc = self.nlp(controlled_text)

            importance = self._calculate_sentence_importance(sentence)
            sentence_importance.append((sentence, importance))
        
        # 按重要性排序
        sentence_importance.sort(key=lambda x: x[1], reverse=True)
        
        # 选择重要句子，直到达到目标长度
        selected_sentences = []
        current_length = 0
        
        for sentence, importance in sentence_importance:
            if current_length + len(sentence) <= target_length:
                selected_sentences.append(sentence)
                current_length += len(sentence)
            else:
                break
        
        # 如果没有选择任何句子，至少选择第一个句子
        if not selected_sentences and sentences:
            selected_sentences = [sentences[0]]
        
        # 重新组合文本
        controlled_text = "".join(selected_sentences)
        
        # 记录删除的句子数量
        removed_count = len(sentences) - len(selected_sentences)
        if removed_count > 0:
            changes.append(f"删除了 {removed_count} 个次要句子")
        
        return controlled_text, changes

        keywords = ["重要", "关键", "主要", "首先", "最后", "总之"]
        for keyword in keywords:
            if keyword in sentence:
                importance += 0.1
        
        # 包含数字的句子可能更重要
        if any(char.isdigit() for char in sentence):
            importance += 0.1
        
        # 较短的句子可能更重要
        if len(sentence) < 20:
            importance += 0.1
        
        return min(importance, 1.0)  # 限制在0-1范围内
```

```python
            self.simple_words = {}
            self.complex_words = {}
    
    def simplify(
        self, 
        text: str, 
        target_age: str = "3-4", 
        simplify_params: Dict[str, Any] = {}
    ) -> SimplifyResult:
        """
        简化文本内容
        
        Args:
            text: 原始文本
            target_age: 目标年龄
            simplify_params: 简化参数
            
        Returns:
            简化结果
        """
        logger.debug(f"开始内容简化，目标年龄: {target_age}")
        
        if not self.nlp:
            logger.warning("NLP模型未加载，返回原始文本")
            return SimplifyResult(
                simplified_text=text,
                simplification_metrics={},
                changes=[]
            )
        
        try:
            # 获取简化参数
            max_sentence_length = simplify_params.get(
                "max_sentence_length", 
                self.config.get("max_sentence_length", 15)
            )
            
            max_word_syllables = simplify_params.get(
                "max_word_syllables", 
                self.config.get("max_word_syllables", 3)
            )
            
            simplify_complex_words = simplify_params.get(
                "simplify_complex_words", 
                self.config.get("simplify_complex_words", True)
            )
            
            simplify_sentence_structure = simplify_params.get(
                "simplify_sentence_structure", 
                self.config.get("simplify_sentence_structure", True)
            )
            
            # 简化文本
            simplified_text = text
            changes = []
            
            # 简化复杂词汇
            if simplify_complex_words:
                simplified_text, changes1 = self._simplify_vocabulary(simplified_text)
                changes.extend(changes1)
            
            # 简化句子结构
            if simplify_sentence_structure:
                simplified_text, changes2 = self._simplify_sentence_structure(
                    simplified_text, max_sentence_length
                )
                changes.extend(changes2)
            
            # 计算简化指标
            simplification_metrics = self._calculate_simplification_metrics(
                original_text=text,
                simplified_text=simplified_text
            )
            
            logger.debug(f"内容简化完成，修改数量: {len(changes)}")
            
            return SimplifyResult(
                simplified_text=simplified_text,
                simplification_metrics=simplification_metrics,
                changes=changes
            )
            
        except Exception as e:
            logger.error(f"内容简化过程中出错: {str(e)}")
            return SimplifyResult(
                simplified_text=text,
                simplification_metrics={},
                changes=[]
            )
    
    def _simplify_vocabulary(self, text: str) -> Tuple[str, List[str]]:
        """简化词汇"""
        changes = []
        simplified_text = text
        
        # 替换复杂词汇为简单词汇
        for complex_word, simple_word in self.complex_words.items():
            if complex_word in simplified_text:
                simplified_text = simplified_text.replace(complex_word, simple_word)
                changes.append(f"简化词汇: {complex_word} -> {simple_word}")
        
        return simplified_text, changes
    
    def _simplify_sentence_structure(self, text: str, max_sentence_length: int) -> Tuple[str, List[str]]:
        """简化句子结构"""
        changes = []
        simplified_text = text
        
        # 使用NLP模型分析句子
        doc = self.nlp(text)
        sentences = [sent.text for sent in doc.sents]
        
        simplified_sentences = []
        for sentence in sentences:
            # 如果句子长度超过限制，尝试简化
            if len(sentence) > max_sentence_length:
                simplified_sentence, sentence_changes = self._simplify_single_sentence(sentence, max_sentence_length)
                simplified_sentences.append(simplified_sentence)
                changes.extend(sentence_changes)
            else:
                simplified_sentences.append(sentence)
        
        # 重新组合文本
        simplified_text = "".join(simplified_sentences)
        
        return simplified_text, changes
    
    def _simplify_single_sentence(self, sentence: str, max_length: int) -> Tuple[str, List[str]]:
        """简化单个句子"""
        changes = []
        simplified_sentence = sentence
        
        # 简化连词
        conjunctions = {
            "而且": "和",
            "并且": "和",
            "以及": "和",
            "或者": "或",
            "否则": "不然",
            "然而": "但是",
            "不过": "但是"
        }
        
        for complex_conj, simple_conj in conjunctions.items():
            if complex_conj in simplified_sentence:
                simplified_sentence = simplified_sentence.replace(complex_conj, simple_conj)
                changes.append(f"简化连词: {complex_conj} -> {simple_conj}")
        
        # 如果仍然太长，尝试拆分句子
        if len(simplified_sentence) > max_length:
            # 找到可以拆分的位置（如逗号）
            split_positions = [i for i, char in enumerate(simplified_sentence) if char in "，。；；"]
            
            if split_positions:
                # 在第一个可拆分位置拆分
                split_pos = split_positions[len(split_positions) // 2]  # 在中间位置拆分
                part1 = simplified_sentence[:split_pos + 1]
                part2 = simplified_sentence[split_pos + 1:].strip()
                
                # 确保两部分都有内容
                if part1 and part2:
                    simplified_sentence = part1 + part2
                    changes.append("拆分长句")
        
        return simplified_sentence, changes
    
    def _calculate_simplification_metrics(self, original_text: str, simplified_text: str) -> Dict[str, float]:
        """计算简化指标"""
        # 计算复杂度降低
        original_complexity = self._calculate_text_complexity(original_text)
        simplified_complexity = self._calculate_text_complexity(simplified_text)
        complexity_reduction = (original_complexity - simplified_complexity) / original_complexity if original_complexity > 0 else 0
        
        # 计算可读性提升
        original_readability = self._calculate_readability(original_text)
        simplified_readability = self._calculate_readability(simplified_text)
        readability_improvement = (simplified_readability - original_readability) / original_readability if original_readability > 0 else 0
        
        return {
            "complexity_reduction": complexity_reduction,
            "readability_improvement": readability_improvement,
            "original_complexity": original_complexity,
            "simplified_complexity": simplified_complexity,
            "original_readability": original_readability,
            "simplified_readability": simplified_readability
        }
    
    def _calculate_text_complexity(self, text: str) -> float:
        """计算文本复杂度"""
        if not self.nlp:
            return 0.5
        
        doc = self.nlp(text)
        
        # 计算平均句子长度
        sentences = list(doc.sents)
        avg_sentence_length = sum(len(sent) for sent in sentences) / len(sentences) if sentences else 0
        
        # 计算复杂词汇比例
        complex_word_count = sum(1 for token in doc if token.text in self.complex_words)
        total_word_count = sum(1 for token in doc if not token.is_punct)
        complex_word_ratio = complex_word_count / total_word_count if total_word_count > 0 else 0
        
        # 计算从句比例
        clause_count = sum(1 for token in doc if token.dep_ in ["acl", "advcl", "relcl"])
        clause_ratio = clause_count / total_word_count if total_word_count > 0 else 0
        
        # 综合复杂度
        complexity = (
            min(avg_sentence_length / 30, 1.0) * 0.4 +  # 句子长度权重40%
            complex_word_ratio * 0.3 +                   # 复杂词汇权重30%
            clause_ratio * 0.3                           # 从句比例权重30%
        )
        
        return complexity
    
    def _calculate_readability(self, text: str) -> float:
        """计算文本可读性"""
        if not self.nlp:
            return 0.5
        
        doc = self.nlp(text)
        
        # 计算简单词汇比例
        simple_word_count = sum(1 for token in doc if token.text in self.simple_words)
        total_word_count = sum(1 for token in doc if not token.is_punct)
        simple_word_ratio = simple_word_count / total_word_count if total_word_count > 0 else 0
        
        # 计算短句比例
        sentences = list(doc.sents)
        short_sentence_count = sum(1 for sent in sentences if len(sent) <= 15)
        short_sentence_ratio = short_sentence_count / len(sentences) if sentences else 0
        
        # 综合可读性
        readability = (
            simple_word_ratio * 0.5 +          # 简单词汇权重50%
            short_sentence_ratio * 0.5         # 短句权重50%
        )
        
        return readability
```
