# 文本过滤模块实现工作流

## 模块信息

### 模块名称
文本过滤模块

### 所属子系统
信号转文字子系统

### 模块功能
文本过滤模块负责对生成和编辑后的文本内容进行安全性和适宜性过滤，确保所有输出给婴儿的文本内容符合儿童安全和教育标准。该模块能够识别和过滤不适宜的内容，包括暴力、恐怖、不当语言、成人内容等，同时还能根据婴儿的年龄、情感状态和文化背景进行个性化过滤。

### 模块职责
1. 安全内容过滤：识别和过滤不适宜的安全内容
2. 教育适宜性过滤：确保内容符合教育标准和价值观
3. 年龄适宜性过滤：根据婴儿年龄调整过滤标准
4. 情感适宜性过滤：过滤可能引起负面情绪的内容
5. 文化适宜性过滤：确保内容符合文化背景和价值观
6. 过滤结果评估：评估过滤后内容的质量和适宜性

### 模块接口
- **输入接口**：待过滤文本、过滤参数、婴儿状态
- **输出接口**：过滤后文本、过滤标签、安全评分
- **依赖模块**：文本生成模块、文本编辑模块
- **被依赖模块**：语音合成模块、交互表达子系统

```python
# src/input_processing/input_processor.py

import re
import logging
from typing import Dict, Any, List

logger = logging.getLogger(__name__)

class InputProcessor:
    """输入处理器，负责处理输入文本和过滤参数"""
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.text_cleaners = [
            self._remove_html_tags,
            self._remove_urls,
            self._remove_emails,
            self._normalize_whitespace,
            self._remove_special_chars
        ]
    
    def process(self, text: str, filter_params: Dict[str, Any], baby_state: Dict[str, Any]) -> Dict[str, Any]:
        """
        处理输入文本和参数
        
        Args:
            text: 输入文本
            filter_params: 过滤参数
            baby_state: 婴儿状态
            
        Returns:
            处理后的输入数据
        """
        try:
            # 清理文本
            cleaned_text = self._clean_text(text)
            
            # 提取和验证过滤参数
            validated_params = self._validate_filter_params(filter_params)
            
            # 分析婴儿状态
            analyzed_state = self._analyze_baby_state(baby_state)
            
            # 构建处理结果
            result = {
                "text": cleaned_text,
                "original_text": text,
                "filter_params": validated_params,
                "baby_state": analyzed_state,
                "processing_metadata": {
                    "original_length": len(text),
                    "cleaned_length": len(cleaned_text),
                    "removed_chars": len(text) - len(cleaned_text)
                }
            }
            
            logger.info(f"Processed input: original_length={len(text)}, cleaned_length={len(cleaned_text)}")
            return result
            
        except Exception as e:
            logger.error(f"Error processing input: {str(e)}")
            raise
    
    def _clean_text(self, text: str) -> str:
        """清理文本"""
        cleaned = text
        for cleaner in self.text_cleaners:
            cleaned = cleaner(cleaned)
        return cleaned
    
    def _remove_html_tags(self, text: str) -> str:
        """移除HTML标签"""
        clean = re.compile('<.*?>')
        return re.sub(clean, '', text)
    
    def _remove_urls(self, text: str) -> str:
        """移除URL"""
        url_pattern = re.compile(r'http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\\(\\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+')
        return re.sub(url_pattern, '', text)
    
    def _remove_emails(self, text: str) -> str:
        """移除电子邮件地址"""
        email_pattern = re.compile(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b')
        return re.sub(email_pattern, '', text)
    
    def _normalize_whitespace(self, text: str) -> str:
        """标准化空白字符"""
        return ' '.join(text.split())
    
    def _remove_special_chars(self, text: str) -> str:
        """移除特殊字符（保留基本标点）"""
        # 保留字母、数字、基本标点和空格
        return re.sub(r'[^a-zA-Z0-9\s.,!?;:\'-]', '', text)
    
    def _validate_filter_params(self, params: Dict[str, Any]) -> Dict[str, Any]:
        """验证过滤参数"""
        default_params = {
            "filter_safety": True,
            "filter_education": True,
            "filter_age": True,
            "filter_emotion": True,
            "filter_culture": True,
            "optimize_filter": True,
            "strictness": "medium"
        }
        
        # 合并默认参数和用户参数
        validated = {**default_params, **params}
        
        # 验证参数值
        if validated["strictness"] not in ["low", "medium", "high"]:
            validated["strictness"] = "medium"
        
        return validated
    
    def _analyze_baby_state(self, state: Dict[str, Any]) -> Dict[str, Any]:
        """分析婴儿状态"""
        analyzed = state.copy()
        
        # 确保年龄字段存在且有效
        if "age" not in analyzed or not isinstance(analyzed["age"], (int, float)):
            analyzed["age"] = 3  # 默认年龄
        
        # 确保文化背景字段存在
        if "culture" not in analyzed or not analyzed["culture"]:
            analyzed["culture"] = self.config.get("culture_filter", {}).get("default_culture", "western")
        
        # 分析情感状态
        if "emotion" not in analyzed:
            analyzed["emotion"] = "neutral"
        
        return analyzed
```

# src/analysis/text_analyzer.py

import spacy
import nltk
from typing import Dict, Any, List
import logging

logger = logging.getLogger(__name__)

class TextAnalyzer:
    """文本分析器，负责分析文本特征和潜在风险"""
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        try:
            self.nlp = spacy.load("en_core_web_sm")
        except OSError:
            logger.error("spaCy model not found. Please run: python -m spacy download en_core_web_sm")
            raise
        
        # 初始化NLTK
        try:
            nltk.data.find('tokenizers/punkt')
        except LookupError:
            nltk.download('punkt')
        
        try:
            nltk.data.find('corpora/stopwords')
        except LookupError:
            nltk.download('stopwords')
    
    def analyze(self, text: str) -> Dict[str, Any]:
        """
        分析文本特征和潜在风险
        
        Args:
            text: 输入文本
            
        Returns:
            文本分析结果
        """
        try:
            # 基本文本统计
            basic_stats = self._get_basic_stats(text)
            
            # 语言特征分析
            linguistic_features = self._extract_linguistic_features(text)
            
            # 情感分析
            sentiment = self._analyze_sentiment(text)
            
            # 主题分析
            topics = self._analyze_topics(text)
            
            # 风险检测
            risks = self._detect_risks(text)
            
            # 构建分析结果
            result = {
                "basic_stats": basic_stats,
                "linguistic_features": linguistic_features,
                "sentiment": sentiment,
                "topics": topics,
                "risks": risks,
                "analysis_metadata": {
                    "text_length": len(text),
                    "word_count": basic_stats["word_count"],
                    "sentence_count": basic_stats["sentence_count"]
                }
            }
            
            logger.info(f"Analyzed text: length={len(text)}, words={basic_stats['word_count']}")
            return result
            
        except Exception as e:
            logger.error(f"Error analyzing text: {str(e)}")
            raise
    
    def _get_basic_stats(self, text: str) -> Dict[str, Any]:
        """获取基本文本统计信息"""
        words = text.split()
        sentences = nltk.sent_tokenize(text)
        
        return {
            "char_count": len(text),
            "word_count": len(words),
            "sentence_count": len(sentences),
            "avg_word_length": sum(len(word) for word in words) / len(words) if words else 0,
            "avg_sentence_length": sum(len(sent.split()) for sent in sentences) / len(sentences) if sentences else 0
        }
    
    def _extract_linguistic_features(self, text: str) -> Dict[str, Any]:
        """提取语言特征"""
        doc = self.nlp(text)
        
        # 词性标注统计
        pos_counts = {}
        for token in doc:
            pos = token.pos_
            pos_counts[pos] = pos_counts.get(pos, 0) + 1
        
        # 命名实体识别
        entities = [{"text": ent.text, "label": ent.label_} for ent in doc.ents]
        
        # 词汇多样性
        tokens = [token.text.lower() for token in doc if not token.is_stop and not token.is_punct]
        unique_tokens = set(tokens)
        lexical_diversity = len(unique_tokens) / len(tokens) if tokens else 0
        
        return {
            "pos_counts": pos_counts,
            "entities": entities,
            "lexical_diversity": lexical_diversity,
            "token_count": len(doc)
        }
    
    def _analyze_sentiment(self, text: str) -> Dict[str, Any]:
        """分析情感"""
        # 这里可以使用更复杂的情感分析模型
        # 简化实现，实际应用中应使用预训练模型
        
        # 简单的关键词情感分析
        positive_words = ["happy", "joy", "love", "good", "great", "wonderful", "excellent"]
        negative_words = ["sad", "angry", "fear", "bad", "terrible", "awful", "hate"]
        
        words = text.lower().split()
        positive_count = sum(1 for word in words if word in positive_words)
        negative_count = sum(1 for word in words if word in negative_words)
        
        # 计算情感分数
        total_sentiment_words = positive_count + negative_count
        if total_sentiment_words == 0:
            sentiment_score = 0.0
        else:
            sentiment_score = (positive_count - negative_count) / total_sentiment_words
        
        # 确定情感类别
        if sentiment_score > 0.1:
            sentiment = "positive"
        elif sentiment_score < -0.1:
            sentiment = "negative"
        else:
            sentiment = "neutral"
        
        return {
            "sentiment": sentiment,
            "sentiment_score": sentiment_score,
            "positive_words": positive_count,
            "negative_words": negative_count
        }
    
    def _analyze_topics(self, text: str) -> List[str]:
        """分析主题"""
        # 简化的主题分析，实际应用中应使用LDA等主题模型
        doc = self.nlp(text)
        
        # 提取名词和形容词作为潜在主题词
        topic_words = [token.text.lower() for token in doc 
                      if token.pos_ in ["NOUN", "ADJ"] and not token.is_stop]
        
        # 返回出现频率最高的主题词
        from collections import Counter
        word_counts = Counter(topic_words)
        return [word for word, count in word_counts.most_common(5)]
    
    def _detect_risks(self, text: str) -> List[Dict[str, Any]]:
        """检测潜在风险"""
        risks = []
        text_lower = text.lower()
        
        # 检测暴力相关词汇
        violence_keywords = ["kill", "fight", "hit", "hurt", "weapon", "gun", "knife"]
        violence_found = [word for word in violence_keywords if word in text_lower]
        if violence_found:
            risks.append({
                "type": "violence",
                "keywords": violence_found,
                "severity": "high"
            })
        
        # 检测恐怖相关词汇
        horror_keywords = ["scary", "monster", "ghost", "devil", "fear", "terror"]
        horror_found = [word for word in horror_keywords if word in text_lower]
        if horror_found:
            risks.append({
                "type": "horror",
                "keywords": horror_found,
                "severity": "medium"
            })
        
        # 检测不当语言
        inappropriate_keywords = ["damn", "hell", "stupid", "idiot", "shut up"]
        inappropriate_found = [word for word in inappropriate_keywords if word in text_lower]
        if inappropriate_found:
            risks.append({
                "type": "inappropriate_language",
                "keywords": inappropriate_found,
                "severity": "medium"
            })
        
        return risks

```python
# src/filtering/safety_filter.py

import json
import re
import logging
from typing import Dict, Any, List
from onnxruntime import InferenceSession
import numpy as np

logger = logging.getLogger(__name__)

class SafetyFilter:
    """安全过滤器，负责过滤不适宜的安全内容"""
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.strictness = config.get("strictness", "medium")
        self.custom_rules_file = config.get("custom_rules_file", "data/safety_rules.json")
        
        # 加载自定义规则
        self.custom_rules = self._load_custom_rules()
        
        # 初始化安全分类模型
        self.model_path = config.get("model_path", "models/safety_classifier.onnx")
        try:
            self.model = InferenceSession(self.model_path)
            logger.info(f"Loaded safety classifier model from {self.model_path}")
        except Exception as e:
            logger.warning(f"Failed to load safety classifier model: {str(e)}")
            self.model = None
        
        # 设置严格度阈值
        self.thresholds = {
            "low": 0.7,
            "medium": 0.5,
            "high": 0.3
        }
        self.current_threshold = self.thresholds.get(self.strictness, 0.5)
    
    def filter(self, text: str) -> Dict[str, Any]:
        """
        过滤文本中的不安全内容
        
        Args:
            text: 输入文本
            
        Returns:
            过滤结果
        """
        try:
            # 检测不安全内容
            safety_issues = self._detect_safety_issues(text)
            
            # 如果没有安全问题，直接返回
            if not safety_issues:
                return {
                    "is_filtered": False,
                    "filtered_text": text,
                    "detected_issues": [],
                    "safety_score": 1.0
                }
            
            # 过滤不安全内容
            filtered_text = self._filter_safety_issues(text, safety_issues)
            
            # 计算安全评分
            safety_score = self._calculate_safety_score(text, safety_issues)
            
            result = {
                "is_filtered": True,
                "filtered_text": filtered_text,
                "detected_issues": safety_issues,
                "safety_score": safety_score
            }
            
            logger.info(f"Filtered safety issues: {len(safety_issues)}, safety_score: {safety_score}")
            return result
            
        except Exception as e:
            logger.error(f"Error filtering safety issues: {str(e)}")
            raise
    
    def _load_custom_rules(self) -> Dict[str, Any]:
        """加载自定义规则"""
        try:
            with open(self.custom_rules_file, 'r', encoding='utf-8') as f:
                rules = json.load(f)
            logger.info(f"Loaded {len(rules)} custom safety rules")
            return rules
        except FileNotFoundError:
            logger.warning(f"Custom rules file not found: {self.custom_rules_file}")
            return {}
        except json.JSONDecodeError as e:
            logger.error(f"Error parsing custom rules file: {str(e)}")
            return {}
    
    def _detect_safety_issues(self, text: str) -> List[Dict[str, Any]]:
        """检测文本中的安全问题"""
        issues = []
        text_lower = text.lower()
        
        # 使用自定义规则检测
        for category, rules in self.custom_rules.items():
            for rule in rules:
                if rule.get("enabled", True):
                    keywords = rule.get("keywords", [])
                    pattern = rule.get("pattern")
                    severity = rule.get("severity", "medium")
                    
                    # 关键词匹配
                    if keywords:
                        matched_keywords = [kw for kw in keywords if kw in text_lower]
                        if matched_keywords:
                            issues.append({
                                "type": category,
                                "rule_id": rule.get("id"),
                                "detection_method": "keyword",
                                "matched_keywords": matched_keywords,
                                "severity": severity,
                                "description": rule.get("description", "")
                            })
                    
                    # 正则表达式匹配
                    if pattern:
                        matches = re.findall(pattern, text, re.IGNORECASE)
                        if matches:
                            issues.append({
                                "type": category,
                                "rule_id": rule.get("id"),
                                "detection_method": "regex",
                                "matched_patterns": matches,
                                "severity": severity,
                                "description": rule.get("description", "")
                            })
        
        # 使用模型检测
        if self.model:
            model_issues = self._detect_with_model(text)
            issues.extend(model_issues)
        
        return issues
    
    def _detect_with_model(self, text: str) -> List[Dict[str, Any]]:
        """使用模型检测安全问题"""
        try:
            # 文本预处理
            # 这里应该使用与模型训练时相同的预处理方法
            # 简化实现
            processed_text = text.lower()[:512]  # 假设模型最大长度为512
            
            # 模型推理
            # 这里需要根据实际模型的输入输出格式进行调整
            # 简化实现，假设模型输出是各类别的概率
            input_data = np.array([processed_text], dtype=np.str_)
            
            # 获取模型输出
            outputs = self.model.run(None, {"input": input_data})
            probabilities = outputs[0][0]  # 假设第一个输出是概率数组
            
            # 定义类别标签
            labels = ["safe", "violence", "adult_content", "hate_speech", "self_harm"]
            
            # 检测超过阈值的问题
            issues = []
            for i, (label, prob) in enumerate(zip(labels, probabilities)):
                if label != "safe" and prob > self.current_threshold:
                    issues.append({
                        "type": label,
                        "rule_id": f"model_{label}",
                        "detection_method": "model",
                        "probability": float(prob),
                        "severity": "high" if prob > 0.8 else "medium"
                    })
            
            return issues
            
        except Exception as e:
            logger.error(f"Error detecting with model: {str(e)}")
            return []
    
    def _filter_safety_issues(self, text: str, issues: List[Dict[str, Any]]) -> str:
        """过滤文本中的安全问题"""
        filtered_text = text
        
        # 按严重程度排序问题
        sorted_issues = sorted(issues, key=lambda x: {"high": 3, "medium": 2, "low": 1}.get(x.get("severity", "low"), 1), reverse=True)
        
        # 处理每个问题
        for issue in sorted_issues:
            method = issue.get("detection_method")
            
            if method == "keyword":
                filtered_text = self._filter_by_keywords(filtered_text, issue.get("matched_keywords", []))
            elif method == "regex":
                filtered_text = self._filter_by_patterns(filtered_text, issue.get("matched_patterns", []))
            elif method == "model":
                # 对于模型检测的问题，可能需要更复杂的处理
                # 这里简化处理，替换整个文本
                filtered_text = self._replace_with_safe_content(filtered_text, issue.get("type"))
        
        return filtered_text
    
    def _filter_by_keywords(self, text: str, keywords: List[str]) -> str:
        """通过关键词过滤文本"""
        filtered_text = text
        for keyword in keywords:
            # 使用正则表达式进行不区分大小写的替换
            pattern = re.compile(re.escape(keyword), re.IGNORECASE)
            filtered_text = pattern.sub("***", filtered_text)
        return filtered_text
    
    def _filter_by_patterns(self, text: str, patterns: List[str]) -> str:
        """通过正则表达式模式过滤文本"""
        filtered_text = text
        for pattern in patterns:
            try:
                filtered_text = re.sub(pattern, "***", filtered_text, flags=re.IGNORECASE)
            except re.error as e:
                logger.warning(f"Invalid regex pattern: {pattern}, error: {str(e)}")
        return filtered_text
    
    def _replace_with_safe_content(self, text: str, issue_type: str) -> str:
        """用安全内容替换有问题的文本"""
        # 这里可以根据不同的问题类型提供不同的替换内容
        safe_replacements = {
            "violence": "This content has been filtered for safety.",
            "adult_content": "This content is not appropriate for children.",
            "hate_speech": "This content has been filtered for safety.",
            "self_harm": "This content has been filtered for safety."
        }
        
        replacement = safe_replacements.get(issue_type, "This content has been filtered for safety.")
        return replacement
    
    def _calculate_safety_score(self, text: str, issues: List[Dict[str, Any]]) -> float:
        """计算文本的安全评分"""
        if not issues:
            return 1.0
        
        # 基础分数
        base_score = 1.0
        
        # 根据问题严重程度扣分
        for issue in issues:
            severity = issue.get("severity", "medium")
            if severity == "high":
                base_score -= 0.3
            elif severity == "medium":
                base_score -= 0.2
            else:  # low
                base_score -= 0.1
        
        # 确保分数在0-1之间
        return max(0.0, min(1.0, base_score))
```

# src/text_filter_manager.py

import logging
from typing import Dict, Any, List
from .input_processing.input_processor import InputProcessor
from .analysis.text_analyzer import TextAnalyzer
from .filtering.safety_filter import SafetyFilter

logger = logging.getLogger(__name__)

class TextFilterManager:
    """文本过滤管理器，负责协调整个过滤流程"""
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        
        # 初始化各组件
        self.input_processor = InputProcessor(config.get("app", {}))
        self.text_analyzer = TextAnalyzer(config.get("app", {}))
        
        # 初始化过滤器
        filter_config = config.get("filter_config", {})
        self.safety_filter = SafetyFilter(filter_config.get("safety_filter", {}))
        self.education_filter = EducationFilter(filter_config.get("education_filter", {}))
        self.age_filter = AgeFilter(filter_config.get("age_filter", {}))
        self.emotion_filter = EmotionFilter(filter_config.get("emotion_filter", {}))
        self.culture_filter = CultureFilter(filter_config.get("culture_filter", {}))
        
        # 初始化优化器和评估器
        self.filter_optimizer = FilterOptimizer(filter_config.get("optimization", {}))
        self.safety_assessor = SafetyAssessor({})
    
    def filter_text(self, text: str, filter_params: Dict[str, Any], baby_state: Dict[str, Any]) -> Dict[str, Any]:
        """
        过滤文本内容
        
        Args:
            text: 输入文本
            filter_params: 过滤参数
            baby_state: 婴儿状态
            
        Returns:
            过滤结果
        """
        try:
            # 输入处理
            processed_input = self.input_processor.process(text, filter_params, baby_state)
            
            # 文本分析
            text_analysis = self.text_analyzer.analyze(processed_input["text"])
            
            # 过滤操作
            filtered_text = processed_input["text"]
            filter_history = []
            
            # 安全过滤
            if filter_params.get("filter_safety", True):
                safety_result = self.safety_filter.filter(filtered_text)
                if safety_result["is_filtered"]:
                    filtered_text = safety_result["filtered_text"]
                    filter_history.append({
                        "filter_type": "safety",
                        "original": processed_input["text"],
                        "filtered": filtered_text,
                        "detected_issues": safety_result["detected_issues"]
                    })
            
            # 教育过滤
            if filter_params.get("filter_education", True):
                education_result = self.education_filter.filter(filtered_text)
                if education_result["is_filtered"]:
                    filtered_text = education_result["filtered_text"]
                    filter_history.append({
                        "filter_type": "education",
                        "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                        "filtered": filtered_text,
                        "detected_issues": education_result["detected_issues"]
                    })
            
            # 年龄过滤
            if "age" in baby_state:
                age_result = self.age_filter.filter(filtered_text, baby_state["age"])
                if age_result["is_filtered"]:
                    filtered_text = age_result["filtered_text"]
                    filter_history.append({
                        "filter_type": "age",
                        "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                        "filtered": filtered_text,
                        "detected_issues": age_result["detected_issues"]
                    })
            
            # 情感过滤
            if filter_params.get("filter_emotion", True):
                emotion_result = self.emotion_filter.filter(filtered_text)
                if emotion_result["is_filtered"]:
                    filtered_text = emotion_result["filtered_text"]
                    filter_history.append({
                        "filter_type": "emotion",
                        "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                        "filtered": filtered_text,
                        "detected_issues": emotion_result["detected_issues"]
                    })
            
            # 文化过滤
            if filter_params.get("filter_culture", True) and "culture" in baby_state:
                culture_result = self.culture_filter.filter(filtered_text, baby_state["culture"])
                if culture_result["is_filtered"]:
                    filtered_text = culture_result["filtered_text"]
                    filter_history.append({
                        "filter_type": "culture",
                        "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                        "filtered": filtered_text,
                        "detected_issues": culture_result["detected_issues"]
                    })
            
            # 过滤优化
            if filter_params.get("optimize_filter", True):
                optimization_result = self.filter_optimizer.optimize(filtered_text, filter_history)
                filtered_text = optimization_result["optimized_text"]
                filter_history.append({
                    "filter_type": "optimization",
                    "original": filter_history[-1]["filtered"] if filter_history else processed_input["text"],
                    "filtered": filtered_text,
                    "optimization_applied": optimization_result["optimization_applied"]
                })
            
            # 安全评估
            safety_assessment = self.safety_assessor.assess(filtered_text, baby_state)
            
            # 构建结果
            result = {
                "original_text": text,
                "filtered_text": filtered_text,
                "is_filtered": len(filter_history) > 0,
                "filter_history": filter_history,
                "safety_assessment": safety_assessment,
                "text_analysis": text_analysis,
                "processing_metadata": {
                    "filter_count": len(filter_history),
                    "safety_score": safety_assessment.get("safety_score", 0.0),
                    "appropriate_for_age": safety_assessment.get("appropriate_for_age", False)
                }
            }
            
            logger.info(f"Text filtered successfully: filter_count={len(filter_history)}, safety_score={safety_assessment.get('safety_score', 0.0)}")
            return result
            
        except Exception as e:
            logger.error(f"Error filtering text: {str(e)}")
            raise
