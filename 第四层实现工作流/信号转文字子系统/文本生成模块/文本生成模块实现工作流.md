# 文本生成模块实现工作流

## 模块信息

### 模块名称
文本生成模块

### 所属子系统
信号转文字子系统

### 模块功能
负责根据认知决策子系统的指令，生成适合婴儿理解和互动的文本内容，包括故事、歌谣、安抚话语等。

### 模块职责
1. 文本规划：根据互动目标和婴儿状态规划文本内容
2. 内容生成：生成适合婴儿的互动文本内容
3. 风格调整：根据婴儿特点调整文本风格和复杂度
4. 情感注入：为生成的文本注入适当的情感色彩
5. 多样性控制：控制生成文本的多样性和新颖性
6. 质量评估：评估生成文本的质量和适宜性

### 模块接口
- **输入接口**：互动指令、婴儿状态、生成参数
- **输出接口**：生成文本、文本标签、质量评分
- **依赖模块**：认知决策子系统
- **被依赖模块**：语音合成模块、交互表达子系统

#### 1. 文本生成管理器 (TextGenerationManager)
创建 `src/text_generation_manager.py`:
```python
import time
from typing import Dict, Any, Optional
from .input_processing.input_processor import InputProcessor
from .planning.text_planner import TextPlanner
from .generation.text_generator import TextGenerator
from .optimization.text_optimizer import TextOptimizer
from .emotion_injection.emotion_injector import EmotionInjector
from .quality_assessment.quality_assessor import QualityAssessor
from .utils.config import load_config

class TextGenerationManager:
    """文本生成模块管理器，负责协调整个文本生成流程"""
    
    def __init__(self, config_path: str = "config/model_config.yaml"):
        """初始化文本生成管理器
        
        Args:
            config_path: 配置文件路径
        """
        self.config = load_config(config_path)
        
        # 初始化各组件
        self.input_processor = InputProcessor(self.config)
        self.text_planner = TextPlanner(self.config)
        self.text_generator = TextGenerator(self.config)
        self.text_optimizer = TextOptimizer(self.config)
        self.emotion_injector = EmotionInjector(self.config)
        self.quality_assessor = QualityAssessor(self.config)
        
        # 统计信息
        self.stats = {
            "total_requests": 0,
            "successful_generations": 0,
            "failed_generations": 0,
            "average_quality_score": 0.0,
            "average_generation_time": 0.0
        }
    
    def generate_text(self, instruction: str, baby_state: Dict[str, Any], 
                     generation_params: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
        """生成文本内容
        
        Args:
            instruction: 互动指令
            baby_state: 婴儿状态信息
            generation_params: 生成参数
            
        Returns:
            包含生成文本和元信息的字典
        """
        start_time = time.time()
        self.stats["total_requests"] += 1
        
        try:
            # 1. 输入处理
            processed_input = self.input_processor.process(instruction, baby_state, generation_params)
            
            # 2. 文本规划
            text_plan = self.text_planner.plan(processed_input)
            
            # 3. 文本生成
            generated_text = self.text_generator.generate(text_plan)
            
            # 4. 文本优化
            optimized_text = self.text_optimizer.optimize(generated_text, text_plan)
            
            # 5. 情感注入
            emotion_text = self.emotion_injector.inject(optimized_text, text_plan)
            
            # 6. 质量评估
            quality_result = self.quality_assessor.assess(emotion_text, text_plan)
            
            # 如果质量不达标，重新生成
            attempts = 1
            max_attempts = self.config["quality"]["max_attempts"]
            while quality_result["score"] < self.config["quality"]["min_score"] and attempts < max_attempts:
                generated_text = self.text_generator.generate(text_plan)
                optimized_text = self.text_optimizer.optimize(generated_text, text_plan)
                emotion_text = self.emotion_injector.inject(optimized_text, text_plan)
                quality_result = self.quality_assessor.assess(emotion_text, text_plan)
                attempts += 1
            
            # 更新统计信息
            generation_time = time.time() - start_time
            self._update_stats(quality_result["score"], generation_time, True)
            
            return {
                "text": emotion_text,
                "type": text_plan["type"],
                "emotion": text_plan.get("emotion", None),
                "quality_score": quality_result["score"],
                "quality_details": quality_result["details"],
                "generation_time": generation_time,
                "attempts": attempts,
                "timestamp": time.time()
            }
            
        except Exception as e:
            # 更新统计信息
            generation_time = time.time() - start_time
            self._update_stats(0, generation_time, False)
            
            # 记录错误
            self._log_error(e, instruction, baby_state)
            
            # 返回错误信息
            return {
                "text": "",
                "type": "error",
                "error": str(e),
                "generation_time": generation_time,
                "timestamp": time.time()
            }
    
    def _update_stats(self, quality_score: float, generation_time: float, success: bool):
        """更新统计信息"""
        if success:
            self.stats["successful_generations"] += 1
            
            # 更新平均质量分数
            total_successful = self.stats["successful_generations"]
            current_avg = self.stats["average_quality_score"]
            self.stats["average_quality_score"] = (
                (current_avg * (total_successful - 1) + quality_score) / total_successful
            )
            
            # 更新平均生成时间
            total_requests = self.stats["total_requests"]
            current_avg_time = self.stats["average_generation_time"]
            self.stats["average_generation_time"] = (
                (current_avg_time * (total_requests - 1) + generation_time) / total_requests
            )
        else:
            self.stats["failed_generations"] += 1
    
    def _log_error(self, error: Exception, instruction: str, baby_state: Dict[str, Any]):
        """记录错误信息"""
        # 在实际应用中，这里应该使用日志系统
        print(f"文本生成错误: {error}")
        print(f"指令: {instruction}")
        print(f"婴儿状态: {baby_state}")
    
    def get_stats(self) -> Dict[str, Any]:
        """获取统计信息"""
        return self.stats.copy()
    
    def reset_stats(self):
        """重置统计信息"""
        self.stats = {
            "total_requests": 0,
            "successful_generations": 0,
            "failed_generations": 0,
            "average_quality_score": 0.0,
            "average_generation_time": 0.0
        }
```

```python
from typing import Dict, Any, Optional
from .state_analyzer import StateAnalyzer

class InputProcessor:
    """输入处理器，负责处理输入指令和婴儿状态信息"""
    
    def __init__(self, config: Dict[str, Any]):
        """初始化输入处理器
        
        Args:
            config: 配置信息
        """
        self.config = config
        self.state_analyzer = StateAnalyzer(config)
    
    def process(self, instruction: str, baby_state: Dict[str, Any], 
                generation_params: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
        """处理输入信息
        
        Args:
            instruction: 互动指令
            baby_state: 婴儿状态信息
            generation_params: 生成参数
            
        Returns:
            处理后的输入信息
        """
        # 解析指令
        parsed_instruction = self._parse_instruction(instruction)
        
        # 分析婴儿状态
        analyzed_state = self.state_analyzer.analyze(baby_state)
        
        # 处理生成参数
        processed_params = self._process_generation_params(generation_params)
        
        return {
            "instruction": parsed_instruction,
            "baby_state": analyzed_state,
            "generation_params": processed_params
        }
    
    def _parse_instruction(self, instruction: str) -> Dict[str, Any]:
        """解析互动指令"""
        # 在实际应用中，这里应该使用NLP技术解析指令
        # 这里简化为基于关键词的解析
        instruction_lower = instruction.lower()
        
        # 确定文本类型
        text_type = "general"
        if "故事" in instruction_lower or "story" in instruction_lower:
            text_type = "story"
        elif "歌谣" in instruction_lower or "rhyme" in instruction_lower:
            text_type = "rhyme"
        elif "问答" in instruction_lower or "question" in instruction_lower:
            text_type = "qa"
        elif "安抚" in instruction_lower or "comfort" in instruction_lower:
            text_type = "comfort"
        
        # 确定情感
        emotion = None
        if "开心" in instruction_lower or "happy" in instruction_lower:
            emotion = "happy"
        elif "安抚" in instruction_lower or "comfort" in instruction_lower:
            emotion = "comforting"
        elif "兴奋" in instruction_lower or "excited" in instruction_lower:
            emotion = "excited"
        
        return {
            "raw_text": instruction,
            "type": text_type,
            "emotion": emotion,
            "keywords": self._extract_keywords(instruction)
        }
    
    def _extract_keywords(self, instruction: str) -> list:
        """提取关键词"""
        # 在实际应用中，这里应该使用关键词提取算法
        # 这里简化为简单的分词和过滤
        import re
        # 移除标点符号
        cleaned = re.sub(r'[^\w\s]', '', instruction)
        # 分词
        words = cleaned.split()
        # 过滤停用词
        stop_words = ["的", "了", "是", "在", "有", "和", "与", "或", "但", "而", "就", "都", "要", "会", "能", "可以", "the", "a", "an", "and", "or", "but", "in", "on", "at", "to", "for", "of", "with", "by", "is", "are", "was", "were", "be", "been", "have", "has", "had", "do", "does", "did", "will", "would", "could", "should", "may", "might", "must", "can", "this", "that", "these", "those"]
        keywords = [word for word in words if word.lower() not in stop_words and len(word) > 1]
        return keywords
    
    def _process_generation_params(self, generation_params: Optional[Dict[str, Any]]) -> Dict[str, Any]:
        """处理生成参数"""
        # 默认参数
        default_params = {
            "max_length": self.config["generation"]["default_length"],
            "temperature": self.config["model"]["temperature"],
            "top_p": self.config["model"]["top_p"],
            "top_k": self.config["model"]["top_k"]
        }
        
        # 如果提供了参数，则更新默认参数
        if generation_params:
            default_params.update(generation_params)
        
        return default_params
```

```python
from typing import Dict, Any
from .content_structurer import ContentStructurer

class TextPlanner:
    """文本规划器，负责规划文本内容、风格和结构"""
    
    def __init__(self, config: Dict[str, Any]):
        """初始化文本规划器
        
        Args:
            config: 配置信息
        """
        self.config = config
        self.content_structurer = ContentStructurer(config)
    
    def plan(self, processed_input: Dict[str, Any]) -> Dict[str, Any]:
        """规划文本内容
        
        Args:
            processed_input: 处理后的输入信息
            
        Returns:
            文本规划信息
        """
        instruction = processed_input["instruction"]
        baby_state = processed_input["baby_state"]
        generation_params = processed_input["generation_params"]
        
        # 确定文本类型
        text_type = instruction["type"]
        
        # 确定文本风格
        style = self._determine_style(baby_state)
        
        # 确定文本复杂度
        complexity = self._determine_complexity(baby_state)
        
        # 确定文本长度
        length = self._determine_length(text_type, baby_state, generation_params)
        
        # 确定情感
        emotion = instruction.get("emotion") or self._determine_emotion(baby_state)
        
        # 规划内容结构
        content_structure = self.content_structurer.structure(text_type, baby_state)
        
        return {
            "type": text_type,
            "style": style,
            "complexity": complexity,
            "length": length,
            "emotion": emotion,
            "content_structure": content_structure,
            "keywords": instruction["keywords"],
            "generation_params": generation_params
        }
    
    def _determine_style(self, baby_state: Dict[str, Any]) -> str:
        """确定文本风格"""
        age = baby_state.get("age_months", 24)
        
        if age < 12:
            return "very_simple"
        elif age < 24:
            return "simple"
        elif age < 36:
            return "moderate"
        else:
            return "advanced"
    
    def _determine_complexity(self, baby_state: Dict[str, Any]) -> str:
        """确定文本复杂度"""
        age = baby_state.get("age_months", 24)
        mood = baby_state.get("mood", "neutral")
        
        # 基础复杂度基于年龄
        if age < 12:
            base_complexity = "low"
        elif age < 24:
            base_complexity = "low"
        elif age < 36:
            base_complexity = "medium"
        else:
            base_complexity = "medium"
        
        # 根据情绪调整复杂度
        if mood in ["tired", "fussy"]:
            # 如果婴儿疲倦或烦躁，降低复杂度
            if base_complexity == "medium":
                return "low"
            elif base_complexity == "high":
                return "medium"
        
        return base_complexity
    
    def _determine_length(self, text_type: str, baby_state: Dict[str, Any], 
                         generation_params: Dict[str, Any]) -> Dict[str, Any]:
        """确定文本长度"""
        age = baby_state.get("age_months", 24)
        attention_span = baby_state.get("attention_span", "normal")
        
        # 基础长度基于年龄
        if age < 12:
            base_min_length = 10
            base_max_length = 50
        elif age < 24:
            base_min_length = 20
            base_max_length = 100
        elif age < 36:
            base_min_length = 30
            base_max_length = 150
        else:
            base_min_length = 40
            base_max_length = 200
        
        # 根据注意力调整长度
        if attention_span == "short":
            base_max_length = int(base_max_length * 0.7)
        elif attention_span == "long":
            base_max_length = int(base_max_length * 1.3)
        
        # 根据文本类型调整长度
        if text_type == "story":
            base_min_length = max(base_min_length, 30)
            base_max_length = max(base_max_length, 100)
        elif text_type == "rhyme":
            base_min_length = max(base_min_length, 20)
            base_max_length = max(base_max_length, 80)
        elif text_type == "qa":
            base_min_length = max(base_min_length, 10)
            base_max_length = max(base_max_length, 50)
        
        # 应用生成参数中的长度设置
        if "max_length" in generation_params:
            base_max_length = min(base_max_length, generation_params["max_length"])
        
        return {
            "min_length": base_min_length,
            "max_length": base_max_length,
            "target_length": (base_min_length + base_max_length) // 2
        }
    
    def _determine_emotion(self, baby_state: Dict[str, Any]) -> str:
        """确定文本情感"""
        mood = baby_state.get("mood", "neutral")
        
        # 根据婴儿情绪确定文本情感
        if mood == "happy":
            return "happy"
        elif mood == "sad":
            return "comforting"
        elif mood == "excited":
            return "excited"
        elif mood == "tired":
            return "calming"
        elif mood == "fussy":
            return "soothing"
        else:
            return "neutral"
```

```python
import torch
from typing import Dict, Any
from transformers import AutoTokenizer, AutoModelForCausalLM
from .story_generator import StoryGenerator
from .rhyme_generator import RhymeGenerator
from .qa_generator import QAGenerator

class TextGenerator:
    """文本生成器，负责生成基础文本内容"""
    
    def __init__(self, config: Dict[str, Any]):
        """初始化文本生成器
        
        Args:
            config: 配置信息
        """
        self.config = config
        
        # 初始化基础模型
        self.base_model_name = config["model"]["base_model"]
        self.tokenizer = AutoTokenizer.from_pretrained(self.base_model_name)
        self.base_model = AutoModelForCausalLM.from_pretrained(self.base_model_name)
        
        # 初始化专用生成器
        self.story_generator = StoryGenerator(config)
        self.rhyme_generator = RhymeGenerator(config)
        self.qa_generator = QAGenerator(config)
        
        # 设置特殊标记
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token
    
    def generate(self, text_plan: Dict[str, Any]) -> str:
        """生成文本
        
        Args:
            text_plan: 文本规划信息
            
        Returns:
            生成的文本
        """
        text_type = text_plan["type"]
        
        # 根据文本类型选择生成器
        if text_type == "story":
            return self.story_generator.generate(text_plan)
        elif text_type == "rhyme":
            return self.rhyme_generator.generate(text_plan)
        elif text_type == "qa":
            return self.qa_generator.generate(text_plan)
        else:
            return self._generate_base(text_plan)
    
    def _generate_base(self, text_plan: Dict[str, Any]) -> str:
        """使用基础模型生成文本
        
        Args:
            text_plan: 文本规划信息
            
        Returns:
            生成的文本
        """
        # 准备输入
        prompt = self._prepare_prompt(text_plan)
        
        # 生成参数
        generation_params = {
            "max_length": text_plan["length"]["max_length"],
            "min_length": text_plan["length"]["min_length"],
            "temperature": text_plan["generation_params"].get("temperature", self.config["model"]["temperature"]),
            "top_p": text_plan["generation_params"].get("top_p", self.config["model"]["top_p"]),
            "top_k": text_plan["generation_params"].get("top_k", self.config["model"]["top_k"]),
            "repetition_penalty": self.config["generation"]["repetition_penalty"],
            "num_beams": self.config["generation"]["num_beams"],
            "early_stopping": self.config["generation"]["early_stopping"],
            "do_sample": True,
            "pad_token_id": self.tokenizer.pad_token_id
        }
        
        # 编码输入
        inputs = self.tokenizer(prompt, return_tensors="pt")
        
        # 生成文本
        with torch.no_grad():
            outputs = self.base_model.generate(
                inputs.input_ids,
                **generation_params
            )
        
        # 解码输出
        generated_text = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        # 移除输入部分
        if generated_text.startswith(prompt):
            generated_text = generated_text[len(prompt):].strip()
        
        return generated_text
    
    def _prepare_prompt(self, text_plan: Dict[str, Any]) -> str:
        """准备生成提示
        
        Args:
            text_plan: 文本规划信息
            
        Returns:
            生成提示
        """
        text_type = text_plan["type"]
        keywords = text_plan["keywords"]
        emotion = text_plan["emotion"]
        
        # 基础提示
        if text_type == "story":
            prompt = "讲一个适合婴儿的故事："
```

```python
from typing import Dict, Any
from .style_adjuster import StyleAdjuster
from .diversity_controller import DiversityController

class TextOptimizer:
    """文本优化器，负责优化文本质量、调整风格和控制多样性"""
    
    def __init__(self, config: Dict[str, Any]):
        """初始化文本优化器
        
        Args:
            config: 配置信息
        """
        self.config = config
        self.style_adjuster = StyleAdjuster(config)
        self.diversity_controller = DiversityController(config)
    
    def optimize(self, text: str, text_plan: Dict[str, Any]) -> str:
        """优化文本
        
        Args:
            text: 原始文本
            text_plan: 文本规划信息
            
        Returns:
            优化后的文本
        """
        # 1. 风格调整
        if "style" in text_plan:
            text = self.style_adjuster.adjust(text, text_plan["style"])
        
        # 2. 复杂度调整
        if "complexity" in text_plan:
            text = self._adjust_complexity(text, text_plan["complexity"])
        
        # 3. 多样性控制
        if "diversity" in text_plan:
            text = self.diversity_controller.control(text, text_plan["diversity"])
        
        # 4. 长度调整
        if "length" in text_plan:
            text = self._adjust_length(text, text_plan["length"])
        
        return text
    
    def _adjust_complexity(self, text: str, complexity_level: str) -> str:
        """调整文本复杂度
        
        Args:
            text: 原始文本
            complexity_level: 复杂度级别 (low, medium, high)
            
        Returns:
            调整后的文本
        """
        if complexity_level == "low":
            # 简化词汇和句子结构
            text = self._simplify_vocabulary(text)
            text = self._simplify_sentences(text)
        elif complexity_level == "medium":
            # 适度调整
            text = self._moderate_adjustment(text)
        elif complexity_level == "high":
            # 保持或增加复杂度
            text = self._maintain_complexity(text)
        
        return text
    
    def _simplify_vocabulary(self, text: str) -> str:
        """简化词汇"""
        # 在实际应用中，这里应该使用词汇替换表
        # 这里简化为一些常见替换
        replacements = {
            "美丽": "好看",
            "漂亮": "好看",
            "聪明": "棒",
            "高兴": "开心",
            "悲伤": "难过",
            "惊讶": "吃惊",
            "害怕": "紧张",
            "生气": "不开心"
        }
        
        for complex_word, simple_word in replacements.items():
            text = text.replace(complex_word, simple_word)
        
        return text
    
    def _simplify_sentences(self, text: str) -> str:
        """简化句子结构"""
        import re
        
        # 分割句子
        sentences = re.split(r'[。！？]', text)
        
        # 简化每个句子
        simplified_sentences = []
        for sentence in sentences:
            sentence = sentence.strip()
            if not sentence:
                continue
            
            # 移除复杂的从句结构
            # 这里简化为移除一些连词
            connectors = ["因为", "所以", "虽然", "但是", "然而", "因此", "而且", "并且"]
            for connector in connectors:
                sentence = sentence.replace(connector, "")
            
            # 确保句子不要太长
            if len(sentence) > 20:
                # 尝试在适当位置分割
                words = sentence.split()
                if len(words) > 10:
                    # 简单地在中间分割
                    mid = len(words) // 2
                    sentence = " ".join(words[:mid]) + "。" + " ".join(words[mid:])
            
            simplified_sentences.append(sentence)
        
        # 重新组合句子
        return "。".join(simplified_sentences) + "。"
    
    def _moderate_adjustment(self, text: str) -> str:
        """适度调整文本"""
        # 保持大部分内容不变，只做少量调整
        return text
    
    def _maintain_complexity(self, text: str) -> str:
        """保持或增加文本复杂度"""
        # 保持原始复杂度
        return text
    
    def _adjust_length(self, text: str, length_config: Dict[str, Any]) -> str:
        """调整文本长度
        
        Args:
            text: 原始文本
            length_config: 长度配置
            
        Returns:
            调整后的文本
        """
        current_length = len(text)
        min_length = length_config["min_length"]
        max_length = length_config["max_length"]
        
        # 如果当前长度在范围内，不做调整
        if min_length <= current_length <= max_length:
            return text
        
        # 如果太短，尝试扩展
        if current_length < min_length:
            return self._expand_text(text, min_length)
        
        # 如果太长，尝试缩减
        if current_length > max_length:
            return self._shorten_text(text, max_length)
        
        return text
    
    def _expand_text(self, text: str, target_length: int) -> str:
        """扩展文本长度"""
        # 在实际应用中，这里应该使用更智能的扩展方法
        # 这里简化为添加一些简单的填充词
        filler_phrases = [
            "宝宝，",
            "你看，",
            "听我说，",
            "哇，",
            "嗯，",
            "哦，"
        ]
        
        # 计算需要添加的长度
        needed_length = target_length - len(text)
        
        # 添加填充词
        expanded_text = text
        while len(expanded_text) < target_length:
            for phrase in filler_phrases:
                if len(expanded_text) + len(phrase) <= target_length:
                    expanded_text = phrase + expanded_text
                else:
                    break
        
        return expanded_text
    
    def _shorten_text(self, text: str, target_length: int) -> str:
        """缩减文本长度"""
        # 简单截断
        return text[:target_length] + "。"
```

常见HTTP状态码：

- 200 OK：请求成功

- 401 Unauthorized：未授权
- 403 Forbidden：禁止访问
- 422 Unprocessable Entity：请求参数验证失败
- 500 Internal Server Error：服务器内部错误
