# src/decision_maker.py

from typing import Dict, List, Any, Optional, Tuple
from langchain.llms import OpenAI
from langchain.chains import LLMChain
from langchain.prompts import PromptTemplate
import numpy as np
import json
import logging
from datetime import datetime

from .information_fusion import InformationFusion
from .context_analyzer import ContextAnalyzer
from .solution_generator import SolutionGenerator
from .evaluator import Evaluator
from .selector import Selector
from .models.decision_model import DecisionContext, DecisionOption, DecisionResult
from ..utils.cache import CacheManager
from ..utils.database import DatabaseManager

class DecisionMaker:
    """
    决策制定主类，负责协调各个组件完成决策过程
    """
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化决策制定器
        
        Args:
            config: 配置字典，包含模型参数、数据库连接等
        """
        self.config = config
        self.logger = logging.getLogger(__name__)
        
        # 初始化语言模型
        self.llm = OpenAI(
            model_name=config.get("model_name", "gpt-3.5-turbo-instruct"),
            temperature=config.get("temperature", 0.3),
            max_tokens=config.get("max_tokens", 1000)
        )
        
        # 初始化组件
        self.information_fusion = InformationFusion(config)
        self.context_analyzer = ContextAnalyzer(config)
        self.solution_generator = SolutionGenerator(config, self.llm)
        self.evaluator = Evaluator(config)
        self.selector = Selector(config)
        
        # 初始化管理器
        self.cache_manager = CacheManager(config.get("cache", {}))
        self.db_manager = DatabaseManager(config.get("database", {}))
        
        # 决策提示模板
        self.decision_prompt = PromptTemplate(
            input_variables=["context", "options", "criteria"],
            template="""
            基于以下情境和选项，请做出最佳决策：
            
            情境描述：{context}
            
            可选方案：
            {options}
            
            评估标准：
            {criteria}
            
            请分析每个方案的优缺点，并推荐最佳方案，同时说明理由。
            """
        )
        
        self.decision_chain = LLMChain(
            llm=self.llm,
            prompt=self.decision_prompt
        )
    
    async def make_decision(
        self, 
        context: DecisionContext, 
        options: Optional[List[DecisionOption]] = None,
        criteria: Optional[List[str]] = None
    ) -> DecisionResult:
        """
        执行决策过程
        
        Args:
            context: 决策情境
            options: 可选决策方案，如果为None则自动生成
            criteria: 评估标准，如果为None则使用默认标准
            
        Returns:
            DecisionResult: 决策结果
        """
        try:
            # 1. 信息融合
            self.logger.info("开始信息融合阶段")
            fused_info = await self.information_fusion.fuse_information(context)
            
            # 2. 情境分析
            self.logger.info("开始情境分析阶段")
            analyzed_context = await self.context_analyzer.analyze_context(fused_info)
            
            # 3. 方案生成（如果未提供）
            if options is None:
                self.logger.info("开始方案生成阶段")
                options = await self.solution_generator.generate_solutions(analyzed_context)
            
            # 4. 方案评估
            self.logger.info("开始方案评估阶段")
            evaluated_options = await self.evaluator.evaluate_options(
                analyzed_context, options, criteria
            )
            
            # 5. 方案选择
            self.logger.info("开始方案选择阶段")
            selected_option = await self.selector.select_option(
                analyzed_context, evaluated_options
            )
            
            # 6. 生成决策结果
            decision_result = DecisionResult(
                context=analyzed_context,
                selected_option=selected_option,
                all_options=evaluated_options,
                decision_time=datetime.now(),
                reasoning=selected_option.reasoning
            )
            
            # 7. 缓存决策结果
            await self.cache_decision_result(decision_result)
            
            # 8. 持久化决策记录
            await self.persist_decision_record(decision_result)
            
            self.logger.info(f"决策完成，选择方案: {selected_option.title}")
            return decision_result
            
        except Exception as e:
            self.logger.error(f"决策过程中发生错误: {str(e)}")
            raise
    
    async def get_decision_history(
        self, 
        context_id: Optional[str] = None, 
        limit: int = 10
    ) -> List[DecisionResult]:
        """
        获取决策历史
        
        Args:
            context_id: 情境ID，如果提供则只返回该情境的决策历史
            limit: 返回的最大记录数
            
        Returns:
            List[DecisionResult]: 决策历史记录
        """
        return await self.db_manager.get_decision_history(context_id, limit)
    
    async def explain_decision(
        self, 
        decision_result: DecisionResult, 
        detail_level: str = "medium"
    ) -> Dict[str, Any]:
        """
        解释决策过程
        
        Args:
            decision_result: 决策结果
            detail_level: 详细程度，可选 "low", "medium", "high"
            
        Returns:
            Dict[str, Any]: 决策解释
        """
        explanation = {
            "decision_id": decision_result.id,
            "selected_option": decision_result.selected_option.title,
            "decision_time": decision_result.decision_time.isoformat(),
            "reasoning": decision_result.reasoning
        }
        
        if detail_level in ["medium", "high"]:
            explanation["context_analysis"] = decision_result.context.analysis_summary
            explanation["option_scores"] = {
                opt.title: opt.evaluation_score for opt in decision_result.all_options
            }
        
        if detail_level == "high":
            explanation["full_reasoning"] = decision_result.selected_option.full_reasoning
            explanation["alternative_options"] = [
                {
                    "title": opt.title,
                    "score": opt.evaluation_score,
                    "reasoning": opt.reasoning
                }
                for opt in decision_result.all_options 
                if opt.id != decision_result.selected_option.id
            ]
        
        return explanation
    
    async def cache_decision_result(self, decision_result: DecisionResult) -> None:
        """缓存决策结果"""
        cache_key = f"decision:{decision_result.context.id}"
        await self.cache_manager.set(cache_key, decision_result.to_dict(), ttl=3600)
    
    async def persist_decision_record(self, decision_result: DecisionResult) -> None:
        """持久化决策记录"""
        await self.db_manager.save_decision_record(decision_result.to_dict())
    
    async def update_decision_feedback(
        self, 
        decision_id: str, 
        feedback: Dict[str, Any]
    ) -> bool:
        """
        更新决策反馈
        
        Args:
            decision_id: 决策ID
            feedback: 反馈信息
            
        Returns:
            bool: 更新是否成功
        """

        except Exception as e:
            self.logger.error(f"更新决策反馈失败: {str(e)}")
            return False

```python
# src/information_fusion.py

from typing import Dict, List, Any, Optional
import numpy as np
import json
import logging
from datetime import datetime

from .models.decision_model import DecisionContext, InformationSource
from ..utils.nlp_processor import NLPProcessor

class InformationFusion:
    """
    信息融合组件，负责整合来自不同来源的信息
    """
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化信息融合器
        
        Args:
            config: 配置字典
        """
        self.config = config
        self.logger = logging.getLogger(__name__)
        self.nlp_processor = NLPProcessor(config.get("nlp", {}))
        
        # 信息源权重配置
        self.source_weights = config.get("source_weights", {
            "sensor_data": 0.3,
            "user_input": 0.4,
            "historical_data": 0.2,
            "external_knowledge": 0.1
        })
        
        # 信息质量阈值
        self.quality_threshold = config.get("quality_threshold", 0.6)
    
    async def fuse_information(self, context: DecisionContext) -> DecisionContext:
        """
        融合信息
        
        Args:
            context: 原始决策情境
            
        Returns:
            DecisionContext: 融合信息后的决策情境
        """
        try:
            # 1. 提取信息源
            information_sources = self._extract_information_sources(context)
            
            # 2. 评估信息质量
            quality_scores = await self._assess_information_quality(information_sources)
            
            # 3. 过滤低质量信息
            filtered_sources = self._filter_low_quality_sources(
                information_sources, quality_scores
            )
            
            # 4. 信息标准化
            normalized_sources = await self._normalize_information(filtered_sources)
            
            # 5. 信息融合
            fused_information = await self._perform_information_fusion(normalized_sources)
            
            # 6. 更新情境
            context.fused_information = fused_information
            context.fusion_timestamp = datetime.now()
            context.information_sources = filtered_sources
            
            self.logger.info("信息融合完成")
            return context
            
        except Exception as e:
            self.logger.error(f"信息融合过程中发生错误: {str(e)}")
            raise
    
    def _extract_information_sources(self, context: DecisionContext) -> List[InformationSource]:
        """
        提取信息源
        
        Args:
            context: 决策情境
            
        Returns:
            List[InformationSource]: 信息源列表
        """
        sources = []
        
        # 传感器数据
        if context.sensor_data:
            sources.append(InformationSource(
                id="sensor_data",
                type="sensor_data",
                content=context.sensor_data,
                timestamp=datetime.now(),
                confidence=0.8
            ))
        
        # 用户输入
        if context.user_input:
            sources.append(InformationSource(
                id="user_input",
                type="user_input",
                content=context.user_input,
                timestamp=datetime.now(),
                confidence=0.9
            ))
        
        # 历史数据
        if context.historical_data:
            sources.append(InformationSource(
                id="historical_data",
                type="historical_data",
                content=context.historical_data,
                timestamp=datetime.now(),
                confidence=0.7
            ))
        
        # 外部知识
        if context.external_knowledge:
            sources.append(InformationSource(
                id="external_knowledge",
                type="external_knowledge",
                content=context.external_knowledge,
                timestamp=datetime.now(),
                confidence=0.6
            ))
        
        return sources
    
    async def _assess_information_quality(
        self, 
        sources: List[InformationSource]
    ) -> Dict[str, float]:
        """
        评估信息质量
        
        Args:
            sources: 信息源列表
            
        Returns:
            Dict[str, float]: 信息源ID到质量分数的映射
        """
        quality_scores = {}
        
        for source in sources:
            # 基础质量分数
            base_score = source.confidence
            
            # 内容质量评估
            content_score = await self._assess_content_quality(source.content)
            
            # 时效性评估
            recency_score = self._assess_recency(source.timestamp)
            
            # 一致性评估
            consistency_score = await self._assess_consistency(source, sources)
            
            # 综合质量分数
            quality_score = (
                base_score * 0.4 + 
                content_score * 0.3 + 
                recency_score * 0.2 + 
                consistency_score * 0.1
            )
            
            quality_scores[source.id] = quality_score
        
        return quality_scores
    
    async def _assess_content_quality(self, content: Any) -> float:
        """
        评估内容质量
        
        Args:
            content: 内容
            
        Returns:
            float: 内容质量分数
        """
        if isinstance(content, str):
            # 文本内容质量评估
            return await self.nlp_processor.assess_text_quality(content)
        elif isinstance(content, dict):
            # 结构化数据质量评估
            return self._assess_structured_data_quality(content)
        else:
            # 其他类型数据质量评估
            return 0.7  # 默认分数
    
    def _assess_structured_data_quality(self, data: Dict[str, Any]) -> float:
        """
        评估结构化数据质量
        
        Args:
            data: 结构化数据
            
        Returns:
            float: 数据质量分数
        """
        # 完整性评估
        completeness = len([v for v in data.values() if v is not None]) / len(data)
        
        # 一致性评估（简化）
        consistency = 0.8  # 默认一致性分数
        
        # 准确性评估（简化）
        accuracy = 0.8  # 默认准确性分数
        
        # 综合分数
        return (completeness * 0.4 + consistency * 0.3 + accuracy * 0.3)
    
    def _assess_recency(self, timestamp: datetime) -> float:
        """
        评估时效性
        
        Args:
            timestamp: 时间戳
            
        Returns:
            float: 时效性分数
        """
        now = datetime.now()
        time_diff = (now - timestamp).total_seconds()
        
        # 时间差越小，时效性越高
        # 1小时内时效性为1.0，24小时后时效性降至0.2
        if time_diff < 3600:  # 1小时内
            return 1.0
        elif time_diff < 86400:  # 24小时内
            return 1.0 - (time_diff - 3600) / 82800 * 0.8
        else:
            return 0.2
    
    async def _assess_consistency(
        self, 
        source: InformationSource, 
        all_sources: List[InformationSource]
    ) -> float:
        """
        评估信息一致性
        
        Args:
            source: 待评估信息源
            all_sources: 所有信息源
            
        Returns:
            float: 一致性分数
        """
        # 简化实现，实际应基于内容相似度计算
        return 0.7
    
    def _filter_low_quality_sources(
        self, 
        sources: List[InformationSource], 
        quality_scores: Dict[str, float]
    ) -> List[InformationSource]:
        """
        过滤低质量信息源
        
        Args:
            sources: 信息源列表
            quality_scores: 质量分数字典
            
        Returns:
            List[InformationSource]: 过滤后的信息源列表
        """
        filtered_sources = []
        
        for source in sources:
            if quality_scores[source.id] >= self.quality_threshold:
                source.quality_score = quality_scores[source.id]
                filtered_sources.append(source)
            else:
                self.logger.warning(
                    f"信息源 {source.id} 质量分数 {quality_scores[source.id]} 低于阈值 "
                    f"{self.quality_threshold}，已过滤"
                )
        
        return filtered_sources
    
    async def _normalize_information(
        self, 
        sources: List[InformationSource]
    ) -> List[InformationSource]:
        """
        标准化信息
        
        Args:
            sources: 信息源列表
            
        Returns:
            List[InformationSource]: 标准化后的信息源列表
        """
        normalized_sources = []
        
        for source in sources:
            # 根据信息源类型进行标准化
            if source.type == "sensor_data":
                normalized_content = self._normalize_sensor_data(source.content)
            elif source.type == "user_input":
                normalized_content = await self._normalize_user_input(source.content)
            elif source.type == "historical_data":
                normalized_content = self._normalize_historical_data(source.content)
            elif source.type == "external_knowledge":
                normalized_content = await self._normalize_external_knowledge(source.content)
            else:
                normalized_content = source.content
            
            # 创建标准化后的信息源
            normalized_source = InformationSource(
                id=source.id,
                type=source.type,
                content=normalized_content,
                timestamp=source.timestamp,
                confidence=source.confidence,
                quality_score=source.quality_score
            )
            
            normalized_sources.append(normalized_source)
        
        return normalized_sources
    
    def _normalize_sensor_data(self, content: Any) -> Dict[str, Any]:
        """标准化传感器数据"""
        # 实现传感器数据标准化逻辑
        return content if isinstance(content, dict) else {"value": content}
    
    async def _normalize_user_input(self, content: Any) -> Dict[str, Any]:
        """标准化用户输入"""
        if isinstance(content, str):
            # 使用NLP处理器处理文本
            processed_text = await self.nlp_processor.process_text(content)
            return {"text": content, "processed": processed_text}
        else:
            return {"data": content}
    
    def _normalize_historical_data(self, content: Any) -> Dict[str, Any]:
        """标准化历史数据"""
        return content if isinstance(content, dict) else {"data": content}
    
    async def _normalize_external_knowledge(self, content: Any) -> Dict[str, Any]:
        """标准化外部知识"""
        if isinstance(content, str):
            # 使用NLP处理器处理文本
            processed_text = await self.nlp_processor.process_text(content)
            return {"text": content, "processed": processed_text}
        else:
            return {"data": content}
    
    async def _perform_information_fusion(
        self, 
        sources: List[InformationSource]
    ) -> Dict[str, Any]:
        """
        执行信息融合
        
        Args:
            sources: 标准化后的信息源列表
            
        Returns:
            Dict[str, Any]: 融合后的信息
        """
        fused_info = {
            "summary": "",
            "key_points": [],
            "confidence": 0.0,
            "sources": []
        }
        
        # 计算加权置信度
        total_weight = 0.0
        weighted_confidence = 0.0
        
        for source in sources:
            weight = self.source_weights.get(source.type, 0.1)
            total_weight += weight
            weighted_confidence += source.confidence * weight
            
            # 添加源信息
            fused_info["sources"].append({
                "id": source.id,
                "type": source.type,
                "confidence": source.confidence,
                "quality_score": source.quality_score
            })
        
        # 计算整体置信度
        if total_weight > 0:
            fused_info["confidence"] = weighted_confidence / total_weight
        
        # 提取关键点
        for source in sources:
            if isinstance(source.content, dict):
                if "key_points" in source.content:
                    fused_info["key_points"].extend(source.content["key_points"])
                elif "processed" in source.content and "key_points" in source.content["processed"]:
                    fused_info["key_points"].extend(source.content["processed"]["key_points"])
        
        # 生成摘要
        fused_info["summary"] = await self._generate_fusion_summary(sources)
        
        return fused_info
    
    async def _generate_fusion_summary(
        self, 
        sources: List[InformationSource]
    ) -> str:
        """
        生成融合摘要
        
        Args:
            sources: 信息源列表
            
        Returns:
            str: 融合摘要
        """
        # 简化实现，实际应使用NLP技术生成摘要
        summaries = []
        
        for source in sources:
            if isinstance(source.content, dict):
                if "summary" in source.content:
                    summaries.append(source.content["summary"])
                elif "text" in source.content:
                    summaries.append(source.content["text"][:100] + "...")
        
        return " ".join(summaries)

        """
        分析情境
        
        Args:
            context: 决策情境
            
        Returns:
            DecisionContext: 分析后的决策情境
        """
        try:
            # 1. 提取情境特征
            features = await self._extract_context_features(context)
            
            # 2. 识别情境类型
            context_type = await self._identify_context_type(features)
            
            # 3. 评估紧急程度
            urgency_level = await self._assess_urgency_level(features, context_type)
            
            # 4. 识别关键因素
            key_factors = await self._identify_key_factors(features)
            
            # 5. 预测情境演化
            evolution_prediction = await self._predict_context_evolution(features)
            
            # 6. 生成情境分析摘要
            analysis_summary = await self._generate_analysis_summary(
                features, context_type, urgency_level, key_factors, evolution_prediction
            )
            
            # 7. 更新情境
            context.context_type = context_type
            context.urgency_level = urgency_level
            context.key_factors = key_factors
            context.evolution_prediction = evolution_prediction
            context.analysis_summary = analysis_summary
            context.analysis_timestamp = datetime.now()
            
            self.logger.info(f"情境分析完成，类型: {context_type}, 紧急程度: {urgency_level}")
            return context
            
        except Exception as e:
            self.logger.error(f"情境分析过程中发生错误: {str(e)}")
            raise
    
    async def _extract_context_features(self, context: DecisionContext) -> Dict[str, Any]:
        """
        提取情境特征
        
        Args:
            context: 决策情境
            
        Returns:
            Dict[str, Any]: 情境特征
        """
        features = {
            "temporal_features": {},
            "content_features": {},
            "emotional_features": {},
            "behavioral_features": {},
            "environmental_features": {}
        }
        
        # 提取时间特征
        features["temporal_features"] = self._extract_temporal_features(context)
        
        # 提取内容特征
        features["content_features"] = await self._extract_content_features(context)
        
        # 提取情感特征
        features["emotional_features"] = await self._extract_emotional_features(context)
        
        # 提取行为特征
        features["behavioral_features"] = await self._extract_behavioral_features(context)
        
        # 提取环境特征
        features["environmental_features"] = self._extract_environmental_features(context)
        
        return features
    
    def _extract_temporal_features(self, context: DecisionContext) -> Dict[str, Any]:
        """
        提取时间特征
        
        Args:
            context: 决策情境
            
        Returns:
            Dict[str, Any]: 时间特征
        """
        now = datetime.now()
        
        # 一天中的时段
        hour = now.hour
        if 6 <= hour < 12:
            time_of_day = "morning"
        elif 12 <= hour < 18:
            time_of_day = "afternoon"
        elif 18 <= hour < 22:
            time_of_day = "evening"
        else:
            time_of_day = "night"
        
        # 星期几
        weekday = now.weekday()  # 0-6，0是周一
        is_weekend = weekday >= 5
        
        return {
            "hour": hour,
            "time_of_day": time_of_day,
            "weekday": weekday,
            "is_weekend": is_weekend,
            "timestamp": now.isoformat()
        }
    
    async def _extract_content_features(self, context: DecisionContext) -> Dict[str, Any]:
        """
        提取内容特征
        
        Args:
            context: 决策情境
            
        Returns:
            Dict[str, Any]: 内容特征
        """
        content_features = {
            "keywords": [],
            "topics": [],
            "entities": [],
            "sentiment": None,
            "complexity": 0.0
        }
        
        # 从融合信息中提取文本内容
        text_content = self._extract_text_content(context)
        
        if text_content:
            # 使用NLP处理器提取特征
            nlp_features = await self.nlp_processor.extract_features(text_content)
            
            content_features["keywords"] = nlp_features.get("keywords", [])
            content_features["topics"] = nlp_features.get("topics", [])
            content_features["entities"] = nlp_features.get("entities", [])
            content_features["sentiment"] = nlp_features.get("sentiment", {})
            content_features["complexity"] = nlp_features.get("complexity", 0.0)
        
        return content_features
    
    def _extract_text_content(self, context: DecisionContext) -> str:
        """
        从情境中提取文本内容
        
        Args:
            context: 决策情境
            
        Returns:
            str: 文本内容
        """
        text_parts = []
        
        # 从用户输入提取
        if context.user_input:
            text_parts.append(context.user_input)
        
        # 从融合信息中提取
        if context.fused_information and "summary" in context.fused_information:
            text_parts.append(context.fused_information["summary"])
        
        # 从外部知识中提取
        if context.external_knowledge:
            if isinstance(context.external_knowledge, str):
                text_parts.append(context.external_knowledge)
            elif isinstance(context.external_knowledge, dict) and "text" in context.external_knowledge:
                text_parts.append(context.external_knowledge["text"])
        
        return " ".join(text_parts)
    
    async def _extract_emotional_features(self, context: DecisionContext) -> Dict[str, Any]:
        """
        提取情感特征
        
        Args:
            context: 决策情境
            
        Returns:
            Dict[str, Any]: 情感特征
        """
        emotional_features = {
            "dominant_emotion": "neutral",
            "emotion_scores": {},
            "arousal": 0.5,
            "valence": 0.5
        }
        
        # 从用户输入中提取情感特征
        if context.user_input:
            emotion_analysis = await self.nlp_processor.analyze_emotion(context.user_input)
            emotional_features.update(emotion_analysis)
        
        # 从传感器数据中提取情感特征（如果有）
        if context.sensor_data and isinstance(context.sensor_data, dict):
            if "emotion" in context.sensor_data:
                sensor_emotion = context.sensor_data["emotion"]
                if isinstance(sensor_emotion, dict):
                    emotional_features.update(sensor_emotion)
        
        return emotional_features
    
    async def _extract_behavioral_features(self, context: DecisionContext) -> Dict[str, Any]:
        """
        提取行为特征
        
        Args:
            context: 决策情境
            
        Returns:
            Dict[str, Any]: 行为特征
        """
        behavioral_features = {
            "activity_type": "unknown",
            "interaction_pattern": "unknown",
            "attention_level": 0.5
        }
        
        # 从历史数据中提取行为特征
        if context.historical_data and isinstance(context.historical_data, dict):
            if "recent_activities" in context.historical_data:
                recent_activities = context.historical_data["recent_activities"]
                if recent_activities:
                    # 确定最近的活动类型
                    behavioral_features["activity_type"] = recent_activities[0].get("type", "unknown")
        
        # 从传感器数据中提取行为特征
        if context.sensor_data and isinstance(context.sensor_data, dict):
            if "movement" in context.sensor_data:
                movement_data = context.sensor_data["movement"]
                if isinstance(movement_data, dict):
                    behavioral_features["interaction_pattern"] = movement_data.get("pattern", "unknown")
            
            if "attention" in context.sensor_data:
                attention_data = context.sensor_data["attention"]
                if isinstance(attention_data, (int, float)):
                    behavioral_features["attention_level"] = min(max(attention_data, 0.0), 1.0)
        
        return behavioral_features
    
    def _extract_environmental_features(self, context: DecisionContext) -> Dict[str, Any]:
        """
        提取环境特征
        
        Args:
            context: 决策情境
            
        Returns:
            Dict[str, Any]: 环境特征
        """
        environmental_features = {
            "location": "unknown",
            "noise_level": 0.5,
            "lighting": 0.5,
            "temperature": 0.5,
            "social_context": "unknown"
        }
        
        # 从传感器数据中提取环境特征
        if context.sensor_data and isinstance(context.sensor_data, dict):
            if "environment" in context.sensor_data:
                env_data = context.sensor_data["environment"]
                if isinstance(env_data, dict):
                    environmental_features.update(env_data)
        
        return environmental_features
    
    async def _identify_context_type(self, features: Dict[str, Any]) -> str:
        """
        识别情境类型
        
        Args:
            features: 情境特征
            
        Returns:
            str: 情境类型
        """
        # 使用模式识别器识别情境类型
        context_type = await self.pattern_recognizer.recognize_context_type(features)
        
        # 如果识别结果不在预定义类型中，使用默认类型
        if context_type not in self.context_types:
            context_type = "routine"  # 默认类型
        
        return context_type
    
    async def _assess_urgency_level(
        self, 
        features: Dict[str, Any], 
        context_type: str
    ) -> str:
        """
        评估紧急程度
        
        Args:
            features: 情境特征
            context_type: 情境类型
            
        Returns:
            str: 紧急程度
        """
        # 基于情境类型的基础紧急程度
        base_urgency = {
            "emergency": 0.9,
            "health": 0.7,
            "social": 0.5,
            "learning": 0.3,
            "routine": 0.1
        }.get(context_type, 0.3)
        
        # 基于情感特征调整紧急程度
        emotional_features = features.get("emotional_features", {})
        if emotional_features:
            dominant_emotion = emotional_features.get("dominant_emotion", "neutral")
            emotion_urgency = {
                "fear": 0.8,
                "anger": 0.7,
                "sadness": 0.6,
                "joy": 0.2,
                "neutral": 0.3
            }.get(dominant_emotion, 0.3)
            
            base_urgency = (base_urgency + emotion_urgency) / 2
        
        # 基于行为特征调整紧急程度
        behavioral_features = features.get("behavioral_features", {})
        if behavioral_features:
            activity_type = behavioral_features.get("activity_type", "unknown")
            activity_urgency = {
                "crying": 0.9,
                "distress": 0.8,
                "playing": 0.2,
                "sleeping": 0.1,
                "eating": 0.3
            }.get(activity_type, 0.3)
            
            base_urgency = (base_urgency + activity_urgency) / 2
        
        # 确定紧急程度级别
        if base_urgency >= self.urgency_thresholds["emergency"]:
            return "emergency"
        elif base_urgency >= self.urgency_thresholds["high"]:
            return "high"
        elif base_urgency >= self.urgency_thresholds["medium"]:
            return "medium"
        else:
            return "low"
    
    async def _identify_key_factors(self, features: Dict[str, Any]) -> List[Dict[str, Any]]:
        """
        识别关键因素
        
        Args:
            features: 情境特征
            
        Returns:
            List[Dict[str, Any]]: 关键因素列表
        """
        key_factors = []
        
        # 从内容特征中提取关键因素
        content_features = features.get("content_features", {})
        if content_features:
            keywords = content_features.get("keywords", [])
            for keyword in keywords[:5]:  # 取前5个关键词
                key_factors.append({
                    "type": "content",
                    "factor": keyword,
                    "importance": 0.8,
                    "description": f"内容关键词: {keyword}"
                })
        
        # 从情感特征中提取关键因素
        emotional_features = features.get("emotional_features", {})
        if emotional_features:
            dominant_emotion = emotional_features.get("dominant_emotion", "neutral")
            if dominant_emotion != "neutral":
                key_factors.append({
                    "type": "emotional",
                    "factor": dominant_emotion,
                    "importance": 0.9,
                    "description": f"主导情感: {dominant_emotion}"
                })
        
        # 从行为特征中提取关键因素
        behavioral_features = features.get("behavioral_features", {})
        if behavioral_features:
            activity_type = behavioral_features.get("activity_type", "unknown")
            if activity_type != "unknown":
                key_factors.append({
                    "type": "behavioral",
                    "factor": activity_type,
                    "importance": 0.7,
                    "description": f"当前活动: {activity_type}"
                })
        
        # 从环境特征中提取关键因素
        environmental_features = features.get("environmental_features", {})
        if environmental_features:
            location = environmental_features.get("location", "unknown")
            if location != "unknown":
                key_factors.append({
                    "type": "environmental",
                    "factor": location,
                    "importance": 0.6,
                    "description": f"当前位置: {location}"
                })
        
        # 按重要性排序
        key_factors.sort(key=lambda x: x["importance"], reverse=True)
        
        return key_factors[:10]  # 返回前10个关键因素
    
    async def _predict_context_evolution(
        self, 
        features: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        预测情境演化
        
        Args:
            features: 情境特征
            
        Returns:
            Dict[str, Any]: 情境演化预测
        """
        evolution_prediction = {
            "likely_changes": [],
            "time_horizon": "5_minutes",
            "confidence": 0.5
        }
        
        # 基于情境类型预测演化
        content_features = features.get("content_features", {})
        emotional_features = features.get("emotional_features", {})
        behavioral_features = features.get("behavioral_features", {})
        
        # 预测情感变化
        if emotional_features:
            dominant_emotion = emotional_features.get("dominant_emotion", "neutral")
            if dominant_emotion == "anger":
                evolution_prediction["likely_changes"].append({
                    "aspect": "emotional",
                    "prediction": "情绪可能逐渐平复",
                    "probability": 0.7
                })
            elif dominant_emotion == "sadness":
                evolution_prediction["likely_changes"].append({
                    "aspect": "emotional",
                    "prediction": "可能需要安抚",
                    "probability": 0.8
                })
        
        # 预测行为变化
        if behavioral_features:
            activity_type = behavioral_features.get("activity_type", "unknown")
            if activity_type == "crying":
                evolution_prediction["likely_changes"].append({
                    "aspect": "behavioral",
                    "prediction": "哭声可能持续或加剧",
                    "probability": 0.6
                })
            elif activity_type == "playing":
                evolution_prediction["likely_changes"].append({
                    "aspect": "behavioral",
                    "prediction": "活动可能继续或变化",
                    "probability": 0.5
                })
        
        # 计算整体置信度
        if evolution_prediction["likely_changes"]:
            probabilities = [
                change["probability"] for change in evolution_prediction["likely_changes"]
            ]
            evolution_prediction["confidence"] = sum(probabilities) / len(probabilities)
        
        return evolution_prediction
    
    async def _generate_analysis_summary(
        self,
        features: Dict[str, Any],
        context_type: str,
        urgency_level: str,
        key_factors: List[Dict[str, Any]],
        evolution_prediction: Dict[str, Any]
    ) -> str:
        """
        生成情境分析摘要
        
        Args:
            features: 情境特征
            context_type: 情境类型
            urgency_level: 紧急程度
            key_factors: 关键因素
            evolution_prediction: 情境演化预测
            
        Returns:
            str: 情境分析摘要
        """
        # 提取关键信息
        content_features = features.get("content_features", {})
        emotional_features = features.get("emotional_features", {})
        behavioral_features = features.get("behavioral_features", {})
        environmental_features = features.get("environmental_features", {})
        
        # 构建摘要
        summary_parts = []
        
        # 情境类型和紧急程度
        summary_parts.append(f"情境类型: {context_type}, 紧急程度: {urgency_level}")
        
        # 关键因素
        if key_factors:
            top_factors = key_factors[:3]  # 取前3个关键因素
            factor_descriptions = [factor["description"] for factor in top_factors]
            summary_parts.append(f"关键因素: {', '.join(factor_descriptions)}")
        
        # 情感状态
        if emotional_features:
            dominant_emotion = emotional_features.get("dominant_emotion", "neutral")
            if dominant_emotion != "neutral":
                summary_parts.append(f"情感状态: {dominant_emotion}")
        
        # 行为状态
        if behavioral_features:
            activity_type = behavioral_features.get("activity_type", "unknown")
            if activity_type != "unknown":
                summary_parts.append(f"当前活动: {activity_type}")
        
        # 环境状态
        if environmental_features:
            location = environmental_features.get("location", "unknown")
            if location != "unknown":
                summary_parts.append(f"位置: {location}")
        
        # 演化预测
        if evolution_prediction and evolution_prediction["likely_changes"]:
            top_prediction = evolution_prediction["likely_changes"][0]
            summary_parts.append(
                f"预测: {top_prediction['prediction']} (概率: {top_prediction['probability']:.1f})"
            )
        
        return "; ".join(summary_parts)
```

## 实现细节

### 决策流程

决策制定模块的决策流程如下：

1. **信息融合**: 整合来自不同来源的信息，评估信息质量，过滤低质量信息
2. **情境分析**: 提取情境特征，识别情境类型，评估紧急程度，识别关键因素
3. **方案生成**: 基于情境分析结果生成可能的决策选项
4. **方案评估**: 基于预设标准评估各选项的优劣
5. **方案选择**: 选择最佳决策选项
6. **结果缓存与持久化**: 缓存决策结果，持久化决策记录

### 关键算法

#### 信息质量评估算法

信息质量评估基于以下维度：

- **基础质量**: 信息源的固有置信度
- **内容质量**: 信息的完整性、准确性和相关性
- **时效性**: 信息的新鲜程度
- **一致性**: 与其他信息的一致性

#### 情境类型识别算法

情境类型识别基于以下特征：

- **时间特征**: 当前时间、星期几、是否周末等
- **内容特征**: 关键词、主题、实体、情感倾向等
- **情感特征**: 主导情感、情感强度、唤醒度和效价
- **行为特征**: 活动类型、交互模式、注意力水平
- **环境特征**: 位置、噪音水平、光照、温度、社交环境

#### 决策选项评估算法

决策选项评估基于以下标准：

- **有效性**: 选项解决问题的有效性
- **可行性**: 选项实施的可行性
- **安全性**: 选项的安全性
- **效率**: 选项的执行效率
- **成本**: 选项的实施成本
