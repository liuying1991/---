```python
from pydantic import BaseModel, Field
from typing import List, Optional, Dict, Any
from datetime import datetime
from enum import Enum

class ModalityType(str, Enum):
    """模态类型枚举"""
    AUDIO = "audio"
    VIDEO = "video"
    TEXT = "text"

class FeatureData(BaseModel):
    """特征数据模型"""
    modality: ModalityType
    features: List[float] = Field(description="特征向量")
    timestamp: float = Field(description="时间戳(ms)")
    metadata: Optional[Dict[str, Any]] = Field(default={}, description="元数据")
    confidence: Optional[float] = Field(default=1.0, description="置信度")
    
class FusionRequest(BaseModel):
    """融合请求模型"""
    session_id: str = Field(description="会话ID")
    features: List[FeatureData] = Field(description="待融合的特征数据")
    fusion_method: Optional[str] = Field(default="attention", description="融合方法")
    options: Optional[Dict[str, Any]] = Field(default={}, description="融合选项")

class FusionResult(BaseModel):
    """融合结果模型"""
    session_id: str = Field(description="会话ID")
    fused_features: List[float] = Field(description="融合后的特征")
    quality_score: float = Field(description="融合质量评分")
    attention_weights: Optional[List[float]] = Field(default=None, description="注意力权重")
    processing_time: float = Field(description="处理时间(ms)")
    timestamp: datetime = Field(default_factory=datetime.now, description="结果时间戳")
```

```python
import asyncio
import logging

logger = logging.getLogger(__name__)

class DataReceiver:
    """多模态数据接收器"""
    
    def __init__(self):
        self.config = get_config()
        self.data_sources = {}
        self.data_buffers = {}
        self.is_running = False
        self.callbacks = {}
        
    def register_data_source(self, modality: ModalityType, source: str):
        """注册数据源"""
        self.data_sources[modality] = source
        logger.info(f"Registered {modality} data source: {source}")
        
    def register_callback(self, modality: ModalityType, callback: Callable):
        """注册数据接收回调"""
        if modality not in self.callbacks:
            self.callbacks[modality] = []
        self.callbacks[modality].append(callback)
        
    async def start_receiving(self):
        """开始接收数据"""
        self.is_running = True
        tasks = []
        
        for modality, source in self.data_sources.items():
            task = asyncio.create_task(self._receive_data(modality, source))
            tasks.append(task)
            
        await asyncio.gather(*tasks)
        
    async def stop_receiving(self):
        """停止接收数据"""
        self.is_running = False
        
    async def _receive_data(self, modality: ModalityType, source: str):
        """接收特定模态的数据"""
        logger.info(f"Starting to receive {modality} data from {source}")
        
        # 这里简化实现，实际应根据source类型实现不同的接收逻辑
        # 例如：WebSocket、TCP、消息队列等
        while self.is_running:
            try:
                # 模拟接收数据
                data = await self._simulate_data_receive(modality)
                
                # 调用回调函数
                if modality in self.callbacks:
                    for callback in self.callbacks[modality]:
                        await callback(data)
                        
                await asyncio.sleep(0.01)  # 控制接收频率
                
            except Exception as e:
                logger.error(f"Error receiving {modality} data: {e}")
                await asyncio.sleep(1)  # 出错后等待一段时间再重试
                
    async def _simulate_data_receive(self, modality: ModalityType) -> FeatureData:

        import time

        elif modality == ModalityType.VIDEO:
            features = [random.random() for _ in range(512)]

            features = [random.random() for _ in range(256)]
            
        return FeatureData(
            modality=modality,
            features=features,
            timestamp=time.time() * 1000,  # 转换为毫秒
            confidence=random.random()
```

#### 时间对齐实现 (src/core/temporal_aligner.py)

```python
import numpy as np
import logging
from typing import Tuple, List, Optional
from scipy import interpolate
from datetime import datetime

from ..data.data_models import FeatureData, ModalityType
from ..utils.exceptions import AlignmentError

logger = logging.getLogger(__name__)

class TemporalAligner:
    """时间对齐器"""
    
    def __init__(self, window_size: int = 1000, max_delay: int = 500):
        """
        初始化时间对齐器
        
        Args:
            window_size: 时间窗口大小(毫秒)
            max_delay: 最大允许延迟(毫秒)
        """
        self.window_size = window_size
        self.max_delay = max_delay
        
    def align_features(
        self, 
        audio_features: List[FeatureData], 
        video_features: List[FeatureData]
    ) -> Tuple[List[FeatureData], List[FeatureData], List[float]]:
        """
        对齐音频和视频特征
        
        Args:
            audio_features: 音频特征列表
            video_features: 视频特征列表
            
        Returns:
            Tuple[List[FeatureData], List[FeatureData], List[float]]: 
                对齐后的音频特征、视频特征和时间网格
        """
        if not audio_features or not video_features:
            raise AlignmentError("音频或视频特征为空")
            
        # 提取时间戳
        audio_timestamps = [f.timestamp for f in audio_features]
        video_timestamps = [f.timestamp for f in video_features]
        
        # 创建时间网格
        min_time = max(min(audio_timestamps), min(video_timestamps))
        max_time = min(max(audio_timestamps), max(video_timestamps))
        
        # 检查时间重叠是否足够
        if max_time - min_time < self.window_size:
            raise AlignmentError(f"时间重叠不足，需要至少{self.window_size}ms")
            
        # 生成时间网格
        time_grid = np.arange(min_time, max_time, self.window_size)
        
        # 对齐音频特征
        aligned_audio = self._interpolate_features(
            audio_features, audio_timestamps, time_grid
        )
        
        # 对齐视频特征
        aligned_video = self._interpolate_features(
            video_features, video_timestamps, time_grid
        )
        
        logger.info(f"对齐完成，时间点数: {len(time_grid)}")
        return aligned_audio, aligned_video, time_grid.tolist()
        
    def _interpolate_features(
        self, 
        features: List[FeatureData], 
        timestamps: List[float], 
        time_grid: np.ndarray
    ) -> List[FeatureData]:
        """
        插值特征到时间网格
        
        Args:
            features: 原始特征列表
            timestamps: 原始时间戳列表
            time_grid: 目标时间网格
            
        Returns:
            List[FeatureData]: 插值后的特征列表
        """
        # 转换为numpy数组
        feature_matrix = np.array([f.features for f in features])
        
        # 对每个特征维度进行插值
        aligned_features = []
        for i, t in enumerate(time_grid):
            # 使用线性插值
            interp_func = interpolate.interp1d(
                timestamps, feature_matrix, axis=0, kind='linear', 
                bounds_error=False, fill_value='extrapolate'
            )
            
            # 插值得到当前时间点的特征
            interp_feature = interp_func(t)
            
            # 创建新的特征数据
            aligned_feature = FeatureData(
                modality=features[0].modality,
                features=interp_feature.tolist(),
                timestamp=float(t),
                metadata=features[0].metadata,
                confidence=self._calculate_confidence(features, timestamps, t)
            )
            
            aligned_features.append(aligned_feature)
            
        return aligned_features
        
    def _calculate_confidence(
        self, 
        features: List[FeatureData], 
        timestamps: List[float], 
        target_time: float
    ) -> float:
        """
        计算插值置信度
        
        Args:
            features: 原始特征列表
            timestamps: 原始时间戳列表
            target_time: 目标时间点
            
        Returns:
            float: 置信度(0-1)
        """
        # 找到最近的时间点
        distances = [abs(t - target_time) for t in timestamps]
        min_distance = min(distances)
        
        # 距离越近，置信度越高
        confidence = max(0, 1 - min_distance / self.max_delay)
        
        return confidence
```

#### 注意力融合实现 (src/core/attention_fusion.py)

```python
import torch
import torch.nn as nn
import numpy as np
import logging
from typing import List, Tuple, Optional, Dict, Any

from ..data.data_models import FeatureData, ModalityType
from ..utils.exceptions import FusionError
from ..utils.config import get_config

logger = logging.getLogger(__name__)

class CrossModalAttentionFusion(nn.Module):
    """跨模态注意力融合模型"""
    
    def __init__(self, audio_dim: int, video_dim: int, hidden_dim: int = 256):
        """
        初始化跨模态注意力融合模型
        
        Args:
            audio_dim: 音频特征维度
            video_dim: 视频特征维度
            hidden_dim: 隐藏层维度
        """
        super(CrossModalAttentionFusion, self).__init__()
        
        self.audio_dim = audio_dim
        self.video_dim = video_dim
        self.hidden_dim = hidden_dim
        
        # 注意力权重计算
        self.audio_attention = nn.Linear(audio_dim, hidden_dim)
        self.video_attention = nn.Linear(video_dim, hidden_dim)
        self.fusion_attention = nn.Linear(hidden_dim, 1)
        
        # 特征变换
        self.audio_transform = nn.Linear(audio_dim, hidden_dim)
        self.video_transform = nn.Linear(video_dim, hidden_dim)
        
        # 初始化权重
        self._init_weights()
        
    def _init_weights(self):
        """初始化模型权重"""
        for m in self.modules():
            if isinstance(m, nn.Linear):
                nn.init.xavier_uniform_(m.weight)
                nn.init.constant_(m.bias, 0)
                
    def forward(
        self, 
        audio_features: torch.Tensor, 
        video_features: torch.Tensor
    ) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        前向传播
        
        Args:
            audio_features: 音频特征张量 [batch_size, seq_len, audio_dim]
            video_features: 视频特征张量 [batch_size, seq_len, video_dim]
            
        Returns:
            Tuple[torch.Tensor, torch.Tensor]: 
                融合特征和注意力权重
        """
        # 计算注意力权重
        audio_att = torch.tanh(self.audio_attention(audio_features))
        video_att = torch.tanh(self.video_attention(video_features))
        
        # 融合注意力
        joint_attention = torch.tanh(audio_att + video_att)
        attention_weights = torch.softmax(self.fusion_attention(joint_attention), dim=1)
        
        # 加权融合
        weighted_audio = audio_features * attention_weights
        weighted_video = video_features * (1 - attention_weights)
        
        # 特征变换
        transformed_audio = self.audio_transform(weighted_audio)
        transformed_video = self.video_transform(weighted_video)
        
        # 拼接融合
        fused_features = torch.cat([transformed_audio, transformed_video], dim=-1)
        
        return fused_features, attention_weights

class AttentionFusionEngine:
    """注意力融合引擎"""
    
    def __init__(self):
        self.config = get_config()
        self.model = None
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self._initialize_model()
        
    def _initialize_model(self):
        """初始化融合模型"""
        audio_dim = self.config.get("fusion.audio_dim", 128)
        video_dim = self.config.get("fusion.video_dim", 512)
        hidden_dim = self.config.get("fusion.hidden_dim", 256)
        
        self.model = CrossModalAttentionFusion(
            audio_dim=audio_dim,
            video_dim=video_dim,
            hidden_dim=hidden_dim
        ).to(self.device)
        
        logger.info(f"初始化注意力融合模型，设备: {self.device}")
        
    def fuse_features(
        self, 
        audio_features: List[FeatureData], 
        video_features: List[FeatureData]
    ) -> Tuple[List[float], float, List[float]]:
        """
        融合音频和视频特征
        
        Args:
            audio_features: 音频特征列表
            video_features: 视频特征列表
            
        Returns:
            Tuple[List[float], float, List[float]]: 
                融合特征、质量评分和注意力权重
        """
        if not audio_features or not video_features:
            raise FusionError("音频或视频特征为空")
            
        if len(audio_features) != len(video_features):
            raise FusionError("音频和视频特征数量不匹配")
            
        # 转换为张量
        audio_tensor = self._features_to_tensor(audio_features)
        video_tensor = self._features_to_tensor(video_features)
        
        # 添加批次维度
        audio_tensor = audio_tensor.unsqueeze(0)  # [1, seq_len, audio_dim]
        video_tensor = video_tensor.unsqueeze(0)  # [1, seq_len, video_dim]
        
        # 移动到设备
        audio_tensor = audio_tensor.to(self.device)
        video_tensor = video_tensor.to(self.device)
        
        # 模型推理
        with torch.no_grad():
            fused_features, attention_weights = self.model(audio_tensor, video_tensor)
            
        # 计算质量评分
        quality_score = self._calculate_quality_score(
            audio_features, video_features, attention_weights
        )
        
        # 转换为列表
        fused_features_list = fused_features.squeeze(0).mean(dim=0).cpu().numpy().tolist()
        attention_weights_list = attention_weights.squeeze(0).cpu().numpy().tolist()
        
        return fused_features_list, quality_score, attention_weights_list
        
    def _features_to_tensor(self, features: List[FeatureData]) -> torch.Tensor:
        """将特征列表转换为张量"""
        feature_matrix = np.array([f.features for f in features])
        return torch.from_numpy(feature_matrix).float()
        
    def _calculate_quality_score(
        self, 
        audio_features: List[FeatureData], 
        video_features: List[FeatureData], 
        attention_weights: torch.Tensor
    ) -> float:
        """
        计算融合质量评分
        
        Args:
            audio_features: 音频特征列表
            video_features: 视频特征列表
            attention_weights: 注意力权重
            
        Returns:
            float: 质量评分(0-1)
        """
        # 计算特征一致性
        audio_confidence = np.mean([f.confidence for f in audio_features])
        video_confidence = np.mean([f.confidence for f in video_features])
        
        # 计算注意力权重分布的均匀性
        attention_entropy = -torch.sum(
            attention_weights * torch.log(attention_weights + 1e-8), dim=1
        ).mean().item()
        
        # 归一化熵值
        max_entropy = np.log(len(audio_features))
        normalized_entropy = attention_entropy / max_entropy if max_entropy > 0 else 0
        
        # 综合质量评分
        quality_score = 0.4 * audio_confidence + 0.4 * video_confidence + 0.2 * normalized_entropy
        
        return float(quality_score)
```

#### 质量控制实现 (src/core/quality_controller.py)

```python
import numpy as np
import logging
from typing import List, Dict, Any, Tuple
from scipy.stats import entropy
from datetime import datetime

from ..data.data_models import FeatureData, ModalityType
from ..utils.exceptions import QualityControlError
from ..utils.config import get_config

logger = logging.getLogger(__name__)

class QualityController:
    """多模态融合质量控制"""
    
    def __init__(self):
        self.config = get_config()
        self.quality_threshold = self.config.get("fusion.quality_threshold", 0.7)
        self.consistency_threshold = self.config.get("fusion.consistency_threshold", 0.8)
        
    def evaluate_fusion_quality(
        self, 
        original_features: Dict[ModalityType, List[FeatureData]], 
        fused_features: List[float], 
        attention_weights: List[float]
    ) -> Dict[str, float]:
        """
        评估融合质量
        
        Args:
            original_features: 原始特征字典
            fused_features: 融合后的特征
            attention_weights: 注意力权重
            
        Returns:
            Dict[str, float]: 质量评估结果
        """
        # 计算各项质量指标
        consistency_score = self._calculate_consistency(original_features, attention_weights)
        redundancy_score = self._calculate_redundancy(original_features)
        completeness_score = self._calculate_completeness(original_features)
        confidence_score = self._calculate_confidence(original_features)
        
        # 计算综合质量评分
        overall_quality = (
            0.3 * consistency_score +
            0.2 * redundancy_score +
            0.2 * completeness_score +
            0.3 * confidence_score
        )
        
        # 返回质量评估结果
        return {
            "overall_quality": overall_quality,
            "consistency": consistency_score,
            "redundancy": redundancy_score,
            "completeness": completeness_score,
            "confidence": confidence_score,
            "is_acceptable": overall_quality >= self.quality_threshold
        }
        
    def _calculate_consistency(
        self, 
        original_features: Dict[ModalityType, List[FeatureData]], 
        attention_weights: List[float]
    ) -> float:
        """计算模态间一致性"""
        if len(original_features) < 2:
            return 1.0
            
        # 获取模态列表
        modalities = list(original_features.keys())
        
        # 计算模态间相关性
        correlations = []
        for i in range(len(modalities)):
            for j in range(i+1, len(modalities)):
                mod1 = modalities[i]
                mod2 = modalities[j]
                
                # 提取特征矩阵
                features1 = np.array([f.features for f in original_features[mod1]])
                features2 = np.array([f.features for f in original_features[mod2]])
                
                # 计算相关性
                corr = self._calculate_correlation(features1, features2)
                correlations.append(corr)
                
        # 平均相关性作为一致性评分
        consistency = np.mean(correlations) if correlations else 0.0
        
        # 考虑注意力权重分布的稳定性
        if attention_weights:
            weight_entropy = entropy(attention_weights)
            max_entropy = np.log(len(attention_weights))
            normalized_entropy = weight_entropy / max_entropy if max_entropy > 0 else 0
            
            # 注意力权重分布越均匀，一致性越高
            consistency = 0.7 * consistency + 0.3 * normalized_entropy
            
        return float(consistency)
        
    def _calculate_redundancy(
        self, 
        original_features: Dict[ModalityType, List[FeatureData]]
    ) -> float:
        """计算模态间冗余度"""
        if len(original_features) < 2:
            return 0.0
            
        # 获取模态列表
        modalities = list(original_features.keys())
        
        # 计算模态间相似度
        similarities = []
        for i in range(len(modalities)):
            for j in range(i+1, len(modalities)):
                mod1 = modalities[i]
                mod2 = modalities[j]
                
                # 提取特征矩阵
                features1 = np.array([f.features for f in original_features[mod1]])
                features2 = np.array([f.features for f in original_features[mod2]])
                
                # 计算相似度
                sim = self._calculate_similarity(features1, features2)
                similarities.append(sim)
                
        # 平均相似度作为冗余度评分
        redundancy = np.mean(similarities) if similarities else 0.0
        
        # 冗余度越高，评分越低（取反）
        return float(1.0 - redundancy)
        
    def _calculate_completeness(
        self, 
        original_features: Dict[ModalityType, List[FeatureData]]
    ) -> float:
        """计算特征完整性"""
        # 检查是否有缺失的模态
        expected_modalities = [ModalityType.AUDIO, ModalityType.VIDEO]
        missing_modalities = [
            mod for mod in expected_modalities 
            if mod not in original_features or not original_features[mod]
        ]
        
        # 缺失模态越多，完整性越低
        completeness = 1.0 - len(missing_modalities) / len(expected_modalities)
        
        # 检查每个模态内的特征完整性
        for modality, features in original_features.items():
            if not features:
                continue
                
            # 计算特征中的NaN比例
            feature_matrix = np.array([f.features for f in features])
            nan_ratio = np.isnan(feature_matrix).mean()
            
            # NaN比例越高，完整性越低
            modality_completeness = 1.0 - nan_ratio
            completeness = min(completeness, modality_completeness)
            
        return float(completeness)
        
    def _calculate_confidence(
        self, 
        original_features: Dict[ModalityType, List[FeatureData]]
    ) -> float:
        """计算特征置信度"""
        confidences = []
        
        # 收集所有特征的置信度
        for modality, features in original_features.items():
            for feature in features:
                confidences.append(feature.confidence)
                
        # 平均置信度
        avg_confidence = np.mean(confidences) if confidences else 0.0
        
        return float(avg_confidence)
        
    def _calculate_correlation(self, features1: np.ndarray, features2: np.ndarray) -> float:
        """计算两组特征间的相关性"""
        # 确保特征矩阵形状一致
        min_len = min(features1.shape[0], features2.shape[0])
        features1 = features1[:min_len]
        features2 = features2[:min_len]
        
        # 计算每个时间点的相关性
        correlations = []
        for i in range(min_len):
            # 计算皮尔逊相关系数
            corr = np.corrcoef(features1[i], features2[i])[0, 1]
            
            # 处理NaN情况
            if not np.isnan(corr):
                correlations.append(corr)
                
        # 平均相关性
        avg_correlation = np.mean(correlations) if correlations else 0.0
        
        # 确保相关性在[0, 1]范围内
        return float(max(0, avg_correlation))
        
    def _calculate_similarity(self, features1: np.ndarray, features2: np.ndarray) -> float:
        """计算两组特征间的相似度"""
        # 确保特征矩阵形状一致
        min_len = min(features1.shape[0], features2.shape[0])
        features1 = features1[:min_len]
        features2 = features2[:min_len]
        
        # 计算余弦相似度
        similarities = []
        for i in range(min_len):
            # 计算余弦相似度
            dot_product = np.dot(features1[i], features2[i])
            norm1 = np.linalg.norm(features1[i])
            norm2 = np.linalg.norm(features2[i])
            
            if norm1 > 0 and norm2 > 0:
                similarity = dot_product / (norm1 * norm2)
                similarities.append(similarity)
                
        # 平均相似度
        avg_similarity = np.mean(similarities) if similarities else 0.0
        
        # 确保相似度在[0, 1]范围内
        return float(max(0, avg_similarity))
```

```python
import asyncio
import logging
import time
from typing import Dict, List, Optional, Tuple
from datetime import datetime

from ..data.data_models import FeatureData, ModalityType, FusionRequest, FusionResult
from ..data.data_receiver import DataReceiver
from .temporal_aligner import TemporalAligner
from .attention_fusion import AttentionFusionEngine
from .quality_controller import QualityController
from ..utils.exceptions import FusionError
from ..utils.config import get_config

logger = logging.getLogger(__name__)

class FusionEngine:
    """多模态融合引擎"""
    
    def __init__(self):
        self.config = get_config()
        self.data_receiver = DataReceiver()
        self.temporal_aligner = TemporalAligner(
            window_size=self.config.get("fusion.window_size", 1000),
            max_delay=self.config.get("fusion.max_delay", 500)
        )
        self.attention_fusion = AttentionFusionEngine()
        self.quality_controller = QualityController()
        
        # 注册数据源
        self._setup_data_sources()
        
        # 数据缓存
        self.feature_cache = {
            ModalityType.AUDIO: [],
            ModalityType.VIDEO: []
        }
        
        # 运行状态
        self.is_running = False
        
    def _setup_data_sources(self):
        """设置数据源"""
        audio_source = self.config.get("data.audio_source")
        video_source = self.config.get("data.video_source")
        
        if audio_source:
            self.data_receiver.register_data_source(ModalityType.AUDIO, audio_source)
        if video_source:
            self.data_receiver.register_data_source(ModalityType.VIDEO, video_source)
            
        # 注册数据接收回调
        self.data_receiver.register_callback(ModalityType.AUDIO, self._handle_audio_data)
        self.data_receiver.register_callback(ModalityType.VIDEO, self._handle_video_data)
        
    async def start(self):
        """启动融合引擎"""
        if self.is_running:
            logger.warning("融合引擎已在运行")
            return
            
        self.is_running = True
        logger.info("启动多模态融合引擎")
        
        # 启动数据接收
        await self.data_receiver.start_receiving()
        
    async def stop(self):
        """停止融合引擎"""
        if not self.is_running:
            return
            
        self.is_running = False
        await self.data_receiver.stop_receiving()
        logger.info("多模态融合引擎已停止")
        
    async def _handle_audio_data(self, data: FeatureData):
        """处理音频数据"""
        self.feature_cache[ModalityType.AUDIO].append(data)
        
        # 限制缓存大小
        max_cache_size = self.config.get("data.buffer_size", 1000)
        if len(self.feature_cache[ModalityType.AUDIO]) > max_cache_size:
            self.feature_cache[ModalityType.AUDIO].pop(0)
            
    async def _handle_video_data(self, data: FeatureData):
        """处理视频数据"""
        self.feature_cache[ModalityType.VIDEO].append(data)
        
        # 限制缓存大小
        max_cache_size = self.config.get("data.buffer_size", 1000)
        if len(self.feature_cache[ModalityType.VIDEO]) > max_cache_size:
            self.feature_cache[ModalityType.VIDEO].pop(0)
            
    async def fuse_features(
        self, 
        request: FusionRequest
    ) -> FusionResult:
        """
        融合特征
        
        Args:
            request: 融合请求
            
        Returns:
            FusionResult: 融合结果
        """
        start_time = time.time()
        
        try:
            # 提取特征数据
            audio_features = []
            video_features = []
            
            for feature_data in request.features:
                if feature_data.modality == ModalityType.AUDIO:
                    audio_features.append(feature_data)
                elif feature_data.modality == ModalityType.VIDEO:
                    video_features.append(feature_data)
                    
            # 检查是否有足够的特征数据
            if not audio_features or not video_features:
                raise FusionError("需要音频和视频特征进行融合")
                
            # 时间对齐
            aligned_audio, aligned_video, time_grid = self.temporal_aligner.align_features(
                audio_features, video_features
            )
            
            # 特征融合
            fused_features, quality_score, attention_weights = self.attention_fusion.fuse_features(
                aligned_audio, aligned_video
            )
            
            # 质量控制
            original_features = {
                ModalityType.AUDIO: aligned_audio,
                ModalityType.VIDEO: aligned_video
            }
            quality_metrics = self.quality_controller.evaluate_fusion_quality(
                original_features, fused_features, attention_weights
            )
            
            # 更新质量评分
            quality_score = quality_metrics["overall_quality"]
            
            # 计算处理时间
            processing_time = (time.time() - start_time) * 1000  # 转换为毫秒
            
            # 创建融合结果
            result = FusionResult(
                session_id=request.session_id,
                fused_features=fused_features,
                quality_score=quality_score,
                attention_weights=attention_weights,
                processing_time=processing_time,
                timestamp=datetime.now()
            )
            
            logger.info(f"融合完成，会话ID: {request.session_id}, 质量评分: {quality_score:.2f}")
            return result
            
        except Exception as e:
            logger.error(f"融合失败: {e}")
            raise FusionError(f"融合失败: {str(e)}")
            
    async def get_cached_features(
        self, 
        modality: ModalityType, 
        count: int = 10
    ) -> List[FeatureData]:
        """
        获取缓存的特征数据
        
        Args:
            modality: 模态类型
            count: 获取数量
            
        Returns:
            List[FeatureData]: 特征数据列表
        """
        if modality not in self.feature_cache:
            return []
            
        # 返回最新的count个特征
        return self.feature_cache[modality][-count:]
        
    async def clear_cache(self, modality: Optional[ModalityType] = None):
        """
        清除缓存
        
        Args:
            modality: 要清除的模态类型，None表示清除所有
        """
        if modality is None:
            for key in self.feature_cache:
                self.feature_cache[key] = []
        elif modality in self.feature_cache:
            self.feature_cache[modality] = []
            
        logger.info(f"清除缓存: {modality if modality else '全部'}")
```

```python
import os
import logging
from logging.handlers import RotatingFileHandler
from typing import Optional

def setup_logger(name: Optional[str] = None, level: Optional[str] = None) -> logging.Logger:
    """
    设置日志记录器
    
    Args:
        name: 日志记录器名称
        level: 日志级别
        
    Returns:
        logging.Logger: 配置好的日志记录器
    """
    from .config import get_config
    
    config = get_config()
    log_config = config.get("logging", {})
    
    # 日志级别
    log_level = level or log_config.get("level", "INFO")
    
    # 日志格式
    log_format = log_config.get("format", "%(asctime)s - %(name)s - %(levelname)s - %(message)s")
    
    # 创建日志记录器
    logger = logging.getLogger(name)
    logger.setLevel(getattr(logging, log_level.upper()))
    
    # 清除现有处理器
    logger.handlers.clear()
    
    # 控制台处理器
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(logging.Formatter(log_format))
    logger.addHandler(console_handler)
    
    # 文件处理器
    log_file = log_config.get("file", "logs/multimodal_fusion.log")
    if log_file:
        # 确保日志目录存在
        log_dir = os.path.dirname(log_file)
        if log_dir and not os.path.exists(log_dir):
            os.makedirs(log_dir)
            
        file_handler = RotatingFileHandler(
            log_file, maxBytes=10*1024*1024, backupCount=5
        )
        file_handler.setFormatter(logging.Formatter(log_format))
        logger.addHandler(file_handler)
        
    return logger
```
