```python
# 自然语言理解器核心类设计
class NaturalLanguageUnderstander:
    """自然语言理解器"""
    
    def __init__(self, config: NLUConfig):
        """初始化自然语言理解器"""
        self.config = config
        self._initialize_components()
    
    def understand(self, input_data: TextInput) -> NLUOutput:
        """
        理解文本内容，提取意图、实体和情感
        
        Args:
            input_data: 包含文本和上下文的输入数据
            
        Returns:
            NLUOutput: 包含意图、实体、情感和语义表示的输出数据
            
        Raises:
            NLUError: 自然语言理解过程中的错误
        """
        # 预处理文本
        preprocessed_text = self.preprocessor.process(input_data.text)
        
        # 语义分析
        semantic_features = self.semantic_analyzer.analyze(preprocessed_text)
        
        # 获取上下文
        context = self.context_manager.get_context(input_data.session_id) if input_data.session_id else None
        
        # 并行处理意图识别、实体提取和情感分析
        with ThreadPoolExecutor(max_workers=3) as executor:
            intent_future = executor.submit(self._recognize_intent, preprocessed_text, semantic_features, context)
            entities_future = executor.submit(self._extract_entities, preprocessed_text, semantic_features, context)
            emotion_future = executor.submit(self._analyze_emotion, preprocessed_text, semantic_features, context)
            
            intent = intent_future.result()
            entities = entities_future.result()
            emotion = emotion_future.result()
        
        # 更新上下文
        if input_data.session_id:
            self.context_manager.update_context(input_data.session_id, {
                "text": preprocessed_text,
                "intent": intent,
                "entities": entities,
                "emotion": emotion,
                "timestamp": time.time()
            })
        
        # 构建语义表示
        semantic_representation = self._build_semantic_representation(
            preprocessed_text, semantic_features, intent, entities, emotion
        )
        
        return NLUOutput(
            intent=intent,
            entities=entities,
            emotion=emotion,
            semantic_representation=semantic_representation,
            confidence=self._calculate_confidence(intent, entities, emotion)
        )
    
    def _initialize_components(self):
        """初始化组件"""
        self.model_manager = ModelManager(self.config.model_config)
        self.preprocessor = TextPreprocessor(self.config.nlp_config)
        self.semantic_analyzer = SemanticAnalyzer(self.model_manager, self.config.nlp_config)
        self.intent_classifier = IntentClassifier(self.model_manager, self.config.model_config)
        self.entity_extractor = EntityExtractor(self.model_manager, self.config.model_config)
        self.emotion_analyzer = EmotionAnalyzer(self.model_manager, self.config.model_config)
        self.context_manager = ContextManager(self.config.context_config)
```

```python
# 数据模型定义
from pydantic import BaseModel, Field
from typing import List, Dict, Optional, Any
from enum import Enum

class IntentType(str, Enum):
    """意图类型枚举"""
    HUNGER = "hunger"
    DISCOMFORT = "discomfort"
    ATTENTION = "attention"
    FATIGUE = "fatigue"
    PLAY = "play"
    UNKNOWN = "unknown"

class EmotionType(str, Enum):
    """情感类型枚举"""
    HAPPY = "happy"
    SAD = "sad"
    ANGRY = "angry"
    FEAR = "fear"
    SURPRISE = "surprise"
    NEUTRAL = "neutral"

class EntityType(str, Enum):
    """实体类型枚举"""
    PERSON = "person"
    TIME = "time"
    LOCATION = "location"
    OBJECT = "object"
    ACTION = "action"

class TextInput(BaseModel):
    """输入数据模型"""
    text: str = Field(..., description="待理解的文本")
    session_id: Optional[str] = Field(None, description="会话ID")
    timestamp: Optional[float] = Field(None, description="时间戳")
    user_id: Optional[str] = Field(None, description="用户ID")
    context: Optional[Dict[str, Any]] = Field(None, description="额外上下文信息")

class Entity(BaseModel):
    """实体模型"""
    text: str = Field(..., description="实体文本")
    type: EntityType = Field(..., description="实体类型")
    start: int = Field(..., description="起始位置")
    end: int = Field(..., description="结束位置")
    confidence: float = Field(..., description="置信度")

class Intent(BaseModel):
    """意图模型"""
    type: IntentType = Field(..., description="意图类型")
    confidence: float = Field(..., description="置信度")
    parameters: Optional[Dict[str, Any]] = Field(None, description="意图参数")

class Emotion(BaseModel):
    """情感模型"""
    type: EmotionType = Field(..., description="情感类型")
    confidence: float = Field(..., description="置信度")
    intensity: Optional[float] = Field(None, description="情感强度")

class SemanticRepresentation(BaseModel):
    """语义表示模型"""
    text_vector: List[float] = Field(..., description="文本向量表示")
    keywords: List[str] = Field(..., description="关键词列表")
    topics: List[str] = Field(..., description="主题列表")
    summary: Optional[str] = Field(None, description="文本摘要")

class NLUOutput(BaseModel):
    """自然语言理解输出模型"""
    intent: Intent = Field(..., description="识别的意图")
    entities: List[Entity] = Field(..., description="提取的实体")
    emotion: Emotion = Field(..., description="分析的情感")
    semantic_representation: SemanticRepresentation = Field(..., description="语义表示")
    confidence: float = Field(..., description="整体置信度")
    processing_time: Optional[float] = Field(None, description="处理时间")

class NLUConfig(BaseModel):
    """自然语言理解配置模型"""
    model_config: Dict[str, Any] = Field(..., description="模型配置")
    nlp_config: Dict[str, Any] = Field(..., description="NLP配置")
    context_config: Dict[str, Any] = Field(..., description="上下文配置")
    service_config: Dict[str, Any] = Field(..., description="服务配置")
```

#### 异常处理
```python
# 自定义异常类
class NLUError(Exception):
    """自然语言理解错误基类"""
    pass

class PreprocessingError(NLUError):
    """预处理错误"""
    pass

class ModelLoadError(NLUError):
    """模型加载错误"""
    pass

class InferenceError(NLUError):
    """推理错误"""
    pass

class ContextError(NLUError):
    """上下文管理错误"""
    pass

# 异常处理示例
def understand(self, input_data: TextInput) -> NLUOutput:
    try:
        # 验证输入
        if not input_data.text or not input_data.text.strip():
            raise ValueError("输入文本不能为空")
        
        start_time = time.time()
        
        # 预处理
        try:
            preprocessed_text = self.preprocessor.process(input_data.text)
        except Exception as e:
            logger.error(f"文本预处理失败: {str(e)}")
            raise PreprocessingError(f"文本预处理失败: {str(e)}")
        
        # 语义分析
        try:
            semantic_features = self.semantic_analyzer.analyze(preprocessed_text)
        except Exception as e:
            logger.error(f"语义分析失败: {str(e)}")
            raise InferenceError(f"语义分析失败: {str(e)}")
        
        # 获取上下文
        try:
            context = self.context_manager.get_context(input_data.session_id) if input_data.session_id else None
        except Exception as e:
            logger.error(f"获取上下文失败: {str(e)}")
            raise ContextError(f"获取上下文失败: {str(e)}")
        
        # 并行处理意图识别、实体提取和情感分析
        try:
            with ThreadPoolExecutor(max_workers=3) as executor:
                intent_future = executor.submit(self._recognize_intent, preprocessed_text, semantic_features, context)
                entities_future = executor.submit(self._extract_entities, preprocessed_text, semantic_features, context)
                emotion_future = executor.submit(self._analyze_emotion, preprocessed_text, semantic_features, context)
                
                intent = intent_future.result()
                entities = entities_future.result()
                emotion = emotion_future.result()
        except Exception as e:
            logger.error(f"并行处理失败: {str(e)}")
            raise InferenceError(f"并行处理失败: {str(e)}")
        
        # 更新上下文
        if input_data.session_id:
            try:
                self.context_manager.update_context(input_data.session_id, {
                    "text": preprocessed_text,
                    "intent": intent,
                    "entities": entities,
                    "emotion": emotion,
                    "timestamp": time.time()
                })
            except Exception as e:
                logger.error(f"更新上下文失败: {str(e)}")
                # 上下文更新失败不应影响主要功能，仅记录日志
        
        # 构建语义表示
        try:
            semantic_representation = self._build_semantic_representation(
                preprocessed_text, semantic_features, intent, entities, emotion
            )
        except Exception as e:
            logger.error(f"构建语义表示失败: {str(e)}")
            raise InferenceError(f"构建语义表示失败: {str(e)}")
        
        processing_time = time.time() - start_time
        
        return NLUOutput(
            intent=intent,
            entities=entities,
            emotion=emotion,
            semantic_representation=semantic_representation,
            confidence=self._calculate_confidence(intent, entities, emotion),
            processing_time=processing_time
        )
        
    except NLUError as e:
        logger.error(f"自然语言理解错误: {str(e)}")
        raise
    except Exception as e:
        logger.error(f"未知错误: {str(e)}")
        raise NLUError(f"自然语言理解操作失败: {str(e)}")
```

#### 事件驱动接口
```python
# 事件驱动接口实现
import asyncio
from typing import Callable, Dict, Any
from pydantic import BaseModel

class NLUEvent(BaseModel):
    """自然语言理解事件"""
    event_type: str
    event_data: Dict[str, Any]
    timestamp: float

class NLUEventHandler:
    """自然语言理解事件处理器"""
    
    def __init__(self, nlu_service: NaturalLanguageUnderstander):
        self.nlu_service = nlu_service
        self.event_bus = EventBus()
        self._register_handlers()
    
    def _register_handlers(self):
        """注册事件处理器"""
        self.event_bus.subscribe("text_understand_request", self._handle_understand_request)
        self.event_bus.subscribe("context_update_request", self._handle_context_update)
        self.event_bus.subscribe("model_reload_request", self._handle_model_reload)
    
    async def _handle_understand_request(self, event_data: Dict[str, Any]):
        """处理文本理解请求事件"""
        try:
            # 从事件数据中提取输入
            input_data = TextInput(**event_data.get("input_data"))
            
            # 执行理解
            result = self.nlu_service.understand(input_data)
            
            # 发布结果事件
            result_event = NLUEvent(
                event_type="text_understand_result",
                event_data={
                    "request_id": event_data.get("request_id"),
                    "result": result.dict(),
                    "status": "success"
                },
                timestamp=time.time()
            )
            self.event_bus.publish(result_event)
            
        except Exception as e:
            # 发布错误事件
            error_event = NLUEvent(
                event_type="text_understand_result",
                event_data={
                    "request_id": event_data.get("request_id"),
                    "error": str(e),
                    "status": "error"
                },
                timestamp=time.time()
            )
            self.event_bus.publish(error_event)
    
    async def _handle_context_update(self, event_data: Dict[str, Any]):
        """处理上下文更新请求事件"""
        try:
            session_id = event_data.get("session_id")
            context_data = event_data.get("context_data")
            
            # 更新上下文
            self.nlu_service.context_manager.update_context(session_id, context_data)
            
            # 发布成功事件
            success_event = NLUEvent(
                event_type="context_update_result",
                event_data={
                    "request_id": event_data.get("request_id"),
                    "session_id": session_id,
                    "status": "success"
                },
                timestamp=time.time()
            )
            self.event_bus.publish(success_event)
            
        except Exception as e:
            # 发布错误事件
            error_event = NLUEvent(
                event_type="context_update_result",
                event_data={
                    "request_id": event_data.get("request_id"),
                    "error": str(e),
                    "status": "error"
                },
                timestamp=time.time()
            )
            self.event_bus.publish(error_event)
    
    async def _handle_model_reload(self, event_data: Dict[str, Any]):
        """处理模型重载请求事件"""
        try:
            model_type = event_data.get("model_type")
            model_path = event_data.get("model_path")
            
            # 重载模型
            self.nlu_service.model_manager.reload_model(model_type, model_path)
            
            # 发布成功事件
            success_event = NLUEvent(
                event_type="model_reload_result",
                event_data={
                    "request_id": event_data.get("request_id"),
                    "model_type": model_type,
                    "status": "success"
                },
                timestamp=time.time()
            )
            self.event_bus.publish(success_event)
            
        except Exception as e:
            # 发布错误事件
            error_event = NLUEvent(
                event_type="model_reload_result",
                event_data={
                    "request_id": event_data.get("request_id"),
                    "model_type": model_type,
                    "error": str(e),
                    "status": "error"
                },
                timestamp=time.time()
            )
            self.event_bus.publish(error_event)

# 简化的事件总线实现
class EventBus:
    """事件总线"""
    
    def __init__(self):
        self._handlers = {}
    
    def subscribe(self, event_type: str, handler: Callable):
        """订阅事件"""
        if event_type not in self._handlers:
            self._handlers[event_type] = []
        self._handlers[event_type].append(handler)
    
    def publish(self, event: NLUEvent):
        """发布事件"""
        if event.event_type in self._handlers:
            for handler in self._handlers[event.event_type]:
                asyncio.create_task(handler(event.event_data))
```

## 概述
自然语言理解API提供婴儿语言理解服务，能够识别婴儿表达的意图、提取关键实体和分析情感状态。

## 认证
API使用JWT令牌进行认证。在请求头中添加`Authorization: Bearer <token>`。

## 接口列表

### 文本理解
**请求方式**：POST

**请求URL**：`/api/nlu/understand`

**请求参数**：
| 参数名 | 类型 | 必填 | 描述 |
|--------|------|------|------|
| text | string | 是 | 待理解的文本 |
| session_id | string | 否 | 会话ID，用于多轮对话上下文管理 |
| timestamp | float | 否 | 时间戳 |
| user_id | string | 否 | 用户ID |
| context | object | 否 | 额外上下文信息 |

**请求示例**：
```json
{
  "text": "宝宝饿了，要喝奶",
  "session_id": "session_123",
  "user_id": "user_456"
}
```

**响应参数**：
| 参数名 | 类型 | 描述 |
|--------|------|------|
| intent | object | 识别的意图 |
| entities | array | 提取的实体列表 |
| emotion | object | 分析的情感 |
| semantic_representation | object | 语义表示 |
| confidence | float | 整体置信度 |
| processing_time | float | 处理时间 |

**响应示例**：
```json
{
  "intent": {
    "type": "hunger",
    "confidence": 0.92,
    "parameters": {
      "food": "奶"
    }
  },
  "entities": [
    {
      "text": "宝宝",
      "type": "person",
      "start": 0,
      "end": 2,
      "confidence": 0.95
    },
    {
      "text": "奶",
      "type": "object",
      "start": 8,
      "end": 9,
      "confidence": 0.9
    }
  ],
  "emotion": {
    "type": "neutral",
    "confidence": 0.85,
    "intensity": 0.3
  },
  "semantic_representation": {
    "text_vector": [0.1, 0.2, 0.3, ...],
    "keywords": ["宝宝", "饿", "喝奶"],
    "topics": ["需求", "饮食"]
  },
  "confidence": 0.89,
  "processing_time": 0.15
}
```

**错误码**：
| 错误码 | 描述 |
|--------|------|
| 400 | 请求参数错误或理解失败 |
| 401 | 认证失败 |
| 500 | 服务器内部错误 |

## 使用指南

### 基本使用
```python
from nlu import NaturalLanguageUnderstander, TextInput, NLUConfig

# 创建配置
config = NLUConfig(
    model_config={
        "intent_model_path": "models/intent_model.pt",
        "entity_model_path": "models/entity_model.pt",
        "emotion_model_path": "models/emotion_model.pt"
    },
    nlp_config={
        "language": "zh",
        "spacy_model": "zh_core_web_sm"
    },
    context_config={
        "cache_ttl": 3600,
        "max_context_length": 10
    },
    service_config={
        "max_workers": 3,
        "timeout": 5.0
    }
)

# 创建自然语言理解器
nlu = NaturalLanguageUnderstander(config)

# 理解文本
input_data = TextInput(
    text="宝宝饿了，要喝奶",
    session_id="session_123"
)

result = nlu.understand(input_data)

# 输出结果
print(f"意图: {result.intent.type}")
print(f"实体: {[entity.text for entity in result.entities]}")
print(f"情感: {result.emotion.type}")
print(f"置信度: {result.confidence}")
```

### 高级使用
```python
# 多轮对话理解
session_id = "multi_turn_session"

# 第一轮对话

# 第二轮对话（基于上下文）

context = nlu.context_manager.get_context(session_id)
```

## 功能说明

### 意图识别
自然语言理解模块能够识别以下基本意图：
- **饥饿**：婴儿表达饥饿或需要食物
- **不适**：婴儿表达身体不适或疼痛
- **需要关注**：婴儿表达需要关注或陪伴
- **疲劳**：婴儿表达疲劳或需要休息
- **玩耍**：婴儿表达想要玩耍或娱乐

### 实体提取
自然语言理解模块能够提取以下类型的实体：
- **人物**：如"宝宝"、"妈妈"、"爸爸"等
- **时间**：如"现在"、"刚才"、"晚上"等
- **地点**：如"这里"、"那里"、"床上"等
- **物品**：如"奶"、"玩具"、"衣服"等
- **动作**：如"吃"、"玩"、"睡"等

### 情感分析
自然语言理解模块能够识别以下基本情感：
- **快乐**：表达积极情绪
- **悲伤**：表达消极情绪
- **愤怒**：表达不满或愤怒
- **恐惧**：表达害怕或不安
- **惊讶**：表达惊讶或意外
- **中性**：无明显情感倾向

### 上下文理解
自然语言理解模块支持多轮对话的上下文理解，能够：
- 跟踪对话历史
- 理解指代关系（如"那个"、"这个"）
- 维持对话状态
- 基于上下文进行理解

## 常见问题

### Q1: 如何提高理解准确率？
A1: 
1. 提供清晰、完整的文本输入
2. 使用session_id维持多轮对话上下文
3. 提供额外的上下文信息
4. 定期更新模型

### Q2: 如何处理低置信度结果？
A2: 
1. 检查输入文本是否清晰
2. 查看置信度分数，低于阈值的结果可能不可靠
3. 尝试提供更多上下文信息
4. 考虑使用其他交互方式确认

### Q3: 如何添加新的意图类型？
A3: 
1. 收集新意图类型的训练数据
2. 训练新的意图分类模型
3. 更新模型配置
4. 重新部署服务

### Q4: 如何处理特殊字符或表情符号？
A4: 
1. 模块会自动预处理特殊字符
2. 表情符号会被转换为文本描述
3. 可以在配置中指定特殊字符处理方式

## 故障排除

### 问题1: 理解结果不准确
**可能原因**：
- 输入文本不清晰或不完整
- 缺乏上下文信息
- 模型未覆盖特定表达方式

**解决方案**：
- 提供更清晰、完整的文本
- 使用session_id维持上下文
- 提供额外的上下文信息
- 考虑使用其他交互方式确认

### 问题2: 处理速度慢
**可能原因**：
- 文本过长或复杂
- 系统资源不足
- 网络延迟

**解决方案**：
- 缩短文本长度
- 增加系统资源
- 优化网络连接
- 考虑使用本地部署

### 问题3: 上下文理解错误
**可能原因**：
- session_id不一致
- 上下文过期
- 对话轮次过多

**解决方案**：
- 确保使用相同的session_id
- 检查上下文缓存配置
- 定期清理或重置上下文
