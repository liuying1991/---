# 推理规划模块实现工作流

## 模块信息

### 模块名称
推理规划模块

### 所属子系统
认知决策子系统

### 模块功能
推理规划模块是真实婴儿AI管家系统的智能推理和规划组件，负责基于当前情境和目标，进行逻辑推理和规划未来行动路径。该模块整合多源信息，运用推理算法和规划技术，为AI系统提供智能推理和长期规划能力。

### 模块职责
1. 收集和整合推理规划所需的多源信息
2. 分析当前情境和目标状态
3. 进行逻辑推理和因果分析
4. 规划达成目标的多步骤行动路径
5. 评估规划方案的可行性和效果
6. 动态调整规划以适应环境变化
7. 记录推理过程和规划结果以供学习和优化

### 模块接口
- **输入接口**：情境信息、目标状态、约束条件、历史经验
- **输出接口**：推理结果、规划路径、行动建议、置信度
- **依赖模块**：感知处理模块、记忆存储模块、决策制定模块
- **被依赖模块**：决策执行模块、交互表达模块

        """从原始数据中提取情境因子"""
        factors = []
        
        # 这里可以实现更复杂的因子提取逻辑
        # 例如，从文本中提取情感因子，从图像中提取对象因子等
        
        return factors

### 目标解析组件 (GoalParsing)

```python
# src/components/goal_parsing.py

from typing import Dict, List, Any, Optional
import logging
from datetime import datetime

from ..models.goal import Goal, GoalPriority, GoalStatus
from ..utils.config_manager import ConfigManager
from ..utils.knowledge_graph import KnowledgeGraphClient

logger = logging.getLogger(__name__)

class GoalParsing:
    """目标解析组件，负责解析和明确系统目标"""
    
    def __init__(self, config_manager: ConfigManager, kg_client: KnowledgeGraphClient):
        self.config = config_manager.get_config("reasoning")
        self.kg_client = kg_client
        self.goal_templates = {}
        self._load_goal_templates()
    
    def _load_goal_templates(self):
        """加载目标模板"""
        # 这里可以从配置文件或数据库加载目标模板
        self.goal_templates = {
            "care": {
                "description": "照护目标",
                "keywords": ["照护", "照顾", "护理", "看护"],
                "default_priority": GoalPriority.HIGH
            },
            "learning": {
                "description": "学习目标",
                "keywords": ["学习", "教育", "发展", "成长"],
                "default_priority": GoalPriority.MEDIUM
            },
            "safety": {
                "description": "安全目标",
                "keywords": ["安全", "危险", "风险", "保护"],
                "default_priority": GoalPriority.HIGH
            },
            "comfort": {
                "description": "舒适目标",
                "keywords": ["舒适", "安抚", "休息", "放松"],
                "default_priority": GoalPriority.MEDIUM
            }
        }
    
    def parse(self, goal_data: Dict[str, Any]) -> List[Goal]:
        """
        解析和明确系统目标
        
        Args:
            goal_data: 输入的目标数据
            
        Returns:
            List[Goal]: 解析后的目标列表
        """
        try:
            logger.info(f"开始解析目标: {goal_data.get('id', 'unknown')}")
            
            goals = []
            
            # 处理显式目标
            if 'goals' in goal_data and isinstance(goal_data['goals'], list):
                for explicit_goal in goal_data['goals']:
                    goal = self._parse_explicit_goal(explicit_goal)
                    goals.append(goal)
            
            # 处理隐式目标（从文本或情境中推断）
            if 'text' in goal_data or 'situation' in goal_data:
                implicit_goals = self._infer_implicit_goals(goal_data)
                goals.extend(implicit_goals)
            
            # 如果没有目标，创建默认目标
            if not goals:
                default_goal = self._create_default_goal(goal_data)
                goals.append(default_goal)
            
            # 确定目标优先级和依赖关系
            self._determine_priorities_and_dependencies(goals)
            
            # 存储目标到知识图谱
            for goal in goals:
                self._store_goal(goal)
            
            logger.info(f"目标解析完成，共解析出 {len(goals)} 个目标")
            return goals
            
        except Exception as e:
            logger.error(f"目标解析失败: {str(e)}")
            raise
    
    def _parse_explicit_goal(self, goal_dict: Dict[str, Any]) -> Goal:
        """解析显式目标"""
        goal = Goal(
            id=goal_dict.get('id', f"goal_{datetime.now().timestamp()}"),
            description=goal_dict.get('description', ''),
            priority=self._parse_priority(goal_dict.get('priority', 'medium')),
            status=GoalStatus.ACTIVE,
            target_state=goal_dict.get('target_state', {}),
            constraints=goal_dict.get('constraints', {}),
            deadline=goal_dict.get('deadline'),
            metadata=goal_dict.get('metadata', {})
        )
        
        return goal
    
    def _infer_implicit_goals(self, goal_data: Dict[str, Any]) -> List[Goal]:
        """推断隐式目标"""
        implicit_goals = []
        
        # 获取文本内容
        text = goal_data.get('text', '')
        situation = goal_data.get('situation', {})
        
        # 基于关键词匹配推断目标
        for goal_type, template in self.goal_templates.items():
            keywords = template['keywords']
            
            # 检查文本中的关键词
            text_match = any(keyword in text for keyword in keywords)
            
            # 检查情境中的关键词
            situation_match = False
            if situation:
                situation_text = ' '.join(str(v) for v in situation.values())
                situation_match = any(keyword in situation_text for keyword in keywords)
            
            if text_match or situation_match:
                goal = Goal(
                    id=f"{goal_type}_goal_{datetime.now().timestamp()}",
                    description=f"推断的{template['description']}",
                    priority=template['default_priority'],
                    status=GoalStatus.ACTIVE,
                    target_state={},
                    constraints={},
                    metadata={
                        'inferred': True,
                        'source': 'text' if text_match else 'situation',
                        'matched_keywords': [k for k in keywords if k in text or (situation and k in ' '.join(str(v) for v in situation.values()))]
                    }
                )
                implicit_goals.append(goal)
        
        return implicit_goals
    
    def _create_default_goal(self, goal_data: Dict[str, Any]) -> Goal:
        """创建默认目标"""
        goal = Goal(
            id=f"default_goal_{datetime.now().timestamp()}",
            description="默认目标：维持当前状态",
            priority=GoalPriority.LOW,
            status=GoalStatus.ACTIVE,
            target_state=goal_data.get('situation', {}),
            constraints={},
            metadata={'default': True}
        )
        
        return goal
    
    def _parse_priority(self, priority_str: str) -> GoalPriority:
        """解析优先级字符串"""
        priority_map = {
            'low': GoalPriority.LOW,
            'medium': GoalPriority.MEDIUM,
            'high': GoalPriority.HIGH,
            'critical': GoalPriority.CRITICAL
        }
        
        return priority_map.get(priority_str.lower(), GoalPriority.MEDIUM)
    
    def _determine_priorities_and_dependencies(self, goals: List[Goal]):
        """确定目标优先级和依赖关系"""
        # 按优先级排序
        goals.sort(key=lambda g: g.priority.value, reverse=True)
        
        # 确定依赖关系（基于目标类型和内容）
        for i, goal in enumerate(goals):
            # 高优先级目标可能依赖于低优先级目标的完成
            if i > 0 and goal.priority == GoalPriority.HIGH:
                # 添加对前一个目标的依赖
                goal.dependencies = [goals[i-1].id]
    
    def _store_goal(self, goal: Goal):
        """存储目标到知识图谱"""
        try:
            self.kg_client.store_goal(goal)
        except Exception as e:
            logger.warning(f"存储目标到知识图谱失败: {str(e)}")
    
    def update_goal(self, goal_id: str, updates: Dict[str, Any]) -> Goal:
        """更新目标信息"""
        try:
            # 获取现有目标
            existing_goal = self.kg_client.get_goal(goal_id)
            
            if not existing_goal:
                raise ValueError(f"目标不存在: {goal_id}")
            
            # 应用更新
            for key, value in updates.items():
                if hasattr(existing_goal, key):
                    setattr(existing_goal, key, value)
            
            # 更新知识图谱中的目标
            self.kg_client.update_goal(goal_id, existing_goal)
            
            return existing_goal
            
        except Exception as e:
            logger.error(f"更新目标失败: {str(e)}")
            raise
```

    def reason(self, situation: Situation, goals: List[Goal], reasoning_type: str = "deductive") -> ReasoningResult:
        """
        执行逻辑推理
        
        Args:
            situation: 当前情境
            goals: 目标列表
            reasoning_type: 推理类型
            
        Returns:
            ReasoningResult: 推理结果
        """
        try:
            logger.info(f"开始执行{reasoning_type}推理，情境ID: {situation.id}")
            
            # 获取推理算法
            reasoning_algorithm = self.reasoning_algorithms.get(reasoning_type)
            if not reasoning_algorithm:
                raise ValueError(f"不支持的推理类型: {reasoning_type}")
            
            # 执行推理
            reasoning_result = reasoning_algorithm.reason(situation, goals)
            
            # 存储推理结果到知识图谱
            self._store_reasoning_result(reasoning_result)
            
            logger.info(f"推理完成，结果ID: {reasoning_result.id}")
            return reasoning_result
            
        except Exception as e:
            logger.error(f"推理失败: {str(e)}")
            raise
    
    def reason_with_multiple_types(self, situation: Situation, goals: List[Goal], 
                                 reasoning_types: List[str]) -> Dict[str, ReasoningResult]:
        """
        使用多种推理类型进行推理
        
        Args:
            situation: 当前情境
            goals: 目标列表
            reasoning_types: 推理类型列表
            
        Returns:
            Dict[str, ReasoningResult]: 各推理类型的结果
        """
        results = {}
        
        for reasoning_type in reasoning_types:
            try:
                result = self.reason(situation, goals, reasoning_type)
                results[reasoning_type] = result
            except Exception as e:
                logger.error(f"{reasoning_type}推理失败: {str(e)}")
                # 创建一个失败的结果
                results[reasoning_type] = ReasoningResult(
                    id=f"failed_{reasoning_type}_{datetime.now().timestamp()}",
                    reasoning_type=ReasoningType(reasoning_type),
                    situation_id=situation.id,
                    goal_ids=[goal.id for goal in goals],
                    conclusions=[],
                    confidence=ReasoningConfidence.VERY_LOW,
                    reasoning_steps=[f"推理失败: {str(e)}"],
                    metadata={"error": str(e)}
                )
        
        return results
    
    def compare_reasoning_results(self, results: Dict[str, ReasoningResult]) -> ReasoningResult:
        """
        比较不同推理类型的结果，选择最佳结果
        
        Args:
            results: 各推理类型的结果
            
        Returns:
            ReasoningResult: 最佳推理结果
        """
        if not results:
            raise ValueError("没有推理结果可供比较")
        
        # 按置信度排序
        sorted_results = sorted(
            results.values(),
            key=lambda r: r.confidence.value,
            reverse=True
        )
        
        # 返回置信度最高的结果
        best_result = sorted_results[0]
        
        # 添加比较信息到元数据
        best_result.metadata["comparison"] = {
            "total_results": len(results),
            "all_confidences": {rt: r.confidence.value for rt, r in results.items()},
            "selected_reasoning_type": best_result.reasoning_type.value
        }
        
        return best_result
    
    def _store_reasoning_result(self, reasoning_result: ReasoningResult):
        """存储推理结果到知识图谱"""
        try:
            self.kg_client.store_reasoning_result(reasoning_result)
        except Exception as e:
            logger.warning(f"存储推理结果到知识图谱失败: {str(e)}")
    
    def get_reasoning_history(self, situation_id: Optional[str] = None, 
                            goal_ids: Optional[List[str]] = None,
                            reasoning_type: Optional[str] = None,
                            limit: int = 10) -> List[ReasoningResult]:
        """
        获取推理历史
        
        Args:
            situation_id: 情境ID（可选）
            goal_ids: 目标ID列表（可选）
            reasoning_type: 推理类型（可选）
            limit: 返回结果数量限制
            
        Returns:
            List[ReasoningResult]: 推理历史结果
        """
        try:
            return self.kg_client.get_reasoning_history(
                situation_id=situation_id,
                goal_ids=goal_ids,
                reasoning_type=reasoning_type,
                limit=limit
            )
        except Exception as e:
            logger.error(f"获取推理历史失败: {str(e)}")
            return []
    
    def evaluate_reasoning_quality(self, reasoning_result: ReasoningResult) -> Dict[str, Any]:
        """
        评估推理质量
        
        Args:
            reasoning_result: 推理结果
            
        Returns:
            Dict[str, Any]: 质量评估结果
        """
        quality_metrics = {
            "confidence": reasoning_result.confidence.value,
            "conclusion_count": len(reasoning_result.conclusions),
            "step_count": len(reasoning_result.reasoning_steps),
            "has_explanation": len(reasoning_result.explanation) > 0 if reasoning_result.explanation else False,
            "consistency_score": self._calculate_consistency_score(reasoning_result),
            "novelty_score": self._calculate_novelty_score(reasoning_result)
        }
        
        # 计算综合质量分数
        quality_score = (
            quality_metrics["confidence"] * 0.3 +
            min(quality_metrics["conclusion_count"] / 5, 1.0) * 0.2 +
            min(quality_metrics["step_count"] / 10, 1.0) * 0.2 +
            (1.0 if quality_metrics["has_explanation"] else 0.0) * 0.1 +
            quality_metrics["consistency_score"] * 0.1 +
            quality_metrics["novelty_score"] * 0.1
        )
        
        quality_metrics["overall_score"] = quality_score
        
        return quality_metrics
    
    def _calculate_consistency_score(self, reasoning_result: ReasoningResult) -> float:
        """计算推理一致性分数"""
        # 这里可以实现更复杂的一致性检查逻辑
        # 例如，检查推理步骤之间是否存在矛盾
        return 0.8  # 简化实现
    
    def _calculate_novelty_score(self, reasoning_result: ReasoningResult) -> float:
        """计算推理新颖性分数"""
        # 这里可以实现更复杂的新颖性检查逻辑
        # 例如，与历史推理结果比较，检查是否提供了新的见解
        return 0.7  # 简化实现

class CausalAnalysis:

    def __init__(self, config_manager: ConfigManager, kg_client: KnowledgeGraphClient):
        self.config = config_manager.get_config("reasoning")

    def analyze(self, situation: Situation, events: List[Dict[str, Any]]) -> CausalAnalysisResult:
        """
        分析因果关系
        
        Args:
            situation: 当前情境
            events: 事件列表
            
        Returns:
            CausalAnalysisResult: 因果分析结果
        """
        try:
            logger.info(f"开始因果分析，情境ID: {situation.id}")
            
            # 提取因果因子
            causal_factors = self._extract_causal_factors(events)
            
            # 识别因果对
            causal_pairs = self._identify_causal_pairs(events, causal_factors)
            
            # 构建因果链
            causal_chains = self._build_causal_chains(causal_pairs)
            
            # 评估因果强度
            for chain in causal_chains:
                self._evaluate_causal_strength(chain)
            
            # 创建分析结果
            analysis_result = CausalAnalysisResult(
                id=f"causal_analysis_{datetime.now().timestamp()}",
                situation_id=situation.id,
                causal_relations=[rel for chain in causal_chains for rel in chain.relations],
                causal_chains=causal_chains,
                confidence=self._calculate_overall_confidence(causal_chains),
                timestamp=datetime.now(),
                metadata={
                    "event_count": len(events),
                    "factor_count": len(causal_factors),
                    "chain_count": len(causal_chains)
                }
            )
            
            # 存储分析结果到知识图谱
            self._store_analysis_result(analysis_result)
            
            logger.info(f"因果分析完成，结果ID: {analysis_result.id}")
            return analysis_result
            
        except Exception as e:
            logger.error(f"因果分析失败: {str(e)}")
            raise
    
    def _extract_causal_factors(self, events: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """提取因果因子"""
        factors = []
        
        for event in events:
            # 提取事件中的关键因子
            if 'factors' in event and isinstance(event['factors'], list):
                for factor in event['factors']:
                    factors.append({
                        'event_id': event.get('id'),
                        'factor_name': factor.get('name'),
                        'factor_value': factor.get('value'),
                        'factor_type': factor.get('type', 'unknown')
                    })
        
        return factors
    
    def _identify_causal_pairs(self, events: List[Dict[str, Any]], 
                             causal_factors: List[Dict[str, Any]]) -> List[Tuple[str, str, float]]:
        """识别因果对"""
        causal_pairs = []
        
        # 按时间排序事件
        sorted_events = sorted(events, key=lambda e: e.get('timestamp', datetime.now()))
        
        # 尝试识别事件间的因果关系
        for i in range(len(sorted_events)):
            for j in range(i + 1, len(sorted_events)):
                event_i = sorted_events[i]
                event_j = sorted_events[j]
                
                # 计算因果可能性
                causal_probability = self._calculate_causal_probability(event_i, event_j, causal_factors)
                
                # 如果可能性超过阈值，则认为是因果对
                threshold = self.config.get('causal_threshold', 0.6)
                if causal_probability >= threshold:
                    causal_pairs.append((event_i.get('id'), event_j.get('id'), causal_probability))
        
        return causal_pairs
    
    def _calculate_causal_probability(self, cause_event: Dict[str, Any], 
                                    effect_event: Dict[str, Any],
                                    causal_factors: List[Dict[str, Any]]) -> float:
        """计算因果可能性"""
        # 这里可以实现更复杂的因果计算逻辑
        # 例如，基于时间序列分析、相关性分析等
        
        # 简化实现：基于时间间隔和事件类型
        cause_time = cause_event.get('timestamp', datetime.now())
        effect_time = effect_event.get('timestamp', datetime.now())
        
        # 计算时间间隔（秒）
        if isinstance(cause_time, str):
            cause_time = datetime.fromisoformat(cause_time)
        if isinstance(effect_time, str):
            effect_time = datetime.fromisoformat(effect_time)
            
        time_diff = (effect_time - cause_time).total_seconds()
        
        # 时间间隔太短或太长都不太可能是因果关系
        if time_diff < 1 or time_diff > 3600:  # 1秒到1小时
            return 0.0
        
        # 基于时间间隔计算基础概率
        base_probability = 1.0 - (time_diff / 3600)
        
        # 基于事件类型调整概率
        cause_type = cause_event.get('type', 'unknown')
        effect_type = effect_event.get('type', 'unknown')
        
        # 这里可以添加事件类型之间的因果先验知识
        type_adjustment = self._get_type_causal_adjustment(cause_type, effect_type)
        
        return min(1.0, base_probability * type_adjustment)
    
    def _get_type_causal_adjustment(self, cause_type: str, effect_type: str) -> float:
        """获取事件类型之间的因果调整因子"""
        # 这里可以定义事件类型之间的因果先验知识
        # 例如，"哭闹" -> "安抚" 的可能性较高
        
        type_pairs = {
            ("cry", "comfort"): 0.9,
            ("hunger", "feeding"): 0.95,
            ("discomfort", "adjustment"): 0.85,
            ("sleepiness", "sleep"): 0.9
        }
        
        return type_pairs.get((cause_type, effect_type), 0.5)
    
    def _build_causal_chains(self, causal_pairs: List[Tuple[str, str, float]]) -> List[CausalChain]:
        """构建因果链"""
        chains = []
        
        # 创建因果关系图
        causal_graph = {}
        for cause_id, effect_id, probability in causal_pairs:
            if cause_id not in causal_graph:
                causal_graph[cause_id] = []
            causal_graph[cause_id].append((effect_id, probability))
        
        # 查找因果链
        visited = set()
        
        for node in causal_graph:
            if node not in visited:
                chain = self._find_chain_from_node(node, causal_graph, visited)
                if chain and len(chain.relations) > 1:  # 只保留长度大于1的链
                    chains.append(chain)
        
        return chains
    
    def _find_chain_from_node(self, node: str, causal_graph: Dict[str, List[Tuple[str, float]]], 
                           visited: set) -> Optional[CausalChain]:
        """从节点开始查找因果链"""
        if node in visited:
            return None
        
        visited.add(node)
        
        relations = []
        
        # 深度优先搜索构建链
        if node in causal_graph:
            for next_node, probability in causal_graph[node]:
                relation = CausalRelation(
                    cause_id=node,
                    effect_id=next_node,
                    strength=probability,
                    timestamp=datetime.now()
                )
                relations.append(relation)
                
                # 递归查找下一级
                if next_node not in visited:
                    sub_chain = self._find_chain_from_node(next_node, causal_graph, visited)
                    if sub_chain:
                        relations.extend(sub_chain.relations)
        
        if relations:
            return CausalChain(
                id=f"causal_chain_{datetime.now().timestamp()}",
                relations=relations,
                start_node=node,
                end_node=relations[-1].effect_id if relations else node,
                length=len(relations),
                confidence=sum(r.strength for r in relations) / len(relations)
            )
        
        return None
    
    def _evaluate_causal_strength(self, chain: CausalChain):
        """评估因果强度"""
        # 这里可以实现更复杂的因果强度评估逻辑
        # 例如，考虑多个证据、先验知识等
        
        # 简化实现：基于链长度和平均强度
        length_factor = min(1.0, chain.length / 5)  # 链越长，强度可能越高
        strength_factor = chain.confidence
        
        chain.confidence = (length_factor + strength_factor) / 2
    
    def _calculate_overall_confidence(self, chains: List[CausalChain]) -> float:
        """计算总体置信度"""
        if not chains:
            return 0.0
        
        # 取所有链的平均置信度
        total_confidence = sum(chain.confidence for chain in chains)
        return total_confidence / len(chains)
    
    def _store_analysis_result(self, analysis_result: CausalAnalysisResult):
        """存储分析结果到知识图谱"""
        try:
            self.kg_client.store_causal_analysis_result(analysis_result)
        except Exception as e:
            logger.warning(f"存储因果分析结果到知识图谱失败: {str(e)}")
    
    def predict_effects(self, cause_event: Dict[str, Any], 
                       situation: Situation) -> List[Dict[str, Any]]:
        """
        预测事件的效果
        
        Args:
            cause_event: 原因事件
            situation: 当前情境
            
        Returns:
            List[Dict[str, Any]]: 预测的效果列表
        """
        try:
            logger.info(f"预测事件效果，事件ID: {cause_event.get('id')}")
            
            # 从知识图谱中获取相关的因果链
            causal_chains = self.kg_client.get_causal_chains_for_event(
                cause_event.get('id'), 
                situation.id
            )
            
            # 预测可能的效果
            predicted_effects = []
            
            for chain in causal_chains:
                # 只考虑以当前事件为起点的链
                if chain.start_node == cause_event.get('id'):
                    for relation in chain.relations:
                        effect = {
                            'event_id': relation.effect_id,
                            'probability': relation.strength,
                            'chain_id': chain.id,
                            'chain_confidence': chain.confidence
                        }
                        predicted_effects.append(effect)
            
            # 按概率排序
            predicted_effects.sort(key=lambda e: e['probability'], reverse=True)
            
            return predicted_effects
            
        except Exception as e:
            logger.error(f"预测事件效果失败: {str(e)}")
            return []
    
    def explain_causality(self, cause_id: str, effect_id: str,

        """
        解释因果关系
        
        Args:
            cause_id: 原因ID

            situation: 当前情境

                    'confidence': 0.0,
                    'evidence': []
                }
            
            # 选择置信度最高的链
            best_chain = max(causal_chains, key=lambda c: c.confidence)
            
            # 生成解释
            explanation = f"事件 {cause_id} 可能导致 {effect_id}，"
            explanation += f"基于 {len(best_chain.relations)} 个因果步骤，"
            explanation += f"置信度为 {best_chain.confidence:.2f}"
            
            # 收集证据
            evidence = []
            for relation in best_chain.relations:
                evidence.append({
                    'cause': relation.cause_id,
                    'effect': relation.effect_id,
                    'strength': relation.strength
                })
            
            return {
                'explanation': explanation,
                'confidence': best_chain.confidence,
                'evidence': evidence,
                'chain_length': best_chain.length
            }
            
        except Exception as e:
            logger.error(f"解释因果关系失败: {str(e)}")

                'explanation': f"解释因果关系时出错: {str(e)}",

from typing import Dict, List, Any, Optional, Union

class PathPlanning:
    """路径规划组件，负责规划达成目标的行动路径"""
    
    def __init__(self, config_manager: ConfigManager, kg_client: KnowledgeGraphClient):

            "backward": BackwardPlanning(self.config, self.kg_client),
            "hierarchical": HierarchicalPlanning(self.config, self.kg_client),
            "temporal": TemporalPlanning(self.config, self.kg_client)
        }
    
    def plan(self, situation: Situation, goals: List[Goal], 
             reasoning_result: Optional[ReasoningResult] = None,
             planning_type: str = "hierarchical") -> Plan:
        """
        生成规划方案
        
        Args:
            situation: 当前情境
            goals: 目标列表
            reasoning_result: 推理结果（可选）
            planning_type: 规划类型
            
        Returns:
            Plan: 规划方案
        """
        try:
            logger.info(f"开始{planning_type}规划，情境ID: {situation.id}")
            
            # 获取规划算法
            planning_algorithm = self.planning_algorithms.get(planning_type)
            if not planning_algorithm:
                raise ValueError(f"不支持的规划类型: {planning_type}")
            
            # 执行规划
            plan = planning_algorithm.plan(situation, goals, reasoning_result)
            
            # 存储规划到知识图谱
            self._store_plan(plan)
            
            logger.info(f"规划完成，规划ID: {plan.id}")
            return plan
            
        except Exception as e:
            logger.error(f"规划失败: {str(e)}")
            raise
    
    def compare_planning_approaches(self, situation: Situation, goals: List[Goal],
                                  reasoning_result: Optional[ReasoningResult] = None,
                                  planning_types: List[str] = None) -> Dict[str, Plan]:
        """
        比较不同规划方法的结果
        
        Args:
            situation: 当前情境
            goals: 目标列表
            reasoning_result: 推理结果（可选）
            planning_types: 要比较的规划类型列表
            
        Returns:
            Dict[str, Plan]: 各规划类型的结果
        """
        if planning_types is None:
            planning_types = ["forward", "backward", "hierarchical"]
        
        results = {}
        
        for planning_type in planning_types:
            try:
                plan = self.plan(situation, goals, reasoning_result, planning_type)
                results[planning_type] = plan
            except Exception as e:
                logger.error(f"{planning_type}规划失败: {str(e)}")
                # 创建一个失败的规划
                results[planning_type] = Plan(
                    id=f"failed_{planning_type}_{datetime.now().timestamp()}",
                    situation_id=situation.id,
                    goal_ids=[goal.id for goal in goals],
                    algorithm=PlanningAlgorithm(planning_type),
                    steps=[],
                    status=PlanStatus.FAILED,
                    created_at=datetime.now(),
                    metadata={"error": str(e)}
                )
        
        return results
    
    def select_best_plan(self, plans: Dict[str, Plan]) -> Plan:
        """
        从多个规划中选择最佳方案
        
        Args:
            plans: 各规划类型的结果
            
        Returns:
            Plan: 最佳规划方案
        """
        if not plans:
            raise ValueError("没有规划方案可供选择")
        
        # 过滤掉失败的规划
        valid_plans = [plan for plan in plans.values() if plan.status != PlanStatus.FAILED]
        
        if not valid_plans:
            # 如果所有规划都失败了，选择一个"失败"状态但包含最多信息的规划
            return max(plans.values(), key=lambda p: len(p.metadata.get('error', '')))
        
        # 评估每个规划的质量
        plan_scores = {}
        for plan_type, plan in valid_plans:
            score = self._evaluate_plan_quality(plan)
            plan_scores[plan_type] = score
        
        # 选择得分最高的规划
        best_plan_type = max(plan_scores, key=plan_scores.get)
        best_plan = plans[best_plan_type]
        
        # 添加选择信息到元数据
        best_plan.metadata["selection"] = {
            "total_plans": len(plans),
            "valid_plans": len(valid_plans),
            "all_scores": plan_scores,
            "selected_algorithm": best_plan_type
        }
        
        return best_plan
    
    def _evaluate_plan_quality(self, plan: Plan) -> float:
        """评估规划质量"""
        # 基础分数
        score = 0.5
        
        # 步骤数量适中加分（不太少也不太多）
        step_count = len(plan.steps)
        if 5 <= step_count <= 20:
            score += 0.2
        elif 20 < step_count <= 30:
            score += 0.1
        
        # 有时间规划的加分
        has_time_planning = any(step.scheduled_time for step in plan.steps)
        if has_time_planning:
            score += 0.1
        
        # 有资源规划的加分
        has_resource_planning = any(step.resources for step in plan.steps)
        if has_resource_planning:
            score += 0.1
        
        # 规划长度适中加分
        if hasattr(plan, 'estimated_duration') and plan.estimated_duration:
            # 假设理想时长是30分钟到2小时
            duration_minutes = plan.estimated_duration.total_seconds() / 60
            if 30 <= duration_minutes <= 120:
                score += 0.1
        
        return min(1.0, score)
    
    def _store_plan(self, plan: Plan):
        """存储规划到知识图谱"""
        try:
            self.kg_client.store_plan(plan)
        except Exception as e:
            logger.warning(f"存储规划到知识图谱失败: {str(e)}")
    
    def get_plan_history(self, situation_id: Optional[str] = None,

        """

        except Exception as e:

            return []
    
    def update_plan(self, plan_id: str, updates: Dict[str, Any]) -> Plan:
        """更新规划信息"""
        try:
            # 获取现有规划
            existing_plan = self.kg_client.get_plan(plan_id)
            
            if not existing_plan:
                raise ValueError(f"规划不存在: {plan_id}")
            
            # 应用更新
            for key, value in updates.items():
                if hasattr(existing_plan, key):
                    setattr(existing_plan, key, value)
            
            # 更新知识图谱中的规划
            self.kg_client.update_plan(plan_id, existing_plan)
            
            return existing_plan
            
        except Exception as e:
            logger.error(f"更新规划失败: {str(e)}")
            raise
    
    def execute_step(self, plan_id: str, step_id: str, 
                    execution_result: Dict[str, Any]) -> Plan:
        """执行规划步骤并更新规划状态"""
        try:
            # 获取规划
            plan = self.kg_client.get_plan(plan_id)
            if not plan:
                raise ValueError(f"规划不存在: {plan_id}")
            
            # 查找步骤
            step = None
            for s in plan.steps:
                if s.id == step_id:
                    step = s
                    break
            
            if not step:
                raise ValueError(f"步骤不存在: {step_id}")
            
            # 更新步骤状态
            step.status = execution_result.get('status', 'completed')
            step.actual_end_time = datetime.now()
            step.execution_result = execution_result
            
            # 检查是否所有前置步骤都已完成
            all_prerequisites_completed = all(
                ps.status == 'completed' 
                for ps in plan.steps 
                if ps.id in step.prerequisites
            )
            
            # 如果所有前置步骤都已完成，更新后续步骤的状态
            if all_prerequisites_completed:
                for s in plan.steps:
                    if step_id in s.prerequisites and s.status == 'pending':
                        s.status = 'ready'
            
            # 检查是否所有步骤都已完成
            all_steps_completed = all(
                s.status in ['completed', 'failed', 'skipped'] 
                for s in plan.steps
            )
            
            if all_steps_completed:
                plan.status = PlanStatus.COMPLETED
                plan.completed_at = datetime.now()
            elif any(s.status == 'failed' for s in plan.steps):
                plan.status = PlanStatus.FAILED
                plan.failed_at = datetime.now()
            
            # 更新知识图谱中的规划
            self.kg_client.update_plan(plan_id, plan)
            
            return plan
            
        except Exception as e:
            logger.error(f"执行规划步骤失败: {str(e)}")
            raise
    
    def get_next_steps(self, plan_id: str) -> List[PlanStep]:
        """获取规划的下一步可执行步骤"""
        try:
            # 获取规划
            plan = self.kg_client.get_plan(plan_id)
            if not plan:
                raise ValueError(f"规划不存在: {plan_id}")
            
            # 获取所有可执行的步骤（状态为ready且所有前置步骤已完成）
            ready_steps = []
            
            for step in plan.steps:
                if step.status == 'ready':
                    # 检查所有前置步骤是否已完成
                    all_prerequisites_completed = all(
                        ps.status == 'completed' 
                        for ps in plan.steps 
                        if ps.id in step.prerequisites
                    )
                    
                    if all_prerequisites_completed:
                        ready_steps.append(step)
            
            # 按优先级排序
            ready_steps.sort(key=lambda s: s.priority, reverse=True)
            
            return ready_steps
            
        except Exception as e:
            logger.error(f"获取下一步步骤失败: {str(e)}")
            return []

        Args:
            plans: 规划方案列表
            situation: 当前情境
            goals: 目标列表

        best_plan_id = max(
            evaluation_results,

        )
        
        best_plan = None
        for plan in plans:
            if plan.id == best_plan_id:
                best_plan = plan
                break
        
        if not best_plan:
            raise ValueError(f"找不到ID为 {best_plan_id} 的规划")
        
        return best_plan, evaluation_results[best_plan_id]
