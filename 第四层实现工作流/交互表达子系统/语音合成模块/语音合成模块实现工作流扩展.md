## 9. 性能优化

### 9.1 模型优化

1. **模型量化**：使用INT8量化减少模型大小和推理时间
2. **模型剪枝**：移除不重要的权重，减少计算量
3. **知识蒸馏**：使用大型模型训练小型模型，保持性能的同时减少计算量
4. **模型融合**：将多个模型合并为一个，减少模型切换开销

```python
# utils/model_utils.py
import torch
import torch.nn as nn
from typing import Dict, Any, Optional
import logging

logger = logging.getLogger(__name__)

def quantize_model(model: nn.Module, calibration_data: torch.Tensor) -> nn.Module:
    """量化模型"""
    # 准备量化
    model.eval()
    model.qconfig = torch.quantization.get_default_qconfig('fbgemm')
    torch.quantization.prepare(model, inplace=True)
    
    # 校准
    with torch.no_grad():
        model(calibration_data)
    
    # 转换为量化模型
    quantized_model = torch.quantization.convert(model, inplace=False)
    
    logger.info("Model quantized successfully")
    return quantized_model

def prune_model(model: nn.Module, pruning_ratio: float = 0.2) -> nn.Module:
    """剪枝模型"""
    import torch.nn.utils.prune as prune
    
    # 全局结构化剪枝
    parameters_to_prune = []
    for name, module in model.named_modules():
        if isinstance(module, (nn.Conv1d, nn.Conv2d, nn.Linear)):
            parameters_to_prune.append((module, 'weight'))
    
    prune.global_unstructured(
        parameters_to_prune,
        pruning_method=prune.L1Unstructured,
        amount=pruning_ratio,
    )
    
    # 移除剪枝重参数化
    for module, param_name in parameters_to_prune:
        prune.remove(module, param_name)
    
    logger.info(f"Model pruned with ratio {pruning_ratio}")
    return model

def optimize_for_inference(model: nn.Module) -> nn.Module:
    """优化模型以加速推理"""
    # 融合卷积、批归一化和激活函数
    model = torch.jit.script(model)
    
    # 优化模型
    model = torch.jit.optimize_for_inference(model)
    
    logger.info("Model optimized for inference")
    return model
```

```python
# storage/cache_storage.py
import os
import hashlib
import pickle
import torch
from typing import Dict, Any, Optional, Tuple
import logging

logger = logging.getLogger(__name__)

class CacheStorage:
    """缓存存储"""
    
    def __init__(self, cache_dir: str = "./cache"):
        self.cache_dir = cache_dir
        os.makedirs(cache_dir, exist_ok=True)
        
        # 内存缓存
        self.memory_cache = {}
        self.max_memory_cache_size = 100
        
        logger.info(f"CacheStorage initialized with cache_dir: {cache_dir}")
    
    def _get_cache_key(self, data: Dict[str, Any]) -> str:
        """生成缓存键"""
        # 将数据序列化为字符串
        data_str = str(sorted(data.items()))
        
        # 生成MD5哈希
        return hashlib.md5(data_str.encode()).hexdigest()
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        # 先检查内存缓存
        if key in self.memory_cache:
            return self.memory_cache[key]
        
        # 检查磁盘缓存
        cache_path = os.path.join(self.cache_dir, f"{key}.pkl")
        if os.path.exists(cache_path):
            try:
                with open(cache_path, 'rb') as f:
                    data = pickle.load(f)
                
                # 添加到内存缓存
                self._add_to_memory_cache(key, data)
                
                return data
            except Exception as e:
                logger.error(f"Error loading cache: {e}")
        
        return None
    
    def set(self, key: str, data: Any) -> None:
        """设置缓存"""
        # 添加到内存缓存
        self._add_to_memory_cache(key, data)
        
        # 保存到磁盘缓存
        cache_path = os.path.join(self.cache_dir, f"{key}.pkl")
        try:
            with open(cache_path, 'wb') as f:
                pickle.dump(data, f)
        except Exception as e:
            logger.error(f"Error saving cache: {e}")
    
    def _add_to_memory_cache(self, key: str, data: Any) -> None:
        """添加到内存缓存"""
        # 如果缓存已满，移除最旧的条目
        if len(self.memory_cache) >= self.max_memory_cache_size:
            oldest_key = next(iter(self.memory_cache))
            del self.memory_cache[oldest_key]
        
        self.memory_cache[key] = data
    
    def clear(self) -> None:
        """清空缓存"""
        self.memory_cache.clear()
        
        # 清空磁盘缓存
        for filename in os.listdir(self.cache_dir):
            if filename.endswith('.pkl'):
                os.remove(os.path.join(self.cache_dir, filename))
        
        logger.info("Cache cleared")
```

```python
# utils/concurrency_utils.py
import asyncio
import threading
from typing import List, Dict, Any, Callable
import logging
import time

logger = logging.getLogger(__name__)

class BatchProcessor:
    """批处理器"""
    
    def __init__(self, batch_size: int = 4, max_wait_time: float = 0.1):
        self.batch_size = batch_size
        self.max_wait_time = max_wait_time
        self.batch = []
        self.batch_lock = threading.Lock()
        self.batch_event = threading.Event()
        self.processing_thread = threading.Thread(target=self._process_batch)
        self.processing_thread.daemon = True
        self.processing_thread.start()
        
        logger.info(f"BatchProcessor initialized with batch_size: {batch_size}")
    
    def add_request(self, request_data: Dict[str, Any]) -> asyncio.Future:
        """添加请求到批处理"""
        future = asyncio.Future()
        
        with self.batch_lock:
            self.batch.append((request_data, future))
            
            # 如果批次已满，触发处理
            if len(self.batch) >= self.batch_size:
                self.batch_event.set()
        
        return future
    
    def _process_batch(self):
        """处理批次"""
        while True:
            # 等待批次事件或超时
            self.batch_event.wait(self.max_wait_time)
            
            with self.batch_lock:
                if not self.batch:
                    self.batch_event.clear()
                    continue
                
                # 获取当前批次
                current_batch = self.batch.copy()
                self.batch = []
                self.batch_event.clear()
            
            # 处理批次
            try:
                self._process_batch_data(current_batch)
            except Exception as e:
                logger.error(f"Error processing batch: {e}")
                
                # 设置所有future为异常
                for _, future in current_batch:
                    if not future.done():
```

```python
import time
import psutil
import torch
from typing import Dict, Any, Optional
import logging
from functools import wraps

logger = logging.getLogger(__name__)

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self):
        self.metrics = {}
        self.start_time = time.time()
    
    def record_metric(self, name: str, value: float) -> None:
        """记录指标"""
        if name not in self.metrics:
            self.metrics[name] = []
        
        self.metrics[name].append(value)
        
        # 只保留最近100个值
        if len(self.metrics[name]) > 100:
            self.metrics[name] = self.metrics[name][-100:]
    
    def get_metric_stats(self, name: str) -> Optional[Dict[str, float]]:
        """获取指标统计"""
        if name not in self.metrics or not self.metrics[name]:
            return None
        
        values = self.metrics[name]
        return {
            "count": len(values),
            "avg": sum(values) / len(values),
            "min": min(values),
            "max": max(values),
            "latest": values[-1]
        }
    
    def get_system_metrics(self) -> Dict[str, Any]:
        """获取系统指标"""
        # CPU使用率
        cpu_percent = psutil.cpu_percent(interval=1)
        
        # 内存使用情况
        memory = psutil.virtual_memory()
        
        # GPU使用情况（如果有）
        gpu_metrics = {}
        if torch.cuda.is_available():
            for i in range(torch.cuda.device_count()):
                gpu_memory = torch.cuda.get_device_properties(i).total_memory
                gpu_memory_used = torch.cuda.memory_allocated(i)
                gpu_utilization = (gpu_memory_used / gpu_memory) * 100
                
                gpu_metrics[f"gpu_{i}_memory_utilization"] = gpu_utilization
        
        # 运行时间
        uptime = time.time() - self.start_time
        
        return {
            "cpu_percent": cpu_percent,
            "memory_percent": memory.percent,
            "memory_used_gb": memory.used / (1024**3),
            "memory_total_gb": memory.total / (1024**3),
            "uptime_seconds": uptime,
            **gpu_metrics
        }

                    logger.error(f"Error in {metric_name}: {error}")
            
            return result
        return wrapper
    return decorator
```

```python
# utils/logging_config.py
import logging
import logging.handlers
import os
from typing import Dict, Any

def setup_logging(
    log_level: str = "INFO",
    log_file: str = "tts.log",
    max_bytes: int = 10 * 1024 * 1024,  # 10MB
    backup_count: int = 5
) -> None:
    """设置日志配置"""
    # 创建日志目录
    log_dir = os.path.dirname(log_file)
    if log_dir and not os.path.exists(log_dir):
        os.makedirs(log_dir)
    
    # 设置日志级别
    numeric_level = getattr(logging, log_level.upper(), None)
    if not isinstance(numeric_level, int):
        raise ValueError(f'Invalid log level: {log_level}')
    
    # 创建根日志记录器
    root_logger = logging.getLogger()
    root_logger.setLevel(numeric_level)
    
    # 创建格式化器
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    
    # 创建文件处理器（带轮转）
    file_handler = logging.handlers.RotatingFileHandler(
        log_file, maxBytes=max_bytes, backupCount=backup_count
    )
    file_handler.setLevel(numeric_level)
    file_handler.setFormatter(formatter)
    root_logger.addHandler(file_handler)
    
    # 创建控制台处理器
    console_handler = logging.StreamHandler()
    console_handler.setLevel(numeric_level)
    console_handler.setFormatter(formatter)
    root_logger.addHandler(console_handler)
    
    # 设置特定模块的日志级别
    logging.getLogger('urllib3.connectionpool').setLevel(logging.WARNING)
    logging.getLogger('transformers').setLevel(logging.WARNING)
    logging.getLogger('torch').setLevel(logging.WARNING)
```
