# 交互控制模块6A工作流

## 模块概述

### 模块名称
交互控制模块

### 所属子系统
交互表达子系统

### 模块功能描述
交互控制模块负责控制交互表达子系统的整体交互流程，管理交互状态转换，处理用户输入，协调各表达模块的执行顺序和时机，确保交互过程的流畅性和一致性。该模块还负责交互策略的执行和调整，以及交互异常的处理和恢复。

### 模块职责
1. 交互流程控制：控制交互流程的执行顺序和时机
2. 状态管理：管理交互状态和状态转换
3. 用户输入处理：处理用户输入和触发事件
4. 模块协调：协调各表达模块的执行
5. 策略执行：执行和调整交互策略
6. 异常处理：处理交互异常和恢复

### 模块接口
- **输入接口**：用户输入、系统事件、交互请求
- **输出接口**：控制指令、状态更新、异常通知
- **依赖模块**：交互管理模块、表情生成模块、语音合成模块、肢体动作生成模块
- **被依赖模块**：用户界面模块、感知处理模块

# 交互控制器实现
class InteractionController:
    def __init__(self, input_handler, state_manager, flow_controller, module_coordinator,
                 strategy_executor, exception_handler, context_manager, history_recorder, output_formatter):
        self.input_handler = input_handler
        self.state_manager = state_manager
        self.flow_controller = flow_controller
        self.module_coordinator = module_coordinator
        self.strategy_executor = strategy_executor
        self.exception_handler = exception_handler
        self.context_manager = context_manager
        self.history_recorder = history_recorder
        self.output_formatter = output_formatter
        
        # 初始化事件循环
        self.event_loop = asyncio.get_event_loop()
        self._lock = asyncio.Lock()
        
        # 初始化状态
        self.active_sessions = {}
        
    async def start_session(self, session_config):
        try:
            # 生成会话ID
            session_id = str(uuid.uuid4())
            
            # 创建会话状态
            session_state = await self.state_manager.create_session_state(session_id, session_config)
            
            # 初始化交互上下文
            await self.context_manager.initialize_context(session_id, session_config)
            
            # 启动交互流程
            flow_config = session_config.get("flow_config", {})
            await self.flow_controller.start_flow(session_id, flow_config)
            
            # 记录会话启动
            await self.history_recorder.record_session_start(session_id, session_config)
            
            # 添加到活跃会话
            async with self._lock:
                self.active_sessions[session_id] = {
                    "state": session_state,
                    "start_time": time.time(),
                    "last_activity": time.time()
                }
            
            return {
                "session_id": session_id,
                "status": "started",
                "timestamp": time.time()
            }
            
        except Exception as e:
            return await self.exception_handler.handle_exception("start_session", e, session_config)
    
    async def process_input(self, session_id, user_input):
        try:
            # 检查会话是否存在
            if session_id not in self.active_sessions:
                return {"error": "Session not found"}
            
            # 更新会话活动时间
            async with self._lock:
                self.active_sessions[session_id]["last_activity"] = time.time()
            
            # 处理用户输入
            processed_input = await self.input_handler.process_input(session_id, user_input)
            
            # 获取当前状态
            current_state = await self.state_manager.get_session_state(session_id)
            
            # 更新上下文
            await self.context_manager.update_context(session_id, {
                "last_input": processed_input,
                "timestamp": time.time()
            })
            
            # 处理输入事件
            control_result = await self._process_input_event(session_id, processed_input, current_state)
            
            # 记录输入处理
            await self.history_recorder.record_input_processing(session_id, user_input, processed_input, control_result)
            
            # 更新会话活动时间
            async with self._lock:
                self.active_sessions[session_id]["last_activity"] = time.time()
            
            return control_result
            
        except Exception as e:
            return await self.exception_handler.handle_exception("process_input", e, {"session_id": session_id, "user_input": user_input})
    
    async def _process_input_event(self, session_id, processed_input, current_state):
        # 获取当前流程
        current_flow = await self.flow_controller.get_current_flow(session_id)
        
        # 获取当前策略
        current_strategy = await self.strategy_executor.get_current_strategy(session_id)
        
        # 处理输入事件
        event_result = await self.flow_controller.process_event(session_id, processed_input, current_state, current_flow)
        
        # 根据事件结果执行相应操作
        if event_result["action"] == "state_transition":
            # 执行状态转换
            transition_result = await self.state_manager.transition_state(
                session_id, current_state["state"], event_result["target_state"]
            )
            
            # 更新上下文
            await self.context_manager.update_context(session_id, {
                "state_transition": transition_result,
                "timestamp": time.time()
            })
            
            # 执行状态转换后的操作
            post_transition_result = await self._execute_post_transition_actions(
                session_id, transition_result, current_strategy
            )
            
            return {
                "action": "state_transition",
                "transition_result": transition_result,
                "post_transition_result": post_transition_result,
                "timestamp": time.time()
            }
            
        elif event_result["action"] == "module_coordination":
            # 协调模块执行
            coordination_result = await self.module_coordinator.coordinate_modules(
                session_id, event_result["modules"], event_result["params"], current_strategy
            )
            
            # 更新上下文
            await self.context_manager.update_context(session_id, {
                "module_coordination": coordination_result,
                "timestamp": time.time()
            })
            
            return {
                "action": "module_coordination",
                "coordination_result": coordination_result,
                "timestamp": time.time()
            }
            
        elif event_result["action"] == "strategy_adjustment":
            # 调整策略
            adjustment_result = await self.strategy_executor.adjust_strategy(
                session_id, current_strategy, event_result["adjustment_params"]
            )
            
            # 更新上下文
            await self.context_manager.update_context(session_id, {
                "strategy_adjustment": adjustment_result,
                "timestamp": time.time()
            })
            
            return {
                "action": "strategy_adjustment",
                "adjustment_result": adjustment_result,
                "timestamp": time.time()
            }
            
        else:
            # 默认操作
            return {
                "action": "default",
                "event_result": event_result,
                "timestamp": time.time()
            }
    
    async def _execute_post_transition_actions(self, session_id, transition_result, strategy):
        # 执行状态转换后的操作
        actions = []
        
        # 根据目标状态执行相应操作
        target_state = transition_result["target_state"]
        
        if target_state == "story_telling":
            # 故事讲述状态
            action = await self.module_coordinator.coordinate_modules(
                session_id, ["expression", "voice", "action"], {"mode": "story_telling"}, strategy
            )
            actions.append(action)
            
        elif target_state == "song_singing":
            # 歌谣演唱状态
            action = await self.module_coordinator.coordinate_modules(
                session_id, ["expression", "voice", "action"], {"mode": "song_singing"}, strategy
            )
            actions.append(action)
            
        elif target_state == "interactive_qa":
            # 互动问答状态
            action = await self.module_coordinator.coordinate_modules(
                session_id, ["expression", "voice"], {"mode": "interactive_qa"}, strategy
            )
            actions.append(action)
            
        elif target_state == "emotion_expression":
            # 情感表达状态
            action = await self.module_coordinator.coordinate_modules(
                session_id, ["expression", "voice", "action"], {"mode": "emotion_expression"}, strategy
            )
            actions.append(action)
        
        return {
            "actions": actions,
            "timestamp": time.time()
        }
    
    async def end_session(self, session_id):
        try:
            # 检查会话是否存在
            if session_id not in self.active_sessions:
                return {"error": "Session not found"}
            
            # 停止交互流程
            await self.flow_controller.stop_flow(session_id)
            
            # 清理会话状态
            await self.state_manager.cleanup_session_state(session_id)
            
            # 清理上下文
            await self.context_manager.cleanup_context(session_id)
            
            # 记录会话结束
            await self.history_recorder.record_session_end(session_id)
            
            # 从活跃会话中移除
            async with self._lock:
                session_info = self.active_sessions.pop(session_id, None)
            
            return {
                "session_id": session_id,
                "status": "ended",
                "duration": time.time() - session_info["start_time"] if session_info else 0,
                "timestamp": time.time()
            }
            
        except Exception as e:
            return await self.exception_handler.handle_exception("end_session", e, {"session_id": session_id})

# 状态管理器实现
class StateManager:
    def __init__(self, redis_client, transition_rules):
        self.redis_client = redis_client
        self.transition_rules = transition_rules
        
    async def create_session_state(self, session_id, session_config):
        # 创建会话状态
        initial_state = {
            "session_id": session_id,
            "state": session_config.get("initial_state", "idle"),
            "created_at": time.time(),
            "updated_at": time.time(),
            "config": session_config
        }
        
        # 存储到Redis
        await self.redis_client.set(
            f"session_state:{session_id}",
            json.dumps(initial_state),
            ex=3600  # 1小时过期
        )
        
        return initial_state
    
    async def get_session_state(self, session_id):
        # 获取会话状态
        state_json = await self.redis_client.get(f"session_state:{session_id}")
        if not state_json:
            return None
        
        return json.loads(state_json)
    
    async def update_session_state(self, session_id, updates):
        # 更新会话状态
        current_state = await self.get_session_state(session_id)
        if not current_state:
            return None
        
        # 应用更新
        current_state.update(updates)
        current_state["updated_at"] = time.time()
        
        # 存储到Redis
        await self.redis_client.set(
            f"session_state:{session_id}",
            json.dumps(current_state),
            ex=3600  # 1小时过期
        )
        
        return current_state
    
    async def transition_state(self, session_id, current_state, target_state):
        # 验证状态转换是否合法
        if not self._is_valid_transition(current_state, target_state):
            return {
                "success": False,
                "error": f"Invalid state transition from {current_state} to {target_state}"
            }
        
        # 执行状态转换
        transition_result = await self.update_session_state(session_id, {
            "state": target_state,
            "previous_state": current_state,
            "transition_time": time.time()
        })
        
        return {
            "success": True,
            "from_state": current_state,
            "to_state": target_state,
            "transition_time": time.time()
        }
    
    def _is_valid_transition(self, current_state, target_state):
        # 检查状态转换是否合法
        if current_state == target_state:
            return True  # 自我转换是合法的
        
        # 检查转换规则
        valid_transitions = self.transition_rules.get(current_state, [])
        return target_state in valid_transitions
    
    async def cleanup_session_state(self, session_id):
        # 清理会话状态
        await self.redis_client.delete(f"session_state:{session_id}")

# 流程控制器实现
class FlowController:
    def __init__(self, flow_definitions, event_processor):
        self.flow_definitions = flow_definitions
        self.event_processor = event_processor
        self.active_flows = {}
        
    async def start_flow(self, session_id, flow_config):
        # 获取流程定义
        flow_type = flow_config.get("type", "default")
        flow_definition = self.flow_definitions.get(flow_type)
        if not flow_definition:
            flow_definition = self.flow_definitions["default"]
        
        # 创建流程实例
        flow_instance = {
            "session_id": session_id,
            "type": flow_type,
            "definition": flow_definition,
            "current_step": 0,
            "started_at": time.time(),
            "config": flow_config
        }
        
        # 存储活跃流程
        self.active_flows[session_id] = flow_instance
        
        return {
            "session_id": session_id,
            "flow_type": flow_type,
            "status": "started",
            "timestamp": time.time()
        }
    
    async def get_current_flow(self, session_id):
        # 获取当前流程
        return self.active_flows.get(session_id)
    
    async def process_event(self, session_id, processed_input, current_state, current_flow):
        # 处理事件
        if not current_flow:
            # 没有活跃流程，使用默认处理
            return {
                "action": "default",
                "processed_input": processed_input,
                "timestamp": time.time()
            }
        
        # 获取当前步骤
        current_step = current_flow["definition"]["steps"][current_flow["current_step"]]
        
        # 处理事件
        event_result = await self.event_processor.process_event(
            session_id, processed_input, current_state, current_step
        )
        
        # 根据事件结果更新流程
        if event_result.get("next_step") is not None:
            # 移动到下一步
            current_flow["current_step"] = event_result["next_step"]
            
            # 检查是否完成
            if current_flow["current_step"] >= len(current_flow["definition"]["steps"]):
                # 流程完成
                current_flow["current_step"] = 0  # 循环回到第一步
                event_result["flow_completed"] = True
        
        return event_result
    
    async def stop_flow(self, session_id):
        # 停止流程
        if session_id in self.active_flows:
            flow_info = self.active_flows.pop(session_id)
            return {
                "session_id": session_id,
                "flow_type": flow_info["type"],
                "status": "stopped",
                "duration": time.time() - flow_info["started_at"],
                "timestamp": time.time()
            }
        else:
            return {
                "session_id": session_id,
                "status": "not_found",
                "timestamp": time.time()
            }

# 模块协调器实现
class ModuleCoordinator:
    def __init__(self, expression_module, voice_module, action_module, synchronization_manager):
        self.expression_module = expression_module
        self.voice_module = voice_module
        self.action_module = action_module
        self.synchronization_manager = synchronization_manager
        
    async def coordinate_modules(self, session_id, modules, params, strategy):
        # 准备模块执行任务
        tasks = []
        module_results = {}
        
        # 表情模块任务
        if "expression" in modules:
            task = asyncio.create_task(
                self.expression_module.execute(
                    session_id, params.get("expression_params", {}), strategy
                )
            )
            tasks.append(("expression", task))
        
        # 语音模块任务
        if "voice" in modules:
            task = asyncio.create_task(
                self.voice_module.execute(
                    session_id, params.get("voice_params", {}), strategy
                )
            )
            tasks.append(("voice", task))
        
        # 动作模块任务
        if "action" in modules:
            task = asyncio.create_task(
                self.action_module.execute(
                    session_id, params.get("action_params", {}), strategy
                )
            )
            tasks.append(("action", task))
        
        # 等待所有任务完成
        for module_name, task in tasks:
            try:
                result = await task
                module_results[module_name] = result
            except Exception as e:
                module_results[module_name] = {"error": str(e)}
        
        # 同步模块执行结果
        synchronized_results = await self.synchronization_manager.synchronize_results(
            session_id, module_results, params, strategy
        )
        
        return {
            "session_id": session_id,
            "modules": modules,
            "params": params,
            "results": synchronized_results,
            "timestamp": time.time()
        }
